47template <class LocalGridType, std::size_t NSD = LocalGridType::num_space_dim>
48std::enable_if_t<3 == NSD, std::vector<int>>
49getTopology( const LocalGridType& local_grid )
50{
51 std::vector<int> topology( 27, -1 );
52 int nr = 0;
53 for ( int k = -1; k < 2; ++k )
54 for ( int j = -1; j < 2; ++j )
55 for ( int i = -1; i < 2; ++i, ++nr )
56 topology[nr] = local_grid.neighborRank( i, j, k );
57 return topology;
58}

82// Locate the particles in the local grid and get their destination rank.
83// Particles are assumed to only migrate to a location in the nearest
84// neighbor halo or stay on this rank. If the particle crosses a global
85// periodic boundary, wrap it's coordinates back into the domain.
86template <class LocalGridType, class PositionSliceType, class NeighborRankView,
87 class DestinationRankView>
88void getMigrateDestinations( const LocalGridType& local_grid,
89 const NeighborRankView& neighbor_ranks,
90 DestinationRankView& destinations,
91 PositionSliceType& positions )
92{
93 static constexpr std::size_t num_space_dim = LocalGridType::num_space_dim;
94 using execution_space = typename PositionSliceType::execution_space;
95
96 // Check within the local domain.
97 const auto& local_mesh = createLocalMesh<Kokkos::HostSpace>( local_grid );
98
99 // Use global domain for periodicity.
100 const auto& global_grid = local_grid.globalGrid();
101 const auto& global_mesh = global_grid.globalMesh();
102
103 Kokkos::Array<double, num_space_dim> local_low{};
104 Kokkos::Array<double, num_space_dim> local_high{};
105 Kokkos::Array<bool, num_space_dim> periodic{};
106 Kokkos::Array<double, num_space_dim> global_low{};
107 Kokkos::Array<double, num_space_dim> global_high{};
108 Kokkos::Array<double, num_space_dim> global_extent{};
109
110 for ( std::size_t d = 0; d < num_space_dim; ++d )
111 {
112 local_low[d] = local_mesh.lowCorner( Own(), d );
113 local_high[d] = local_mesh.highCorner( Own(), d );
114 periodic[d] = global_grid.isPeriodic( d );
115 global_low[d] = global_mesh.lowCorner( d );
116 global_high[d] = global_mesh.highCorner( d );
117 global_extent[d] = global_mesh.extent( d );
118 }
119
120 Kokkos::parallel_for(
121 "Cabana::Grid::ParticleGridMigrate::get_destinations",
122 Kokkos::RangePolicy<execution_space>( 0, positions.size() ),
123 KOKKOS_LAMBDA( const int p ) {
124 // Compute the logical index of the neighbor we are sending to.
125 int nid[num_space_dim];
126 for ( std::size_t d = 0; d < num_space_dim; ++d )
127 {
128 nid[d] = 1;
129 if ( positions( p, d ) < local_low[d] )
130 nid[d] = 0;
131 else if ( positions( p, d ) > local_high[d] )
132 nid[d] = 2;
133 }
134
135 // Compute the destination MPI rank [ni + 3*(nj + 3*nk) in 3d].
136 int neighbor_index = nid[0];
137 for ( std::size_t d = 1; d < num_space_dim; ++d )
138 {
139 int npower = 1;
140 for ( std::size_t dp = 1; dp <= d; ++dp )
141 npower *= 3;
142 neighbor_index += npower * nid[d];
143 }
144 destinations( p ) = neighbor_ranks( neighbor_index );
145
146 // Shift particles through periodic boundaries.
147 for ( std::size_t d = 0; d < num_space_dim; ++d )
148 {
149 if ( periodic[d] )
150 {
151 if ( positions( p, d ) > global_high[d] )
152 positions( p, d ) -= global_extent[d];
153 else if ( positions( p, d ) < global_low[d] )
154 positions( p, d ) += global_extent[d];
155 }
156 }
157 } );
158}

175template <class LocalGridType, class PositionSliceType>
176int migrateCount( const LocalGridType& local_grid,
177 const PositionSliceType& positions,
178 const int minimum_halo_width )
179{
180 using grid_type = LocalGridType;
181 static constexpr std::size_t num_space_dim = grid_type::num_space_dim;
182 using mesh_type = typename grid_type::mesh_type;
183 using scalar_type = typename mesh_type::scalar_type;
184 using uniform_type = UniformMesh<scalar_type, num_space_dim>;
185 static_assert( std::is_same<mesh_type, uniform_type>::value,
186 "Migrate count requires a uniform mesh." );
187
188 using execution_space = typename PositionSliceType::execution_space;
189
190 // Check within the halo width, within the ghosted domain.
191 const auto& local_mesh = createLocalMesh<Kokkos::HostSpace>( local_grid );
192
193 Kokkos::Array<double, num_space_dim> local_low{};
194 Kokkos::Array<double, num_space_dim> local_high{};
195 for ( std::size_t d = 0; d < num_space_dim; ++d )
196 {
197 auto dx = local_grid.globalGrid().globalMesh().cellSize( d );
198 local_low[d] =
199 local_mesh.lowCorner( Ghost(), d ) + minimum_halo_width * dx;
200 local_high[d] =
201 local_mesh.highCorner( Ghost(), d ) - minimum_halo_width * dx;
202 }
203
204 int comm_count = 0;
205 Kokkos::parallel_reduce(
206 "Cabana::Grid::ParticleGridMigrate::count",
207 Kokkos::RangePolicy<execution_space>( 0, positions.size() ),
208 KOKKOS_LAMBDA( const int p, int& result ) {
209 for ( std::size_t d = 0; d < num_space_dim; ++d )
210 if ( positions( p, d ) < local_low[d] ||
211 positions( p, d ) > local_high[d] )
212 {
213 result += 1;
214 break;
215 }
216 },
217 comm_count );
218
219 MPI_Allreduce( MPI_IN_PLACE, &comm_count, 1, MPI_INT, MPI_SUM,
220 local_grid.globalGrid().comm() );
221
222 return comm_count;
223}

240template <class LocalGridType, class PositionSliceType>
241Cabana::Distributor<typename PositionSliceType::memory_space>
242createParticleGridDistributor( const LocalGridType& local_grid,
243 PositionSliceType& positions )
244{
245 using memory_space = typename PositionSliceType::memory_space;
246
247 // Get all 26 neighbor ranks.
248 auto topology = getTopology( local_grid );
249
250 Kokkos::View<int*, Kokkos::HostSpace, Kokkos::MemoryUnmanaged>
251 topology_host( topology.data(), topology.size() );
252 auto topology_mirror =
253 Kokkos::create_mirror_view_and_copy( memory_space(), topology_host );
254 Kokkos::View<int*, memory_space> destinations(
255 Kokkos::ViewAllocateWithoutInitializing( "destinations" ),
256 positions.size() );
257
258 // Determine destination ranks for all particles and wrap positions across
259 // periodic boundaries.
260 Impl::getMigrateDestinations( local_grid, topology_mirror, destinations,
261 positions );
262
263 // Create the Cabana distributor.
264 Cabana::Distributor<memory_space> distributor(
265 local_grid.globalGrid().comm(), destinations, topology );
266 return distributor;
267}

288template <class LocalGridType, class ParticlePositions, class ParticleContainer>
289bool particleGridMigrate( const LocalGridType& local_grid,
290 const ParticlePositions& positions,
291 ParticleContainer& particles,
292 const int min_halo_width,
293 const bool force_migrate = false )
294{
295 // When false, this option checks that any particles are nearly outside the
296 // ghosted halo region (outside the min_halo_width) before initiating
297 // migration. Otherwise, anything outside the local domain is migrated
298 // regardless of position in the halo.
299 if ( !force_migrate )
300 {
301 // Check to see if we need to communicate.
302 auto comm_count = migrateCount( local_grid, positions, min_halo_width );
303
304 // If we have no particles near the ghosted boundary, then exit.
305 if ( 0 == comm_count )
306 return false;
307 }
308
309 auto distributor = createParticleGridDistributor( local_grid, positions );
310
311 // Redistribute the particles.
312 migrate( distributor, particles );
313 return true;
314}

337template <class LocalGridType, class ParticlePositions, class ParticleContainer>
338bool particleGridMigrate( const LocalGridType& local_grid,
339 const ParticlePositions& positions,
340 const ParticleContainer& src_particles,
341 ParticleContainer& dst_particles,
342 const int min_halo_width,
343 const bool force_migrate = false )
344{
345 // When false, this option checks that any particles are nearly outside the
346 // ghosted halo region (outside the min_halo_width) before initiating
347 // migration. Otherwise, anything outside the local domain is migrated
348 // regardless of position in the halo.
349 if ( !force_migrate )
350 {
351 // Check to see if we need to communicate.
352 auto comm_count = migrateCount( local_grid, positions, min_halo_width );
353
354 // If we have no particles near the ghosted boundary, copy, then exit.
355 if ( 0 == comm_count )
356 {
357 Cabana::deep_copy( dst_particles, src_particles );
358 return false;
359 }
360 }
361
362 auto distributor = createParticleGridDistributor( local_grid, positions );
363
364 // Resize as needed.
365 dst_particles.resize( distributor.totalNumImport() );
366
367 // Redistribute the particles.
368 migrate( distributor, src_particles, dst_particles );
369 return true;
370}
