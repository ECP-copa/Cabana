123 template <class FunctorType, int VectorLength, class... ExecParameters>
124 inline void simd_parallel_for(
125  const SimdPolicy<VectorLength, ExecParameters...>& exec_policy,
126  const FunctorType& functor, const std::string& str = "" )
127 {
128  using simd_policy = SimdPolicy<VectorLength, ExecParameters...>;
129 
130  using work_tag = typename simd_policy::work_tag;
131 
132  using team_policy = typename simd_policy::base_type;
133 
134  using index_type = typename team_policy::index_type;
135 
136  auto simd_func =
137  KOKKOS_LAMBDA( const typename team_policy::member_type& team )
138  {
139  index_type s = team.league_rank() + exec_policy.structBegin();
140  Kokkos::parallel_for(
141  Kokkos::ThreadVectorRange( team, exec_policy.arrayBegin( s ),
142  exec_policy.arrayEnd( s ) ),
143  [&]( const index_type a ) {
144  Impl::functorTagDispatch<work_tag>( functor, s, a );
145  } );
146  };
147  if ( str.empty() )
148  Kokkos::parallel_for( dynamic_cast<const team_policy&>( exec_policy ),
149  simd_func );
150  else
151  Kokkos::parallel_for(
152  str, dynamic_cast<const team_policy&>( exec_policy ), simd_func );
153 }
