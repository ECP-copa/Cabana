36// No work tag was provided so call without a tag argument.
37template <class WorkTag, class FunctorType, class... IndexTypes>
38KOKKOS_FORCEINLINE_FUNCTION
39 typename std::enable_if<std::is_same<WorkTag, void>::value>::type
40 functorTagDispatch( const FunctorType& functor, IndexTypes&&... indices )
41{
42 functor( std::forward<IndexTypes>( indices )... );
43}
44
45// The user gave us a tag so call the version using that.
46template <class WorkTag, class FunctorType, class... IndexTypes>
47KOKKOS_FORCEINLINE_FUNCTION
48 typename std::enable_if<!std::is_same<WorkTag, void>::value>::type
49 functorTagDispatch( const FunctorType& functor, IndexTypes&&... indices )
50{
51 const WorkTag t{};
52 functor( t, std::forward<IndexTypes>( indices )... );
53}
54
55// No work tag was provided so call reduce without a tag argument.
56template <class WorkTag, class FunctorType, class... IndexTypes,
57 class ReduceType>
58KOKKOS_FORCEINLINE_FUNCTION
59 typename std::enable_if<std::is_same<WorkTag, void>::value>::type
60 functorTagDispatch( const FunctorType& functor, IndexTypes&&... indices,
61 ReduceType& reduce_val )
62{
63 functor( std::forward<IndexTypes>( indices )..., reduce_val );
64}

78template <class ExecutionPolicy, class Functor>
79struct ParallelFor;
80
81template <class Functor, int VectorLength, class... Properties>
82struct ParallelFor<SimdPolicy<VectorLength, Properties...>, Functor>
83{
84 using simd_policy = SimdPolicy<VectorLength, Properties...>;
85 using team_policy = typename simd_policy::base_type;
86 using work_tag = typename team_policy::work_tag;
87 using index_type = typename team_policy::index_type;
88 using member_type = typename team_policy::member_type;
89
90 simd_policy exec_policy_;
91 Functor functor_;
92
93 ParallelFor( std::string label, simd_policy exec_policy, Functor functor )
94 : exec_policy_( std::move( exec_policy ) )
95 , functor_( std::move( functor ) )
96 {
97 if ( label.empty() )
98 Kokkos::parallel_for(
99 dynamic_cast<const team_policy&>( exec_policy_ ), *this );
100 else
101 Kokkos::parallel_for(
102 label, dynamic_cast<const team_policy&>( exec_policy_ ),
103 *this );
104 }
105
106 template <class WorkTag>
107 KOKKOS_FUNCTION std::enable_if_t<!std::is_void<WorkTag>::value &&
108 std::is_same<WorkTag, work_tag>::value>
109 operator()( WorkTag, member_type const& team ) const
110 {
111 this->operator()( team );
112 }
113
114 KOKKOS_FUNCTION void operator()( member_type const& team ) const
115 {
116 index_type s = team.league_rank() + exec_policy_.structBegin();
117 Kokkos::parallel_for(
118 Kokkos::ThreadVectorRange( team, exec_policy_.arrayBegin( s ),
119 exec_policy_.arrayEnd( s ) ),
120 [&]( index_type a )
121 { Impl::functorTagDispatch<work_tag>( functor_, s, a ); } );
122 }
123};
124
126} // end namespace Impl

249template <class FunctorType, class NeighborListType, class... ExecParameters>
250inline void neighbor_parallel_for(
251 const Kokkos::RangePolicy<ExecParameters...>& exec_policy,
252 const FunctorType& functor, const NeighborListType& list,
253 const FirstNeighborsTag, const SerialOpTag, const std::string& str = "" )
254{
255 Kokkos::Profiling::ScopedRegion region( "Cabana::neighbor_parallel_for" );
256
257 using work_tag = typename Kokkos::RangePolicy<ExecParameters...>::work_tag;
258
259 using execution_space =
260 typename Kokkos::RangePolicy<ExecParameters...>::execution_space;
261
262 using index_type =
263 typename Kokkos::RangePolicy<ExecParameters...>::index_type;
264
265 using neighbor_list_traits = NeighborList<NeighborListType>;
266
267 using memory_space = typename neighbor_list_traits::memory_space;
268
269 auto begin = exec_policy.begin();
270 auto end = exec_policy.end();
271 using linear_policy_type = Kokkos::RangePolicy<execution_space, void, void>;
272 linear_policy_type linear_exec_policy( begin, end );
273
274 static_assert( is_accessible_from<memory_space, execution_space>{}, "" );
275
276 auto neigh_func = KOKKOS_LAMBDA( const index_type i )
277 {
278 for ( index_type n = 0;
279 n < neighbor_list_traits::numNeighbor( list, i ); ++n )
280 Impl::functorTagDispatch<work_tag>(
281 functor, i,
282 static_cast<index_type>(
283 neighbor_list_traits::getNeighbor( list, i, n ) ) );
284 };
285 if ( str.empty() )
286 Kokkos::parallel_for( linear_exec_policy, neigh_func );
287 else
288 Kokkos::parallel_for( str, linear_exec_policy, neigh_func );
289}
