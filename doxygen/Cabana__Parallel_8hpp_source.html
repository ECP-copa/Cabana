35// No work tag was provided so call without a tag argument.
36template <class WorkTag, class FunctorType, class... IndexTypes>
37KOKKOS_FORCEINLINE_FUNCTION
38 typename std::enable_if<std::is_same<WorkTag, void>::value>::type
39 functorTagDispatch( const FunctorType& functor, IndexTypes&&... indices )
40{
41 functor( std::forward<IndexTypes>( indices )... );
42}
43
44// The user gave us a tag so call the version using that.
45template <class WorkTag, class FunctorType, class... IndexTypes>
46KOKKOS_FORCEINLINE_FUNCTION
47 typename std::enable_if<!std::is_same<WorkTag, void>::value>::type
48 functorTagDispatch( const FunctorType& functor, IndexTypes&&... indices )
49{
50 const WorkTag t{};
51 functor( t, std::forward<IndexTypes>( indices )... );
52}
53
54// No work tag was provided so call reduce without a tag argument.
55template <class WorkTag, class FunctorType, class... IndexTypes,
56 class ReduceType>
57KOKKOS_FORCEINLINE_FUNCTION
58 typename std::enable_if<std::is_same<WorkTag, void>::value>::type
59 functorTagDispatch( const FunctorType& functor, IndexTypes&&... indices,
60 ReduceType& reduce_val )
61{
62 functor( std::forward<IndexTypes>( indices )..., reduce_val );
63}

78struct ParallelFor;
79
80template <class Functor, int VectorLength, class... Properties>
81struct ParallelFor<SimdPolicy<VectorLength, Properties...>, Functor>
82{
83 using simd_policy = SimdPolicy<VectorLength, Properties...>;
84 using team_policy = typename simd_policy::base_type;
85 using work_tag = typename team_policy::work_tag;
86 using index_type = typename team_policy::index_type;
87 using member_type = typename team_policy::member_type;
88
89 simd_policy exec_policy_;
90 Functor functor_;
91
92 ParallelFor( std::string label, simd_policy exec_policy, Functor functor )
93 : exec_policy_( std::move( exec_policy ) )
94 , functor_( std::move( functor ) )
95 {
96 if ( label.empty() )
97 Kokkos::parallel_for(
98 dynamic_cast<const team_policy&>( exec_policy_ ), *this );
99 else
100 Kokkos::parallel_for(
101 label, dynamic_cast<const team_policy&>( exec_policy_ ),
102 *this );
103 }
104
105 template <class WorkTag>
106 KOKKOS_FUNCTION std::enable_if_t<!std::is_void<WorkTag>::value &&
107 std::is_same<WorkTag, work_tag>::value>
108 operator()( WorkTag, member_type const& team ) const
109 {
110 this->operator()( team );
111 }
112
113 KOKKOS_FUNCTION void operator()( member_type const& team ) const
114 {
115 index_type s = team.league_rank() + exec_policy_.structBegin();
116 Kokkos::parallel_for(
117 Kokkos::ThreadVectorRange( team, exec_policy_.arrayBegin( s ),
118 exec_policy_.arrayEnd( s ) ),
119 [&]( index_type a )
120 { Impl::functorTagDispatch<work_tag>( functor_, s, a ); } );
121 }
122};
123
125} // end namespace Impl

246template <class FunctorType, class NeighborListType, class... ExecParameters>
247inline void neighbor_parallel_for(
248 const Kokkos::RangePolicy<ExecParameters...>& exec_policy,
249 const FunctorType& functor, const NeighborListType& list,
250 const FirstNeighborsTag, const SerialOpTag, const std::string& str = "" )
251{
252 using work_tag = typename Kokkos::RangePolicy<ExecParameters...>::work_tag;
253
254 using execution_space =
255 typename Kokkos::RangePolicy<ExecParameters...>::execution_space;
256
257 using index_type =
258 typename Kokkos::RangePolicy<ExecParameters...>::index_type;
259
260 using neighbor_list_traits = NeighborList<NeighborListType>;
261
262 using memory_space = typename neighbor_list_traits::memory_space;
263
264 auto begin = exec_policy.begin();
265 auto end = exec_policy.end();
266 using linear_policy_type = Kokkos::RangePolicy<execution_space, void, void>;
267 linear_policy_type linear_exec_policy( begin, end );
268
269 static_assert( is_accessible_from<memory_space, execution_space>{}, "" );
270
271 auto neigh_func = KOKKOS_LAMBDA( const index_type i )
272 {
273 for ( index_type n = 0;
274 n < neighbor_list_traits::numNeighbor( list, i ); ++n )
275 Impl::functorTagDispatch<work_tag>(
276 functor, i,
277 static_cast<index_type>(
278 neighbor_list_traits::getNeighbor( list, i, n ) ) );
279 };
280 if ( str.empty() )
281 Kokkos::parallel_for( linear_exec_policy, neigh_func );
282 else
283 Kokkos::parallel_for( str, linear_exec_policy, neigh_func );
284}
