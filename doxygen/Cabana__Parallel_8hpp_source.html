37 // Given a begin and end index get the starting and ending struct indices.
38 template<int array_size>
39 KOKKOS_INLINE_FUNCTION
40 Kokkos::pair<int,int>
41 getStructBounds( const std::size_t begin, const std::size_t end )
42 {
43  Kokkos::pair<int,int> struct_bounds;
44 
45  // Get the AoSoA Indices for the begin and end.
46  auto begin_s = Impl::Index<array_size>::s( begin );
47  auto end_s = Impl::Index<array_size>::s( end );
48  auto end_a = Impl::Index<array_size>::a( end );
49 
50  // The first struct is the struct index of the beginning.
51  struct_bounds.first = begin_s;
52 
53  // If the end is also at the front of an array that means the struct index
54  // of end is also the ending struct index. If not, we are not iterating
55  // all the way through the arrays of the last struct. In this case we add
56  // 1 to ensure that the loop over structs loops through all structs with
57  // data.
58  struct_bounds.second =
59  (0 == end_a) ? end_s : end_s + 1;
60 
61  return struct_bounds;
62 }

64 //---------------------------------------------------------------------------//
65 // Given a begin and end index, struct bounds, and a struct index get the
66 // starting and ending array indices.
67 template<int array_size>
68 KOKKOS_INLINE_FUNCTION
69 Kokkos::pair<int,int>
70 getArrayBounds( const int begin,
71  const int end,
72  const Kokkos::pair<int,int>& struct_bounds,
73  const int s )
74 {
75  Kokkos::pair<int,int> array_bounds;
76 
77  // Get the AoSoA Indices for the begin and end.
78  auto begin_a = Impl::Index<array_size>::a( begin );
79  auto end_a = Impl::Index<array_size>::a( end );
80 
81  // If the given struct index is also the index of the struct index in
82  // begin, use the starting array index. If not, that means we have passed
83  // the first struct and all subsequent structs start at array index 0.
84  array_bounds.first = (s == struct_bounds.first) ? begin_a : 0;
85 
86  // If we are in the last struct unfilled struct then use the array index
87  // of end. If not, we are looping through the current array all the way to
88  // the end so use the array size.
89  array_bounds.second =
90  ((s == struct_bounds.second - 1) && (end_a != 0))
91  ? end_a : array_size;
92 
93  return array_bounds;
94 }

209 template<class ExecutionPolicy, class FunctorType>
210 inline void parallel_for( const ExecutionPolicy& exec_policy,
211  const FunctorType& functor,
212  const StructParallelTag&,
213  const std::string& str = "" )
214 {
215  // Create a range policy over the structs. If the end is not at a struct
216  // boundary we need to add an extra struct so we loop through the last
217  // unfilled struct.
218  auto begin = exec_policy.begin();
219  auto end = exec_policy.end();
220  auto struct_bounds =
221  Impl::getStructBounds<ExecutionPolicy::array_size>( begin, end );
222 
223  // Create a wrapper for the functor. Each struct is given a thread and
224  // each thread loops over the inner arrays.
225  auto functor_wrapper =
226  KOKKOS_LAMBDA( const std::size_t s )
227  {
228  auto array_bounds = Impl::getArrayBounds<ExecutionPolicy::array_size>(
229  begin, end, struct_bounds, s );
230  for ( int a = array_bounds.first;
231  a < array_bounds.second;
232  ++a )
233  {
234  functor(
235  Impl::Index<ExecutionPolicy::array_size>::i(s,a) );
236  }
237  };
238 
239  // Create the kokkos execution policy
240  using kokkos_policy =
241  Kokkos::RangePolicy<typename ExecutionPolicy::execution_space>;
242  kokkos_policy k_policy( struct_bounds.first, struct_bounds.second );
243 
244  // Execute the functor.
245  Kokkos::parallel_for( str, k_policy, functor_wrapper );
246 
247  // Fence.
248  Kokkos::fence();
249 }

273 inline void parallel_for( const ExecutionPolicy& exec_policy,
274  const FunctorType& functor,
275  const ArrayParallelTag&,
276  const std::string& str = "" )
277 {
278  // Kokkos execution policy type.
279  using kokkos_policy =
280  Kokkos::RangePolicy<typename ExecutionPolicy::execution_space>;
281 
282  // Loop over structs. If the end is not at a struct boundary we need to
283  // add an extra struct so we loop through the last unfilled struct.
284  auto begin = exec_policy.begin();
285  auto end = exec_policy.end();
286  auto struct_bounds =
287  Impl::getStructBounds<ExecutionPolicy::array_size>( begin, end );
288  for ( int s = struct_bounds.first;
289  s < struct_bounds.second;
290  ++s )
291  {
292  // Create a range policy over the array.
293  auto array_bounds =
294  Impl::getArrayBounds<ExecutionPolicy::array_size>(

298  // each thread loops over the inner arrays.

302  functor(
303  Impl::Index<ExecutionPolicy::array_size>::i(s,a) );
304  };
305

307  kokkos_policy k_policy( array_bounds.first, array_bounds.second );

337 template<class ExecutionPolicy, class FunctorType>
338 inline void parallel_for( const ExecutionPolicy& exec_policy,
339  const FunctorType& functor,
340  const StructAndArrayParallelTag&,
341  const std::string& str = "" )
342 {
343  // Type aliases.
344  constexpr auto kokkos_iterate= Kokkos::Iterate::Right;
345  using kokkos_policy =
346  Kokkos::MDRangePolicy<typename ExecutionPolicy::execution_space,
347  Kokkos::Rank<2,kokkos_iterate,kokkos_iterate>,
348  Kokkos::IndexType<int> >;
349  using point_type = typename kokkos_policy::point_type;
350 
351  // Make a 2D execution policy.
352  auto begin = exec_policy.begin();
353  auto end = exec_policy.end();
354  auto struct_bounds =
355  Impl::getStructBounds<ExecutionPolicy::array_size>( begin, end );
356  point_type lower = { struct_bounds.first, 0 };
357  point_type upper = { struct_bounds.second, ExecutionPolicy::array_size };
358  kokkos_policy k_policy( lower, upper );
359 
360  // Create a wrapper for the functor.
361  auto functor_wrapper =
362  KOKKOS_LAMBDA( const std::size_t s, const int a )
363  {
364  auto array_bounds =
365  Impl::getArrayBounds<ExecutionPolicy::array_size>(
366  begin, end, struct_bounds, s );
367  if ( a < array_bounds.second )
368  functor(
369  Impl::Index<ExecutionPolicy::array_size>::i(s,a) );
370  };
371 
372  // Execute the functor.
373  Kokkos::parallel_for( str, k_policy, functor_wrapper );
374 
375  // Fence.
376  Kokkos::fence();
377 }
