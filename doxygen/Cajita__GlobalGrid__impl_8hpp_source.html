21 //---------------------------------------------------------------------------//
22 // Constructor.
23 template <class MeshType>
24 GlobalGrid<MeshType>::GlobalGrid(
25  MPI_Comm comm, const std::shared_ptr<GlobalMesh<MeshType>>& global_mesh,
26  const std::array<bool, 3>& periodic, const Partitioner& partitioner )
27  : _global_mesh( global_mesh )
28  , _periodic( periodic )
29 {
30  // Partition the problem.
31  std::array<int, 3> global_num_cell = {
32  _global_mesh->globalNumCell( Dim::I ),
33  _global_mesh->globalNumCell( Dim::J ),
34  _global_mesh->globalNumCell( Dim::K ) };
35  _ranks_per_dim = partitioner.ranksPerDimension( comm, global_num_cell );
36 
37  // Extract the periodicity of the boundary as integers.
38  std::array<int, 3> periodic_dims = { _periodic[Dim::I], _periodic[Dim::J],
39  _periodic[Dim::K] };
40 
41  // Generate a communicator with a Cartesian topology.
42  int reorder_cart_ranks = 1;
43  MPI_Cart_create( comm, 3, _ranks_per_dim.data(), periodic_dims.data(),
44  reorder_cart_ranks, &_cart_comm );
45 
46  // Get the Cartesian topology index of this rank.
47  int linear_rank;
48  MPI_Comm_rank( _cart_comm, &linear_rank );
49  MPI_Cart_coords( _cart_comm, linear_rank, 3, _cart_rank.data() );
50 
51  // Get the cells per dimension and the remainder.
52  std::array<int, 3> cells_per_dim;
53  std::array<int, 3> dim_remainder;
54  for ( int d = 0; d < 3; ++d )
55  {
56  cells_per_dim[d] = global_num_cell[d] / _ranks_per_dim[d];
57  dim_remainder[d] = global_num_cell[d] % _ranks_per_dim[d];
58  }
59 
60  // Compute the global cell offset and the local low corner on this rank by
61  // computing the starting global cell index via exclusive scan.
62  _global_cell_offset = { 0, 0, 0 };
63  for ( int d = 0; d < 3; ++d )
64  {
65  for ( int r = 0; r < _cart_rank[d]; ++r )
66  {
67  _global_cell_offset[d] += cells_per_dim[d];
68  if ( dim_remainder[d] > r )
69  ++_global_cell_offset[d];
70  }
71  }
72 
73  // Compute the number of local cells in this rank in each dimension.
74  for ( int d = 0; d < 3; ++d )
75  {
76  _owned_num_cell[d] = cells_per_dim[d];
77  if ( dim_remainder[d] > _cart_rank[d] )
78  ++_owned_num_cell[d];
79  }
80 }

150 //---------------------------------------------------------------------------//
151 // Get the MPI rank of a block with the given indices. If the rank is out
152 // of bounds and the boundary is not periodic, return -1 to indicate an
153 // invalid rank.
154 template <class MeshType>
155 int GlobalGrid<MeshType>::blockRank( const int i, const int j,
156  const int k ) const
157 {
158  // Get the indices.
159  std::array<int, 3> cr = { i, j, k };
160 
161  // Check for invalid indices. An index is invalid if it is out of bounds
162  // and the dimension is not periodic. An out of bound index in a periodic
163  // dimension is valid because it will wrap around to a valid index.
164  for ( int d = 0; d < 3; ++d )
165  if ( !_periodic[d] && ( cr[d] < 0 || _ranks_per_dim[d] <= cr[d] ) )
166  return -1;
167 
168  // If we have indices get their rank.
169  int lr;
170  MPI_Cart_rank( _cart_comm, cr.data(), &lr );
171  return lr;
172 }

182 //---------------------------------------------------------------------------//
183 // Get the global number of nodes in a given dimension.
184 template <class MeshType>
185 int GlobalGrid<MeshType>::globalNumEntity( Node, const int dim ) const
186 {
187  // If this dimension is periodic that last node in the dimension is
188  // repeated across the periodic boundary.
189  if ( _periodic[dim] )
190  return globalNumEntity( Cell(), dim );
191  else
192  return globalNumEntity( Cell(), dim ) + 1;
193 }
