21//---------------------------------------------------------------------------//
22// Constructor.
23template <class MeshType>
24GlobalGrid<MeshType>::GlobalGrid(
25 MPI_Comm comm, const std::shared_ptr<GlobalMesh<MeshType>>& global_mesh,
26 const std::array<bool, num_space_dim>& periodic,
27 const BlockPartitioner<num_space_dim>& partitioner )
28 : _global_mesh( global_mesh )
29 , _periodic( periodic )
30{
31 // Partition the problem.
32 std::array<int, num_space_dim> global_num_cell;
33 for ( std::size_t d = 0; d < num_space_dim; ++d )
34 global_num_cell[d] = _global_mesh->globalNumCell( d );
35 _ranks_per_dim = partitioner.ranksPerDimension( comm, global_num_cell );
36
37 // Extract the periodicity of the boundary as integers.
38 std::array<int, num_space_dim> periodic_dims;
39 for ( std::size_t d = 0; d < num_space_dim; ++d )
40 periodic_dims[d] = _periodic[d];
41
42 // Generate a communicator with a Cartesian topology.
43 int reorder_cart_ranks = 1;
44
45 auto num_space_dim_copy = num_space_dim;
46 auto ranks_per_dim_copy = _ranks_per_dim.data();
47 auto periodic_dims_copy = periodic_dims.data();
48 _cart_comm_ptr.reset(
49 // Duplicate the communicator and store in a std::shared_ptr so that
50 // all copies point to the same object
51 [comm, num_space_dim_copy, ranks_per_dim_copy, periodic_dims_copy,
52 reorder_cart_ranks]()
53 {
54 auto p = std::make_unique<MPI_Comm>();
55 MPI_Cart_create( comm, num_space_dim_copy, ranks_per_dim_copy,
56 periodic_dims_copy, reorder_cart_ranks, p.get() );
57 return p.release();
58 }(),
59 // Custom deleter to mark the communicator for deallocation
60 []( MPI_Comm* p )
61 {
62 MPI_Comm_free( p );
63 delete p;
64 } );
65
66 // Get the Cartesian topology index of this rank.
67 int linear_rank;
68 MPI_Comm_rank( this->comm(), &linear_rank );
69 MPI_Cart_coords( this->comm(), linear_rank, num_space_dim,
70 _cart_rank.data() );
71
72 // Get the cells per dimension and the remainder.
73 partitioner.ownedCellInfo( this->comm(), global_num_cell, _owned_num_cell,
74 _global_cell_offset );
75
76 // Determine if a block is on the low or high boundaries.
77 for ( std::size_t d = 0; d < num_space_dim; ++d )
78 {
79 _boundary_lo[d] = ( 0 == _cart_rank[d] );
80 _boundary_hi[d] = ( _ranks_per_dim[d] - 1 == _cart_rank[d] );
81 }
82}

167//---------------------------------------------------------------------------//
168// Get the MPI rank of a block with the given indices. If the rank is out
169// of bounds and the boundary is not periodic, return -1 to indicate an
170// invalid rank.
171template <class MeshType>
172int GlobalGrid<MeshType>::blockRank(
173 const std::array<int, num_space_dim>& ijk ) const
174{
175 // Check for invalid indices. An index is invalid if it is out of bounds
176 // and the dimension is not periodic. An out of bound index in a periodic
177 // dimension is valid because it will wrap around to a valid index.
178 for ( std::size_t d = 0; d < num_space_dim; ++d )
179 if ( !_periodic[d] && ( ijk[d] < 0 || _ranks_per_dim[d] <= ijk[d] ) )
180 return -1;
181
182 // If we have indices get their rank.
183 int lr;
184 MPI_Cart_rank( this->comm(), ijk.data(), &lr );
185 return lr;
186}

222//---------------------------------------------------------------------------//
223// Get the global number of nodes in a given dimension.
224template <class MeshType>
225int GlobalGrid<MeshType>::globalNumEntity( Node, const int dim ) const
226{
227 // If this dimension is periodic that last node in the dimension is
228 // repeated across the periodic boundary.
229 if ( _periodic[dim] )
230 return globalNumEntity( Cell(), dim );
231 else
232 return globalNumEntity( Cell(), dim ) + 1;
233}
