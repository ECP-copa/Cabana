21 //---------------------------------------------------------------------------//
22 // Constructor.
23 template <class MeshType>
24 GlobalGrid<MeshType>::GlobalGrid(
25  MPI_Comm comm, const std::shared_ptr<GlobalMesh<MeshType>>& global_mesh,
26  const std::array<bool, num_space_dim>& periodic,
27  const BlockPartitioner<num_space_dim>& partitioner )
28  : _global_mesh( global_mesh )
29  , _periodic( periodic )
30 {
31  // Partition the problem.
32  std::array<int, num_space_dim> global_num_cell;
33  for ( std::size_t d = 0; d < num_space_dim; ++d )
34  global_num_cell[d] = _global_mesh->globalNumCell( d );
35  _ranks_per_dim = partitioner.ranksPerDimension( comm, global_num_cell );
36 
37  // Extract the periodicity of the boundary as integers.
38  std::array<int, num_space_dim> periodic_dims;
39  for ( std::size_t d = 0; d < num_space_dim; ++d )
40  periodic_dims[d] = _periodic[d];
41 
42  // Generate a communicator with a Cartesian topology.
43  int reorder_cart_ranks = 1;
44  MPI_Cart_create( comm, num_space_dim, _ranks_per_dim.data(),
45  periodic_dims.data(), reorder_cart_ranks, &_cart_comm );
46 
47  // Get the Cartesian topology index of this rank.
48  int linear_rank;
49  MPI_Comm_rank( _cart_comm, &linear_rank );
50  MPI_Cart_coords( _cart_comm, linear_rank, num_space_dim,
51  _cart_rank.data() );
52 
53  // Get the cells per dimension and the remainder.
54  partitioner.ownedCellInfo( _cart_comm, global_num_cell, _owned_num_cell,
55  _global_cell_offset );
56 
57  // Determine if a block is on the low or high boundaries.
58  for ( std::size_t d = 0; d < num_space_dim; ++d )
59  {
60  _boundary_lo[d] = ( 0 == _cart_rank[d] );
61  _boundary_hi[d] = ( _ranks_per_dim[d] - 1 == _cart_rank[d] );
62  }
63 }

149 //---------------------------------------------------------------------------//
150 // Get the MPI rank of a block with the given indices. If the rank is out
151 // of bounds and the boundary is not periodic, return -1 to indicate an
152 // invalid rank.
153 template <class MeshType>
154 int GlobalGrid<MeshType>::blockRank(
155  const std::array<int, num_space_dim>& ijk ) const
156 {
157  // Check for invalid indices. An index is invalid if it is out of bounds
158  // and the dimension is not periodic. An out of bound index in a periodic
159  // dimension is valid because it will wrap around to a valid index.
160  for ( std::size_t d = 0; d < num_space_dim; ++d )
161  if ( !_periodic[d] && ( ijk[d] < 0 || _ranks_per_dim[d] <= ijk[d] ) )
162  return -1;
163 
164  // If we have indices get their rank.
165  int lr;
166  MPI_Cart_rank( _cart_comm, ijk.data(), &lr );
167  return lr;
168 }

205 // Get the global number of nodes in a given dimension.
206 template <class MeshType>
207 int GlobalGrid<MeshType>::globalNumEntity( Node, const int dim ) const
208 {
209  // If this dimension is periodic that last node in the dimension is
210  // repeated across the periodic boundary.

213  else
214  return globalNumEntity( Cell(), dim ) + 1;
