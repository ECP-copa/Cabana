21 //---------------------------------------------------------------------------//
22 // Constructor.
23 template <class MeshType>
24 GlobalGrid<MeshType>::GlobalGrid(
25  MPI_Comm comm, const std::shared_ptr<GlobalMesh<MeshType>>& global_mesh,
26  const std::array<bool, num_space_dim>& periodic,
27  const BlockPartitioner<num_space_dim>& partitioner )
28  : _global_mesh( global_mesh )
29  , _periodic( periodic )
30 {
31  // Partition the problem.
32  std::array<int, num_space_dim> global_num_cell;
33  for ( std::size_t d = 0; d < num_space_dim; ++d )
34  global_num_cell[d] = _global_mesh->globalNumCell( d );
35  _ranks_per_dim = partitioner.ranksPerDimension( comm, global_num_cell );
36 
37  // Extract the periodicity of the boundary as integers.
38  std::array<int, num_space_dim> periodic_dims;
39  for ( std::size_t d = 0; d < num_space_dim; ++d )
40  periodic_dims[d] = _periodic[d];
41 
42  // Generate a communicator with a Cartesian topology.
43  int reorder_cart_ranks = 1;
44  MPI_Cart_create( comm, num_space_dim, _ranks_per_dim.data(),
45  periodic_dims.data(), reorder_cart_ranks, &_cart_comm );
46 
47  // Get the Cartesian topology index of this rank.
48  int linear_rank;
49  MPI_Comm_rank( _cart_comm, &linear_rank );
50  MPI_Cart_coords( _cart_comm, linear_rank, num_space_dim,
51  _cart_rank.data() );
52 
53  // Get the cells per dimension and the remainder.
54  std::array<int, num_space_dim> cells_per_dim;
55  std::array<int, num_space_dim> dim_remainder;
56  for ( std::size_t d = 0; d < num_space_dim; ++d )
57  {
58  cells_per_dim[d] = global_num_cell[d] / _ranks_per_dim[d];
59  dim_remainder[d] = global_num_cell[d] % _ranks_per_dim[d];
60  }
61 
62  // Compute the global cell offset and the local low corner on this rank by
63  // computing the starting global cell index via exclusive scan.
64  for ( std::size_t d = 0; d < num_space_dim; ++d )
65  {
66  _global_cell_offset[d] = 0;
67  for ( int r = 0; r < _cart_rank[d]; ++r )
68  {
69  _global_cell_offset[d] += cells_per_dim[d];
70  if ( dim_remainder[d] > r )
71  ++_global_cell_offset[d];
72  }
73  }
74 
75  // Compute the number of local cells in this rank in each dimension.
76  for ( std::size_t d = 0; d < num_space_dim; ++d )
77  {
78  _owned_num_cell[d] = cells_per_dim[d];
79  if ( dim_remainder[d] > _cart_rank[d] )
80  ++_owned_num_cell[d];
81  }
82 
83  // Determine if a block is on the low or high boundaries.
84  for ( std::size_t d = 0; d < num_space_dim; ++d )
85  {
86  _boundary_lo[d] = ( 0 == _cart_rank[d] );
87  _boundary_hi[d] = ( _ranks_per_dim[d] - 1 == _cart_rank[d] );
88  }
89 }

175 //---------------------------------------------------------------------------//
176 // Get the MPI rank of a block with the given indices. If the rank is out
177 // of bounds and the boundary is not periodic, return -1 to indicate an
178 // invalid rank.
179 template <class MeshType>
180 int GlobalGrid<MeshType>::blockRank(
181  const std::array<int, num_space_dim>& ijk ) const
182 {
183  // Check for invalid indices. An index is invalid if it is out of bounds
184  // and the dimension is not periodic. An out of bound index in a periodic
185  // dimension is valid because it will wrap around to a valid index.
186  for ( std::size_t d = 0; d < num_space_dim; ++d )
187  if ( !_periodic[d] && ( ijk[d] < 0 || _ranks_per_dim[d] <= ijk[d] ) )
188  return -1;
189 
190  // If we have indices get their rank.
191  int lr;
192  MPI_Cart_rank( _cart_comm, ijk.data(), &lr );
193  return lr;
194 }
