73 template <class ArrayLayout_t>
74 HypreSemiStructuredSolver( const ArrayLayout_t& layout, int n_vars,
75 const bool is_preconditioner = false )
76 : _comm( layout.localGrid()->globalGrid().comm() )
77 , _is_preconditioner( is_preconditioner )
78 {
79 HYPRE_Init();
80
81 static_assert( is_array_layout<ArrayLayout_t>::value,
82 "Must use an array layout" );
83 static_assert(
84 std::is_same<typename ArrayLayout_t::entity_type,
85 entity_type>::value,
86 "Array layout entity type mush match solver entity type" );
87
88 // Spatial dimension.
89 const std::size_t num_space_dim = ArrayLayout_t::num_space_dim;
90
91 // Only a single part grid is supported initially
92 int n_parts = 1;
93 int part = 0;
94
95 // Only create data structures if this is not a preconditioner.
96 if ( !_is_preconditioner )
97 {
98 // Create the grid.
99 auto error = HYPRE_SStructGridCreate( _comm, num_space_dim, n_parts,
100 &_grid );
101 checkHypreError( error );
102
103 // Get the global index space spanned by the local grid on this
104 // rank. Note that the upper bound is not a bound but rather the
105 // last index as this is what Hypre wants. Note that we reordered
106 // this to KJI from IJK to be consistent with HYPRE ordering. By
107 // setting up the grid like this, HYPRE will then want layout-right
108 // data indexed as (i,j,k) or (i,j,k,l) which will allow us to
109 // directly use Kokkos::deep_copy to move data between Cajita arrays
110 // and HYPRE data structures.
111 auto global_space = layout.indexSpace( Own(), Global() );
112 _lower.resize( num_space_dim );
113 _upper.resize( num_space_dim );
114 for ( std::size_t d = 0; d < num_space_dim; ++d )
115 {
116 _lower[d] = static_cast<HYPRE_Int>(
117 global_space.min( num_space_dim - d - 1 ) );
118 _upper[d] = static_cast<HYPRE_Int>(
119 global_space.max( num_space_dim - d - 1 ) - 1 );
120 }
121 error = HYPRE_SStructGridSetExtents( _grid, part, _lower.data(),
122 _upper.data() );
123 checkHypreError( error );
124
125 // Get periodicity. Note we invert the order of this to KJI as well.
126 const auto& global_grid = layout.localGrid()->globalGrid();
127 HYPRE_Int periodic[num_space_dim];
128 for ( std::size_t d = 0; d < num_space_dim; ++d )
129 periodic[num_space_dim - 1 - d] =
130 global_grid.isPeriodic( d )
131 ? layout.localGrid()->globalGrid().globalNumEntity(
132 EntityType(), d )
133 : 0;
134 error = HYPRE_SStructGridSetPeriodic( _grid, part, periodic );
135 checkHypreError( error );
136
137 // Set the variables on the HYPRE grid
138 std::vector<HYPRE_SStructVariable> vartypes;
139 vartypes.resize( n_vars );
140 for ( int i = 0; i < n_vars; ++i )
141 {
142 vartypes[i] = HYPRE_SSTRUCT_VARIABLE_CELL;
143 }
144 error = HYPRE_SStructGridSetVariables( _grid, part, n_vars,
145 vartypes.data() );
146
147 // Assemble the grid.
148 error = HYPRE_SStructGridAssemble( _grid );
149 checkHypreError( error );
150
151 // Allocate LHS and RHS vectors and initialize to zero. Note that we
152 // are fixing the views under these vectors to layout-right.
153
154 std::array<long, num_space_dim + 1> reorder_size;
155 for ( std::size_t d = 0; d < num_space_dim; ++d )
156 {
157 reorder_size[d] = global_space.extent( d );
158 }
159 reorder_size.back() = n_vars;
160 IndexSpace<num_space_dim + 1> reorder_space( reorder_size );
161 auto vector_values =
162 createView<HYPRE_Complex, Kokkos::LayoutRight, memory_space>(
163 "vector_values0", reorder_space );
164 Kokkos::deep_copy( vector_values, 0.0 );
165
166 _stencils.resize( n_vars );
167 _stencil_size.resize( n_vars );
168 _stencil_index.resize( n_vars,
169 std::vector<unsigned>( n_vars + 1 ) );
170
171 error = HYPRE_SStructVectorCreate( _comm, _grid, &_b );
172 checkHypreError( error );
173 error = HYPRE_SStructVectorSetObjectType( _b, object_type );
174 checkHypreError( error );
175 error = HYPRE_SStructVectorInitialize( _b );
176 checkHypreError( error );
177 for ( int i = 0; i < n_vars; ++i )
178 {
179 error = HYPRE_SStructVectorSetBoxValues(
180 _b, part, _lower.data(), _upper.data(), i,
181 vector_values.data() );
182 checkHypreError( error );
183 }
184 error = HYPRE_SStructVectorAssemble( _b );
185 checkHypreError( error );
186
187 error = HYPRE_SStructVectorCreate( _comm, _grid, &_x );
188 checkHypreError( error );
189 error = HYPRE_SStructVectorSetObjectType( _x, object_type );
190 checkHypreError( error );
191 error = HYPRE_SStructVectorInitialize( _x );
192 checkHypreError( error );
193 for ( int i = 0; i < n_vars; ++i )
194 {
195 error = HYPRE_SStructVectorSetBoxValues(
196 _x, part, _lower.data(), _upper.data(), i,
197 vector_values.data() );
198 checkHypreError( error );
199 }
200 checkHypreError( error );
201 error = HYPRE_SStructVectorAssemble( _x );
202 checkHypreError( error );
203 }
204 }

239 void createMatrixStencil( int NumSpaceDim, int var = 0, int n_vars = 3,
240 std::vector<unsigned> stencil_length = { 7, 7,
241 7 } )
242 {
243 // This function is only valid for non-preconditioners.
244 if ( _is_preconditioner )
245 throw std::logic_error(
246 "Cannot call createMatrixStencil() on preconditioners" );
247
248 // Generate the stencil indexing
249 unsigned index = 0;
250 for ( int i = 0; i < n_vars; ++i )
251 {
252 _stencil_index[var][i] = index;
253 index += stencil_length[i];
254 }
255 _stencil_index[var][n_vars] = index;
256
257 // Create the stencil.
258 _stencil_size[var] = index;
259 auto error = HYPRE_SStructStencilCreate(
260 NumSpaceDim, _stencil_size[var], &_stencils[var] );
261 checkHypreError( error );
262 }

274 void
275 setMatrixStencil( const std::vector<std::array<int, NumSpaceDim>>& stencil,
276 int var = 0, int dep = 0 )
277 {
278 // This function is only valid for non-preconditioners.
279 if ( _is_preconditioner )

362 template <class Array_t>
363 void setMatrixValues( const Array_t& values, int v_x, int v_h )
364 {
365 static_assert( is_array<Array_t>::value, "Must use an array" );
366 static_assert(
367 std::is_same<typename Array_t::entity_type, entity_type>::value,
368 "Array entity type mush match solver entity type" );
369 static_assert(
370 std::is_same<typename Array_t::memory_space, MemorySpace>::value,
371 "Array device type and solver device type are different." );
372
373 static_assert(
374 std::is_same<typename Array_t::value_type, value_type>::value,
375 "Array value type and solver value type are different." );
376
377 // This function is only valid for non-preconditioners.
378 if ( _is_preconditioner )
379 throw std::logic_error(
380 "Cannot call setMatrixValues() on preconditioners" );
381
382 int index_size =
383 _stencil_index[v_h][v_x + 1] - _stencil_index[v_h][v_x];
384
385 // Ensure the values array matches up in dimension with the stencil size
386 if ( values.layout()->dofsPerEntity() !=
387 static_cast<int>( index_size ) )
388 throw std::runtime_error(
389 "Number of matrix values does not match stencil size" );
390
391 // Spatial dimension.
392 const std::size_t num_space_dim = Array_t::num_space_dim;
393
394 int part = 0;
395
396 // Copy the matrix entries into HYPRE. The HYPRE layout is fixed as
397 // layout-right.
398 auto owned_space = values.layout()->indexSpace( Own(), Local() );
399 std::array<long, num_space_dim + 1> reorder_size;
400 for ( std::size_t d = 0; d < num_space_dim; ++d )
401 {
402 reorder_size[d] = owned_space.extent( d );
403 }
404
405 reorder_size.back() = index_size;
406 IndexSpace<num_space_dim + 1> reorder_space( reorder_size );
407 auto a_values =
408 createView<HYPRE_Complex, Kokkos::LayoutRight, memory_space>(
409 "a_values", reorder_space );
410
411 auto values_subv = createSubview( values.view(), owned_space );
412 Kokkos::deep_copy( a_values, values_subv );
413
414 // Insert values into the HYPRE matrix.
415 std::vector<HYPRE_Int> indices( index_size );
416 int start = _stencil_index[v_h][v_x];
417 std::iota( indices.begin(), indices.end(), start );
418 auto error = HYPRE_SStructMatrixSetBoxValues(
419 _A, part, _lower.data(), _upper.data(), v_h, indices.size(),
420 indices.data(), a_values.data() );
421 checkHypreError( error );
422 }

500 template <class Array_t>
501 void solve( const Array_t& b, Array_t& x, int n_vars = 3 )
502 {
503 Kokkos::Profiling::pushRegion(
504 "Cabana::Grid::HypreSemiStructuredSolver::solve" );
505
506 static_assert( is_array<Array_t>::value, "Must use an array" );
507 static_assert(
508 std::is_same<typename Array_t::entity_type, entity_type>::value,
509 "Array entity type mush match solver entity type" );
510 static_assert(
511 std::is_same<typename Array_t::memory_space, MemorySpace>::value,
512 "Array device type and solver device type are different." );
513
514 static_assert(
515 std::is_same<typename Array_t::value_type, value_type>::value,
516 "Array value type and solver value type are different." );
517
518 // This function is only valid for non-preconditioners.
519 if ( _is_preconditioner )
520 throw std::logic_error( "Cannot call solve() on preconditioners" );
521
522 // Spatial dimension.
523 const std::size_t num_space_dim = Array_t::num_space_dim;
524
525 int part = 0;
526
527 // Initialize the RHS.
528 auto error = HYPRE_SStructVectorInitialize( _b );
529 checkHypreError( error );
530
531 // Copy the RHS into HYPRE. The HYPRE layout is fixed as layout-right.
532 auto owned_space = b.layout()->indexSpace( Own(), Local() );
533 std::array<long, num_space_dim + 1> reorder_min;
534 std::array<long, num_space_dim + 1> reorder_max;
535 for ( std::size_t d = 0; d < num_space_dim; ++d )
536 {
537 reorder_min[d] = owned_space.min( d );
538 reorder_max[d] = owned_space.max( d );
539 }
540
541 // Insert b values into the HYPRE vector.
542 // The process of creating the view and then deep copying each
543 // variable is functional, but we should avoid this process
544 // for performance if possible
545 for ( int var = 0; var < n_vars; ++var )
546 {
547 reorder_min.back() = var;
548 reorder_max.back() = var + 1;
549
550 IndexSpace<num_space_dim + 1> reorder_space( reorder_min,
551 reorder_max );
552 auto b_values =
553 createView<HYPRE_Complex, Kokkos::LayoutRight, memory_space>(
554 "vector_values", reorder_space );
555 // Extract one variable at at time.
556 auto b_subv = createSubview( b.view(), reorder_space );
557
558 Kokkos::deep_copy( b_values, b_subv );
559 error = HYPRE_SStructVectorSetBoxValues(
560 _b, part, _lower.data(), _upper.data(), var, b_values.data() );
561 checkHypreError( error );
562 }
563
564 error = HYPRE_SStructVectorAssemble( _b );
565 checkHypreError( error );
566
567 // Solve the problem
568 this->solveImpl( _A, _b, _x );
569
570 // Extract the solution from the LHS
571 for ( int var = 0; var < n_vars; ++var )
572 {
573 reorder_min.back() = var;
574 reorder_max.back() = var + 1;
575
576 IndexSpace<num_space_dim + 1> reorder_space( reorder_min,
577 reorder_max );
578 auto x_values =
579 createView<HYPRE_Complex, Kokkos::LayoutRight, memory_space>(
580 "vector_values", reorder_space );
581
582 // Extract one variable at at time.
583 // Use a pair here to retain the view rank.
584
585 error = HYPRE_SStructVectorGetBoxValues(
586 _x, part, _lower.data(), _upper.data(), var, x_values.data() );
587 checkHypreError( error );
588
589 // Copy the HYPRE solution to the LHS.
590 auto x_subv = createSubview( x.view(), reorder_space );
591 Kokkos::deep_copy( x_subv, x_values );
592 }
593
594 Kokkos::Profiling::popRegion();
595 }
