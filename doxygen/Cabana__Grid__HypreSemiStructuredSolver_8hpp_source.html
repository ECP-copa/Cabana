74 template <class ArrayLayout_t>
75 HypreSemiStructuredSolver( const ArrayLayout_t& layout, int n_vars,
76 const bool is_preconditioner = false )
77 : _comm( layout.localGrid()->globalGrid().comm() )
78 , _is_preconditioner( is_preconditioner )
79 {
80 HYPRE_Init();
81
82 static_assert( is_array_layout<ArrayLayout_t>::value,
83 "Must use an array layout" );
84 static_assert(
85 std::is_same<typename ArrayLayout_t::entity_type,
86 entity_type>::value,
87 "Array layout entity type mush match solver entity type" );
88
89 // Spatial dimension.
90 const std::size_t num_space_dim = ArrayLayout_t::num_space_dim;
91
92 // Only a single part grid is supported initially
93 int n_parts = 1;
94 int part = 0;
95
96 // Only create data structures if this is not a preconditioner.
97 if ( !_is_preconditioner )
98 {
99 // Create the grid.
100 auto error = HYPRE_SStructGridCreate( _comm, num_space_dim, n_parts,
101 &_grid );
102 checkHypreError( error );
103
104 // Get the global index space spanned by the local grid on this
105 // rank. Note that the upper bound is not a bound but rather the
106 // last index as this is what Hypre wants. Note that we reordered
107 // this to KJI from IJK to be consistent with HYPRE ordering. By
108 // setting up the grid like this, HYPRE will then want layout-right
109 // data indexed as (i,j,k) or (i,j,k,l) which will allow us to
110 // directly use Kokkos::deep_copy to move data between Cajita arrays
111 // and HYPRE data structures.
112 auto global_space = layout.indexSpace( Own(), Global() );
113 _lower.resize( num_space_dim );
114 _upper.resize( num_space_dim );
115 for ( std::size_t d = 0; d < num_space_dim; ++d )
116 {
117 _lower[d] = static_cast<HYPRE_Int>(
118 global_space.min( num_space_dim - d - 1 ) );
119 _upper[d] = static_cast<HYPRE_Int>(
120 global_space.max( num_space_dim - d - 1 ) - 1 );
121 }
122 error = HYPRE_SStructGridSetExtents( _grid, part, _lower.data(),
123 _upper.data() );
124 checkHypreError( error );
125
126 // Get periodicity. Note we invert the order of this to KJI as well.
127 const auto& global_grid = layout.localGrid()->globalGrid();
128 HYPRE_Int periodic[num_space_dim];
129 for ( std::size_t d = 0; d < num_space_dim; ++d )
130 periodic[num_space_dim - 1 - d] =
131 global_grid.isPeriodic( d )
132 ? layout.localGrid()->globalGrid().globalNumEntity(
133 EntityType(), d )
134 : 0;
135 error = HYPRE_SStructGridSetPeriodic( _grid, part, periodic );
136 checkHypreError( error );
137
138 // Set the variables on the HYPRE grid
139 std::vector<HYPRE_SStructVariable> vartypes;
140 vartypes.resize( n_vars );
141 for ( int i = 0; i < n_vars; ++i )
142 {
143 vartypes[i] = HYPRE_SSTRUCT_VARIABLE_CELL;
144 }
145 error = HYPRE_SStructGridSetVariables( _grid, part, n_vars,
146 vartypes.data() );
147
148 // Assemble the grid.
149 error = HYPRE_SStructGridAssemble( _grid );
150 checkHypreError( error );
151
152 // Allocate LHS and RHS vectors and initialize to zero. Note that we
153 // are fixing the views under these vectors to layout-right.
154
155 std::array<long, num_space_dim + 1> reorder_size;
156 for ( std::size_t d = 0; d < num_space_dim; ++d )
157 {
158 reorder_size[d] = global_space.extent( d );
159 }
160 reorder_size.back() = n_vars;
161 IndexSpace<num_space_dim + 1> reorder_space( reorder_size );
162 auto vector_values =
163 createView<HYPRE_Complex, Kokkos::LayoutRight, memory_space>(
164 "vector_values0", reorder_space );
165 Kokkos::deep_copy( vector_values, 0.0 );
166
167 _stencils.resize( n_vars );
168 _stencil_size.resize( n_vars );
169 _stencil_index.resize( n_vars,
170 std::vector<unsigned>( n_vars + 1 ) );
171
172 error = HYPRE_SStructVectorCreate( _comm, _grid, &_b );
173 checkHypreError( error );
174 error = HYPRE_SStructVectorSetObjectType( _b, object_type );
175 checkHypreError( error );
176 error = HYPRE_SStructVectorInitialize( _b );
177 checkHypreError( error );
178 for ( int i = 0; i < n_vars; ++i )
179 {
180 error = HYPRE_SStructVectorSetBoxValues(
181 _b, part, _lower.data(), _upper.data(), i,
182 vector_values.data() );
183 checkHypreError( error );
184 }
185 error = HYPRE_SStructVectorAssemble( _b );
186 checkHypreError( error );
187
188 error = HYPRE_SStructVectorCreate( _comm, _grid, &_x );
189 checkHypreError( error );
190 error = HYPRE_SStructVectorSetObjectType( _x, object_type );
191 checkHypreError( error );
192 error = HYPRE_SStructVectorInitialize( _x );
193 checkHypreError( error );
194 for ( int i = 0; i < n_vars; ++i )
195 {
196 error = HYPRE_SStructVectorSetBoxValues(
197 _x, part, _lower.data(), _upper.data(), i,
198 vector_values.data() );
199 checkHypreError( error );
200 }
201 checkHypreError( error );
202 error = HYPRE_SStructVectorAssemble( _x );
203 checkHypreError( error );
204 }
205 }

240 void createMatrixStencil( int NumSpaceDim, int var = 0, int n_vars = 3,
241 std::vector<unsigned> stencil_length = { 7, 7,
242 7 } )
243 {
244 // This function is only valid for non-preconditioners.
245 if ( _is_preconditioner )
246 throw std::logic_error(
247 "Cannot call createMatrixStencil() on preconditioners" );
248
249 // Generate the stencil indexing
250 unsigned index = 0;
251 for ( int i = 0; i < n_vars; ++i )
252 {
253 _stencil_index[var][i] = index;
254 index += stencil_length[i];
255 }
256 _stencil_index[var][n_vars] = index;
257
258 // Create the stencil.
259 _stencil_size[var] = index;
260 auto error = HYPRE_SStructStencilCreate(
261 NumSpaceDim, _stencil_size[var], &_stencils[var] );
262 checkHypreError( error );
263 }

274 template <std::size_t NumSpaceDim>
275 void
276 setMatrixStencil( const std::vector<std::array<int, NumSpaceDim>>& stencil,
277 int var = 0, int dep = 0 )
278 {
279 // This function is only valid for non-preconditioners.
280 if ( _is_preconditioner )
281 throw std::logic_error(
282 "Cannot call setMatrixStencil() on preconditioners" );
283
284 std::array<HYPRE_Int, NumSpaceDim> offset;
285
286 auto index = _stencil_index[var][dep];
287 for ( unsigned n = index; n < index + stencil.size(); ++n )
288 {
289 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
290 offset[d] = stencil[n - index][d];
291 auto error = HYPRE_SStructStencilSetEntry( _stencils[var], n,
292 offset.data(), dep );
293 checkHypreError( error );
294 }

357 template <class Array_t>
358 void setMatrixValues( const Array_t& values, int v_x, int v_h )
359 {
360 static_assert( is_array<Array_t>::value, "Must use an array" );
361 static_assert(
362 std::is_same<typename Array_t::entity_type, entity_type>::value,
363 "Array entity type mush match solver entity type" );
364 static_assert(
365 std::is_same<typename Array_t::memory_space, MemorySpace>::value,
366 "Array device type and solver device type are different." );
367
368 static_assert(
369 std::is_same<typename Array_t::value_type, value_type>::value,
370 "Array value type and solver value type are different." );
371
372 // This function is only valid for non-preconditioners.
373 if ( _is_preconditioner )
374 throw std::logic_error(
375 "Cannot call setMatrixValues() on preconditioners" );
376
377 int index_size =
378 _stencil_index[v_h][v_x + 1] - _stencil_index[v_h][v_x];
379
380 // Ensure the values array matches up in dimension with the stencil size
381 if ( values.layout()->dofsPerEntity() !=
382 static_cast<int>( index_size ) )
383 throw std::runtime_error(
384 "Number of matrix values does not match stencil size" );
385
386 // Spatial dimension.
387 const std::size_t num_space_dim = Array_t::num_space_dim;
388
389 int part = 0;
390
391 // Copy the matrix entries into HYPRE. The HYPRE layout is fixed as
392 // layout-right.
393 auto owned_space = values.layout()->indexSpace( Own(), Local() );
394 std::array<long, num_space_dim + 1> reorder_size;
395 for ( std::size_t d = 0; d < num_space_dim; ++d )
396 {
397 reorder_size[d] = owned_space.extent( d );
398 }
399
400 reorder_size.back() = index_size;
401 IndexSpace<num_space_dim + 1> reorder_space( reorder_size );
402 auto a_values =
403 createView<HYPRE_Complex, Kokkos::LayoutRight, memory_space>(
404 "a_values", reorder_space );
405
406 auto values_subv = createSubview( values.view(), owned_space );
407 Kokkos::deep_copy( a_values, values_subv );
408
409 // Insert values into the HYPRE matrix.
410 std::vector<HYPRE_Int> indices( index_size );
411 int start = _stencil_index[v_h][v_x];
412 std::iota( indices.begin(), indices.end(), start );
413 auto error = HYPRE_SStructMatrixSetBoxValues(
414 _A, part, _lower.data(), _upper.data(), v_h, indices.size(),
415 indices.data(), a_values.data() );
416 checkHypreError( error );
417 }

495 template <class Array_t>
496 void solve( const Array_t& b, Array_t& x, int n_vars = 3 )
497 {
498 Kokkos::Profiling::ScopedRegion region(
499 "Cabana::Grid::HypreSemiStructuredSolver::solve" );
500
501 static_assert( is_array<Array_t>::value, "Must use an array" );
502 static_assert(
503 std::is_same<typename Array_t::entity_type, entity_type>::value,
504 "Array entity type mush match solver entity type" );
505 static_assert(
506 std::is_same<typename Array_t::memory_space, MemorySpace>::value,
507 "Array device type and solver device type are different." );
508
509 static_assert(
510 std::is_same<typename Array_t::value_type, value_type>::value,
511 "Array value type and solver value type are different." );
512
513 // This function is only valid for non-preconditioners.
514 if ( _is_preconditioner )
515 throw std::logic_error( "Cannot call solve() on preconditioners" );
516
517 // Spatial dimension.
518 const std::size_t num_space_dim = Array_t::num_space_dim;
519
520 int part = 0;
521
522 // Copy the RHS into HYPRE. The HYPRE layout is fixed as layout-right.
523 auto owned_space = b.layout()->indexSpace( Own(), Local() );
524 std::array<long, num_space_dim + 1> reorder_min;
525 std::array<long, num_space_dim + 1> reorder_max;
526 for ( std::size_t d = 0; d < num_space_dim; ++d )
527 {
528 reorder_min[d] = owned_space.min( d );
529 reorder_max[d] = owned_space.max( d );
530 }
531
532 // Insert b values into the HYPRE vector.
533 // The process of creating the view and then deep copying each
534 // variable is functional, but we should avoid this process
535 // for performance if possible
536 int error;
537 for ( int var = 0; var < n_vars; ++var )
538 {
539 reorder_min.back() = var;
540 reorder_max.back() = var + 1;
541
542 IndexSpace<num_space_dim + 1> reorder_space( reorder_min,
543 reorder_max );
544 auto b_values =
545 createView<HYPRE_Complex, Kokkos::LayoutRight, memory_space>(
546 "vector_values", reorder_space );
547 // Extract one variable at at time.
548 auto b_subv = createSubview( b.view(), reorder_space );
549
550 Kokkos::deep_copy( b_values, b_subv );
551 error = HYPRE_SStructVectorSetBoxValues(
552 _b, part, _lower.data(), _upper.data(), var, b_values.data() );
553 checkHypreError( error );
554 }
555
556 error = HYPRE_SStructVectorAssemble( _b );
557 checkHypreError( error );
558
559 // Solve the problem
560 this->solveImpl();
561
562 // Extract the solution from the LHS
563 for ( int var = 0; var < n_vars; ++var )
564 {
565 reorder_min.back() = var;
566 reorder_max.back() = var + 1;
567
568 IndexSpace<num_space_dim + 1> reorder_space( reorder_min,
569 reorder_max );
570 auto x_values =
571 createView<HYPRE_Complex, Kokkos::LayoutRight, memory_space>(
572 "vector_values", reorder_space );
573
574 // Extract one variable at at time.
575 // Use a pair here to retain the view rank.
576
577 error = HYPRE_SStructVectorGetBoxValues(
578 _x, part, _lower.data(), _upper.data(), var, x_values.data() );
579 checkHypreError( error );
580
581 // Copy the HYPRE solution to the LHS.
582 auto x_subv = createSubview( x.view(), reorder_space );
583 Kokkos::deep_copy( x_subv, x_values );
584 }
585 }
