145template <class EntityType, class MeshType, class Scalar, class MemorySpace,
146 class Derived>
147class FastFourierTransform
148{
149 public:
151 using entity_type = EntityType;
153 using mesh_type = MeshType;
155 using value_type = Scalar;
156 // FIXME: extracting the self type for backwards compatibility with previous
157 // template on DeviceType. Should simply be MemorySpace after next release.
159 using memory_space = typename MemorySpace::memory_space;
161 using execution_space = typename memory_space::execution_space;
163 using exec_space [[deprecated]] = execution_space;
165 using device_type [[deprecated]] = typename memory_space::device_type;
166
168 static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;
169
171 std::array<int, num_space_dim> global_high;
173 std::array<int, num_space_dim> global_low;
174
179 FastFourierTransform( const ArrayLayout<EntityType, MeshType>& layout )
180 {
181 checkArrayDofs( layout.dofsPerEntity() );
182
183 // Get the local dimensions of the problem.
184 auto entity_space =
185 layout.localGrid()->indexSpace( Own(), EntityType(), Local() );
186 // Get the global grid.
187 const auto& global_grid = layout.localGrid()->globalGrid();
188
189 for ( std::size_t d = 0; d < num_space_dim; ++d )
190 {
191 // Get the low corner of the global index space on this rank.
192 global_low[d] =
193 (int)global_grid.globalOffset( num_space_dim - d - 1 );
194
195 // Get the high corner of the global index space on this rank.
196 int local_num_entity =
197 (int)entity_space.extent( num_space_dim - d - 1 );
198 global_high[d] = global_low[d] + local_num_entity - 1;
199 }
200 }
201
206 inline void checkArrayDofs( const int dof )
207 {
208 if ( 2 != dof )
209 throw std::logic_error(
210 "Only 1 complex value per entity allowed in FFT" );
211 }

264 template <class ExecutionSpace, class IndexSpaceType, class LViewType,
265 class LGViewType, std::size_t NSD = num_space_dim>
266 std::enable_if_t<3 == NSD, void>
267 copyToLocal( ExecutionSpace exec_space, const IndexSpaceType own_space,
268 LViewType& l_view, const LGViewType lg_view )
269 {
270 Kokkos::parallel_for(
271 "fft_copy_to_work", createExecutionPolicy( own_space, exec_space ),
272 KOKKOS_LAMBDA( const int i, const int j, const int k ) {
273 auto iw = i - own_space.min( Dim::I );
274 auto jw = j - own_space.min( Dim::J );
275 auto kw = k - own_space.min( Dim::K );
276 l_view( iw, jw, kw, 0 ) = lg_view( i, j, k, 0 );
277 l_view( iw, jw, kw, 1 ) = lg_view( i, j, k, 1 );
278 } );
279 }

471template <class EntityType, class MeshType, class Scalar, class MemorySpace,
472 class ExecSpace, class BackendType>
473class HeffteFastFourierTransform
474 : public FastFourierTransform<
475 EntityType, MeshType, Scalar, MemorySpace,
476 HeffteFastFourierTransform<EntityType, MeshType, Scalar, MemorySpace,
477 ExecSpace, BackendType>>
478{
479 public:
481 using value_type = Scalar;
482 // FIXME: extracting the self type for backwards compatibility with previous
483 // template on DeviceType. Should simply be MemorySpace after next release.
485 using memory_space = typename MemorySpace::memory_space;
487 using execution_space = ExecSpace;
489 using exec_space [[deprecated]] = execution_space;
491 using device_type [[deprecated]] = typename memory_space::device_type;
493 using backend_type = BackendType;
495 using mesh_type = MeshType;
496
498 static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;
499
501 using heffte_backend_type =
502 typename Impl::HeffteBackendTraits<execution_space,
503 backend_type>::backend_type;
504
506 execution_space heffte_execution_space;
507
514 HeffteFastFourierTransform( execution_space exec_space,
515 const ArrayLayout<EntityType, MeshType>& layout,
516 const FastFourierTransformParams& params )
517 : FastFourierTransform<
518 EntityType, MeshType, Scalar, MemorySpace,
519 HeffteFastFourierTransform<EntityType, MeshType, Scalar,
520 MemorySpace, ExecSpace, BackendType>>(
521 layout )
522 , heffte_execution_space( exec_space )
523 {
524 // heFFTe correctly handles 2D or 3D domains within "box3d"
525 heffte::box3d<> inbox = { this->global_low, this->global_high };
526 heffte::box3d<> outbox = { this->global_low, this->global_high };
527
528 heffte::plan_options heffte_params =
529 heffte::default_options<heffte_backend_type>();
530 // TODO: use all three heffte options for algorithm
531 bool alltoall = params.getAllToAll();
532 if ( alltoall )
533 heffte_params.algorithm = heffte::reshape_algorithm::alltoallv;
534 else
535 heffte_params.algorithm = heffte::reshape_algorithm::p2p;
536 heffte_params.use_pencils = params.getPencils();
537 heffte_params.use_reorder = params.getReorder();
538
539 // Create the heFFTe main class (separated to handle SYCL queue
540 // correctly).
541 _fft = Impl::createHeffteFft3d(
542 heffte_execution_space, heffte_backend_type{}, inbox, outbox,
543 layout.localGrid()->globalGrid().comm(), heffte_params );
544 int fftsize = std::max( _fft->size_outbox(), _fft->size_inbox() );
545
546 // Check the size.
547 auto entity_space =
548 layout.localGrid()->indexSpace( Own(), EntityType(), Local() );
549 if ( fftsize < (int)entity_space.size() )
550 throw std::logic_error( "Expected FFT allocation size smaller "
551 "than local grid size" );
552
553 _fft_work = Kokkos::View<Scalar*, MemorySpace>(
554 Kokkos::ViewAllocateWithoutInitializing( "fft_work" ),
555 2 * fftsize );
556 _workspace = Kokkos::View<Scalar* [2], MemorySpace>(
557 Kokkos::ViewAllocateWithoutInitializing( "workspace" ),
558 4 * fftsize );
559 }

589 template <class Array_t>
590 void compute( const Array_t& x, const int flag, const heffte::scale scale )
591 {
592 // Create a subview of the work array to write the local data into.
593 auto own_space =
594 x.layout()->localGrid()->indexSpace( Own(), EntityType(), Local() );
595 auto local_view_space = appendDimension( own_space, 2 );
596 auto local_view = createView<Scalar, Kokkos::LayoutRight, MemorySpace>(
597 local_view_space, _fft_work.data() );
598
599 // TODO: pull this out to template function
600 // Copy to the work array. The work array only contains owned data.
601 auto localghost_view = x.view();
602
603 this->copyToLocal( heffte_execution_space, own_space, local_view,
604 localghost_view );
605
606 if ( flag == 1 )
607 {
608 _fft->forward(
609 reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
610 reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
611 reinterpret_cast<std::complex<Scalar>*>( _workspace.data() ),
612 scale );
613 }
614 else if ( flag == -1 )
615 {
616 _fft->backward(
617 reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
618 reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
619 reinterpret_cast<std::complex<Scalar>*>( _workspace.data() ),
620 scale );
621 }
622 else
623 {
624 throw std::logic_error(
625 "Only 1:forward and -1:backward are allowed as compute flag" );
626 }
627
628 // Copy back to output array.
629 this->copyFromLocal( heffte_execution_space, own_space, local_view,
630 localghost_view );
631 }
632
633 private:
634 // heFFTe correctly handles 2D or 3D FFTs within "fft3d"
635 std::shared_ptr<heffte::fft3d<heffte_backend_type>> _fft;
636 Kokkos::View<Scalar*, memory_space> _fft_work;
637 Kokkos::View<Scalar* [2], memory_space> _workspace;
638};
