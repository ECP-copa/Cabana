144template <class EntityType, class MeshType, class Scalar, class DeviceType,
145 class Derived>
146class FastFourierTransform
147{
148 public:
150 using entity_type = EntityType;
152 using mesh_type = MeshType;
154 using value_type = Scalar;
156 using device_type = DeviceType;
158 using exec_space = typename device_type::execution_space;
159
161 static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;
162
164 std::array<int, num_space_dim> global_high;
166 std::array<int, num_space_dim> global_low;
167
172 FastFourierTransform( const ArrayLayout<EntityType, MeshType>& layout )
173 {
174 checkArrayDofs( layout.dofsPerEntity() );
175
176 // Get the local dimensions of the problem.
177 auto entity_space =
178 layout.localGrid()->indexSpace( Own(), EntityType(), Local() );
179 // Get the global grid.
180 const auto& global_grid = layout.localGrid()->globalGrid();
181
182 for ( std::size_t d = 0; d < num_space_dim; ++d )
183 {
184 // Get the low corner of the global index space on this rank.
185 global_low[d] =
186 (int)global_grid.globalOffset( num_space_dim - d - 1 );
187
188 // Get the high corner of the global index space on this rank.
189 int local_num_entity =
190 (int)entity_space.extent( num_space_dim - d - 1 );
191 global_high[d] = global_low[d] + local_num_entity - 1;
192 }
193 }
194
199 inline void checkArrayDofs( const int dof )
200 {
201 if ( 2 != dof )
202 throw std::logic_error(
203 "Only 1 complex value per entity allowed in FFT" );
204 }

257 template <class IndexSpaceType, class LViewType, class LGViewType,
258 std::size_t NSD = num_space_dim>
259 std::enable_if_t<3 == NSD, void>
260 copyToLocal( const IndexSpaceType own_space, LViewType& l_view,
261 const LGViewType lg_view )
262 {
263 Kokkos::parallel_for(
264 "fft_copy_to_work",
265 createExecutionPolicy( own_space, exec_space() ),
266 KOKKOS_LAMBDA( const int i, const int j, const int k ) {
267 auto iw = i - own_space.min( Dim::I );
268 auto jw = j - own_space.min( Dim::J );
269 auto kw = k - own_space.min( Dim::K );
270 l_view( iw, jw, kw, 0 ) = lg_view( i, j, k, 0 );
271 l_view( iw, jw, kw, 1 ) = lg_view( i, j, k, 1 );
272 } );
273 }

417template <class EntityType, class MeshType, class Scalar, class DeviceType,
418 class BackendType>
419class HeffteFastFourierTransform
420 : public FastFourierTransform<
421 EntityType, MeshType, Scalar, DeviceType,
422 HeffteFastFourierTransform<EntityType, MeshType, Scalar, DeviceType,
423 BackendType>>
424{
425 public:
427 using value_type = Scalar;
429 using device_type = DeviceType;
431 using backend_type = BackendType;
433 using exec_space = typename device_type::execution_space;
435 using mesh_type = MeshType;
436
438 static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;
439
441 using heffte_backend_type =
442 typename Impl::HeffteBackendTraits<exec_space,
443 backend_type>::backend_type;
444
450 HeffteFastFourierTransform( const ArrayLayout<EntityType, MeshType>& layout,
451 const FastFourierTransformParams& params )
452 : FastFourierTransform<
453 EntityType, MeshType, Scalar, DeviceType,
454 HeffteFastFourierTransform<EntityType, MeshType, Scalar,
455 DeviceType, BackendType>>( layout )
456 {
457 // heFFTe correctly handles 2D or 3D domains within "box3d"
458 heffte::box3d<> inbox = { this->global_low, this->global_high };
459 heffte::box3d<> outbox = { this->global_low, this->global_high };
460
461 heffte::plan_options heffte_params =
462 heffte::default_options<heffte_backend_type>();
463 heffte_params.use_alltoall = params.getAllToAll();
464 heffte_params.use_pencils = params.getPencils();
465 heffte_params.use_reorder = params.getReorder();
466
467 // Set FFT options from given parameters
468 // heFFTe correctly handles 2D or 3D FFTs within "fft3d"
469 _fft = std::make_shared<heffte::fft3d<heffte_backend_type>>(
470 inbox, outbox, layout.localGrid()->globalGrid().comm(),
471 heffte_params );
472
473 int fftsize = std::max( _fft->size_outbox(), _fft->size_inbox() );
474
475 // Check the size.
476 auto entity_space =
477 layout.localGrid()->indexSpace( Own(), EntityType(), Local() );
478 if ( fftsize < (int)entity_space.size() )
479 throw std::logic_error( "Expected FFT allocation size smaller "
480 "than local grid size" );
481
482 _fft_work = Kokkos::View<Scalar*, DeviceType>(
483 Kokkos::ViewAllocateWithoutInitializing( "fft_work" ),
484 2 * fftsize );
485 _workspace = Kokkos::View<Scalar* [2], DeviceType>(
486 Kokkos::ViewAllocateWithoutInitializing( "workspace" ),
487 4 * fftsize );
488 }

495 template <class Array_t, class ScaleType>
496 void forwardImpl( const Array_t& x, const ScaleType )
497 {
498 compute( x, 1, Impl::HeffteScalingTraits<ScaleType>().scaling_type );
499 }

506 template <class Array_t, class ScaleType>
507 void reverseImpl( const Array_t& x, const ScaleType )
508 {
509 compute( x, -1, Impl::HeffteScalingTraits<ScaleType>().scaling_type );
510 }

518 template <class Array_t>
519 void compute( const Array_t& x, const int flag, const heffte::scale scale )
520 {
521 // Create a subview of the work array to write the local data into.
522 auto own_space =
523 x.layout()->localGrid()->indexSpace( Own(), EntityType(), Local() );
524 auto local_view_space = appendDimension( own_space, 2 );
525 auto local_view = createView<Scalar, Kokkos::LayoutRight, DeviceType>(
526 local_view_space, _fft_work.data() );
527
528 // TODO: pull this out to template function
529 // Copy to the work array. The work array only contains owned data.
530 auto localghost_view = x.view();
531
532 this->copyToLocal( own_space, local_view, localghost_view );
533
534 if ( flag == 1 )
535 {
536 _fft->forward(
537 reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
538 reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
539 reinterpret_cast<std::complex<Scalar>*>( _workspace.data() ),
540 scale );
541 }
542 else if ( flag == -1 )
543 {
544 _fft->backward(
545 reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
546 reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
547 reinterpret_cast<std::complex<Scalar>*>( _workspace.data() ),
548 scale );
549 }
550 else
551 {
552 throw std::logic_error(
553 "Only 1:forward and -1:backward are allowed as compute flag" );
554 }
555
556 // Copy back to output array.
557 this->copyFromLocal( own_space, local_view, localghost_view );
558 }
559
560 private:
561 // heFFTe correctly handles 2D or 3D FFTs within "fft3d"
562 std::shared_ptr<heffte::fft3d<heffte_backend_type>> _fft;
563 Kokkos::View<Scalar*, DeviceType> _fft_work;
564 Kokkos::View<Scalar* [2], DeviceType> _workspace;
565};
