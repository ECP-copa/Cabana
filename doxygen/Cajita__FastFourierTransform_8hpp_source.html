164  std::array<int, num_space_dim> global_high;

172  FastFourierTransform( const ArrayLayout<EntityType, MeshType>& layout )
173  {
174  checkArrayDofs( layout.dofsPerEntity() );
175 
176  // Get the local dimensions of the problem.
177  auto entity_space =
178  layout.localGrid()->indexSpace( Own(), EntityType(), Local() );
179  // Get the global grid.
180  const auto& global_grid = layout.localGrid()->globalGrid();
181 
182  for ( std::size_t d = 0; d < num_space_dim; ++d )
183  {
184  // Get the low corner of the global index space on this rank.
185  global_low[d] =
186  (int)global_grid.globalOffset( num_space_dim - d - 1 );
187 
188  // Get the high corner of the global index space on this rank.
189  int local_num_entity =
190  (int)entity_space.extent( num_space_dim - d - 1 );
191  global_high[d] = global_low[d] + local_num_entity - 1;
192  }
193  }

401 template <class EntityType, class MeshType, class Scalar, class DeviceType,
402  class BackendType>
403 class HeffteFastFourierTransform
404  : public FastFourierTransform<
405  EntityType, MeshType, Scalar, DeviceType,
406  HeffteFastFourierTransform<EntityType, MeshType, Scalar, DeviceType,
407  BackendType>>
408 {
409  public:
411  using value_type = Scalar;
413  using device_type = DeviceType;
415  using backend_type = BackendType;
417  using exec_space = typename device_type::execution_space;
419  using mesh_type = MeshType;
420 
422  static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;
423 
425  using heffte_backend_type =
426  typename Impl::HeffteBackendTraits<exec_space,
427  backend_type>::backend_type;
428 
434  HeffteFastFourierTransform( const ArrayLayout<EntityType, MeshType>& layout,
435  const FastFourierTransformParams& params )
436  : FastFourierTransform<
437  EntityType, MeshType, Scalar, DeviceType,
438  HeffteFastFourierTransform<EntityType, MeshType, Scalar,
439  DeviceType, BackendType>>( layout )
440  {
441  // heFFTe correctly handles 2D or 3D domains within "box3d"
442  heffte::box3d inbox = { this->global_low, this->global_high };
443  heffte::box3d outbox = { this->global_low, this->global_high };
444 
445  heffte::plan_options heffte_params =
446  heffte::default_options<heffte_backend_type>();
447  heffte_params.use_alltoall = params.getAllToAll();
448  heffte_params.use_pencils = params.getPencils();
449  heffte_params.use_reorder = params.getReorder();
450 
451  // Set FFT options from given parameters
452  // heFFTe correctly handles 2D or 3D FFTs within "fft3d"
453  _fft = std::make_shared<heffte::fft3d<heffte_backend_type>>(
454  inbox, outbox, layout.localGrid()->globalGrid().comm(),
455  heffte_params );
456 
457  int fftsize = std::max( _fft->size_outbox(), _fft->size_inbox() );
458 
459  // Check the size.
460  auto entity_space =
461  layout.localGrid()->indexSpace( Own(), EntityType(), Local() );
462  if ( fftsize < (int)entity_space.size() )
463  throw std::logic_error( "Expected FFT allocation size smaller "
464  "than local grid size" );
465 
466  _fft_work = Kokkos::View<Scalar*, DeviceType>(
467  Kokkos::ViewAllocateWithoutInitializing( "fft_work" ),
468  2 * fftsize );
469  }

476  template <class Array_t, class ScaleType>
477  void forwardImpl( const Array_t& x, const ScaleType )
478  {
479  compute( x, 1, Impl::HeffteScalingTraits<ScaleType>().scaling_type );
480  }

487  template <class Array_t, class ScaleType>
488  void reverseImpl( const Array_t& x, const ScaleType )
489  {
490  compute( x, -1, Impl::HeffteScalingTraits<ScaleType>().scaling_type );
491  }

499  template <class Array_t>
500  void compute( const Array_t& x, const int flag, const heffte::scale scale )
501  {
502  // Create a subview of the work array to write the local data into.
503  auto own_space =
504  x.layout()->localGrid()->indexSpace( Own(), EntityType(), Local() );
505  auto local_view_space = appendDimension( own_space, 2 );
506  auto local_view = createView<Scalar, Kokkos::LayoutRight, DeviceType>(
507  local_view_space, _fft_work.data() );
508 
509  // TODO: pull this out to template function
510  // Copy to the work array. The work array only contains owned data.
511  auto localghost_view = x.view();
512 
513  this->copyToLocal( own_space, local_view, localghost_view );
514 
515  if ( flag == 1 )
516  {
517  _fft->forward(
518  reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
519  reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
520  scale );
521  }
522  else if ( flag == -1 )
523  {
524  _fft->backward(
525  reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
526  reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
527  scale );
528  }
529  else
530  {
531  throw std::logic_error(
532  "Only 1:forward and -1:backward are allowed as compute flag" );
533  }
534 
535  // Copy back to output array.
536  this->copyFromLocal( own_space, local_view, localghost_view );
537  }
538 
539  private:
540  // heFFTe correctly handles 2D or 3D FFTs within "fft3d"
541  std::shared_ptr<heffte::fft3d<heffte_backend_type>> _fft;
542  Kokkos::View<Scalar*, DeviceType> _fft_work;
543 };
