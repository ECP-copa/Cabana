148  public:

154  using value_type = Scalar;

161  static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;

164  std::array<int, num_space_dim> global_high;
166  std::array<int, num_space_dim> global_low;
167 
172  FastFourierTransform( const ArrayLayout<EntityType, MeshType>& layout )
173  {
174  checkArrayDofs( layout.dofsPerEntity() );
175 
176  // Get the local dimensions of the problem.
177  auto entity_space =
178  layout.localGrid()->indexSpace( Own(), EntityType(), Local() );
179  // Get the global grid.
180  const auto& global_grid = layout.localGrid()->globalGrid();
181 
182  for ( std::size_t d = 0; d < num_space_dim; ++d )
183  {
184  // Get the low corner of the global index space on this rank.
185  global_low[d] =
186  (int)global_grid.globalOffset( num_space_dim - d - 1 );
187 
188  // Get the high corner of the global index space on this rank.
189  int local_num_entity =
190  (int)entity_space.extent( num_space_dim - d - 1 );
191  global_high[d] = global_low[d] + local_num_entity - 1;
192  }
193  }

409 template <class EntityType, class MeshType, class Scalar, class DeviceType,
410  class BackendType>
411 class HeffteFastFourierTransform
412  : public FastFourierTransform<
413  EntityType, MeshType, Scalar, DeviceType,
414  HeffteFastFourierTransform<EntityType, MeshType, Scalar, DeviceType,
415  BackendType>>
416 {
417  public:
419  using value_type = Scalar;
421  using device_type = DeviceType;
423  using backend_type = BackendType;
425  using exec_space = typename device_type::execution_space;
427  using mesh_type = MeshType;
428 
430  static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;
431 
433  using heffte_backend_type =
434  typename Impl::HeffteBackendTraits<exec_space,
435  backend_type>::backend_type;
436 
442  HeffteFastFourierTransform( const ArrayLayout<EntityType, MeshType>& layout,
443  const FastFourierTransformParams& params )
444  : FastFourierTransform<
445  EntityType, MeshType, Scalar, DeviceType,
446  HeffteFastFourierTransform<EntityType, MeshType, Scalar,
447  DeviceType, BackendType>>( layout )
448  {
449  // heFFTe correctly handles 2D or 3D domains within "box3d"
450  heffte::box3d<> inbox = { this->global_low, this->global_high };
451  heffte::box3d<> outbox = { this->global_low, this->global_high };
452 
453  heffte::plan_options heffte_params =
454  heffte::default_options<heffte_backend_type>();
455  heffte_params.use_alltoall = params.getAllToAll();
456  heffte_params.use_pencils = params.getPencils();
457  heffte_params.use_reorder = params.getReorder();
458 
459  // Set FFT options from given parameters
460  // heFFTe correctly handles 2D or 3D FFTs within "fft3d"
461  _fft = std::make_shared<heffte::fft3d<heffte_backend_type>>(
462  inbox, outbox, layout.localGrid()->globalGrid().comm(),
463  heffte_params );
464 
465  int fftsize = std::max( _fft->size_outbox(), _fft->size_inbox() );
466 
467  // Check the size.
468  auto entity_space =
469  layout.localGrid()->indexSpace( Own(), EntityType(), Local() );
470  if ( fftsize < (int)entity_space.size() )
471  throw std::logic_error( "Expected FFT allocation size smaller "
472  "than local grid size" );
473 
474  _fft_work = Kokkos::View<Scalar*, DeviceType>(
475  Kokkos::ViewAllocateWithoutInitializing( "fft_work" ),
476  2 * fftsize );
477  _workspace = Kokkos::View<Scalar* [2], DeviceType>(
478  Kokkos::ViewAllocateWithoutInitializing( "workspace" ),
479  4 * fftsize );
480  }

487  template <class Array_t, class ScaleType>
488  void forwardImpl( const Array_t& x, const ScaleType )
489  {
490  compute( x, 1, Impl::HeffteScalingTraits<ScaleType>().scaling_type );
491  }

498  template <class Array_t, class ScaleType>
499  void reverseImpl( const Array_t& x, const ScaleType )
500  {
501  compute( x, -1, Impl::HeffteScalingTraits<ScaleType>().scaling_type );
502  }

510  template <class Array_t>
511  void compute( const Array_t& x, const int flag, const heffte::scale scale )
512  {
513  // Create a subview of the work array to write the local data into.
514  auto own_space =
515  x.layout()->localGrid()->indexSpace( Own(), EntityType(), Local() );
516  auto local_view_space = appendDimension( own_space, 2 );
517  auto local_view = createView<Scalar, Kokkos::LayoutRight, DeviceType>(
518  local_view_space, _fft_work.data() );
519 
520  // TODO: pull this out to template function
521  // Copy to the work array. The work array only contains owned data.
522  auto localghost_view = x.view();
523 
524  this->copyToLocal( own_space, local_view, localghost_view );
525 
526  if ( flag == 1 )
527  {
528  _fft->forward(
529  reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
530  reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
531  reinterpret_cast<std::complex<Scalar>*>( _workspace.data() ),
532  scale );
533  }
534  else if ( flag == -1 )
535  {
536  _fft->backward(
537  reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
538  reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
539  reinterpret_cast<std::complex<Scalar>*>( _workspace.data() ),
540  scale );
541  }
542  else
543  {
544  throw std::logic_error(
545  "Only 1:forward and -1:backward are allowed as compute flag" );
546  }
547 
548  // Copy back to output array.
549  this->copyFromLocal( own_space, local_view, localghost_view );
550  }
551 
552  private:
553  // heFFTe correctly handles 2D or 3D FFTs within "fft3d"
554  std::shared_ptr<heffte::fft3d<heffte_backend_type>> _fft;
555  Kokkos::View<Scalar*, DeviceType> _fft_work;
556  Kokkos::View<Scalar* [2], DeviceType> _workspace;
557 };
