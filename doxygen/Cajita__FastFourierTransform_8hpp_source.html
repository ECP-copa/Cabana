147template <class EntityType, class MeshType, class Scalar, class MemorySpace,
148 class Derived>
149class FastFourierTransform
150{
151 public:
153 using entity_type = EntityType;
155 using mesh_type = MeshType;
157 using value_type = Scalar;
158
159 // FIXME: extracting the self type for backwards compatibility with previous
160 // template on DeviceType. Should simply be MemorySpace after next release.
162 using memory_space = typename MemorySpace::memory_space;
163 // FIXME: replace warning with memory space assert after next release.
164 static_assert( Cabana::Impl::warn( Kokkos::is_device<MemorySpace>() ) );
165
167 using execution_space = typename memory_space::execution_space;
169 using exec_space [[deprecated]] = execution_space;
171 using device_type [[deprecated]] = typename memory_space::device_type;
172
174 static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;
175
177 std::array<int, num_space_dim> global_high;
179 std::array<int, num_space_dim> global_low;
180
185 FastFourierTransform( const ArrayLayout<EntityType, MeshType>& layout )
186 {
187 checkArrayDofs( layout.dofsPerEntity() );
188
189 // Get the local dimensions of the problem.
190 auto entity_space =
191 layout.localGrid()->indexSpace( Own(), EntityType(), Local() );
192 // Get the global grid.
193 const auto& global_grid = layout.localGrid()->globalGrid();
194
195 for ( std::size_t d = 0; d < num_space_dim; ++d )
196 {
197 // Get the low corner of the global index space on this rank.
198 global_low[d] =
199 (int)global_grid.globalOffset( num_space_dim - d - 1 );
200
201 // Get the high corner of the global index space on this rank.
202 int local_num_entity =
203 (int)entity_space.extent( num_space_dim - d - 1 );
204 global_high[d] = global_low[d] + local_num_entity - 1;
205 }
206 }
207
212 inline void checkArrayDofs( const int dof )
213 {
214 if ( 2 != dof )
215 throw std::logic_error(
216 "Only 1 complex value per entity allowed in FFT" );
217 }

270 template <class ExecutionSpace, class IndexSpaceType, class LViewType,
271 class LGViewType, std::size_t NSD = num_space_dim>
272 std::enable_if_t<3 == NSD, void>
273 copyToLocal( ExecutionSpace exec_space, const IndexSpaceType own_space,
274 LViewType& l_view, const LGViewType lg_view )
275 {
276 Kokkos::parallel_for(
277 "fft_copy_to_work", createExecutionPolicy( own_space, exec_space ),
278 KOKKOS_LAMBDA( const int i, const int j, const int k ) {
279 auto iw = i - own_space.min( Dim::I );
280 auto jw = j - own_space.min( Dim::J );
281 auto kw = k - own_space.min( Dim::K );
282 l_view( iw, jw, kw, 0 ) = lg_view( i, j, k, 0 );
283 l_view( iw, jw, kw, 1 ) = lg_view( i, j, k, 1 );
284 } );
285 }

477template <class EntityType, class MeshType, class Scalar, class MemorySpace,
478 class ExecSpace, class BackendType>
479class HeffteFastFourierTransform
480 : public FastFourierTransform<
481 EntityType, MeshType, Scalar, MemorySpace,
482 HeffteFastFourierTransform<EntityType, MeshType, Scalar, MemorySpace,
483 ExecSpace, BackendType>>
484{
485 public:
487 using value_type = Scalar;
488
489 // FIXME: extracting the self type for backwards compatibility with previous
490 // template on DeviceType. Should simply be MemorySpace after next release.
492 using memory_space = typename MemorySpace::memory_space;
493 // FIXME: replace warning with memory space assert after next release.
494 static_assert( Cabana::Impl::warn( Kokkos::is_device<MemorySpace>() ) );
495
497 using execution_space = ExecSpace;
499 using exec_space [[deprecated]] = execution_space;
501 using device_type [[deprecated]] = typename memory_space::device_type;
503 using backend_type = BackendType;
505 using mesh_type = MeshType;
506
508 static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;
509
511 using heffte_backend_type =
512 typename Impl::HeffteBackendTraits<execution_space,
513 backend_type>::backend_type;
514
516 execution_space heffte_execution_space;
517
524 HeffteFastFourierTransform( execution_space exec_space,
525 const ArrayLayout<EntityType, MeshType>& layout,
526 const FastFourierTransformParams& params )
527 : FastFourierTransform<
528 EntityType, MeshType, Scalar, MemorySpace,
529 HeffteFastFourierTransform<EntityType, MeshType, Scalar,
530 MemorySpace, ExecSpace, BackendType>>(
531 layout )
532 , heffte_execution_space( exec_space )
533 {
534 // heFFTe correctly handles 2D or 3D domains within "box3d"
535 heffte::box3d<> inbox = { this->global_low, this->global_high };
536 heffte::box3d<> outbox = { this->global_low, this->global_high };
537
538 heffte::plan_options heffte_params =
539 heffte::default_options<heffte_backend_type>();
540 // TODO: use all three heffte options for algorithm
541 bool alltoall = params.getAllToAll();
542 if ( alltoall )
543 heffte_params.algorithm = heffte::reshape_algorithm::alltoallv;
544 else
545 heffte_params.algorithm = heffte::reshape_algorithm::p2p;
546 heffte_params.use_pencils = params.getPencils();
547 heffte_params.use_reorder = params.getReorder();
548
549 // Create the heFFTe main class (separated to handle SYCL queue
550 // correctly).
551 _fft = Impl::createHeffteFft3d(
552 heffte_execution_space, heffte_backend_type{}, inbox, outbox,
553 layout.localGrid()->globalGrid().comm(), heffte_params );
554 int fftsize = std::max( _fft->size_outbox(), _fft->size_inbox() );
555
556 // Check the size.
557 auto entity_space =
558 layout.localGrid()->indexSpace( Own(), EntityType(), Local() );
559 if ( fftsize < (int)entity_space.size() )
560 throw std::logic_error( "Expected FFT allocation size smaller "
561 "than local grid size" );
562
563 _fft_work = Kokkos::View<Scalar*, memory_space>(
564 Kokkos::ViewAllocateWithoutInitializing( "fft_work" ),
565 2 * fftsize );
566 _workspace = Kokkos::View<Scalar* [2], memory_space>(
567 Kokkos::ViewAllocateWithoutInitializing( "workspace" ),
568 4 * fftsize );
569 }

599 template <class Array_t>
600 void compute( const Array_t& x, const int flag, const heffte::scale scale )
601 {
602 // Create a subview of the work array to write the local data into.
603 auto own_space =
604 x.layout()->localGrid()->indexSpace( Own(), EntityType(), Local() );
605 auto local_view_space = appendDimension( own_space, 2 );
606 auto local_view = createView<Scalar, Kokkos::LayoutRight, memory_space>(
607 local_view_space, _fft_work.data() );
608
609 // TODO: pull this out to template function
610 // Copy to the work array. The work array only contains owned data.
611 auto localghost_view = x.view();
612
613 this->copyToLocal( heffte_execution_space, own_space, local_view,
614 localghost_view );
615
616 if ( flag == 1 )
617 {
618 _fft->forward(
619 reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
620 reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
621 reinterpret_cast<std::complex<Scalar>*>( _workspace.data() ),
622 scale );
623 }
624 else if ( flag == -1 )
625 {
626 _fft->backward(
627 reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
628 reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
629 reinterpret_cast<std::complex<Scalar>*>( _workspace.data() ),
630 scale );
631 }
632 else
633 {
634 throw std::logic_error(
635 "Only 1:forward and -1:backward are allowed as compute flag" );
636 }
637
638 // Copy back to output array.
639 this->copyFromLocal( heffte_execution_space, own_space, local_view,
640 localghost_view );
641 }
642
643 private:
644 // heFFTe correctly handles 2D or 3D FFTs within "fft3d"
645 std::shared_ptr<heffte::fft3d<heffte_backend_type>> _fft;
646 Kokkos::View<Scalar*, memory_space> _fft_work;
647 Kokkos::View<Scalar* [2], memory_space> _workspace;
648};
