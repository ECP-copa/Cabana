152 template <class EntityType, class MeshType, class Scalar, class DeviceType,
153  class Derived>
154 class FastFourierTransform
155 {
156  public:
157  using entity_type = EntityType;
158  using mesh_type = MeshType;
159  using value_type = Scalar;
160  using device_type = DeviceType;
161  using exec_space = typename device_type::execution_space;
162 
163  // Spatial dimension.
164  static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;
165 
166  std::array<int, num_space_dim> global_high;
167  std::array<int, num_space_dim> global_low;
168 
173  FastFourierTransform( const ArrayLayout<EntityType, MeshType>& layout )
174  {
175  checkArrayDofs( layout.dofsPerEntity() );
176 
177  // Get the local dimensions of the problem.
178  auto entity_space =
179  layout.localGrid()->indexSpace( Own(), EntityType(), Local() );
180  // Get the global grid.
181  const auto& global_grid = layout.localGrid()->globalGrid();
182 
183  for ( std::size_t d = 0; d < num_space_dim; ++d )
184  {
185  // Get the low corner of the global index space on this rank.
186  global_low[d] =
187  (int)global_grid.globalOffset( num_space_dim - d - 1 );
188 
189  // Get the high corner of the global index space on this rank.
190  int local_num_entity =
191  (int)entity_space.extent( num_space_dim - d - 1 );
192  global_high[d] = global_low[d] + local_num_entity - 1;
193  }
194  }

200  inline void checkArrayDofs( const int dof )
201  {
202  if ( 2 != dof )
203  throw std::logic_error(
204  "Only 1 complex value per entity allowed in FFT" );

247  template <class IndexSpaceType, class LViewType, class LGViewType,
248  std::size_t NSD = num_space_dim>
249  std::enable_if_t<3 == NSD, void>
250  copyToLocal( const IndexSpaceType own_space, LViewType& l_view,
251  const LGViewType lg_view )
252  {
253  Kokkos::parallel_for(
254  "fft_copy_to_work",
255  createExecutionPolicy( own_space, exec_space() ),
256  KOKKOS_LAMBDA( const int i, const int j, const int k ) {
257  auto iw = i - own_space.min( Dim::I );
258  auto jw = j - own_space.min( Dim::J );
259  auto kw = k - own_space.min( Dim::K );
260  l_view( iw, jw, kw, 0 ) = lg_view( i, j, k, 0 );
261  l_view( iw, jw, kw, 1 ) = lg_view( i, j, k, 1 );
262  } );
263  }

389 template <class EntityType, class MeshType, class Scalar, class DeviceType,
390  class BackendType>
391 class HeffteFastFourierTransform
392  : public FastFourierTransform<
393  EntityType, MeshType, Scalar, DeviceType,
394  HeffteFastFourierTransform<EntityType, MeshType, Scalar, DeviceType,
395  BackendType>>
396 {
397  public:
398  // Types.
399  using value_type = Scalar;
400  using device_type = DeviceType;
401  using backend_type = BackendType;
402  using exec_space = typename device_type::execution_space;
403  using mesh_type = MeshType;
404  static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;
405 
406  using heffte_backend_type =
407  typename Impl::HeffteBackendTraits<exec_space,
408  backend_type>::backend_type;
409 
415  HeffteFastFourierTransform( const ArrayLayout<EntityType, MeshType>& layout,
416  const FastFourierTransformParams& params )
417  : FastFourierTransform<
418  EntityType, MeshType, Scalar, DeviceType,
419  HeffteFastFourierTransform<EntityType, MeshType, Scalar,
420  DeviceType, BackendType>>( layout )
421  {
422  // heFFTe correctly handles 2D or 3D domains within "box3d"
423  heffte::box3d inbox = { this->global_low, this->global_high };
424  heffte::box3d outbox = { this->global_low, this->global_high };
425 
426  heffte::plan_options heffte_params =
427  heffte::default_options<heffte_backend_type>();
428  heffte_params.use_alltoall = params.getAllToAll();
429  heffte_params.use_pencils = params.getPencils();
430  heffte_params.use_reorder = params.getReorder();
431 
432  // Set FFT options from given parameters
433  // heFFTe correctly handles 2D or 3D FFTs within "fft3d"
434  _fft = std::make_shared<heffte::fft3d<heffte_backend_type>>(
435  inbox, outbox, layout.localGrid()->globalGrid().comm(),
436  heffte_params );
437 
438  int fftsize = std::max( _fft->size_outbox(), _fft->size_inbox() );
439 
440  // Check the size.
441  auto entity_space =
442  layout.localGrid()->indexSpace( Own(), EntityType(), Local() );
443  if ( fftsize < (int)entity_space.size() )
444  throw std::logic_error( "Expected FFT allocation size smaller "
445  "than local grid size" );
446 
447  _fft_work = Kokkos::View<Scalar*, DeviceType>(
448  Kokkos::ViewAllocateWithoutInitializing( "fft_work" ),
449  2 * fftsize );
450  }

480  template <class Array_t>
481  void compute( const Array_t& x, const int flag, const heffte::scale scale )
482  {
483  // Create a subview of the work array to write the local data into.
484  auto own_space =
485  x.layout()->localGrid()->indexSpace( Own(), EntityType(), Local() );
486  auto local_view_space = appendDimension( own_space, 2 );
487  auto local_view = createView<Scalar, Kokkos::LayoutRight, DeviceType>(
488  local_view_space, _fft_work.data() );
489 
490  // TODO: pull this out to template function
491  // Copy to the work array. The work array only contains owned data.
492  auto localghost_view = x.view();
493 
494  this->copyToLocal( own_space, local_view, localghost_view );
495 
496  if ( flag == 1 )
497  {
498  _fft->forward(
499  reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
500  reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
501  scale );
502  }
503  else if ( flag == -1 )
504  {
505  _fft->backward(
506  reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
507  reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
508  scale );
509  }
510  else
511  {
512  throw std::logic_error(
513  "Only 1:forward and -1:backward are allowed as compute flag" );
514  }
515 
516  // Copy back to output array.
517  this->copyFromLocal( own_space, local_view, localghost_view );
518  }
519 
520  private:
521  // heFFTe correctly handles 2D or 3D FFTs within "fft3d"
522  std::shared_ptr<heffte::fft3d<heffte_backend_type>> _fft;
523  Kokkos::View<Scalar*, DeviceType> _fft_work;
524 };
