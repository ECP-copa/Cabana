206 template <class EntityType, class MeshType, class Scalar, class DeviceType,
207  class Derived>
208 class FastFourierTransform
209 {
210  public:
211  using entity_type = EntityType;
212  using mesh_type = MeshType;
213  using value_type = Scalar;
214  using device_type = DeviceType;
215 
216  std::array<int, 3> global_high;
217  std::array<int, 3> global_low;
218 
223  FastFourierTransform( const ArrayLayout<EntityType, MeshType>& layout )
224  {
225  checkArrayDofs( layout.dofsPerEntity() );
226 
227  // Get the local dimensions of the problem.
228  auto entity_space =
229  layout.localGrid()->indexSpace( Own(), EntityType(), Local() );
230  std::array<int, 3> local_num_entity = {
231  (int)entity_space.extent( Dim::K ),
232  (int)entity_space.extent( Dim::J ),
233  (int)entity_space.extent( Dim::I ) };
234 
235  // Get the global grid.
236  const auto& global_grid = layout.localGrid()->globalGrid();
237 
238  // Get the low corner of the global index space on this rank.
239  global_low = { (int)global_grid.globalOffset( Dim::K ),
240  (int)global_grid.globalOffset( Dim::J ),
241  (int)global_grid.globalOffset( Dim::I ) };
242 
243  // Get the high corner of the global index space on this rank.
244  global_high = { global_low[Dim::I] + local_num_entity[Dim::I] - 1,
245  global_low[Dim::J] + local_num_entity[Dim::J] - 1,
246  global_low[Dim::K] + local_num_entity[Dim::K] - 1 };
247  }
248 
253  inline void checkArrayDofs( const int dof )
254  {
255  if ( 1 != dof )
256  throw std::logic_error(
257  "Only 1 complex value per entity allowed in FFT" );
258  }
259 
265  template <class Array_t, class ScaleType>
266  void forward(
267  const Array_t& x, const ScaleType scaling,
268  typename std::enable_if<
269  ( is_array<Array_t>::value &&
270  is_matching_array<
271  typename Array_t::entity_type, typename Array_t::mesh_type,
272  typename Array_t::device_type, typename Array_t::value_type,
273  entity_type, mesh_type, device_type, value_type>::value ),
274  int>::type* = 0 )
275  {
276  checkArrayDofs( x.layout()->dofsPerEntity() );
277  static_cast<Derived*>( this )->forwardImpl( x, scaling );
278  }
279 
285  template <class Array_t, class ScaleType>
286  void reverse(
287  const Array_t& x, const ScaleType scaling,
288  typename std::enable_if<
289  ( is_array<Array_t>::value &&
290  is_matching_array<
291  typename Array_t::entity_type, typename Array_t::mesh_type,
292  typename Array_t::device_type, typename Array_t::value_type,
293  entity_type, mesh_type, device_type, value_type>::value ),
294  int>::type* = 0 )
295  {
296  checkArrayDofs( x.layout()->dofsPerEntity() );
297  static_cast<Derived*>( this )->reverseImpl( x, scaling );
298  }

384 template <class EntityType, class MeshType, class Scalar, class DeviceType,
385  class BackendType>
386 class HeffteFastFourierTransform
387  : public FastFourierTransform<
388  EntityType, MeshType, Scalar, DeviceType,
389  HeffteFastFourierTransform<EntityType, MeshType, Scalar, DeviceType,
390  BackendType>>
391 {
392  public:
393  // Types.
394  using value_type = Scalar;
395  using device_type = DeviceType;
396  using backend_type = BackendType;
397  using exec_space = typename device_type::execution_space;
398  using heffte_backend_type =
399  typename Impl::HeffteBackendTraits<exec_space, value_type,
400  backend_type>::backend_type;
401  using complex_type =
402  typename Impl::HeffteBackendTraits<exec_space, value_type,
403  backend_type>::complex_type;
404 
410  HeffteFastFourierTransform( const ArrayLayout<EntityType, MeshType>& layout,
411  const FastFourierTransformParams& params )
412  : FastFourierTransform<
413  EntityType, MeshType, Scalar, DeviceType,
414  HeffteFastFourierTransform<EntityType, MeshType, Scalar,
415  DeviceType, BackendType>>( layout )
416  {
417  heffte::box3d inbox = { this->global_low, this->global_high };
418  heffte::box3d outbox = { this->global_low, this->global_high };
419 
420  heffte::plan_options heffte_params =
421  heffte::default_options<heffte_backend_type>();
422  heffte_params.use_alltoall = params.getAllToAll();
423  heffte_params.use_pencils = params.getPencils();
424  heffte_params.use_reorder = params.getReorder();
425 
426  // Set FFT options from given parameters
427  _fft = std::make_shared<heffte::fft3d<heffte_backend_type>>(
428  inbox, outbox, layout.localGrid()->globalGrid().comm(),
429  heffte_params );
430 
431  int fftsize = std::max( _fft->size_outbox(), _fft->size_inbox() );
432 
433  // Check the size.
434  auto entity_space =
435  layout.localGrid()->indexSpace( Own(), EntityType(), Local() );
436  if ( fftsize < (int)entity_space.size() )
437  throw std::logic_error( "Expected FFT allocation size smaller "
438  "than local grid size" );
439 
440  _fft_work = Kokkos::View<complex_type*, DeviceType>(
441  Kokkos::ViewAllocateWithoutInitializing( "fft_work" ), fftsize );
442  }

534  template <class Array_t>
535  void compute( const Array_t& x, const int flag, const heffte::scale scale )
536  {
537  // Create a subview of the work array to write the local data into.
538  auto own_space =
539  x.layout()->localGrid()->indexSpace( Own(), EntityType(), Local() );
540 
541  auto work_view =
542  createView<complex_type, Kokkos::LayoutRight, DeviceType>(
543  own_space, _fft_work.data() );
544 
545  // TODO: pull this out to template function
546  // Copy to the work array. The work array only contains owned data.
547  auto x_view = x.view();
548 
549  Kokkos::parallel_for(
550  "fft_copy_x_to_cufft_work",
551  createExecutionPolicy( own_space, exec_space() ),
552  KOKKOS_LAMBDA( const int i, const int j, const int k ) {
553  auto iw = i - own_space.min( Dim::I );
554  auto jw = j - own_space.min( Dim::J );
555  auto kw = k - own_space.min( Dim::K );
556  work_view( iw, jw, kw ) = copyFromKokkosComplex(
557  x_view( i, j, k, 0 ), work_view( iw, jw, kw ) );
558  } );
559 
560  if ( flag == 1 )
561  {
562  _fft->forward( _fft_work.data(), _fft_work.data(), scale );
563  }
564  else if ( flag == -1 )
565  {
566  _fft->backward( _fft_work.data(), _fft_work.data(), scale );
567  }
568  else
569  {
570  throw std::logic_error(
571  "Only 1:forward and -1:backward are allowed as compute flag" );
572  }
573 
574  // Copy back to output array.
575  Kokkos::parallel_for(
576  "fft_copy_work_to_x",
577  createExecutionPolicy( own_space,
578  typename DeviceType::execution_space() ),
579  KOKKOS_LAMBDA( const int i, const int j, const int k ) {
580  auto iw = i - own_space.min( Dim::I );
581  auto jw = j - own_space.min( Dim::J );
582  auto kw = k - own_space.min( Dim::K );
583  x_view( i, j, k, 0 ) = copyToKokkosComplex(
584  work_view( iw, jw, kw ), x_view( i, j, k, 0 ) );
585  } );
586  }
587 
588  private:
589  std::shared_ptr<heffte::fft3d<heffte_backend_type>> _fft;
590  Kokkos::View<complex_type*, DeviceType> _fft_work;
591 };
