53template <class GlobalGridType, class CoordSliceType, class... FieldSliceTypes>
54void writePartialRangeTimeStep( const std::string& prefix,
55 const GlobalGridType& global_grid,
56 const int time_step_index, const double time,
57 const std::size_t begin, const std::size_t end,
58 const CoordSliceType& coords,
59 FieldSliceTypes&&... fields )
60{
61 // Pick a number of groups. We want to write approximately the N^3 blocks
62 // to N^2 groups. Pick the block dimension with the largest number of
63 // ranks as the number of groups. We may want to tweak this as an optional
64 // input later with this behavior as the default.
65 int num_group = 0;
66 for ( int d = 0; d < 3; ++d )
67 if ( global_grid.dimNumBlock( d ) > num_group )
68 num_group = global_grid.dimNumBlock( d );
69
70 Cabana::Experimental::SiloParticleOutput::writePartialRangeTimeStep(
71 prefix, global_grid.comm(), num_group, time_step_index, time, begin,
72 end, coords, fields... );
73}
