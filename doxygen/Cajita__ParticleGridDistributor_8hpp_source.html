38 // Build neighbor topology of 27 nearest 3D neighbors. Some of the ranks in
39 // this list may be invalid.
40 template <class LocalGridType, std::size_t NSD = LocalGridType::num_space_dim>
41 std::enable_if_t<3 == NSD, std::vector<int>>
42 getTopology( const LocalGridType& local_grid )
43 {
44  std::vector<int> topology( 27, -1 );
45  int nr = 0;
46  for ( int k = -1; k < 2; ++k )
47  for ( int j = -1; j < 2; ++j )
48  for ( int i = -1; i < 2; ++i, ++nr )
49  topology[nr] = local_grid.neighborRank( i, j, k );
50  return topology;
51 }

67 // Locate the particles in the local grid and get their destination rank.
68 // Particles are assumed to only migrate to a location in the nearest
69 // neighbor halo or stay on this rank. If the particle crosses a global
70 // periodic boundary, wrap it's coordinates back into the domain.
71 template <class LocalGridType, class PositionSliceType, class NeighborRankView,
72  class DestinationRankView>
73 void getMigrateDestinations( const LocalGridType& local_grid,
74  const NeighborRankView& neighbor_ranks,
75  DestinationRankView& destinations,
76  PositionSliceType& positions )
77 {
78  static constexpr std::size_t num_space_dim = LocalGridType::num_space_dim;
79  using execution_space = typename PositionSliceType::execution_space;
80 
81  // Check within the local domain.
82  const auto& local_mesh =
83  Cajita::createLocalMesh<Kokkos::HostSpace>( local_grid );
84 
85  // Use global domain for periodicity.
86  const auto& global_grid = local_grid.globalGrid();
87  const auto& global_mesh = global_grid.globalMesh();
88 
89  Kokkos::Array<double, num_space_dim> local_low{};
90  Kokkos::Array<double, num_space_dim> local_high{};
91  Kokkos::Array<bool, num_space_dim> periodic{};
92  Kokkos::Array<double, num_space_dim> global_low{};
93  Kokkos::Array<double, num_space_dim> global_high{};
94  Kokkos::Array<double, num_space_dim> global_extent{};
95 
96  for ( std::size_t d = 0; d < num_space_dim; ++d )
97  {
98  local_low[d] = local_mesh.lowCorner( Cajita::Own(), d );
99  local_high[d] = local_mesh.highCorner( Cajita::Own(), d );
100  periodic[d] = global_grid.isPeriodic( d );
101  global_low[d] = global_mesh.lowCorner( d );
102  global_high[d] = global_mesh.highCorner( d );
103  global_extent[d] = global_mesh.extent( d );
104  }
105 
106  Kokkos::parallel_for(
107  "get_migrate_destinations",
108  Kokkos::RangePolicy<execution_space>( 0, positions.size() ),
109  KOKKOS_LAMBDA( const int p ) {
110  // Compute the logical index of the neighbor we are sending to.
111  int nid[num_space_dim];
112  for ( std::size_t d = 0; d < num_space_dim; ++d )
113  {
114  nid[d] = 1;
115  if ( positions( p, d ) < local_low[d] )
116  nid[d] = 0;
117  else if ( positions( p, d ) > local_high[d] )
118  nid[d] = 2;
119  }
120 
121  // Compute the destination MPI rank [ni + 3*(nj + 3*nk) in 3d].
122  int neighbor_index = nid[0];
123  for ( std::size_t d = 1; d < num_space_dim; ++d )
124  {
125  int npower = 1;
126  for ( std::size_t dp = 1; dp <= d; ++dp )
127  npower *= 3;
128  neighbor_index += npower * nid[d];
129  }
130  destinations( p ) = neighbor_ranks( neighbor_index );
131 
132  // Shift particles through periodic boundaries.
133  for ( std::size_t d = 0; d < num_space_dim; ++d )
134  {
135  if ( periodic[d] )
136  {
137  if ( positions( p, d ) > global_high[d] )
138  positions( p, d ) -= global_extent[d];
139  else if ( positions( p, d ) < global_low[d] )
140  positions( p, d ) += global_extent[d];
141  }
142  }
143  } );
144 }
146 } // namespace Impl

164 template <class LocalGridType, class PositionSliceType>
165 int migrateCount( const LocalGridType& local_grid,
166  const PositionSliceType& positions,
167  const int minimum_halo_width )
168 {
169  using grid_type = LocalGridType;
170  static constexpr std::size_t num_space_dim = grid_type::num_space_dim;
171  using mesh_type = typename grid_type::mesh_type;
172  using scalar_type = typename mesh_type::scalar_type;
173  using uniform_type = UniformMesh<scalar_type, num_space_dim>;
174  static_assert( std::is_same<mesh_type, uniform_type>::value,
175  "Migrate count requires a uniform mesh." );
176 
177  using execution_space = typename PositionSliceType::execution_space;
178 
179  // Check within the halo width, within the ghosted domain.
180  const auto& local_mesh =
181  Cajita::createLocalMesh<Kokkos::HostSpace>( local_grid );
182 
183  Kokkos::Array<double, num_space_dim> local_low{};
184  Kokkos::Array<double, num_space_dim> local_high{};
185  for ( std::size_t d = 0; d < num_space_dim; ++d )
186  {
187  auto dx = local_grid.globalGrid().globalMesh().cellSize( d );
188  local_low[d] = local_mesh.lowCorner( Cajita::Ghost(), d ) +
189  minimum_halo_width * dx;
190  local_high[d] = local_mesh.highCorner( Cajita::Ghost(), d ) -
191  minimum_halo_width * dx;
192  }
193 
194  int comm_count = 0;
195  Kokkos::parallel_reduce(
196  "redistribute_count",
197  Kokkos::RangePolicy<execution_space>( 0, positions.size() ),
198  KOKKOS_LAMBDA( const int p, int& result ) {
199  for ( std::size_t d = 0; d < num_space_dim; ++d )
200  if ( positions( p, d ) < local_low[d] ||
201  positions( p, d ) > local_high[d] )
202  {
203  result += 1;
204  break;
205  }
206  },
207  comm_count );
208 
209  MPI_Allreduce( MPI_IN_PLACE, &comm_count, 1, MPI_INT, MPI_SUM,
210  local_grid.globalGrid().comm() );
211 
212  return comm_count;
213 }

232 template <class LocalGridType, class PositionSliceType>
233 Cabana::Distributor<typename PositionSliceType::device_type>
234 createParticleGridDistributor( const LocalGridType& local_grid,
235  PositionSliceType& positions )
236 {
237  using device_type = typename PositionSliceType::device_type;
238 
239  // Get all 26 neighbor ranks.
240  auto topology = Impl::getTopology( local_grid );
241 
242  Kokkos::View<int*, Kokkos::HostSpace, Kokkos::MemoryUnmanaged>
243  topology_host( topology.data(), topology.size() );
244  auto topology_mirror =
245  Kokkos::create_mirror_view_and_copy( device_type(), topology_host );
246  Kokkos::View<int*, device_type> destinations(
247  Kokkos::ViewAllocateWithoutInitializing( "destinations" ),
248  positions.size() );
249 
250  // Determine destination ranks for all particles and wrap positions across
251  // periodic boundaries.
252  Impl::getMigrateDestinations( local_grid, topology_mirror, destinations,
253  positions );
254 
255  // Create the Cabana distributor.
256  Cabana::Distributor<device_type> distributor(
257  local_grid.globalGrid().comm(), destinations, topology );
258  return distributor;
259 }

285 template <class LocalGridType, class ParticlePositions, class ParticleContainer>
286 void particleGridMigrate( const LocalGridType& local_grid,
287  ParticlePositions& positions,
288  ParticleContainer& particles,
289  const int min_halo_width,
290  const bool force_migrate = false )
291 {
292  // When false, this option checks that any particles are nearly outside the
293  // ghosted halo region (outside the min_halo_width) before initiating
294  // migration. Otherwise, anything outside the local domain is migrated
295  // regardless of position in the halo.
296  if ( !force_migrate )
297  {
298  // Check to see if we need to communicate.
299  auto comm_count = migrateCount( local_grid, positions, min_halo_width );
300 
301  // If we have no particles near the ghosted boundary, then exit.
302  if ( 0 == comm_count )
303  return;
304  }
305 
306  auto distributor = createParticleGridDistributor( local_grid, positions );
307 
308  // Redistribute the particles.
309  migrate( distributor, particles );
310 }

339 template <class LocalGridType, class ParticlePositions, class ParticleContainer>
340 void particleGridMigrate( const LocalGridType& local_grid,
341  const ParticlePositions& positions,
342  const ParticleContainer& src_particles,
343  ParticleContainer& dst_particles,
344  const int min_halo_width,
345  const bool force_migrate = false )
346 {
347  // When false, this option checks that any particles are nearly outside the
348  // ghosted halo region (outside the min_halo_width) before initiating
349  // migration. Otherwise, anything outside the local domain is migrated
350  // regardless of position in the halo.
351  if ( !force_migrate )
352  {
353  // Check to see if we need to communicate.
354  auto comm_count = migrateCount( local_grid, positions, min_halo_width );
355 
356  // If we have no particles near the ghosted boundary, copy, then exit.
357  if ( 0 == comm_count )
358  {
359  Cabana::deep_copy( dst_particles, src_particles );
360  return;
361  }
362  }
363 
364  auto distributor = createParticleGridDistributor( local_grid, positions );
365 
366  // Resize as needed.
367  dst_particles.resize( distributor.totalNumImport() );
368 
369  // Redistribute the particles.
370  migrate( distributor, src_particles, dst_particles );
371 }
