154 // Create a permutation vector over a range subset using a comparator over the
155 // given Kokkos View of keys.
156 template <class KeyViewType, class Comparator,
157  class DeviceType = typename KeyViewType::device_type>
158 BinningData<DeviceType>
159 kokkosBinSort( KeyViewType keys, Comparator comp, const bool sort_within_bins,
160  const std::size_t begin, const std::size_t end )
161 {
162  Kokkos::BinSort<KeyViewType, Comparator, DeviceType> bin_sort(
163  keys, begin, end, comp, sort_within_bins );
164  bin_sort.create_permute_vector();
165  return BinningData<DeviceType>( begin, end, bin_sort.get_bin_count(),
166  bin_sort.get_bin_offsets(),
167  bin_sort.get_permute_vector() );
168 }

171 // Given a set of keys, find the minimum and maximum over the given range.
172 template <class KeyViewType,
173  class DeviceType = typename KeyViewType::device_type>
174 Kokkos::MinMaxScalar<typename KeyViewType::non_const_value_type>
175 keyMinMax( KeyViewType keys, const std::size_t begin, const std::size_t end )
176 {
177  Kokkos::MinMaxScalar<typename KeyViewType::non_const_value_type> result;
178  Kokkos::MinMax<typename KeyViewType::non_const_value_type> reducer(
179  result );
180  Kokkos::parallel_reduce(
181  "Cabana::keyMinMax",
182  Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
183  Kokkos::Impl::min_max_functor<KeyViewType>( keys ), reducer );
184  Kokkos::fence();
185  return result;
186 }

188 //---------------------------------------------------------------------------//
189 // Sort an AoSoA over a subset of its range using the given Kokkos View of
190 // keys.
191 template <class KeyViewType,
192  class DeviceType = typename KeyViewType::device_type>
193 BinningData<DeviceType>
194 kokkosBinSort1d( KeyViewType keys, const int nbin, const bool sort_within_bins,
195  const std::size_t begin, const std::size_t end )
196 {
197  // Find the minimum and maximum key values.
198  auto key_bounds =
199  Impl::keyMinMax<KeyViewType, DeviceType>( keys, begin, end );
200 
201  // Create a sorting comparator.
202  Kokkos::BinOp1D<KeyViewType> comp( nbin, key_bounds.min_val,
203  key_bounds.max_val );
204 
205  // BinSort
206  return kokkosBinSort<KeyViewType, decltype( comp ), DeviceType>(
207  keys, comp, sort_within_bins, begin, end );
208 }

210 //---------------------------------------------------------------------------//
211 // Copy the a 1D slice into a Kokkos view.
212 template <class SliceType, class DeviceType = typename SliceType::device_type>
213 Kokkos::View<typename SliceType::value_type*, typename SliceType::device_type>
214 copySliceToKeys( SliceType slice )
215 {
216  using KeyViewType =
217  Kokkos::View<typename SliceType::value_type*, DeviceType>;
218  KeyViewType keys( Kokkos::ViewAllocateWithoutInitializing( "slice_keys" ),
219  slice.size() );
220  Kokkos::RangePolicy<typename DeviceType::execution_space> exec_policy(
221  0, slice.size() );
222  auto copy_op = KOKKOS_LAMBDA( const std::size_t i )
223  {
224  keys( i ) = slice( i );
225  };
226  Kokkos::parallel_for( "Cabana::copySliceToKeys::copy_op", exec_policy,
227  copy_op );
228  Kokkos::fence();
229  return keys;
230 }

570 template <class BinningDataType, class AoSoA_t,
571  class DeviceType = typename BinningDataType::device_type>
572 void permute(
573  const BinningDataType& binning_data, AoSoA_t& aosoa,
574  typename std::enable_if<( is_binning_data<BinningDataType>::value &&
575  is_aosoa<AoSoA_t>::value ),
576  int>::type* = 0 )
577 {
578  auto begin = binning_data.rangeBegin();
579  auto end = binning_data.rangeEnd();
580 
581  Kokkos::View<typename AoSoA_t::tuple_type*, DeviceType> scratch_tuples(
582  Kokkos::ViewAllocateWithoutInitializing( "scratch_tuples" ),
583  end - begin );
584 
585  auto permute_to_scratch = KOKKOS_LAMBDA( const std::size_t i )
586  {
587  scratch_tuples( i - begin ) =
588  aosoa.getTuple( binning_data.permutation( i - begin ) );
589  };
590  Kokkos::parallel_for(
591  "Cabana::kokkosBinSort::permute_to_scratch",
592  Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
593  permute_to_scratch );
594  Kokkos::fence();
595 
596  auto copy_back = KOKKOS_LAMBDA( const std::size_t i )
597  {
598  aosoa.setTuple( i, scratch_tuples( i - begin ) );
599  };
600  Kokkos::parallel_for(
601  "Cabana::kokkosBinSort::copy_back",
602  Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
603  copy_back );
604  Kokkos::fence();
605 }

619 template <class BinningDataType, class SliceType,
620  class DeviceType = typename BinningDataType::device_type>
621 void permute(
622  const BinningDataType& binning_data, SliceType& slice,
623  typename std::enable_if<( is_binning_data<BinningDataType>::value &&
624  is_slice<SliceType>::value ),
625  int>::type* = 0 )
626 {
627 
628  auto begin = binning_data.rangeBegin();
629  auto end = binning_data.rangeEnd();
630 
631  // Get the number of components in the slice.
632  std::size_t num_comp = 1;
633  for ( std::size_t d = 2; d < slice.rank(); ++d )
634  num_comp *= slice.extent( d );
635 
636  // Get the raw slice data.
637  auto slice_data = slice.data();
638 
639  Kokkos::View<typename SliceType::value_type**, DeviceType> scratch_array(
640  Kokkos::ViewAllocateWithoutInitializing( "scratch_array" ), end - begin,
641  num_comp );
642 
643  auto permute_to_scratch = KOKKOS_LAMBDA( const std::size_t i )
644  {
645  auto permute_i = binning_data.permutation( i - begin );
646  auto s = SliceType::index_type::s( permute_i );
647  auto a = SliceType::index_type::a( permute_i );
648  std::size_t slice_offset = s * slice.stride( 0 ) + a;
649  for ( std::size_t n = 0; n < num_comp; ++n )
650  scratch_array( i - begin, n ) =
651  slice_data[slice_offset + SliceType::vector_length * n];
652  };
653  Kokkos::parallel_for(
654  "Cabana::kokkosBinSort::permute_to_scratch",
655  Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
656  permute_to_scratch );
657  Kokkos::fence();
658 
659  auto copy_back = KOKKOS_LAMBDA( const std::size_t i )
660  {
661  auto s = SliceType::index_type::s( i );
662  auto a = SliceType::index_type::a( i );
663  std::size_t slice_offset = s * slice.stride( 0 ) + a;
664  for ( std::size_t n = 0; n < num_comp; ++n )
665  slice_data[slice_offset + SliceType::vector_length * n] =
666  scratch_array( i - begin, n );
667  };
668  Kokkos::parallel_for(
669  "Cabana::kokkosBinSort::copy_back",
670  Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
671  copy_back );
672  Kokkos::fence();
673 }
