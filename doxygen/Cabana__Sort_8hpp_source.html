142  Comparator comp,
143  const bool sort_within_bins,
144  const std::size_t begin,
145  const std::size_t end )
146 {
147  Kokkos::BinSort<KeyViewType,Comparator,DeviceType> bin_sort(

159 // Given a set of keys, find the minimum and maximum over the given range.
160 template<class KeyViewType,
161  class DeviceType = typename KeyViewType::device_type>
162 Kokkos::MinMaxScalar<typename KeyViewType::non_const_value_type>
163 keyMinMax( KeyViewType keys, const std::size_t begin, const std::size_t end )
164 {
165  Kokkos::MinMaxScalar<typename KeyViewType::non_const_value_type> result;
166  Kokkos::MinMax<typename KeyViewType::non_const_value_type> reducer(result);
167  Kokkos::parallel_reduce(
168  "Cabana::keyMinMax",
169  Kokkos::RangePolicy<typename DeviceType::execution_space>(begin,end),
170  Kokkos::Impl::min_max_functor<KeyViewType>(keys),
171  reducer );
172  Kokkos::fence();
173  return result;
174 }

176 //---------------------------------------------------------------------------//
177 // Sort an AoSoA over a subset of its range using the given Kokkos View of
178 // keys.
179 template<class KeyViewType,
180  class DeviceType = typename KeyViewType::device_type>
181 BinningData<DeviceType>
182 kokkosBinSort1d( KeyViewType keys,
183  const int nbin,
184  const bool sort_within_bins,
185  const std::size_t begin,
186  const std::size_t end )
187 {
188  // Find the minimum and maximum key values.
189  auto key_bounds =
190  Impl::keyMinMax<KeyViewType,DeviceType>( keys, begin, end );
191 
192  // Create a sorting comparator.
193  Kokkos::BinOp1D<KeyViewType> comp(
194  nbin, key_bounds.min_val, key_bounds.max_val );
195 
196  // BinSort
197  return kokkosBinSort<KeyViewType,decltype(comp),DeviceType>(
198  keys, comp, sort_within_bins, begin, end );
199 }

202 // Copy the a 1D slice into a Kokkos view.
203 template<class SliceType,
204  class DeviceType = typename SliceType::device_type>
205 Kokkos::View<typename SliceType::value_type*,
206  typename SliceType::device_type>
207 copySliceToKeys( SliceType slice )
208 {
209  using KeyViewType =
210  Kokkos::View<typename SliceType::value_type*,DeviceType>;
211  KeyViewType keys( "slice_keys", slice.size() );
212  Kokkos::RangePolicy<typename DeviceType::execution_space>
213  exec_policy( 0, slice.size() );
214  auto copy_op = KOKKOS_LAMBDA( const std::size_t i ) { keys(i) = slice(i); };
215  Kokkos::parallel_for( "Cabana::copySliceToKeys::copy_op",
216  exec_policy,
217  copy_op );
218  Kokkos::fence();
219  return keys;
220 }

586 template<class BinningDataType,
587  class AoSoA_t,
588  class DeviceType = typename BinningDataType::device_type>
589 void permute( const BinningDataType& binning_data,
590  AoSoA_t& aosoa,
591  typename std::enable_if<(is_binning_data<BinningDataType>::value &&
592  is_aosoa<AoSoA_t>::value),
593  int>::type * = 0)
594 {
595  auto begin = binning_data.rangeBegin();
596  auto end = binning_data.rangeEnd();
597 
598  Kokkos::View<typename AoSoA_t::tuple_type*,DeviceType>
599  scratch_tuples( "scratch_tuples", end - begin );
600 
601  auto permute_to_scratch =
602  KOKKOS_LAMBDA( const std::size_t i )
603  {
604  scratch_tuples( i - begin ) =
605  aosoa.getTuple( binning_data.permutation(i-begin) );
606  };
607  Kokkos::parallel_for(
608  "Cabana::kokkosBinSort::permute_to_scratch",
609  Kokkos::RangePolicy<typename DeviceType::execution_space>(begin,end),
610  permute_to_scratch );
611  Kokkos::fence();
612 
613  auto copy_back = KOKKOS_LAMBDA( const std::size_t i )
614  { aosoa.setTuple( i, scratch_tuples(i-begin) ); };
615  Kokkos::parallel_for(
616  "Cabana::kokkosBinSort::copy_back",
617  Kokkos::RangePolicy<typename DeviceType::execution_space>(begin,end),
618  copy_back );
619  Kokkos::fence();
620 }

634 template<class BinningDataType,
635  class SliceType,
636  class DeviceType = typename BinningDataType::device_type>
637 void permute( const BinningDataType& binning_data,
638  SliceType& slice,
639  typename std::enable_if<(is_binning_data<BinningDataType>::value &&
640  is_slice<SliceType>::value),
641  int>::type * = 0)
642 {
643 
644  auto begin = binning_data.rangeBegin();
645  auto end = binning_data.rangeEnd();
646 
647  // Get the number of components in the slice.
648  std::size_t num_comp = 1;
649  for ( std::size_t d = 2; d < slice.rank(); ++d )
650  num_comp *= slice.extent(d);
651 
652  // Get the raw slice data.
653  auto slice_data = slice.data();
654 
655  Kokkos::View<typename SliceType::value_type**,DeviceType>
656  scratch_array( "scratch_array", end - begin, num_comp );
657 
658  auto permute_to_scratch =
659  KOKKOS_LAMBDA( const std::size_t i )
660  {
661  auto permute_i = binning_data.permutation(i-begin);
662  auto s = SliceType::index_type::s( permute_i );
663  auto a = SliceType::index_type::a( permute_i );
664  std::size_t slice_offset = s*slice.stride(0) + a;
665  for ( std::size_t n = 0; n < num_comp; ++n )
666  scratch_array( i-begin, n ) =
667  slice_data[ slice_offset + SliceType::vector_length * n ];
668  };
669  Kokkos::parallel_for(
670  "Cabana::kokkosBinSort::permute_to_scratch",
671  Kokkos::RangePolicy<typename DeviceType::execution_space>(begin,end),
672  permute_to_scratch );
673  Kokkos::fence();
674 
675  auto copy_back =
676  KOKKOS_LAMBDA( const std::size_t i )
677  {
678  auto s = SliceType::index_type::s( i );
679  auto a = SliceType::index_type::a( i );
680  std::size_t slice_offset = s*slice.stride(0) + a;
681  for ( std::size_t n = 0; n < num_comp; ++n )
682  slice_data[ slice_offset + SliceType::vector_length * n ] =
683  scratch_array( i-begin, n );
684  };
685  Kokkos::parallel_for(
686  "Cabana::kokkosBinSort::copy_back",
687  Kokkos::RangePolicy<typename DeviceType::execution_space>(begin,end),
688  copy_back );
689  Kokkos::fence();
690 }
