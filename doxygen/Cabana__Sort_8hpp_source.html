195 template <class KeyViewType,
196  class DeviceType = typename KeyViewType::device_type>
197 BinningData<DeviceType>
198 kokkosBinSort1d( KeyViewType keys, const int nbin, const bool sort_within_bins,
199  const std::size_t begin, const std::size_t end )
200 {
201  // Find the minimum and maximum key values.
202  auto key_bounds =
203  Impl::keyMinMax<KeyViewType, DeviceType>( keys, begin, end );
204 
205  // Create a sorting comparator.
206  Kokkos::BinOp1D<KeyViewType> comp( nbin, key_bounds.min_val,
207  key_bounds.max_val );
208 
209  // BinSort
210  return kokkosBinSort<KeyViewType, decltype( comp ), DeviceType>(
211  keys, comp, sort_within_bins, begin, end );
212 }

217 Kokkos::View<typename SliceType::value_type*, typename SliceType::device_type>
218 copySliceToKeys( SliceType slice )
219 {
220  using KeyViewType =
221  Kokkos::View<typename SliceType::value_type*, DeviceType>;
222  KeyViewType keys( Kokkos::ViewAllocateWithoutInitializing( "slice_keys" ),
223  slice.size() );
224  Kokkos::RangePolicy<typename DeviceType::execution_space> exec_policy(
225  0, slice.size() );
226  auto copy_op = KOKKOS_LAMBDA( const std::size_t i )
227  {
228  keys( i ) = slice( i );
229  };
230  Kokkos::parallel_for( "Cabana::copySliceToKeys::copy_op", exec_policy,
231  copy_op );
232  Kokkos::fence();
233  return keys;
234 }

574 template <class BinningDataType, class AoSoA_t,
575  class DeviceType = typename BinningDataType::device_type>
576 void permute(
577  const BinningDataType& binning_data, AoSoA_t& aosoa,
578  typename std::enable_if<( is_binning_data<BinningDataType>::value &&
579  is_aosoa<AoSoA_t>::value ),
580  int>::type* = 0 )
581 {
582  auto begin = binning_data.rangeBegin();
583  auto end = binning_data.rangeEnd();
584 
585  Kokkos::View<typename AoSoA_t::tuple_type*, DeviceType> scratch_tuples(
586  Kokkos::ViewAllocateWithoutInitializing( "scratch_tuples" ),
587  end - begin );
588 
589  auto permute_to_scratch = KOKKOS_LAMBDA( const std::size_t i )
590  {
591  scratch_tuples( i - begin ) =
592  aosoa.getTuple( binning_data.permutation( i - begin ) );
593  };
594  Kokkos::parallel_for(
595  "Cabana::kokkosBinSort::permute_to_scratch",
596  Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
597  permute_to_scratch );
598  Kokkos::fence();
599 
600  auto copy_back = KOKKOS_LAMBDA( const std::size_t i )
601  {
602  aosoa.setTuple( i, scratch_tuples( i - begin ) );
603  };
604  Kokkos::parallel_for(
605  "Cabana::kokkosBinSort::copy_back",
606  Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
607  copy_back );
608  Kokkos::fence();

623 template <class BinningDataType, class SliceType,
624  class DeviceType = typename BinningDataType::device_type>
625 void permute(
626  const BinningDataType& binning_data, SliceType& slice,
627  typename std::enable_if<( is_binning_data<BinningDataType>::value &&
628  is_slice<SliceType>::value ),
629  int>::type* = 0 )
630 {
631 
632  auto begin = binning_data.rangeBegin();
633  auto end = binning_data.rangeEnd();
634 
635  // Get the number of components in the slice.
636  std::size_t num_comp = 1;
637  for ( std::size_t d = 2; d < slice.rank(); ++d )
638  num_comp *= slice.extent( d );
639 
640  // Get the raw slice data.
641  auto slice_data = slice.data();
642 
643  Kokkos::View<typename SliceType::value_type**, DeviceType> scratch_array(
644  Kokkos::ViewAllocateWithoutInitializing( "scratch_array" ), end - begin,
645  num_comp );
646 
647  auto permute_to_scratch = KOKKOS_LAMBDA( const std::size_t i )
648  {
649  auto permute_i = binning_data.permutation( i - begin );
650  auto s = SliceType::index_type::s( permute_i );
651  auto a = SliceType::index_type::a( permute_i );
652  std::size_t slice_offset = s * slice.stride( 0 ) + a;
653  for ( std::size_t n = 0; n < num_comp; ++n )
654  scratch_array( i - begin, n ) =
655  slice_data[slice_offset + SliceType::vector_length * n];
656  };
657  Kokkos::parallel_for(
658  "Cabana::kokkosBinSort::permute_to_scratch",
659  Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
660  permute_to_scratch );
661  Kokkos::fence();
662 
663  auto copy_back = KOKKOS_LAMBDA( const std::size_t i )
664  {
665  auto s = SliceType::index_type::s( i );
666  auto a = SliceType::index_type::a( i );
667  std::size_t slice_offset = s * slice.stride( 0 ) + a;
668  for ( std::size_t n = 0; n < num_comp; ++n )
669  slice_data[slice_offset + SliceType::vector_length * n] =
670  scratch_array( i - begin, n );
671  };
672  Kokkos::parallel_for(
673  "Cabana::kokkosBinSort::copy_back",
674  Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
675  copy_back );
676  Kokkos::fence();
677 }
