137 // Create a permutation vector over a range subset using a comparator over the
138 // given Kokkos View of keys.
139 template <class KeyViewType, class Comparator,
140  class DeviceType = typename KeyViewType::device_type>
141 BinningData<DeviceType>
142 kokkosBinSort( KeyViewType keys, Comparator comp, const bool sort_within_bins,
143  const std::size_t begin, const std::size_t end )
144 {
145  Kokkos::BinSort<KeyViewType, Comparator, DeviceType> bin_sort(
146  keys, begin, end, comp, sort_within_bins );
147  bin_sort.create_permute_vector();
148  return BinningData<DeviceType>( begin, end, bin_sort.get_bin_count(),
149  bin_sort.get_bin_offsets(),
150  bin_sort.get_permute_vector() );
151 }

154 // Given a set of keys, find the minimum and maximum over the given range.
155 template <class KeyViewType,
156  class DeviceType = typename KeyViewType::device_type>
157 Kokkos::MinMaxScalar<typename KeyViewType::non_const_value_type>
158 keyMinMax( KeyViewType keys, const std::size_t begin, const std::size_t end )
159 {
160  Kokkos::MinMaxScalar<typename KeyViewType::non_const_value_type> result;
161  Kokkos::MinMax<typename KeyViewType::non_const_value_type> reducer(
162  result );
163  Kokkos::parallel_reduce(
164  "Cabana::keyMinMax",
165  Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
166  Kokkos::Impl::min_max_functor<KeyViewType>( keys ), reducer );
167  Kokkos::fence();
168  return result;
169 }

171 //---------------------------------------------------------------------------//
172 // Sort an AoSoA over a subset of its range using the given Kokkos View of
173 // keys.
174 template <class KeyViewType,
175  class DeviceType = typename KeyViewType::device_type>
176 BinningData<DeviceType>
177 kokkosBinSort1d( KeyViewType keys, const int nbin, const bool sort_within_bins,
178  const std::size_t begin, const std::size_t end )
179 {
180  // Find the minimum and maximum key values.
181  auto key_bounds =
182  Impl::keyMinMax<KeyViewType, DeviceType>( keys, begin, end );
183 
184  // Create a sorting comparator.
185  Kokkos::BinOp1D<KeyViewType> comp( nbin, key_bounds.min_val,
186  key_bounds.max_val );
187 
188  // BinSort
189  return kokkosBinSort<KeyViewType, decltype( comp ), DeviceType>(
190  keys, comp, sort_within_bins, begin, end );
191 }

193 //---------------------------------------------------------------------------//
194 // Copy the a 1D slice into a Kokkos view.
195 template <class SliceType, class DeviceType = typename SliceType::device_type>
196 Kokkos::View<typename SliceType::value_type*, typename SliceType::device_type>
197 copySliceToKeys( SliceType slice )
198 {
199  using KeyViewType =
200  Kokkos::View<typename SliceType::value_type*, DeviceType>;
201  KeyViewType keys( Kokkos::ViewAllocateWithoutInitializing( "slice_keys" ),
202  slice.size() );
203  Kokkos::RangePolicy<typename DeviceType::execution_space> exec_policy(
204  0, slice.size() );
205  auto copy_op = KOKKOS_LAMBDA( const std::size_t i )
206  {
207  keys( i ) = slice( i );
208  };
209  Kokkos::parallel_for( "Cabana::copySliceToKeys::copy_op", exec_policy,
210  copy_op );
211  Kokkos::fence();
212  return keys;
213 }

553 template <class BinningDataType, class AoSoA_t,
554  class DeviceType = typename BinningDataType::device_type>
555 void permute(
556  const BinningDataType& binning_data, AoSoA_t& aosoa,
557  typename std::enable_if<( is_binning_data<BinningDataType>::value &&
558  is_aosoa<AoSoA_t>::value ),
559  int>::type* = 0 )
560 {
561  auto begin = binning_data.rangeBegin();
562  auto end = binning_data.rangeEnd();
563 
564  Kokkos::View<typename AoSoA_t::tuple_type*, DeviceType> scratch_tuples(
565  Kokkos::ViewAllocateWithoutInitializing( "scratch_tuples" ),
566  end - begin );
567 
568  auto permute_to_scratch = KOKKOS_LAMBDA( const std::size_t i )
569  {
570  scratch_tuples( i - begin ) =
571  aosoa.getTuple( binning_data.permutation( i - begin ) );
572  };
573  Kokkos::parallel_for(
574  "Cabana::kokkosBinSort::permute_to_scratch",
575  Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
576  permute_to_scratch );
577  Kokkos::fence();
578 
579  auto copy_back = KOKKOS_LAMBDA( const std::size_t i )
580  {
581  aosoa.setTuple( i, scratch_tuples( i - begin ) );
582  };
583  Kokkos::parallel_for(
584  "Cabana::kokkosBinSort::copy_back",
585  Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
586  copy_back );
587  Kokkos::fence();
588 }

602 template <class BinningDataType, class SliceType,
603  class DeviceType = typename BinningDataType::device_type>
604 void permute(
605  const BinningDataType& binning_data, SliceType& slice,
606  typename std::enable_if<( is_binning_data<BinningDataType>::value &&
607  is_slice<SliceType>::value ),
608  int>::type* = 0 )
609 {
610 
611  auto begin = binning_data.rangeBegin();
612  auto end = binning_data.rangeEnd();
613 
614  // Get the number of components in the slice.
615  std::size_t num_comp = 1;
616  for ( std::size_t d = 2; d < slice.rank(); ++d )
617  num_comp *= slice.extent( d );
618 
619  // Get the raw slice data.
620  auto slice_data = slice.data();
621 
622  Kokkos::View<typename SliceType::value_type**, DeviceType> scratch_array(
623  Kokkos::ViewAllocateWithoutInitializing( "scratch_array" ), end - begin,
624  num_comp );
625 
626  auto permute_to_scratch = KOKKOS_LAMBDA( const std::size_t i )
627  {
628  auto permute_i = binning_data.permutation( i - begin );
629  auto s = SliceType::index_type::s( permute_i );
630  auto a = SliceType::index_type::a( permute_i );
631  std::size_t slice_offset = s * slice.stride( 0 ) + a;
632  for ( std::size_t n = 0; n < num_comp; ++n )
633  scratch_array( i - begin, n ) =
634  slice_data[slice_offset + SliceType::vector_length * n];
635  };
636  Kokkos::parallel_for(
637  "Cabana::kokkosBinSort::permute_to_scratch",
638  Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
639  permute_to_scratch );
640  Kokkos::fence();
641 
642  auto copy_back = KOKKOS_LAMBDA( const std::size_t i )
643  {
644  auto s = SliceType::index_type::s( i );
645  auto a = SliceType::index_type::a( i );
646  std::size_t slice_offset = s * slice.stride( 0 ) + a;
647  for ( std::size_t n = 0; n < num_comp; ++n )
648  slice_data[slice_offset + SliceType::vector_length * n] =
649  scratch_array( i - begin, n );
650  };
651  Kokkos::parallel_for(
652  "Cabana::kokkosBinSort::copy_back",
653  Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
654  copy_back );
655  Kokkos::fence();
656 }
