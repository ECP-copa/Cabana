165{
166 Kokkos::Profiling::pushRegion( "Cabana::BinSort" );
167 Kokkos::BinSort<KeyViewType, Comparator, DeviceType> bin_sort(

169 bin_sort.create_permute_vector();

179template <class KeyViewType,
180 class DeviceType = typename KeyViewType::device_type>
181Kokkos::MinMaxScalar<typename KeyViewType::non_const_value_type>
182keyMinMax( KeyViewType keys, const std::size_t begin, const std::size_t end )
183{
184 Kokkos::Profiling::pushRegion( "Cabana::keyMinMax" );
185
186 using KeyValueType = typename KeyViewType::non_const_value_type;
187 Kokkos::MinMaxScalar<KeyValueType> result;
188 Kokkos::MinMax<KeyValueType> reducer( result );
189 Kokkos::parallel_reduce(
190 "Cabana::keyMinMax",
191 Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
192 KOKKOS_LAMBDA( std::size_t i, decltype( result )& local_minmax ) {
193 auto const val = keys( i );
194 if ( val < local_minmax.min_val )
195 {
196 local_minmax.min_val = val;
197 }
198 if ( val > local_minmax.max_val )
199 {
200 local_minmax.max_val = val;
201 }
202 },
203 reducer );
204 Kokkos::fence();
205
206 Kokkos::Profiling::popRegion();
207
208 return result;
209}

214template <class KeyViewType,
215 class DeviceType = typename KeyViewType::device_type>
216BinningData<DeviceType>
217kokkosBinSort1d( KeyViewType keys, const int nbin, const bool sort_within_bins,
218 const std::size_t begin, const std::size_t end )
219{
220 // Find the minimum and maximum key values.
221 auto key_bounds =
222 Impl::keyMinMax<KeyViewType, DeviceType>( keys, begin, end );
223
224 // Create a sorting comparator.
225 Kokkos::BinOp1D<KeyViewType> comp( nbin, key_bounds.min_val,
226 key_bounds.max_val );
227
228 // BinSort
229 return kokkosBinSort<KeyViewType, decltype( comp ), DeviceType>(
230 keys, comp, sort_within_bins, begin, end );
231}

541template <class BinningDataType, class AoSoA_t,
542 class DeviceType = typename BinningDataType::device_type>
543void permute(
544 const BinningDataType& binning_data, AoSoA_t& aosoa,
545 typename std::enable_if<( is_binning_data<BinningDataType>::value &&
546 is_aosoa<AoSoA_t>::value ),
547 int>::type* = 0 )
548{
549 Kokkos::Profiling::pushRegion( "Cabana::permute" );
550
551 auto begin = binning_data.rangeBegin();
552 auto end = binning_data.rangeEnd();
553
554 Kokkos::View<typename AoSoA_t::tuple_type*, DeviceType> scratch_tuples(
555 Kokkos::ViewAllocateWithoutInitializing( "scratch_tuples" ),
556 end - begin );
557
558 auto permute_to_scratch = KOKKOS_LAMBDA( const std::size_t i )
559 {
560 scratch_tuples( i - begin ) =
561 aosoa.getTuple( binning_data.permutation( i - begin ) );
562 };
563 Kokkos::parallel_for(
564 "Cabana::kokkosBinSort::permute_to_scratch",
565 Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
566 permute_to_scratch );
567 Kokkos::fence();
568
569 auto copy_back = KOKKOS_LAMBDA( const std::size_t i )
570 {
571 aosoa.setTuple( i, scratch_tuples( i - begin ) );
572 };
573 Kokkos::parallel_for(
574 "Cabana::kokkosBinSort::copy_back",
575 Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
576 copy_back );
577 Kokkos::fence();
578
579 Kokkos::Profiling::popRegion();
580}

592template <class BinningDataType, class SliceType,
593 class DeviceType = typename BinningDataType::device_type>
594void permute(
595 const BinningDataType& binning_data, SliceType& slice,
596 typename std::enable_if<( is_binning_data<BinningDataType>::value &&
597 is_slice<SliceType>::value ),
598 int>::type* = 0 )
599{
600 Kokkos::Profiling::pushRegion( "Cabana::permute" );
601
602 auto begin = binning_data.rangeBegin();
603 auto end = binning_data.rangeEnd();
604
605 // Get the number of components in the slice.
606 std::size_t num_comp = 1;
607 for ( std::size_t d = 2; d < slice.viewRank(); ++d )
608 num_comp *= slice.extent( d );
609
610 // Get the raw slice data.
611 auto slice_data = slice.data();
612
613 Kokkos::View<typename SliceType::value_type**, DeviceType> scratch_array(
614 Kokkos::ViewAllocateWithoutInitializing( "scratch_array" ), end - begin,
615 num_comp );
616
617 auto permute_to_scratch = KOKKOS_LAMBDA( const std::size_t i )
618 {
619 auto permute_i = binning_data.permutation( i - begin );
620 auto s = SliceType::index_type::s( permute_i );
621 auto a = SliceType::index_type::a( permute_i );
622 std::size_t slice_offset = s * slice.stride( 0 ) + a;
623 for ( std::size_t n = 0; n < num_comp; ++n )
624 scratch_array( i - begin, n ) =
625 slice_data[slice_offset + SliceType::vector_length * n];
626 };
627 Kokkos::parallel_for(
628 "Cabana::kokkosBinSort::permute_to_scratch",
629 Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
630 permute_to_scratch );
631 Kokkos::fence();
632
633 auto copy_back = KOKKOS_LAMBDA( const std::size_t i )
634 {
635 auto s = SliceType::index_type::s( i );
636 auto a = SliceType::index_type::a( i );
637 std::size_t slice_offset = s * slice.stride( 0 ) + a;
638 for ( std::size_t n = 0; n < num_comp; ++n )
639 slice_data[slice_offset + SliceType::vector_length * n] =
640 scratch_array( i - begin, n );
641 };
642 Kokkos::parallel_for(
643 "Cabana::kokkosBinSort::copy_back",
644 Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
645 copy_back );
646 Kokkos::fence();
647
648 Kokkos::Profiling::popRegion();
649}
