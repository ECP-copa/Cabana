163kokkosBinSort( KeyViewType keys, Comparator comp, const bool sort_within_bins,
164 const std::size_t begin, const std::size_t end )
165{
166 Kokkos::BinSort<KeyViewType, Comparator, DeviceType> bin_sort(
167 keys, begin, end, comp, sort_within_bins );
168 bin_sort.create_permute_vector();
169 return BinningData<DeviceType>( begin, end, bin_sort.get_bin_count(),
170 bin_sort.get_bin_offsets(),
171 bin_sort.get_permute_vector() );

176template <class KeyViewType,
177 class DeviceType = typename KeyViewType::device_type>
178Kokkos::MinMaxScalar<typename KeyViewType::non_const_value_type>
179keyMinMax( KeyViewType keys, const std::size_t begin, const std::size_t end )
180{
181 using KeyValueType = typename KeyViewType::non_const_value_type;
182 Kokkos::MinMaxScalar<KeyValueType> result;
183 Kokkos::MinMax<KeyValueType> reducer( result );
184 Kokkos::parallel_reduce(
185 "Cabana::keyMinMax",
186 Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
187 KOKKOS_LAMBDA( std::size_t i, decltype( result )& local_minmax ) {
188 auto const val = keys( i );
189 if ( val < local_minmax.min_val )
190 {
191 local_minmax.min_val = val;
192 }
193 if ( val > local_minmax.max_val )
194 {
195 local_minmax.max_val = val;
196 }
197 },
198 reducer );
199 Kokkos::fence();
200 return result;
201}

206template <class KeyViewType,
207 class DeviceType = typename KeyViewType::device_type>
208BinningData<DeviceType>
209kokkosBinSort1d( KeyViewType keys, const int nbin, const bool sort_within_bins,
210 const std::size_t begin, const std::size_t end )
211{
212 // Find the minimum and maximum key values.
213 auto key_bounds =
214 Impl::keyMinMax<KeyViewType, DeviceType>( keys, begin, end );
215
216 // Create a sorting comparator.
217 Kokkos::BinOp1D<KeyViewType> comp( nbin, key_bounds.min_val,
218 key_bounds.max_val );
219
220 // BinSort
221 return kokkosBinSort<KeyViewType, decltype( comp ), DeviceType>(
222 keys, comp, sort_within_bins, begin, end );
223}

227template <class SliceType, class DeviceType = typename SliceType::device_type>
228Kokkos::View<typename SliceType::value_type*, typename SliceType::device_type>
229copySliceToKeys( SliceType slice )
230{
231 using KeyViewType =
232 Kokkos::View<typename SliceType::value_type*, DeviceType>;
233 KeyViewType keys( Kokkos::ViewAllocateWithoutInitializing( "slice_keys" ),
234 slice.size() );
235 Kokkos::RangePolicy<typename DeviceType::execution_space> exec_policy(
236 0, slice.size() );
237 auto copy_op = KOKKOS_LAMBDA( const std::size_t i )
238 {
239 keys( i ) = slice( i );
240 };
241 Kokkos::parallel_for( "Cabana::copySliceToKeys::copy_op", exec_policy,
242 copy_op );
243 Kokkos::fence();
244 return keys;
245}

585template <class BinningDataType, class AoSoA_t,
586 class DeviceType = typename BinningDataType::device_type>
587void permute(
588 const BinningDataType& binning_data, AoSoA_t& aosoa,
589 typename std::enable_if<( is_binning_data<BinningDataType>::value &&
590 is_aosoa<AoSoA_t>::value ),
591 int>::type* = 0 )
592{
593 auto begin = binning_data.rangeBegin();
594 auto end = binning_data.rangeEnd();
595
596 Kokkos::View<typename AoSoA_t::tuple_type*, DeviceType> scratch_tuples(
597 Kokkos::ViewAllocateWithoutInitializing( "scratch_tuples" ),
598 end - begin );
599
600 auto permute_to_scratch = KOKKOS_LAMBDA( const std::size_t i )
601 {
602 scratch_tuples( i - begin ) =
603 aosoa.getTuple( binning_data.permutation( i - begin ) );
604 };
605 Kokkos::parallel_for(
606 "Cabana::kokkosBinSort::permute_to_scratch",
607 Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
608 permute_to_scratch );
609 Kokkos::fence();
610
611 auto copy_back = KOKKOS_LAMBDA( const std::size_t i )
612 {
613 aosoa.setTuple( i, scratch_tuples( i - begin ) );
614 };
615 Kokkos::parallel_for(
616 "Cabana::kokkosBinSort::copy_back",
617 Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
618 copy_back );
619 Kokkos::fence();
620}

634template <class BinningDataType, class SliceType,
635 class DeviceType = typename BinningDataType::device_type>
636void permute(
637 const BinningDataType& binning_data, SliceType& slice,
638 typename std::enable_if<( is_binning_data<BinningDataType>::value &&
639 is_slice<SliceType>::value ),
640 int>::type* = 0 )
641{
642
643 auto begin = binning_data.rangeBegin();
644 auto end = binning_data.rangeEnd();
645
646 // Get the number of components in the slice.
647 std::size_t num_comp = 1;
648 for ( std::size_t d = 2; d < slice.rank(); ++d )
649 num_comp *= slice.extent( d );
650
651 // Get the raw slice data.
652 auto slice_data = slice.data();
653
654 Kokkos::View<typename SliceType::value_type**, DeviceType> scratch_array(
655 Kokkos::ViewAllocateWithoutInitializing( "scratch_array" ), end - begin,
656 num_comp );
657
658 auto permute_to_scratch = KOKKOS_LAMBDA( const std::size_t i )
659 {
660 auto permute_i = binning_data.permutation( i - begin );
661 auto s = SliceType::index_type::s( permute_i );
662 auto a = SliceType::index_type::a( permute_i );
663 std::size_t slice_offset = s * slice.stride( 0 ) + a;
664 for ( std::size_t n = 0; n < num_comp; ++n )
665 scratch_array( i - begin, n ) =
666 slice_data[slice_offset + SliceType::vector_length * n];
667 };
668 Kokkos::parallel_for(
669 "Cabana::kokkosBinSort::permute_to_scratch",
670 Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
671 permute_to_scratch );
672 Kokkos::fence();
673
674 auto copy_back = KOKKOS_LAMBDA( const std::size_t i )
675 {
676 auto s = SliceType::index_type::s( i );
677 auto a = SliceType::index_type::a( i );
678 std::size_t slice_offset = s * slice.stride( 0 ) + a;
679 for ( std::size_t n = 0; n < num_comp; ++n )
680 slice_data[slice_offset + SliceType::vector_length * n] =
681 scratch_array( i - begin, n );
682 };
683 Kokkos::parallel_for(
684 "Cabana::kokkosBinSort::copy_back",
685 Kokkos::RangePolicy<typename DeviceType::execution_space>( begin, end ),
686 copy_back );
687 Kokkos::fence();
688}
