139 BinningData<

141 kokkosBinSort( KeyViewType keys,
142  Comparator comp,
143  const bool sort_within_bins,
144  const std::size_t begin,
145  const std::size_t end )
146 {

159 //---------------------------------------------------------------------------//
160 // Given a set of keys, find the minimum and maximum over the given range.
161 template<class KeyViewType>
162 Kokkos::MinMaxScalar<typename KeyViewType::non_const_value_type>
163 keyMinMax( KeyViewType keys, const std::size_t begin, const std::size_t end )
164 {
165  Kokkos::MinMaxScalar<typename KeyViewType::non_const_value_type> result;
166  Kokkos::MinMax<typename KeyViewType::non_const_value_type> reducer(result);
167  Kokkos::parallel_reduce(
168  "Cabana::keyMinMax",
169  Kokkos::RangePolicy<typename KeyViewType::execution_space>(begin,end),
170  Kokkos::Impl::min_max_functor<KeyViewType>(keys),
171  reducer );
172  Kokkos::fence();
173  return result;
174 }

176 //---------------------------------------------------------------------------//
177 // Sort an AoSoA over a subset of its range using the given Kokkos View of
178 // keys.
179 template<class KeyViewType>
180 BinningData<
181  typename KokkosSpaceToCabana<typename KeyViewType::memory_space>::type>
182 kokkosBinSort1d( KeyViewType keys,
183  const int nbin,
184  const bool sort_within_bins,
185  const std::size_t begin,
186  const std::size_t end )
187 {
188  // Find the minimum and maximum key values.
189  auto key_bounds = Impl::keyMinMax( keys, begin, end );
190 
191  // Create a sorting comparator.
192  Kokkos::BinOp1D<KeyViewType> comp(
193  nbin, key_bounds.min_val, key_bounds.max_val );
194 
195  // BinSort
196  return kokkosBinSort( keys, comp, sort_within_bins, begin, end );
197 }

200 // Copy the a 1D slice into a Kokkos view.
201 template<class SliceType>
202 Kokkos::View<typename SliceType::value_type*,
203  typename SliceType::kokkos_memory_space>
204 copySliceToKeys( SliceType slice )
205 {
206  using KeyViewType = Kokkos::View<typename SliceType::value_type*,
207  typename SliceType::kokkos_memory_space>;
208  KeyViewType keys( "slice_keys", slice.size() );
209  Kokkos::RangePolicy<typename SliceType::kokkos_execution_space>
210  exec_policy( 0, slice.size() );
211  auto copy_op = KOKKOS_LAMBDA( const std::size_t i ) { keys(i) = slice(i); };
212  Kokkos::parallel_for( "Cabana::copySliceToKeys::copy_op",
213  exec_policy,
214  copy_op );
215  Kokkos::fence();
216  return keys;
217 }

570 template<class BinningDataType, class AoSoA_t>
571 void permute( const BinningDataType& binning_data,
572  AoSoA_t& aosoa,
573  typename std::enable_if<(is_binning_data<BinningDataType>::value &&
574  is_aosoa<AoSoA_t>::value),
575  int>::type * = 0)
576 {
577  auto begin = binning_data.rangeBegin();
578  auto end = binning_data.rangeEnd();
579 
580  Kokkos::View<typename AoSoA_t::tuple_type*,
581  typename BinningDataType::KokkosMemorySpace>
582  scratch_tuples( "scratch_tuples", end - begin );
583 
584  auto permute_to_scratch =
585  KOKKOS_LAMBDA( const std::size_t i )
586  {
587  scratch_tuples( i - begin ) =
588  aosoa.getTuple( binning_data.permutation(i-begin) );
589  };
590  Kokkos::parallel_for(
591  "Cabana::kokkosBinSort::permute_to_scratch",
592  Kokkos::RangePolicy<typename BinningDataType::KokkosExecutionSpace>(begin,end),
593  permute_to_scratch );
594  Kokkos::fence();
595 
596  auto copy_back = KOKKOS_LAMBDA( const std::size_t i )
597  { aosoa.setTuple( i, scratch_tuples(i-begin) ); };
598  Kokkos::parallel_for(
599  "Cabana::kokkosBinSort::copy_back",
600  Kokkos::RangePolicy<typename BinningDataType::KokkosExecutionSpace>(begin,end),
601  copy_back );
602  Kokkos::fence();
603 }
