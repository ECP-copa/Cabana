188template <class KeyViewType,
189 class ExecutionSpace = typename KeyViewType::execution_space>
190Kokkos::MinMaxScalar<typename KeyViewType::non_const_value_type>
191keyMinMax( KeyViewType keys, const std::size_t begin, const std::size_t end )
192{
193 Kokkos::Profiling::pushRegion( "Cabana::keyMinMax" );
194
195 using memory_space = typename KeyViewType::memory_space;
196 static_assert( is_accessible_from<memory_space, ExecutionSpace>{}, "" );
197
198 using KeyValueType = typename KeyViewType::non_const_value_type;
199 Kokkos::MinMaxScalar<KeyValueType> result;
200 Kokkos::MinMax<KeyValueType> reducer( result );
201 Kokkos::parallel_reduce(
202 "Cabana::keyMinMax", Kokkos::RangePolicy<ExecutionSpace>( begin, end ),
203 KOKKOS_LAMBDA( std::size_t i, decltype( result )& local_minmax ) {
204 auto const val = keys( i );
205 if ( val < local_minmax.min_val )
206 {
207 local_minmax.min_val = val;
208 }
209 if ( val > local_minmax.max_val )
210 {
211 local_minmax.max_val = val;
212 }
213 },
214 reducer );
215 Kokkos::fence();
216
217 Kokkos::Profiling::popRegion();
218
219 return result;
220}

225template <class KeyViewType,
226 class ExecutionSpace = typename KeyViewType::execution_space>
227auto kokkosBinSort1d( KeyViewType keys, const int nbin,
228 const bool sort_within_bins, const std::size_t begin,
229 const std::size_t end )
230{
231 // Find the minimum and maximum key values.
232 auto key_bounds =
233 Impl::keyMinMax<KeyViewType, ExecutionSpace>( keys, begin, end );
234
235 // Create a sorting comparator.
236 Kokkos::BinOp1D<KeyViewType> comp( nbin, key_bounds.min_val,
237 key_bounds.max_val );
238
239 // BinSort
240 return kokkosBinSort<KeyViewType, decltype( comp ), ExecutionSpace>(
241 keys, comp, sort_within_bins, begin, end );
242}

556template <class BinningDataType, class AoSoA_t,
557 class ExecutionSpace = typename BinningDataType::execution_space>
558void permute(
559 const BinningDataType& binning_data, AoSoA_t& aosoa,
560 typename std::enable_if<( is_binning_data<BinningDataType>::value &&
561 is_aosoa<AoSoA_t>::value ),
562 int>::type* = 0 )
563{
564 Kokkos::Profiling::pushRegion( "Cabana::permute" );
565
566 using memory_space = typename BinningDataType::memory_space;
567 static_assert( is_accessible_from<memory_space, ExecutionSpace>{}, "" );
568
569 auto begin = binning_data.rangeBegin();
570 auto end = binning_data.rangeEnd();
571
572 using memory_space = typename BinningDataType::memory_space;
573 Kokkos::View<typename AoSoA_t::tuple_type*, memory_space> scratch_tuples(
574 Kokkos::ViewAllocateWithoutInitializing( "scratch_tuples" ),
575 end - begin );
576
577 auto permute_to_scratch = KOKKOS_LAMBDA( const std::size_t i )
578 {
579 scratch_tuples( i - begin ) =
580 aosoa.getTuple( binning_data.permutation( i - begin ) );
581 };
582 Kokkos::parallel_for( "Cabana::kokkosBinSort::permute_to_scratch",
583 Kokkos::RangePolicy<ExecutionSpace>( begin, end ),
584 permute_to_scratch );
585 Kokkos::fence();
586
587 auto copy_back = KOKKOS_LAMBDA( const std::size_t i )
588 {
589 aosoa.setTuple( i, scratch_tuples( i - begin ) );
590 };
591 Kokkos::parallel_for( "Cabana::kokkosBinSort::copy_back",
592 Kokkos::RangePolicy<ExecutionSpace>( begin, end ),
593 copy_back );
594 Kokkos::fence();
595
596 Kokkos::Profiling::popRegion();
597}

609template <class BinningDataType, class SliceType,
610 class ExecutionSpace = typename BinningDataType::execution_space>
611void permute(
612 const BinningDataType& binning_data, SliceType& slice,
613 typename std::enable_if<( is_binning_data<BinningDataType>::value &&
614 is_slice<SliceType>::value ),
615 int>::type* = 0 )
616{
617 Kokkos::Profiling::pushRegion( "Cabana::permute" );
618
619 using memory_space = typename BinningDataType::memory_space;
620 static_assert( is_accessible_from<memory_space, ExecutionSpace>{}, "" );
621
622 auto begin = binning_data.rangeBegin();
623 auto end = binning_data.rangeEnd();
624
625 // Get the number of components in the slice.
626 std::size_t num_comp = 1;
627 for ( std::size_t d = 2; d < slice.viewRank(); ++d )
628 num_comp *= slice.extent( d );
629
630 // Get the raw slice data.
631 auto slice_data = slice.data();
632
633 Kokkos::View<typename SliceType::value_type**, memory_space> scratch_array(
634 Kokkos::ViewAllocateWithoutInitializing( "scratch_array" ), end - begin,
635 num_comp );
636
637 auto permute_to_scratch = KOKKOS_LAMBDA( const std::size_t i )
638 {
639 auto permute_i = binning_data.permutation( i - begin );
640 auto s = SliceType::index_type::s( permute_i );
641 auto a = SliceType::index_type::a( permute_i );
642 std::size_t slice_offset = s * slice.stride( 0 ) + a;
643 for ( std::size_t n = 0; n < num_comp; ++n )
644 scratch_array( i - begin, n ) =
645 slice_data[slice_offset + SliceType::vector_length * n];
646 };
647 Kokkos::parallel_for( "Cabana::kokkosBinSort::permute_to_scratch",
648 Kokkos::RangePolicy<ExecutionSpace>( begin, end ),
649 permute_to_scratch );
650 Kokkos::fence();
651
652 auto copy_back = KOKKOS_LAMBDA( const std::size_t i )
653 {
654 auto s = SliceType::index_type::s( i );
655 auto a = SliceType::index_type::a( i );
656 std::size_t slice_offset = s * slice.stride( 0 ) + a;
657 for ( std::size_t n = 0; n < num_comp; ++n )
658 slice_data[slice_offset + SliceType::vector_length * n] =
659 scratch_array( i - begin, n );
660 };
661 Kokkos::parallel_for( "Cabana::kokkosBinSort::copy_back",
662 Kokkos::RangePolicy<ExecutionSpace>( begin, end ),
663 copy_back );
664 Kokkos::fence();
665
666 Kokkos::Profiling::popRegion();
667}
