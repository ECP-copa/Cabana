281 template <typename Scalar>
282 KOKKOS_INLINE_FUNCTION static void value( const Scalar x0,
283 Scalar values[num_knot] )
284 {
285 // Constants
286 Scalar nine_eights = 9.0 / 8.0;
287
288 // Knot at i - 1
289 Scalar xn = x0 - static_cast<int>( x0 ) + 0.5;
290 values[0] = 0.5 * xn * xn - 1.5 * xn + nine_eights;
291
292 // Knot at i
293 xn -= 1.0;
294 values[1] = -xn * xn + 0.75;
295
296 // Knot at i + 1
297 xn -= 1.0;
298 values[2] = 0.5 * xn * xn + 1.5 * xn + nine_eights;
299 }
300
310 template <typename Scalar>
311 KOKKOS_INLINE_FUNCTION static void
312 gradient( const Scalar x0, const Scalar rdx, Scalar gradients[num_knot] )
313 {
314 // Knot at i - 1
315 Scalar xn = x0 - static_cast<int>( x0 ) + 0.5;
316 gradients[0] = ( xn - 1.5 ) * rdx;
317
318 // Knot at i
319 xn -= 1.0;
320 gradients[1] = ( -2.0 * xn ) * rdx;
321
322 // Knot at i + 1
323 xn -= 1.0;
324 gradients[2] = ( xn + 1.5 ) * rdx;
325 }

365 {
366 indices[0] = -1;
367 indices[1] = 0;
368 indices[2] = 1;
369 indices[3] = 2;
370 }
371
377 template <class Scalar>
378 KOKKOS_INLINE_FUNCTION static void stencil( const Scalar x0,
379 int indices[num_knot] )
380 {
381 indices[0] = static_cast<int>( x0 ) - 1;
382 indices[1] = indices[0] + 1;
383 indices[2] = indices[1] + 1;
384 indices[3] = indices[2] + 1;
385 }
386
394 template <typename Scalar>
395 KOKKOS_INLINE_FUNCTION static void value( const Scalar x0,
396 Scalar values[num_knot] )
397 {
398 // Constants
399 Scalar one_sixth = 1.0 / 6.0;
400 Scalar two_thirds = one_sixth * 4.0;
401 Scalar four_thirds = two_thirds * 2.0;
402
403 // Knot at i - 1
404 Scalar xn = x0 - static_cast<int>( x0 ) + 1.0;
405 Scalar xn2 = xn * xn;
406 values[0] = -xn * xn2 * one_sixth + xn2 - 2.0 * xn + four_thirds;
407
408 // Knot at i
409 xn -= 1.0;
410 xn2 = xn * xn;
411 values[1] = 0.5 * xn * xn2 - xn2 + two_thirds;
412
413 // Knot at i + 1
414 xn -= 1.0;
415 xn2 = xn * xn;
416 values[2] = -0.5 * xn * xn2 - xn2 + two_thirds;
417
418 // Knot at i + 2
419 xn -= 1.0;
420 xn2 = xn * xn;
421 values[3] = xn * xn2 * one_sixth + xn2 + 2.0 * xn + four_thirds;
422 }
423
433 template <typename Scalar>
434 KOKKOS_INLINE_FUNCTION static void
435 gradient( const Scalar x0, const Scalar rdx, Scalar gradients[num_knot] )
436 {
437 // Knot at i - 1
438 Scalar xn = x0 - static_cast<int>( x0 ) + 1.0;
439 gradients[0] = ( -0.5 * xn * xn + 2.0 * xn - 2.0 ) * rdx;
440
441 // Knot at i
442 xn -= 1.0;
443 gradients[1] = ( 1.5 * xn * xn - 2.0 * xn ) * rdx;
444
445 // Knot at i + 1
446 xn -= 1.0;
447 gradients[2] = ( -1.5 * xn * xn - 2.0 * xn ) * rdx;
448
449 // Knot at i + 2
450 xn -= 1.0;
451 gradients[3] = ( 0.5 * xn * xn + 2.0 * xn + 2.0 ) * rdx;
452 }

797template <typename Scalar, int Order, std::size_t NumSpaceDim,
798 class MemorySpace, class EntityType, class DataTags>
799KOKKOS_INLINE_FUNCTION void evaluateSpline(
800 const LocalMesh<MemorySpace, UniformMesh<Scalar, NumSpaceDim>>& local_mesh,
801 const Scalar p[NumSpaceDim],
802 SplineData<Scalar, Order, NumSpaceDim, EntityType, DataTags>& data )
803{
804 // data type
805 using sd_type =
806 SplineData<Scalar, Order, NumSpaceDim, EntityType, DataTags>;
807
808 // spline type.
809
810 // Get the low corner of the mesh.
811 Scalar low_x[NumSpaceDim];
812 Scalar low_x_p1[NumSpaceDim];
813 int low_id[NumSpaceDim];
814 int low_id_p1[NumSpaceDim];
815 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
816 {
817 low_id[d] = 0;
818 low_id_p1[d] = 1;
819 }
820 local_mesh.coordinates( EntityType(), low_id, low_x );
821 local_mesh.coordinates( EntityType(), low_id_p1, low_x_p1 );
822
823 // Compute the physical cell size.
824 Scalar dx[NumSpaceDim];
825 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
826 {
827 dx[d] = low_x_p1[d] - low_x[d];
828 setSplineData( SplinePhysicalCellSize(), data, d, dx[d] );
829 }
830
831 // Compute the inverse physicall cell size.
832 Scalar rdx[NumSpaceDim];
833 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
834 rdx[d] = 1.0 / dx[d];
835
836 // Compute the reference coordinates.
837 Scalar x[NumSpaceDim];
838 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
839 {
840 x[d] = sd_type::spline_type::mapToLogicalGrid( p[d], rdx[d], low_x[d] );
841 setSplineData( SplineLogicalPosition(), data, d, x[d] );
842 }
843
844 // Compute the stencil.
845 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
846 {
847 sd_type::spline_type::stencil( x[d], data.s[d] );
848 }
849
850 // Compute the weight values.
851 setSplineData( SplineWeightValues(), data, x );
852
853 // Compute the weight gradients.
854 setSplineData( SplineWeightPhysicalGradients(), data, x, rdx );
855
856 // Compute the physical distance.
857 setSplineData( SplinePhysicalDistance(), data, low_x, p, dx );
858}
