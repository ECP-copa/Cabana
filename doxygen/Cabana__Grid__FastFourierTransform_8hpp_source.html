150template <class EntityType, class MeshType, class Scalar, class MemorySpace,
151 class Derived>
152class FastFourierTransform
153{
154 public:
156 using entity_type = EntityType;
158 using mesh_type = MeshType;
160 using value_type = Scalar;
161
162 // FIXME: extracting the self type for backwards compatibility with previous
163 // template on DeviceType. Should simply be MemorySpace after next release.
165 using memory_space = typename MemorySpace::memory_space;
166 // FIXME: replace warning with memory space assert after next release.
167 static_assert(
168 Cabana::Impl::deprecated( Kokkos::is_device<MemorySpace>() ) );
169
171 using execution_space = typename memory_space::execution_space;
173 using exec_space [[deprecated]] = execution_space;
175 using device_type [[deprecated]] = typename memory_space::device_type;
176
178 static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;
179
181 std::array<int, num_space_dim> global_high;
183 std::array<int, num_space_dim> global_low;
184
189 FastFourierTransform( const ArrayLayout<EntityType, MeshType>& layout )
190 {
191 checkArrayDofs( layout.dofsPerEntity() );
192
193 // Get the local dimensions of the problem.
194 auto entity_space =
195 layout.localGrid()->indexSpace( Own(), EntityType(), Local() );
196 // Get the global grid.
197 const auto& global_grid = layout.localGrid()->globalGrid();
198
199 for ( std::size_t d = 0; d < num_space_dim; ++d )
200 {
201 // Get the low corner of the global index space on this rank.
202 global_low[d] =
203 (int)global_grid.globalOffset( num_space_dim - d - 1 );
204
205 // Get the high corner of the global index space on this rank.
206 int local_num_entity =
207 (int)entity_space.extent( num_space_dim - d - 1 );
208 global_high[d] = global_low[d] + local_num_entity - 1;
209 }
210 }
211
216 inline void checkArrayDofs( const int dof )
217 {
218 if ( 2 != dof )
219 throw std::logic_error(
220 "Only 1 complex value per entity allowed in FFT" );
221 }

270 template <class ExecutionSpace, class IndexSpaceType, class LViewType,
271 class LGViewType, std::size_t NSD = num_space_dim>
272 std::enable_if_t<3 == NSD, void>
273 copyToLocal( ExecutionSpace exec_space, const IndexSpaceType own_space,
274 LViewType& l_view, const LGViewType lg_view )
275 {
276 Kokkos::parallel_for(
277 "fft_copy_to_work", createExecutionPolicy( own_space, exec_space ),
278 KOKKOS_LAMBDA( const int i, const int j, const int k ) {
279 auto iw = i - own_space.min( Dim::I );
280 auto jw = j - own_space.min( Dim::J );
281 auto kw = k - own_space.min( Dim::K );
282 l_view( iw, jw, kw, 0 ) = lg_view( i, j, k, 0 );
283 l_view( iw, jw, kw, 1 ) = lg_view( i, j, k, 1 );
284 } );
285 }

477template <class EntityType, class MeshType, class Scalar, class MemorySpace,
478 class ExecSpace, class BackendType>
479class HeffteFastFourierTransform
480 : public FastFourierTransform<
481 EntityType, MeshType, Scalar, MemorySpace,
482 HeffteFastFourierTransform<EntityType, MeshType, Scalar, MemorySpace,
483 ExecSpace, BackendType>>
484{
485 public:
487 using value_type = Scalar;
488
489 // FIXME: extracting the self type for backwards compatibility with previous
490 // template on DeviceType. Should simply be MemorySpace after next release.
492 using memory_space = typename MemorySpace::memory_space;
493 // FIXME: replace warning with memory space assert after next release.
494 static_assert(
495 Cabana::Impl::deprecated( Kokkos::is_device<MemorySpace>() ) );
496
498 using execution_space = ExecSpace;
500 using exec_space [[deprecated]] = execution_space;
502 using device_type [[deprecated]] = typename memory_space::device_type;
504 using backend_type = BackendType;
506 using mesh_type = MeshType;
507
509 static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;
510
512 using heffte_backend_type =
513 typename Impl::HeffteBackendTraits<execution_space,
514 backend_type>::backend_type;
515
517 execution_space heffte_execution_space;
518
525 HeffteFastFourierTransform( execution_space exec_space,
526 const ArrayLayout<EntityType, MeshType>& layout,
527 const FastFourierTransformParams& params )
528 : FastFourierTransform<
529 EntityType, MeshType, Scalar, MemorySpace,
530 HeffteFastFourierTransform<EntityType, MeshType, Scalar,
531 MemorySpace, ExecSpace, BackendType>>(
532 layout )
533 , heffte_execution_space( exec_space )
534 {
535 // heFFTe correctly handles 2D or 3D domains within "box3d"
536 heffte::box3d<> inbox = { this->global_low, this->global_high };
537 heffte::box3d<> outbox = { this->global_low, this->global_high };
538
539 heffte::plan_options heffte_params =
540 heffte::default_options<heffte_backend_type>();
541 // TODO: use all three heffte options for algorithm
542 bool alltoall = params.getAllToAll();
543 if ( alltoall )
544 heffte_params.algorithm = heffte::reshape_algorithm::alltoallv;
545 else
546 heffte_params.algorithm = heffte::reshape_algorithm::p2p;
547 heffte_params.use_pencils = params.getPencils();
548 heffte_params.use_reorder = params.getReorder();
549
550 // Create the heFFTe main class (separated to handle SYCL queue
551 // correctly).
552 _fft = Impl::createHeffteFft3d(
553 heffte_execution_space, heffte_backend_type{}, inbox, outbox,
554 layout.localGrid()->globalGrid().comm(), heffte_params );
555 long long fftsize = std::max( _fft->size_outbox(), _fft->size_inbox() );
556
557 // Check the size.
558 auto entity_space =
559 layout.localGrid()->indexSpace( Own(), EntityType(), Local() );
560 if ( fftsize < (int)entity_space.size() )
561 throw std::logic_error( "Expected FFT allocation size smaller "
562 "than local grid size" );
563
564 _fft_work = Kokkos::View<Scalar*, memory_space>(
565 Kokkos::ViewAllocateWithoutInitializing( "fft_work" ),
566 2 * fftsize );
567 _workspace = Kokkos::View<Scalar* [2], memory_space>(
568 Kokkos::ViewAllocateWithoutInitializing( "workspace" ),
569 _fft->size_workspace() );
570 }

600 template <class Array_t>
601 void compute( const Array_t& x, const int flag, const heffte::scale scale )
602 {
603 // Create a subview of the work array to write the local data into.
604 auto own_space =
605 x.layout()->localGrid()->indexSpace( Own(), EntityType(), Local() );
606 auto local_view_space = appendDimension( own_space, 2 );
607 auto local_view = createView<Scalar, Kokkos::LayoutRight, memory_space>(
608 local_view_space, _fft_work.data() );
609
610 // TODO: pull this out to template function
611 // Copy to the work array. The work array only contains owned data.
612 auto localghost_view = x.view();
613
614 this->copyToLocal( heffte_execution_space, own_space, local_view,
615 localghost_view );
616
617 if ( flag == 1 )
618 {
619 _fft->forward(
620 reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
621 reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
622 reinterpret_cast<std::complex<Scalar>*>( _workspace.data() ),
623 scale );
624 }
625 else if ( flag == -1 )
626 {
627 _fft->backward(
628 reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
629 reinterpret_cast<std::complex<Scalar>*>( _fft_work.data() ),
630 reinterpret_cast<std::complex<Scalar>*>( _workspace.data() ),
631 scale );
632 }
633 else
634 {
635 throw std::logic_error(
636 "Only 1:forward and -1:backward are allowed as compute flag" );
637 }
638
639 // Copy back to output array.
640 this->copyFromLocal( heffte_execution_space, own_space, local_view,
641 localghost_view );
642 }
643
644 private:
645 // heFFTe correctly handles 2D or 3D FFTs within "fft3d"
646 std::shared_ptr<heffte::fft3d<heffte_backend_type>> _fft;
647 Kokkos::View<Scalar*, memory_space> _fft_work;
648 Kokkos::View<Scalar* [2], memory_space> _workspace;
649};
