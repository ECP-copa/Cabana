208  // This function should be private but we need to expose it as public to
209  // launch CUDA kernels with class data.
210  template<class SliceType>
211  void build( SliceType positions,
212  const std::size_t begin,
213  const std::size_t end,
214  const typename SliceType::value_type grid_delta[3],
215  const typename SliceType::value_type grid_min[3],
216  const typename SliceType::value_type grid_max[3] )
217  {
218  // Allocate the binning data. Note that the permutation vector spans
219  // only the length of begin-end;
220  std::size_t ncell = totalBins();
221  Kokkos::View<int*,KokkosMemorySpace> counts( "counts", ncell );
222  OffsetView offsets( "offsets", ncell );
223  OffsetView permute( "permute", end - begin );
224 
225  // Get a local copy of the grid because it is class data and a lambda
226  // function will not capture it otherwise via CUDA.
227  auto grid = _grid;
228 
229  // Count.
230  Kokkos::RangePolicy<typename OffsetView::execution_space>
231  particle_range( begin, end );
232  Kokkos::deep_copy( counts, 0 );
233  auto cell_count =
234  KOKKOS_LAMBDA( const std::size_t p )
235  {
236  int i, j, k;
237  grid.locatePoint(
238  positions(p,0), positions(p,1), positions(p,2), i , j, k );
239  Kokkos::atomic_increment(
240  &counts(grid.cardinalCellIndex(i,j,k)) );
241  };
242  Kokkos::parallel_for( "Cabana::LinkedCellList::build::cell_count",
243  particle_range,
244  cell_count );
245  Kokkos::fence();
246 
247  // Compute offsets.
248  Kokkos::RangePolicy<typename OffsetView::execution_space>
249  cell_range( 0, ncell );
250  auto offset_scan =
251  KOKKOS_LAMBDA( const std::size_t c, int& update, const bool final_pass )
252  {
253  if ( final_pass ) offsets( c ) = update;
254  update += counts( c );
255  };
256  Kokkos::parallel_scan( "Cabana::LinkedCellList::build::offset_scan",
257  cell_range,
258  offset_scan );
259  Kokkos::fence();
260 
261  // Reset counts.
262  Kokkos::deep_copy( counts, 0 );
263 
264  // Compute the permutation vector.
265  auto create_permute =
266  KOKKOS_LAMBDA( const std::size_t p )
267  {
268  int i, j, k;
269  grid.locatePoint(
270  positions(p,0), positions(p,1), positions(p,2), i , j, k );
271  auto cell_id = grid.cardinalCellIndex(i,j,k);
272  int c = Kokkos::atomic_fetch_add( &counts(cell_id), 1 );
273  permute( offsets(cell_id) + c ) = p;
274  };
275  Kokkos::parallel_for( "Cabana::LinkedCellList::build::create_permute",
276  particle_range,
277  create_permute );
278  Kokkos::fence();
279 
280  // Create the binning data.
281  _bin_data =
282  BinningData<MemorySpace>( begin, end, counts, offsets, permute );
283  }
284 
285  private:
286 
287  BinningData<MemorySpace> _bin_data;
288  Impl::CartesianGrid<double> _grid;
