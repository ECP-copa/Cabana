241 template <class ExecutionSpace, class SliceType>
242 void build( ExecutionSpace, SliceType positions, const std::size_t begin,
243 const std::size_t end )
244 {
245 Kokkos::Profiling::pushRegion( "Cabana::LinkedCellList::build" );
246
247 static_assert( is_accessible_from<memory_space, ExecutionSpace>{}, "" );
248 assert( end >= begin );
249 assert( end <= positions.size() );
250
251 // Resize the binning data. Note that the permutation vector spans
252 // only the length of begin-end;
253 std::size_t ncell = totalBins();
254 if ( _counts.extent( 0 ) != ncell )
255 {
256 Kokkos::resize( _counts, ncell );
257 Kokkos::resize( _offsets, ncell );
258 }
259 std::size_t nparticles = end - begin;
260 if ( _permutes.extent( 0 ) != nparticles )
261 {
262 Kokkos::resize( _permutes, nparticles );
263 }
264
265 // Get local copies of class data for lambda function capture.
266 auto grid = _grid;
267 auto counts = _counts;
268 auto offsets = _offsets;
269 auto permutes = _permutes;
270
271 // Count.
272 Kokkos::RangePolicy<ExecutionSpace> particle_range( begin, end );
273 Kokkos::deep_copy( _counts, 0 );
274 auto counts_sv = Kokkos::Experimental::create_scatter_view( _counts );
275 auto cell_count = KOKKOS_LAMBDA( const std::size_t p )
276 {
277 int i, j, k;
278 grid.locatePoint( positions( p, 0 ), positions( p, 1 ),
279 positions( p, 2 ), i, j, k );
280 auto counts_data = counts_sv.access();
281 counts_data( grid.cardinalCellIndex( i, j, k ) ) += 1;
282 };
283 Kokkos::parallel_for( "Cabana::LinkedCellList::build::cell_count",
284 particle_range, cell_count );
285 Kokkos::fence();
286 Kokkos::Experimental::contribute( _counts, counts_sv );
287
288 // Compute offsets.
289 Kokkos::RangePolicy<ExecutionSpace> cell_range( 0, ncell );
290 auto offset_scan = KOKKOS_LAMBDA( const std::size_t c, int& update,
291 const bool final_pass )
292 {
293 if ( final_pass )
294 offsets( c ) = update;
295 update += counts( c );
296 };
297 Kokkos::parallel_scan( "Cabana::LinkedCellList::build::offset_scan",
298 cell_range, offset_scan );
299 Kokkos::fence();
300
301 // Reset counts.
302 Kokkos::deep_copy( _counts, 0 );
303
304 // Compute the permutation vector.
305 auto create_permute = KOKKOS_LAMBDA( const std::size_t p )
306 {
307 int i, j, k;
308 grid.locatePoint( positions( p, 0 ), positions( p, 1 ),
309 positions( p, 2 ), i, j, k );
310 auto cell_id = grid.cardinalCellIndex( i, j, k );
311 int c = Kokkos::atomic_fetch_add( &counts( cell_id ), 1 );
312 permutes( offsets( cell_id ) + c ) = p;
313 };
314 Kokkos::parallel_for( "Cabana::LinkedCellList::build::create_permute",
315 particle_range, create_permute );
316 Kokkos::fence();
317
318 // Create the binning data.
319 _bin_data = BinningData<MemorySpace>( begin, end, _counts, _offsets,
320 _permutes );
321
322 Kokkos::Profiling::popRegion();
323 }
