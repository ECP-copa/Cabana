222  template <class SliceType>
223  void build( SliceType positions, const std::size_t begin,
224  const std::size_t end )
225  {
226  // Resize the binning data. Note that the permutation vector spans
227  // only the length of begin-end;
228  std::size_t ncell = totalBins();
229  if ( _counts.extent( 0 ) != ncell )
230  {
231  Kokkos::resize( _counts, ncell );
232  Kokkos::resize( _offsets, ncell );
233  }
234  std::size_t nparticles = end - begin;
235  if ( _permutes.extent( 0 ) != nparticles )
236  {
237  Kokkos::resize( _permutes, nparticles );
238  }
239 
240  // Get local copies of class data for lambda function capture.
241  auto grid = _grid;
242  auto counts = _counts;
243  auto offsets = _offsets;
244  auto permutes = _permutes;
245 
246  // Count.
247  Kokkos::RangePolicy<execution_space> particle_range( begin, end );
248  Kokkos::deep_copy( _counts, 0 );
249  auto counts_sv = Kokkos::Experimental::create_scatter_view( _counts );
250  auto cell_count = KOKKOS_LAMBDA( const std::size_t p )
251  {
252  int i, j, k;
253  grid.locatePoint( positions( p, 0 ), positions( p, 1 ),
254  positions( p, 2 ), i, j, k );
255  auto counts_data = counts_sv.access();
256  counts_data( grid.cardinalCellIndex( i, j, k ) ) += 1;
257  };
258  Kokkos::parallel_for( "Cabana::LinkedCellList::build::cell_count",
259  particle_range, cell_count );
260  Kokkos::fence();
261  Kokkos::Experimental::contribute( _counts, counts_sv );
262 
263  // Compute offsets.
264  Kokkos::RangePolicy<execution_space> cell_range( 0, ncell );
265  auto offset_scan = KOKKOS_LAMBDA( const std::size_t c, int& update,
266  const bool final_pass )
267  {
268  if ( final_pass )
269  offsets( c ) = update;
270  update += counts( c );
271  };
272  Kokkos::parallel_scan( "Cabana::LinkedCellList::build::offset_scan",
273  cell_range, offset_scan );
274  Kokkos::fence();
275 
276  // Reset counts.
277  Kokkos::deep_copy( _counts, 0 );
278 
279  // Compute the permutation vector.
280  auto create_permute = KOKKOS_LAMBDA( const std::size_t p )
281  {
282  int i, j, k;
283  grid.locatePoint( positions( p, 0 ), positions( p, 1 ),
284  positions( p, 2 ), i, j, k );
285  auto cell_id = grid.cardinalCellIndex( i, j, k );
286  int c = Kokkos::atomic_fetch_add( &counts( cell_id ), 1 );
287  permutes( offsets( cell_id ) + c ) = p;
288  };
289  Kokkos::parallel_for( "Cabana::LinkedCellList::build::create_permute",
290  particle_range, create_permute );
291  Kokkos::fence();
292 
293  // Create the binning data.
294  _bin_data =
295  BinningData<DeviceType>( begin, end, _counts, _offsets, _permutes );
296  }
