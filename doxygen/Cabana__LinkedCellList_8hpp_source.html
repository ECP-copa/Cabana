206  // This function should be private but we need to expose it as public to
207  // launch CUDA kernels with class data.
208  template<class SliceType>
209  void build( SliceType positions,
210  const std::size_t begin,
211  const std::size_t end )
212  {
213  // Allocate the binning data. Note that the permutation vector spans
214  // only the length of begin-end;
215  std::size_t ncell = totalBins();
216  Kokkos::View<int*,memory_space> counts( "counts", ncell );
217  OffsetView offsets( "offsets", ncell );
218  OffsetView permute( "permute", end - begin );
219 
220  // Get a local copy of the grid because it is class data and a lambda
221  // function will not capture it otherwise via CUDA.
222  auto grid = _grid;
223 
224  // Count.
225  Kokkos::RangePolicy<typename OffsetView::execution_space>
226  particle_range( begin, end );
227  Kokkos::deep_copy( counts, 0 );
228  auto cell_count =
229  KOKKOS_LAMBDA( const std::size_t p )
230  {
231  int i, j, k;
232  grid.locatePoint(
233  positions(p,0), positions(p,1), positions(p,2), i , j, k );
234  Kokkos::atomic_increment(
235  &counts(grid.cardinalCellIndex(i,j,k)) );
236  };
237  Kokkos::parallel_for( "Cabana::LinkedCellList::build::cell_count",
238  particle_range,
239  cell_count );
240  Kokkos::fence();
241 
242  // Compute offsets.
243  Kokkos::RangePolicy<typename OffsetView::execution_space>
244  cell_range( 0, ncell );
245  auto offset_scan =
246  KOKKOS_LAMBDA( const std::size_t c, int& update, const bool final_pass )
247  {
248  if ( final_pass ) offsets( c ) = update;
249  update += counts( c );
250  };
251  Kokkos::parallel_scan( "Cabana::LinkedCellList::build::offset_scan",
252  cell_range,
253  offset_scan );
254  Kokkos::fence();
255 
256  // Reset counts.
257  Kokkos::deep_copy( counts, 0 );
258 
259  // Compute the permutation vector.
260  auto create_permute =
261  KOKKOS_LAMBDA( const std::size_t p )
262  {
263  int i, j, k;
264  grid.locatePoint(
265  positions(p,0), positions(p,1), positions(p,2), i , j, k );
266  auto cell_id = grid.cardinalCellIndex(i,j,k);
267  int c = Kokkos::atomic_fetch_add( &counts(cell_id), 1 );
268  permute( offsets(cell_id) + c ) = p;
269  };
270  Kokkos::parallel_for( "Cabana::LinkedCellList::build::create_permute",
271  particle_range,
272  create_permute );
273  Kokkos::fence();
274 
275  // Create the binning data.
276  _bin_data =
277  BinningData<MemorySpace>( begin, end, counts, offsets, permute );
278  }
279 
280  private:
281
