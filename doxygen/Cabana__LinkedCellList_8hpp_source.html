231  template <class SliceType>
232  void build( SliceType positions, const std::size_t begin,
233  const std::size_t end )
234  {
235  // Resize the binning data. Note that the permutation vector spans
236  // only the length of begin-end;
237  std::size_t ncell = totalBins();
238  if ( _counts.extent( 0 ) != ncell )
239  {
240  Kokkos::resize( _counts, ncell );
241  Kokkos::resize( _offsets, ncell );
242  }
243  std::size_t nparticles = end - begin;
244  if ( _permutes.extent( 0 ) != nparticles )
245  {
246  Kokkos::resize( _permutes, nparticles );
247  }
248 
249  // Get local copies of class data for lambda function capture.
250  auto grid = _grid;
251  auto counts = _counts;
252  auto offsets = _offsets;
253  auto permutes = _permutes;
254 
255  // Count.
256  Kokkos::RangePolicy<execution_space> particle_range( begin, end );
257  Kokkos::deep_copy( _counts, 0 );
258  auto counts_sv = Kokkos::Experimental::create_scatter_view( _counts );
259  auto cell_count = KOKKOS_LAMBDA( const std::size_t p )
260  {
261  int i, j, k;
262  grid.locatePoint( positions( p, 0 ), positions( p, 1 ),
263  positions( p, 2 ), i, j, k );
264  auto counts_data = counts_sv.access();
265  counts_data( grid.cardinalCellIndex( i, j, k ) ) += 1;
266  };
267  Kokkos::parallel_for( "Cabana::LinkedCellList::build::cell_count",
268  particle_range, cell_count );
269  Kokkos::fence();
270  Kokkos::Experimental::contribute( _counts, counts_sv );
271 
272  // Compute offsets.
273  Kokkos::RangePolicy<execution_space> cell_range( 0, ncell );
274  auto offset_scan = KOKKOS_LAMBDA( const std::size_t c, int& update,
275  const bool final_pass )
276  {
277  if ( final_pass )
278  offsets( c ) = update;
279  update += counts( c );
280  };
281  Kokkos::parallel_scan( "Cabana::LinkedCellList::build::offset_scan",
282  cell_range, offset_scan );
283  Kokkos::fence();
284 
285  // Reset counts.
286  Kokkos::deep_copy( _counts, 0 );
287 
288  // Compute the permutation vector.
289  auto create_permute = KOKKOS_LAMBDA( const std::size_t p )
290  {
291  int i, j, k;
292  grid.locatePoint( positions( p, 0 ), positions( p, 1 ),
293  positions( p, 2 ), i, j, k );
294  auto cell_id = grid.cardinalCellIndex( i, j, k );
295  int c = Kokkos::atomic_fetch_add( &counts( cell_id ), 1 );
296  permutes( offsets( cell_id ) + c ) = p;
297  };
298  Kokkos::parallel_for( "Cabana::LinkedCellList::build::create_permute",
299  particle_range, create_permute );
300  Kokkos::fence();
301 
302  // Create the binning data.
303  _bin_data =
304  BinningData<DeviceType>( begin, end, _counts, _offsets, _permutes );
305  }
