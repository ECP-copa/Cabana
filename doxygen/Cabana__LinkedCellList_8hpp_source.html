233 template <class SliceType>
234 void build( SliceType positions, const std::size_t begin,
235 const std::size_t end )
236 {
237 Kokkos::Profiling::pushRegion( "Cabana::LinkedCellList::build" );
238
239 assert( end >= begin );
240 assert( end <= positions.size() );
241
242 // Resize the binning data. Note that the permutation vector spans
243 // only the length of begin-end;
244 std::size_t ncell = totalBins();
245 if ( _counts.extent( 0 ) != ncell )
246 {
247 Kokkos::resize( _counts, ncell );
248 Kokkos::resize( _offsets, ncell );
249 }
250 std::size_t nparticles = end - begin;
251 if ( _permutes.extent( 0 ) != nparticles )
252 {
253 Kokkos::resize( _permutes, nparticles );
254 }
255
256 // Get local copies of class data for lambda function capture.
257 auto grid = _grid;
258 auto counts = _counts;
259 auto offsets = _offsets;
260 auto permutes = _permutes;
261
262 // Count.
263 Kokkos::RangePolicy<execution_space> particle_range( begin, end );
264 Kokkos::deep_copy( _counts, 0 );
265 auto counts_sv = Kokkos::Experimental::create_scatter_view( _counts );
266 auto cell_count = KOKKOS_LAMBDA( const std::size_t p )
267 {
268 int i, j, k;
269 grid.locatePoint( positions( p, 0 ), positions( p, 1 ),
270 positions( p, 2 ), i, j, k );
271 auto counts_data = counts_sv.access();
272 counts_data( grid.cardinalCellIndex( i, j, k ) ) += 1;
273 };
274 Kokkos::parallel_for( "Cabana::LinkedCellList::build::cell_count",
275 particle_range, cell_count );
276 Kokkos::fence();
277 Kokkos::Experimental::contribute( _counts, counts_sv );
278
279 // Compute offsets.
280 Kokkos::RangePolicy<execution_space> cell_range( 0, ncell );
281 auto offset_scan = KOKKOS_LAMBDA( const std::size_t c, int& update,
282 const bool final_pass )
283 {
284 if ( final_pass )
285 offsets( c ) = update;
286 update += counts( c );
287 };
288 Kokkos::parallel_scan( "Cabana::LinkedCellList::build::offset_scan",
289 cell_range, offset_scan );
290 Kokkos::fence();
291
292 // Reset counts.
293 Kokkos::deep_copy( _counts, 0 );
294
295 // Compute the permutation vector.
296 auto create_permute = KOKKOS_LAMBDA( const std::size_t p )
297 {
298 int i, j, k;
299 grid.locatePoint( positions( p, 0 ), positions( p, 1 ),
300 positions( p, 2 ), i, j, k );
301 auto cell_id = grid.cardinalCellIndex( i, j, k );
302 int c = Kokkos::atomic_fetch_add( &counts( cell_id ), 1 );
303 permutes( offsets( cell_id ) + c ) = p;
304 };
305 Kokkos::parallel_for( "Cabana::LinkedCellList::build::create_permute",
306 particle_range, create_permute );
307 Kokkos::fence();
308
309 // Create the binning data.
310 _bin_data =
311 BinningData<DeviceType>( begin, end, _counts, _offsets, _permutes );
312
313 Kokkos::Profiling::popRegion();
314 }
