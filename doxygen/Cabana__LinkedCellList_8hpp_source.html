209  // This function should be private but we need to expose it as public to
210  // launch CUDA kernels with class data.
211  template<class SliceType>
212  void build( SliceType positions,
213  const std::size_t begin,
214  const std::size_t end )
215  {
216  // Allocate the binning data. Note that the permutation vector spans
217  // only the length of begin-end;
218  std::size_t ncell = totalBins();
219  Kokkos::View<int*,memory_space> counts( "counts", ncell );
220  OffsetView offsets( "offsets", ncell );
221  OffsetView permute( "permute", end - begin );
222 
223  // Get a local copy of the grid because it is class data and a lambda
224  // function will not capture it otherwise via CUDA.
225  auto grid = _grid;
226 
227  // Count.
228  Kokkos::RangePolicy<execution_space> particle_range( begin, end );
229  Kokkos::deep_copy( counts, 0 );
230  auto counts_sv = Kokkos::Experimental::create_scatter_view( counts );
231  auto cell_count =
232  KOKKOS_LAMBDA( const std::size_t p )
233  {
234  int i, j, k;
235  grid.locatePoint(
236  positions(p,0), positions(p,1), positions(p,2), i , j, k );
237  auto counts_data = counts_sv.access();
238  counts_data(grid.cardinalCellIndex(i,j,k)) += 1;
239  };
240  Kokkos::parallel_for( "Cabana::LinkedCellList::build::cell_count",
241  particle_range,
242  cell_count );
243  Kokkos::fence();
244  Kokkos::Experimental::contribute( counts, counts_sv );
245 
246  // Compute offsets.
247  Kokkos::RangePolicy<execution_space> cell_range( 0, ncell );
248  auto offset_scan =
249  KOKKOS_LAMBDA( const std::size_t c, int& update, const bool final_pass )
250  {
251  if ( final_pass ) offsets( c ) = update;
252  update += counts( c );
253  };
254  Kokkos::parallel_scan( "Cabana::LinkedCellList::build::offset_scan",
255  cell_range,
256  offset_scan );
257  Kokkos::fence();
258 
259  // Reset counts.
260  Kokkos::deep_copy( counts, 0 );
261 
262  // Compute the permutation vector.
263  auto create_permute =
264  KOKKOS_LAMBDA( const std::size_t p )
265  {
266  int i, j, k;
267  grid.locatePoint(
268  positions(p,0), positions(p,1), positions(p,2), i , j, k );
269  auto cell_id = grid.cardinalCellIndex(i,j,k);
270  int c = Kokkos::atomic_fetch_add( &counts(cell_id), 1 );
271  permute( offsets(cell_id) + c ) = p;
272  };
273  Kokkos::parallel_for( "Cabana::LinkedCellList::build::create_permute",
274  particle_range,
275  create_permute );
276  Kokkos::fence();
277 
278  // Create the binning data.
279  _bin_data =
280  BinningData<DeviceType>( begin, end, counts, offsets, permute );
281  }
282 
283  private:
284
