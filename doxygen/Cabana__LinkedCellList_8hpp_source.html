208  // This function should be private but we need to expose it as public to
209  // launch CUDA kernels with class data.
210  template <class SliceType>
211  void build( SliceType positions, const std::size_t begin,
212  const std::size_t end )
213  {
214  // Allocate the binning data. Note that the permutation vector spans
215  // only the length of begin-end;
216  std::size_t ncell = totalBins();
217  Kokkos::View<int*, memory_space> counts(
218  Kokkos::view_alloc( Kokkos::WithoutInitializing, "counts" ),
219  ncell );
220  OffsetView offsets(
221  Kokkos::view_alloc( Kokkos::WithoutInitializing, "offsets" ),
222  ncell );
223  OffsetView permute(
224  Kokkos::view_alloc( Kokkos::WithoutInitializing, "permute" ),
225  end - begin );
226 
227  // Get a local copy of the grid because it is class data and a lambda
228  // function will not capture it otherwise via CUDA.
229  auto grid = _grid;
230 
231  // Count.
232  Kokkos::RangePolicy<execution_space> particle_range( begin, end );
233  Kokkos::deep_copy( counts, 0 );
234  auto counts_sv = Kokkos::Experimental::create_scatter_view( counts );
235  auto cell_count = KOKKOS_LAMBDA( const std::size_t p )
236  {
237  int i, j, k;
238  grid.locatePoint( positions( p, 0 ), positions( p, 1 ),
239  positions( p, 2 ), i, j, k );
240  auto counts_data = counts_sv.access();
241  counts_data( grid.cardinalCellIndex( i, j, k ) ) += 1;
242  };
243  Kokkos::parallel_for( "Cabana::LinkedCellList::build::cell_count",
244  particle_range, cell_count );
245  Kokkos::fence();
246  Kokkos::Experimental::contribute( counts, counts_sv );
247 
248  // Compute offsets.
249  Kokkos::RangePolicy<execution_space> cell_range( 0, ncell );
250  auto offset_scan = KOKKOS_LAMBDA( const std::size_t c, int& update,
251  const bool final_pass )
252  {
253  if ( final_pass )
254  offsets( c ) = update;
255  update += counts( c );
256  };
257  Kokkos::parallel_scan( "Cabana::LinkedCellList::build::offset_scan",
258  cell_range, offset_scan );
259  Kokkos::fence();
260 
261  // Reset counts.
262  Kokkos::deep_copy( counts, 0 );
263 
264  // Compute the permutation vector.
265  auto create_permute = KOKKOS_LAMBDA( const std::size_t p )
266  {
267  int i, j, k;
268  grid.locatePoint( positions( p, 0 ), positions( p, 1 ),
269  positions( p, 2 ), i, j, k );
270  auto cell_id = grid.cardinalCellIndex( i, j, k );
271  int c = Kokkos::atomic_fetch_add( &counts( cell_id ), 1 );
272  permute( offsets( cell_id ) + c ) = p;
273  };
274  Kokkos::parallel_for( "Cabana::LinkedCellList::build::create_permute",
275  particle_range, create_permute );
276  Kokkos::fence();
277 
278  // Create the binning data.
279  _bin_data =
280  BinningData<DeviceType>( begin, end, counts, offsets, permute );
281  }
282 
283  private:
284  BinningData<DeviceType> _bin_data;
285  Impl::CartesianGrid<double> _grid;
286 };
