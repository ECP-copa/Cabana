233  template <class SliceType>
234  void build( SliceType positions, const std::size_t begin,
235  const std::size_t end )
236  {
237  assert( end >= begin );
238  assert( end <= positions.size() );
239 
240  // Resize the binning data. Note that the permutation vector spans
241  // only the length of begin-end;
242  std::size_t ncell = totalBins();
243  if ( _counts.extent( 0 ) != ncell )
244  {
245  Kokkos::resize( _counts, ncell );
246  Kokkos::resize( _offsets, ncell );
247  }
248  std::size_t nparticles = end - begin;
249  if ( _permutes.extent( 0 ) != nparticles )
250  {
251  Kokkos::resize( _permutes, nparticles );
252  }
253 
254  // Get local copies of class data for lambda function capture.
255  auto grid = _grid;
256  auto counts = _counts;
257  auto offsets = _offsets;
258  auto permutes = _permutes;
259 
260  // Count.
261  Kokkos::RangePolicy<execution_space> particle_range( begin, end );
262  Kokkos::deep_copy( _counts, 0 );
263  auto counts_sv = Kokkos::Experimental::create_scatter_view( _counts );
264  auto cell_count = KOKKOS_LAMBDA( const std::size_t p )
265  {
266  int i, j, k;
267  grid.locatePoint( positions( p, 0 ), positions( p, 1 ),
268  positions( p, 2 ), i, j, k );
269  auto counts_data = counts_sv.access();
270  counts_data( grid.cardinalCellIndex( i, j, k ) ) += 1;
271  };
272  Kokkos::parallel_for( "Cabana::LinkedCellList::build::cell_count",
273  particle_range, cell_count );
274  Kokkos::fence();
275  Kokkos::Experimental::contribute( _counts, counts_sv );
276 
277  // Compute offsets.
278  Kokkos::RangePolicy<execution_space> cell_range( 0, ncell );
279  auto offset_scan = KOKKOS_LAMBDA( const std::size_t c, int& update,
280  const bool final_pass )
281  {
282  if ( final_pass )
283  offsets( c ) = update;
284  update += counts( c );
285  };
286  Kokkos::parallel_scan( "Cabana::LinkedCellList::build::offset_scan",
287  cell_range, offset_scan );
288  Kokkos::fence();
289 
290  // Reset counts.
291  Kokkos::deep_copy( _counts, 0 );
292 
293  // Compute the permutation vector.
294  auto create_permute = KOKKOS_LAMBDA( const std::size_t p )
295  {
296  int i, j, k;
297  grid.locatePoint( positions( p, 0 ), positions( p, 1 ),
298  positions( p, 2 ), i, j, k );
299  auto cell_id = grid.cardinalCellIndex( i, j, k );
300  int c = Kokkos::atomic_fetch_add( &counts( cell_id ), 1 );
301  permutes( offsets( cell_id ) + c ) = p;
302  };
303  Kokkos::parallel_for( "Cabana::LinkedCellList::build::create_permute",
304  particle_range, create_permute );
305  Kokkos::fence();
306 
307  // Create the binning data.
308  _bin_data =
309  BinningData<DeviceType>( begin, end, _counts, _offsets, _permutes );
310  }
