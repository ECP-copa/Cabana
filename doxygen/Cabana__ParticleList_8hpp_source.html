177template <class MemorySpace, int VectorLength, class... FieldTags>
178class ParticleList
179{
180 public:
182 using memory_space = MemorySpace;
184 using traits = ParticleTraits<FieldTags...>;
186 using member_types = typename traits::member_types;
188 using aosoa_type = Cabana::AoSoA<member_types, memory_space, VectorLength>;
190 using tuple_type = typename aosoa_type::tuple_type;
195 template <std::size_t M>
196 using slice_type = typename aosoa_type::template member_slice_type<M>;
198 using particle_type = Particle<FieldTags...>;
200 using particle_view_type =
201 ParticleView<aosoa_type::vector_length, FieldTags...>;
202
204 ParticleList( const std::string& label )
205 : _aosoa( label )
206 {
207 }
208
210 ParticleList( const aosoa_type& aosoa )
211 : _aosoa( aosoa )
212 {
213 }
214
216 std::size_t size() const { return _aosoa.size(); }
217
219 aosoa_type& aosoa() { return _aosoa; }
221 const aosoa_type& aosoa() const { return _aosoa; }
222
224 template <class IndexType>
225 KOKKOS_INLINE_FUNCTION auto getParticle( IndexType p ) const
226 {
227 return particle_type( _aosoa.getTuple( p ) );
228 }
229
231 template <class ParticleType, class IndexType>
232 KOKKOS_INLINE_FUNCTION void setParticle( ParticleType particle,
233 IndexType p ) const
234 {
235 _aosoa.setTuple( p, particle.tuple() );
236 }
237
239 template <class IndexType>
240 KOKKOS_INLINE_FUNCTION auto getParticleView( IndexType p ) const
241 {
242 auto s = Cabana::Impl::Index<particle_view_type::vector_length>::s( p );
243 auto a = Cabana::Impl::Index<particle_view_type::vector_length>::a( p );
244 return particle_view_type( _aosoa.access( s ), a );
245 }
246
248 const std::string& label() const { return _aosoa.label(); }
249
251 template <class FieldTag>
252 slice_type<TypeIndexer<FieldTag, FieldTags...>::index>
253 slice( FieldTag ) const
254 {
255 return Cabana::slice<TypeIndexer<FieldTag, FieldTags...>::index>(
256 _aosoa, FieldTag::label() );
257 }
258
259 protected:
261 aosoa_type _aosoa;
262};
