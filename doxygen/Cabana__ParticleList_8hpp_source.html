177template <class MemorySpace, class... FieldTags>
178class ParticleList
179{
180 public:
182 using memory_space = MemorySpace;
184 using traits = ParticleTraits<FieldTags...>;
186 using aosoa_type =
187 Cabana::AoSoA<typename traits::member_types, memory_space>;
189 using tuple_type = typename aosoa_type::tuple_type;
194 template <std::size_t M>
195 using slice_type = typename aosoa_type::template member_slice_type<M>;
197 using particle_type = Particle<FieldTags...>;
199 using particle_view_type =
200 ParticleView<aosoa_type::vector_length, FieldTags...>;
201
203 ParticleList( const std::string& label )
204 : _aosoa( label )
205 {
206 }
207
209 std::size_t size() const { return _aosoa.size(); }
210
212 aosoa_type& aosoa() { return _aosoa; }
214 const aosoa_type& aosoa() const { return _aosoa; }
215
217 template <class IndexType>
218 KOKKOS_INLINE_FUNCTION auto getParticle( IndexType p ) const
219 {
220 return particle_type( _aosoa.getTuple( p ) );
221 }
222
224 template <class ParticleType, class IndexType>
225 KOKKOS_INLINE_FUNCTION void setParticle( ParticleType particle,
226 IndexType p ) const
227 {
228 _aosoa.setTuple( p, particle.tuple() );
229 }
230
232 template <class IndexType>
233 KOKKOS_INLINE_FUNCTION auto getParticleView( IndexType p ) const
234 {
235 auto s = Cabana::Impl::Index<particle_view_type::vector_length>::s( p );
236 auto a = Cabana::Impl::Index<particle_view_type::vector_length>::a( p );
237 return particle_view_type( _aosoa.access( s ), a );
238 }
239
241 const std::string& label() const { return _aosoa.label(); }
242
244 template <class FieldTag>
245 slice_type<TypeIndexer<FieldTag, FieldTags...>::index>
246 slice( FieldTag ) const
247 {
248 return Cabana::slice<TypeIndexer<FieldTag, FieldTags...>::index>(
249 _aosoa, FieldTag::label() );
250 }
251
252 protected:
254 aosoa_type _aosoa;
255};
