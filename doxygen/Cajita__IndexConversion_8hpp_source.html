32template <class MeshType, class EntityType>
33struct L2G
34{
36 using mesh_type = MeshType;
37
39 static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;
40
42 using entity_type = EntityType;

46
48 Kokkos::Array<int, num_space_dim> local_own_max;

54 Kokkos::Array<int, num_space_dim> global_num_entity;
55
57 Kokkos::Array<bool, num_space_dim> periodic;

60 Kokkos::Array<bool, num_space_dim> boundary_lo;

66 L2G( const LocalGrid<MeshType>& local_grid )
67 {
68 // Local index set of owned entities.
69 auto local_own_space =
70 local_grid.indexSpace( Own(), EntityType(), Local() );
71
72 // Get the local owned min.
73 for ( std::size_t d = 0; d < num_space_dim; ++d )
74 local_own_min[d] = local_own_space.min( d );
75
76 // Get the local owned max.
77 for ( std::size_t d = 0; d < num_space_dim; ++d )
78 local_own_max[d] = local_own_space.max( d );
79
80 // Global index set of owned entities.
81 auto global_own_space =
82 local_grid.indexSpace( Own(), EntityType(), Global() );
83
84 // Get the global owned min.
85 for ( std::size_t d = 0; d < num_space_dim; ++d )
86 global_own_min[d] = global_own_space.min( d );
87
88 // Get the global grid.
89 const auto& global_grid = local_grid.globalGrid();
90
91 // Global number of entities.
92 for ( std::size_t d = 0; d < num_space_dim; ++d )
93 global_num_entity[d] =
94 global_grid.globalNumEntity( EntityType(), d );
95
96 // Periodicity
97 for ( std::size_t d = 0; d < num_space_dim; ++d )
98 periodic[d] = global_grid.isPeriodic( d );
99
100 // Determine if a block is on the low or high boundaries.
101 for ( std::size_t d = 0; d < num_space_dim; ++d )
102 {
103 boundary_lo[d] = global_grid.onLowBoundary( d );
104 boundary_hi[d] = global_grid.onHighBoundary( d );
105 }
106 }
107
109 KOKKOS_INLINE_FUNCTION
110 void operator()( const int lijk[num_space_dim],
111 int gijk[num_space_dim] ) const
112 {
113 for ( std::size_t d = 0; d < num_space_dim; ++d )
114 {
115 // Compute periodic wrap-around on low boundary.
116 if ( periodic[d] && lijk[d] < local_own_min[d] && boundary_lo[d] )
117 {
118 gijk[d] = global_num_entity[d] - local_own_min[d] + lijk[d];
119 }
120
121 // Compute periodic wrap-around on high boundary.
122 else if ( periodic[d] && local_own_max[d] <= lijk[d] &&
123 boundary_hi[d] )
124 {
125 gijk[d] = lijk[d] - local_own_max[d];
126 }
127
128 // Otherwise compute I indices as normal.
129 else
130 {
131 gijk[d] = lijk[d] - local_own_min[d] + global_own_min[d];
132 }
133 }
134 }
135
137 template <std::size_t NSD = num_space_dim>
138 KOKKOS_INLINE_FUNCTION std::enable_if_t<3 == NSD, void>
139 operator()( const int li, const int lj, const int lk, int& gi, int& gj,
140 int& gk ) const
141 {
142 int lijk[num_space_dim] = { li, lj, lk };
143 int gijk[num_space_dim];
144 this->operator()( lijk, gijk );
145 gi = gijk[0];
146 gj = gijk[1];
147 gk = gijk[2];
148 }
149
151 template <std::size_t NSD = num_space_dim>
152 KOKKOS_INLINE_FUNCTION std::enable_if_t<2 == NSD, void>
153 operator()( const int li, const int lj, int& gi, int& gj ) const
154 {
155 int lijk[num_space_dim] = { li, lj };
156 int gijk[num_space_dim];
157 this->operator()( lijk, gijk );
158 gi = gijk[0];
159 gj = gijk[1];
160 }
161};
