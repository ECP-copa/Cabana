24 // Checks if an integer is a power of two. N must be greater than 0.
25 template <int N>
26 struct IsPowerOfTwo
27 {
28  static_assert( N > 0, "Vector length must be greather than 0" );
29  static constexpr bool value = ( ( N & ( N - 1 ) ) == 0 );
30 };

32 //---------------------------------------------------------------------------//
33 // Calculate the base-2 logarithm of an integer which must be a power of 2 and
34 // greater than 0.
35 template <int N>
36 struct LogBase2
37 {
38  static_assert( IsPowerOfTwo<N>::value,
39  "Vector length must be a power of two" );
40  static constexpr int value = 1 + LogBase2<( N >> 1U )>::value;
41 };
