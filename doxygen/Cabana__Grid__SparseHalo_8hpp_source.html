122 template <class SparseArrayType>
123 SparseHalo( const halo_pattern_type pattern,
124 const std::shared_ptr<SparseArrayType>& sparse_array )
125 : _pattern( pattern )
126 {
127 // Function to get the local id of the neighbor.
128 auto neighbor_id = []( const std::array<int, num_space_dim>& ijk )
129 {
130 int id = ijk[0];
131 for ( std::size_t d = 1; d < num_space_dim; ++d )
132 id += num_space_dim * id + ijk[d];
133 return id;
134 };
135
136 // Neighbor id flip function. This lets us compute what neighbor we
137 // are relative to a given neighbor.
138 auto flip_id = [=]( const std::array<int, num_space_dim>& ijk )
139 {
140 std::array<int, num_space_dim> flip_ijk;
141 for ( std::size_t d = 0; d < num_space_dim; ++d )
142 flip_ijk[d] = -ijk[d];
143 return flip_ijk;
144 };
145
146 // compute total number of bytes in each SoA structure of the sparse
147 // grid (the spasre grid is in AoSoA manner)
148 auto soa_byte_array = compute_member_size_list();
149 for ( std::size_t i = 0; i < member_num; ++i )
150 _soa_member_bytes[i] = soa_byte_array[i];
151 _soa_total_bytes = std::max(
152 std::accumulate( soa_byte_array.begin(), soa_byte_array.end(), 0 ),
153 static_cast<int>( sizeof( tuple_type ) ) );
154
155 // Get the local grid the array uses.
156 auto local_grid = sparse_array->layout().localGrid();
157
158 // linear MPI rank ID of the current working rank
159 _self_rank =
160 local_grid->neighborRank( std::array<int, 3>( { 0, 0, 0 } ) );
161
162 // set the linear neighbor rank ID
163 // set up correspondence between sending and receiving buffers
164 auto neighbors = _pattern.getNeighbors();
165 for ( const auto& n : neighbors )
166 {
167 // neighbor rank linear ID
168 int rank = local_grid->neighborRank( n );
169
170 // if neighbor is valid
171 if ( rank >= 0 )
172 {
173 // add neighbor to the list
174 _neighbor_ranks.push_back( rank );
175 // set the tag to use for sending data to this neighbor
176 // the corresponding neighbor will have the same recieving tag
177 _send_tags.push_back( neighbor_id( n ) );
178 // set the tag to use for receiving data from this neighbor
179 // the corresponding neighbor will have the same sending tag
180 _receive_tags.push_back( neighbor_id( flip_id( n ) ) );
181
182 // build communication data for owned entries
183 buildCommData( Own(), local_grid, n, _owned_buffers,
184 _owned_tile_steering, _owned_tile_spaces );
185 // build communication data for ghosted entries
186 buildCommData( Ghost(), local_grid, n, _ghosted_buffers,
187 _ghosted_tile_steering, _ghosted_tile_spaces );
188
189 auto& own_index_space = _owned_tile_spaces.back();
190 auto& ghost_index_space = _ghosted_tile_spaces.back();
191 int tmp_steering_size =
192 own_index_space.sizeTile() > ghost_index_space.sizeTile()
193 ? own_index_space.sizeTile()
194 : ghost_index_space.sizeTile();
195 _tmp_tile_steering.push_back(
196 steering_view( "tmp_tile_steering", tmp_steering_size ) );
197
198 _valid_counting.push_back(
199 counting_view( "halo_valid_counting" ) );
200 _neighbor_counting.push_back(
201 counting_view( "halo_neighbor_valid_counting" ) );
202 Kokkos::deep_copy( _valid_counting.back(), 0 );
203 Kokkos::deep_copy( _neighbor_counting.back(), 0 );
204
205 _valid_neighbor_ids.emplace_back( n );
206 }
207 }
208 }

228 template <class DecompositionTag, class LocalGridType>
229 void buildCommData( DecompositionTag decomposition_tag,
230 const std::shared_ptr<LocalGridType>& local_grid,
231 const std::array<int, num_space_dim>& nid,
232 std::vector<buffer_view>& buffers,
233 std::vector<steering_view>& steering,
234 std::vector<tile_index_space>& spaces )
235 {
236 // get the halo sparse tile index space sharsed with the neighbor
237 spaces.push_back(
238 local_grid->template sharedTileIndexSpace<cell_bits_per_tile_dim>(
239 decomposition_tag, entity_type(), nid ) );
240 auto& index_space = spaces.back();
241
242 // allocate the buffer to store shared data with given neighbor
243 buffers.push_back(
244 buffer_view( "halo_buffer", index_space.sizeCell() ) );
245 // allocate the steering to guide the communication with the given
246 // neighbor
247 steering.push_back(
248 steering_view( "halo_tile_steering", index_space.sizeTile() ) );
249 // clear steering (init)
250 Kokkos::deep_copy( steering.back(), invalid_key );
251 }

259 template <class LocalGridType>
260 void updateTileSpace( const std::shared_ptr<LocalGridType>& local_grid )
261 {
262 // clear index space array first
263 _owned_tile_spaces.clear();
264 _ghosted_tile_spaces.clear();
265
266 // loop over all neighbors and update the shared tile index space
267 auto neighbors = _pattern.getNeighbors();
268 for ( std::size_t i = 0; i < _valid_neighbor_ids.size(); ++i )
269 {
270 // get neighbor relative id
271 auto& n = _valid_neighbor_ids[i];
272 // get neighbor linear MPI rank ID
273 int rank = local_grid->neighborRank( n );
274 // check if neighbor rank is valid
275 // the neighbor id should always be valid (as all should be
276 // well-prepared during construction/initialization)
277 if ( rank == _neighbor_ranks[i] )
278 {
279 // get shared tile index spcae from local grid
280 _owned_tile_spaces.push_back(
281 local_grid
282 ->template sharedTileIndexSpace<cell_bits_per_tile_dim>(
283 Own(), entity_type(), n ) );
284 _ghosted_tile_spaces.push_back(
285 local_grid
286 ->template sharedTileIndexSpace<cell_bits_per_tile_dim>(
287 Ghost(), entity_type(), n ) );
288
289 // reference to tile index spaces
290 auto& own_index_space = _owned_tile_spaces.back();
291 auto& ghost_index_space = _ghosted_tile_spaces.back();
292
293 // number of tiles inside each shared tile space
294 int own_tile_size = own_index_space.sizeTile();
295 int ghost_tile_size = ghost_index_space.sizeTile();
296 int tmp_steering_size = own_tile_size > ghost_tile_size
297 ? own_tile_size
298 : ghost_tile_size;
299
300 // check if the data steering and buffers require resize
301 // since the partition may changes during simulation process
302 // the size of shared spaces may also change accordingly
303 if ( own_tile_size > _owned_tile_steering[i].extent( 0 ) )
304 {
305 Kokkos::resize( _owned_tile_steering[i], own_tile_size );
306 Kokkos::resize( _owned_buffers[i],
307 own_index_space.sizeCell() );
308 }
309 if ( ghost_tile_size > _ghosted_tile_steering[i].extent( 0 ) )
310 {
311 Kokkos::resize( _ghosted_tile_steering[i],
312 ghost_tile_size );
313 Kokkos::resize( _ghosted_buffers[i],
314 ghost_index_space.sizeCell() );
315 }
316 if ( tmp_steering_size > _tmp_tile_steering[i].extent( 0 ) )
317 Kokkos::resize( _tmp_tile_steering[i], tmp_steering_size );
318 }
319 else
320 std::runtime_error( "neighbor rank doesn't match id" );
321 }
322 }

333 template <class ExecSpace, class SparseMapType>
334 void register_halo( SparseMapType& map )
335 {
336 // return if there's no valid neighbors
337 int num_n = _neighbor_ranks.size();
338 if ( 0 == num_n )
339 return;
340
341 // loop for all valid neighbors
342 for ( int nid = 0; nid < num_n; nid++ )
343 {
344 auto& owned_space = _owned_tile_spaces[nid];
345 auto& owned_steering = _owned_tile_steering[nid];
346
347 auto& ghosted_space = _ghosted_tile_spaces[nid];
348 auto& ghosted_steering = _ghosted_tile_steering[nid];
349
350 auto& counting = _valid_counting[nid];
351
352 // check the sparse map, if there are valid tiles in corresponding
353 // shared tile index space, add the tile key into the steering view,
354 // this steering keys will later be used for halo data collection
355
356 Kokkos::parallel_for(
357 Kokkos::RangePolicy<ExecSpace>( 0, map.capacity() ),
358 KOKKOS_LAMBDA( const int index ) {
359 if ( map.valid_at( index ) )
360 {
361 auto tile_key = map.key_at( index );
362 int ti, tj, tk;
363 map.key2ijk( tile_key, ti, tj, tk );
364 if ( owned_space.tileInRange( ti, tj, tk ) )
365 {
366 owned_steering( counting( Index::own )++ ) =
367 tile_key;
368 }
369 else if ( ghosted_space.tileInRange( ti, tj, tk ) )
370 {
371 ghosted_steering( counting( Index::ghost )++ ) =
372 tile_key;
373 }
374 }
375 } );
376 Kokkos::fence();
377 }
378 }

388 void clear( MPI_Comm comm )
389 {
390 // clear counting
391 for ( std::size_t i = 0; i < _valid_counting.size(); ++i )
392 Kokkos::deep_copy( _valid_counting[i], 0 );
393 for ( std::size_t i = 0; i < _neighbor_counting.size(); ++i )
394 Kokkos::deep_copy( _neighbor_counting[i], 0 );
395 // clear steering
396 for ( std::size_t i = 0; i < _owned_tile_steering.size(); ++i )
397 Kokkos::deep_copy( _owned_tile_steering[i], invalid_key );
398 for ( std::size_t i = 0; i < _ghosted_tile_steering.size(); ++i )
399 Kokkos::deep_copy( _ghosted_tile_steering[i], invalid_key );
400 for ( std::size_t i = 0; i < _tmp_tile_steering.size(); ++i )
401 Kokkos::deep_copy( _tmp_tile_steering[i], invalid_key );
402 // sync
403 MPI_Barrier( comm );
404 }

415 void collectNeighborCounting(
416 MPI_Comm comm, const bool is_neighbor_counting_collected = false ) const
417 {
418 // the valid halo size is already counted, no need to recount
419 if ( is_neighbor_counting_collected )
420 return;
421
422 // number of neighbors
423 int num_n = _neighbor_ranks.size();
424
425 // MPI request to collect counting information in shared owned and
426 // shared ghosted space
427 std::vector<MPI_Request> counting_requests( 2 * num_n,
428 MPI_REQUEST_NULL );
429 const int mpi_tag_counting = 1234;
430
431 // receive from all neighbors
432 for ( int nid = 0; nid < num_n; ++nid )
433 {
434 MPI_Irecv( _neighbor_counting[nid].data(),
435 Index::total * sizeof( int ), MPI_BYTE,
436 _neighbor_ranks[nid],
437 mpi_tag_counting + _receive_tags[nid], comm,
438 &counting_requests[nid] );
439 }
440 // send to all valid neighbors
441 for ( int nid = 0; nid < num_n; ++nid )
442 {
443 MPI_Isend( _valid_counting[nid].data(),
444 Index::total * sizeof( int ), MPI_BYTE,
445 _neighbor_ranks[nid], mpi_tag_counting + _send_tags[nid],
446 comm, &counting_requests[nid + num_n] );
447 }
448
449 // wait until all counting data sending finished
450 const int ec = MPI_Waitall( num_n, counting_requests.data() + num_n,
451 MPI_STATUSES_IGNORE );
452
453 // check if the counting communication succeed
454 if ( MPI_SUCCESS != ec )
455 throw std::logic_error( "sparse_halo: counting sending failed." );
456 MPI_Barrier( comm );
457 }

468 void scatterValidSendAndRecvRanks(
469 MPI_Comm comm, std::vector<int>& valid_sends,
470 std::vector<int>& valid_recvs,
471 const bool is_neighbor_counting_collected = false ) const
472 {
473 // collect neighbor counting if needed
474 collectNeighborCounting( comm, is_neighbor_counting_collected );
475
476 // loop over all valid neighbors to check if there's data communication
477 // needed
478 for ( std::size_t nid = 0; nid < _neighbor_ranks.size(); ++nid )
479 {
480 // reference to counting in owned and ghosted share spaces
481 auto h_counting = Kokkos::create_mirror_view_and_copy(
482 Kokkos::HostSpace(), _valid_counting[nid] );
483 auto h_neighbor_counting = Kokkos::create_mirror_view_and_copy(
484 Kokkos::HostSpace(), _neighbor_counting[nid] );
485
486 // if the current ghosted space and the neighbor's owned share
487 // space is non-emty, we need to send data to the neighbor
488 if ( !( h_counting( Index::ghost ) == 0 ||
489 h_neighbor_counting( Index::own ) == 0 ) )
490 {
491 valid_sends.push_back( nid );
492 }
493 // if the current owned share space and the neighbor's ghosted share
494 // space is non-emty, the neighbor will send data to us and we need
495 // to recieve data accordingly
496 if ( !( h_counting( Index::own ) == 0 ||
497 h_neighbor_counting( Index::ghost ) == 0 ) )
498 {
499 valid_recvs.push_back( nid );
500 }
501 }
502 }

559 template <class ExecSpace, class SparseArrayType>
560 void gather( const ExecSpace& exec_space, SparseArrayType& sparse_array,
561 const bool is_neighbor_counting_collected = false ) const
562 {
563 // return if no valid neighbor
564 if ( 0 == _neighbor_ranks.size() )
565 return;
566
567 // Get the MPI communicator.
568 auto comm = getComm( sparse_array );
569
570 const auto& map = sparse_array.layout().sparseMap();
571
572 // communicate "counting" among neighbors, to decide if the grid data
573 // communication is needed
574 std::vector<int> valid_sends;
575 std::vector<int> valid_recvs;
576 gatherValidSendAndRecvRanks( comm, valid_sends, valid_recvs,
577 is_neighbor_counting_collected );
578 MPI_Barrier( comm );
579
580 // ------------------------------------------------------------------
581 // communicate steering (array keys) for all valid sends and recieves
582 std::vector<MPI_Request> steering_requests(
583 valid_recvs.size() + valid_sends.size(), MPI_REQUEST_NULL );
584 const int mpi_tag_steering = 3214;
585
586 // get the steering keys from valid neighbors to get all grids that
587 // we need to receive
588 // loop over all neighbors that will send data to the current rank
589 for ( std::size_t i = 0; i < valid_recvs.size(); ++i )
590 {
591 int nid = valid_recvs[i];
592 Kokkos::View<int[2], Kokkos::HostSpace> h_neighbor_counting(
593 "tmp_host_neighbor_counting" );
594 Kokkos::deep_copy( h_neighbor_counting, _neighbor_counting[nid] );
595
596 MPI_Irecv( _tmp_tile_steering[nid].data(),
597 h_neighbor_counting( Index::own ) * sizeof( key_type ),
598 MPI_BYTE, _neighbor_ranks[nid],
599 mpi_tag_steering + _receive_tags[nid], comm,
600 &steering_requests[i] );
601 }
602
603 // send the steering keys to valid neighbors
604 // loop over all neighbors that requires our owned data
605 for ( std::size_t i = 0; i < valid_sends.size(); ++i )
606 {
607 int nid = valid_sends[i];
608 Kokkos::View<int[2], Kokkos::HostSpace> h_counting(
609 "tmp_host_counting" );
610 Kokkos::deep_copy( h_counting, _valid_counting[nid] );
611
612 MPI_Isend( _owned_tile_steering[nid].data(),
613 h_counting( Index::own ) * sizeof( key_type ), MPI_BYTE,
614 _neighbor_ranks[nid], mpi_tag_steering + _send_tags[nid],
615 comm, &steering_requests[i + valid_recvs.size()] );
616 }
617
618 // wait for all sending work finish
619 const int ec_ss = MPI_Waitall(
620 valid_sends.size(), steering_requests.data() + valid_recvs.size(),
621 MPI_STATUSES_IGNORE );
622 if ( MPI_SUCCESS != ec_ss )
623 throw std::logic_error(
624 "sparse_halo_gather: steering sending failed." );
625 MPI_Barrier( comm );
626
627 // ------------------------------------------------------------------
628 // communicate sparse array data
629 // Pick a tag to use for communication. This object has its own
630 // communication space so any tag will do.
631 std::vector<MPI_Request> requests(
632 valid_recvs.size() + valid_sends.size(), MPI_REQUEST_NULL );
633 const int mpi_tag = 2345;
634
635 // post receives
636 for ( std::size_t i = 0; i < valid_recvs.size(); ++i )
637 {
638 int nid = valid_recvs[i];
639 Kokkos::View<int[2], Kokkos::HostSpace> h_neighbor_counting(
640 "tmp_host_neighbor_counting" );
641 ;
642 Kokkos::deep_copy( h_neighbor_counting, _neighbor_counting[nid] );
643
644 MPI_Irecv( _ghosted_buffers[nid].data(),
645 h_neighbor_counting( Index::own ) * cell_num_per_tile *
646 _soa_total_bytes,
647 MPI_BYTE, _neighbor_ranks[nid],
648 mpi_tag + _receive_tags[nid], comm, &requests[i] );
649 }
650
651 // pack send buffers and post sends
652 for ( std::size_t i = 0; i < valid_sends.size(); ++i )
653 {
654 int nid = valid_sends[i];
655 Kokkos::View<int[2], Kokkos::HostSpace> h_counting(
656 "tmp_host_counting" );
657 ;
658 Kokkos::deep_copy( h_counting, _valid_counting[nid] );
659
660 packBuffer( exec_space, _owned_buffers[nid],
661 _owned_tile_steering[nid], sparse_array,
662 h_counting( Index::own ) );
663 Kokkos::fence();
664
665 MPI_Isend(
666 _owned_buffers[nid].data(),
667 h_counting( Index::own ) * cell_num_per_tile * _soa_total_bytes,
668 MPI_BYTE, _neighbor_ranks[nid], mpi_tag + _send_tags[nid], comm,
669 &requests[i + valid_recvs.size()] );
670 }
671
672 // unpack receive buffers
673 for ( std::size_t i = 0; i < valid_recvs.size(); ++i )
674 {
675 // get the next buffer to unpack
676 int unpack_index = MPI_UNDEFINED;
677 MPI_Waitany( valid_recvs.size(), requests.data(), &unpack_index,
678 MPI_STATUS_IGNORE );
679
680 // in theory we should receive enough buffers to unpack
681 // if not there could be some problems
682 if ( MPI_UNDEFINED == unpack_index )
683 std::runtime_error(
684 std::string( "sparse_halo_gather: data receiving failed, "
685 "get only " ) +
686 std::to_string( i ) + ", need " +
687 std::to_string( valid_recvs.size() ) );
688 // otherwise unpack the next buffer
689 else
690 {
691 int nid = valid_recvs[unpack_index];
692 auto h_neighbor_counting = Kokkos::create_mirror_view_and_copy(
693 Kokkos::HostSpace(), _neighbor_counting[nid] );
694 unpackBuffer( ScatterReduce::Replace(), exec_space,
695 _ghosted_buffers[nid], _tmp_tile_steering[nid],
696 sparse_array, map,
697 h_neighbor_counting( Index::own ) );
698 Kokkos::fence();
699 }
700 }
701
702 // wait to finish all send requests
703 const int ec_data = MPI_Waitall( valid_sends.size(),
704 requests.data() + valid_recvs.size(),
705 MPI_STATUSES_IGNORE );
706 if ( MPI_SUCCESS != ec_data )
707 throw std::logic_error(
708 "sparse_halo_gather: data sending failed." );
709
710 // reinit steerings for next round of communication
711 for ( std::size_t i = 0; i < _tmp_tile_steering.size(); ++i )
712 Kokkos::deep_copy( _tmp_tile_steering[i], invalid_key );
713 MPI_Barrier( comm );
714 }

730 template <class ExecSpace, class ReduceOp, class SparseArrayType>
731 void scatter( const ExecSpace& exec_space, const ReduceOp& reduce_op,
732 SparseArrayType& sparse_array,
733 const bool is_neighbor_counting_collected = false ) const
734 {
735 // return if no valid neighbor
736 if ( 0 == _neighbor_ranks.size() )
737 return;
738
739 // Get the MPI communicator.
740 auto comm = getComm( sparse_array );
741
742 const auto& map = sparse_array.layout().sparseMap();
743
744 // communicate "counting" among neighbors, to decide if the grid data
745 // transfer is needed
746 std::vector<int> valid_sends;
747 std::vector<int> valid_recvs;
748 scatterValidSendAndRecvRanks( comm, valid_sends, valid_recvs,
749 is_neighbor_counting_collected );
750 MPI_Barrier( comm );
751
752 // ------------------------------------------------------------------
753 // communicate steering (array keys) for all valid sends and recieves
754 std::vector<MPI_Request> steering_requests(
755 valid_recvs.size() + valid_sends.size(), MPI_REQUEST_NULL );
756 const int mpi_tag_steering = 214;
757
758 // get the steering keys from valid neighbors to know all grids that
759 // we need to receive
760 // loop over all neighbors that will send data to the current rank
761 for ( std::size_t i = 0; i < valid_recvs.size(); ++i )
762 {
763 int nid = valid_recvs[i];
764 Kokkos::View<int[2], Kokkos::HostSpace> h_neighbor_counting(
765 "tmp_host_neighbor_counting" );
766 ;
767 Kokkos::deep_copy( h_neighbor_counting, _neighbor_counting[nid] );
768
769 MPI_Irecv( _tmp_tile_steering[nid].data(),
770 h_neighbor_counting( Index::ghost ) * sizeof( key_type ),
771 MPI_BYTE, _neighbor_ranks[nid],
772 mpi_tag_steering + _receive_tags[nid], comm,
773 &steering_requests[i] );
774 }
775
776 // send the steering keys to valid neighbors
777 // loop over all neighbors that requires our owned data
778 for ( std::size_t i = 0; i < valid_sends.size(); ++i )
779 {
780 int nid = valid_sends[i];
781 Kokkos::View<int[2], Kokkos::HostSpace> h_counting(
782 "tmp_host_counting" );
783 ;
784 Kokkos::deep_copy( h_counting, _valid_counting[nid] );
785
786 MPI_Isend( _ghosted_tile_steering[nid].data(),
787 h_counting( Index::ghost ) * sizeof( key_type ),
788 MPI_BYTE, _neighbor_ranks[nid],
789 mpi_tag_steering + _send_tags[nid], comm,
790 &steering_requests[i + valid_recvs.size()] );
791 }
792
793 // wait for all sending work finish
794 const int ec_ss = MPI_Waitall(
795 valid_sends.size(), steering_requests.data() + valid_recvs.size(),
796 MPI_STATUSES_IGNORE );
797 if ( MPI_SUCCESS != ec_ss )
798 throw std::logic_error(
799 "sparse_halo_scatter: steering sending failed." );
800 MPI_Barrier( comm );
801
802 // ------------------------------------------------------------------
803 // communicate sparse array data
804 // Pick a tag to use for communication. This object has its own
805 // communication space so any tag will do.
806 std::vector<MPI_Request> requests(
807 valid_recvs.size() + valid_sends.size(), MPI_REQUEST_NULL );
808 const int mpi_tag = 345;
809
810 // post receives
811 for ( std::size_t i = 0; i < valid_recvs.size(); ++i )
812 {
813 int nid = valid_recvs[i];
814 Kokkos::View<int[2], Kokkos::HostSpace> h_neighbor_counting(
815 "tmp_host_neighbor_counting" );
816 ;
817 Kokkos::deep_copy( h_neighbor_counting, _neighbor_counting[nid] );
818
819 MPI_Irecv( _owned_buffers[nid].data(),
820 h_neighbor_counting( Index::ghost ) * cell_num_per_tile *
821 _soa_total_bytes,
822 MPI_BYTE, _neighbor_ranks[nid],
823 mpi_tag + _receive_tags[nid], comm, &requests[i] );
824 }
825
826 // pack send buffers and post sends
827 for ( std::size_t i = 0; i < valid_sends.size(); ++i )
828 {
829 int nid = valid_sends[i];
830 Kokkos::View<int[2], Kokkos::HostSpace> h_counting(
831 "tmp_host_counting" );
832 ;
833 Kokkos::deep_copy( h_counting, _valid_counting[nid] );
834 packBuffer( exec_space, _ghosted_buffers[nid],
835 _ghosted_tile_steering[nid], sparse_array,
836 h_counting( Index::ghost ) );
837 Kokkos::fence();
838
839 MPI_Isend( _ghosted_buffers[nid].data(),
840 h_counting( Index::ghost ) * cell_num_per_tile *
841 _soa_total_bytes,
842 MPI_BYTE, _neighbor_ranks[nid],
843 mpi_tag + _send_tags[nid], comm,
844 &requests[i + valid_recvs.size()] );
845 }
846
847 // unpack receive buffers
848 for ( std::size_t i = 0; i < valid_recvs.size(); ++i )
849 {
850 // get the next buffer to unpack
851 int unpack_index = MPI_UNDEFINED;
852 MPI_Waitany( valid_recvs.size(), requests.data(), &unpack_index,
853 MPI_STATUS_IGNORE );
854
855 // in theory we should receive enough buffers to unpack
856 // if not there could be some problems
857 if ( MPI_UNDEFINED == unpack_index )
858 std::runtime_error(
859 std::string( "sparse_halo_scatter: data receiving failed, "
860 "get only " ) +
861 std::to_string( i ) + ", need " +
862 std::to_string( valid_recvs.size() ) );
863 // otherwise unpack the next buffer with the given reduce operator
864 else
865 {
866 int nid = valid_recvs[unpack_index];
867 auto h_neighbor_counting = Kokkos::create_mirror_view_and_copy(
868 Kokkos::HostSpace(), _neighbor_counting[nid] );
869 unpackBuffer( reduce_op, exec_space, _owned_buffers[nid],
870 _tmp_tile_steering[nid], sparse_array, map,
871 h_neighbor_counting( Index::ghost ) );
872 Kokkos::fence();
873 }
874 }
875
876 // wait to finish all send requests
877 const int ec_data = MPI_Waitall( valid_sends.size(),
878 requests.data() + valid_recvs.size(),
879 MPI_STATUSES_IGNORE );
880 if ( MPI_SUCCESS != ec_data )
881 throw std::logic_error(
882 "sparse_halo_scatter: data sending failed." );
883
884 // reinit steerings for next round of communication
885 for ( std::size_t i = 0; i < _tmp_tile_steering.size(); ++i )
886 Kokkos::deep_copy( _tmp_tile_steering[i], invalid_key );
887 MPI_Barrier( comm );
888 }
