62  result += view( sd.s[Dim::I][i], sd.s[Dim::J][j],
63  sd.s[Dim::K][k], 0 ) *
64  sd.w[Dim::I][i] * sd.w[Dim::J][j] * sd.w[Dim::K][k];
65 }

221 template <class ViewType, class SplineDataType, class PointDataType>
222 KOKKOS_INLINE_FUNCTION
223  std::enable_if_t<3 == SplineDataType::num_space_dim, void>
224  gradient( const ViewType& view, const SplineDataType& sd,
225  PointDataType result[3][3],
226  typename std::enable_if<( std::rank<PointDataType>::value == 0 ),
227  void*>::type = 0 )
228 {
229  static_assert( SplineDataType::has_weight_values,
230  "G2P::gradient requires spline weight values" );
231  static_assert( SplineDataType::has_weight_physical_gradients,
232  "G2P::gradient requires spline weight physical gradients" );
233 
234  for ( int d0 = 0; d0 < 3; ++d0 )
235  for ( int d1 = 0; d1 < 3; ++d1 )
236  result[d0][d1] = 0.0;
237 
238  for ( int i = 0; i < SplineDataType::num_knot; ++i )
239  for ( int j = 0; j < SplineDataType::num_knot; ++j )
240  for ( int k = 0; k < SplineDataType::num_knot; ++k )
241  {
242  typename SplineDataType::scalar_type rg[3] = {
243  sd.g[Dim::I][i] * sd.w[Dim::J][j] * sd.w[Dim::K][k],
244  sd.w[Dim::I][i] * sd.g[Dim::J][j] * sd.w[Dim::K][k],
245  sd.w[Dim::I][i] * sd.w[Dim::J][j] * sd.g[Dim::K][k] };
246 
247  for ( int d0 = 0; d0 < 3; ++d0 )
248  {
249  typename ViewType::value_type mg = rg[d0];
250  for ( int d1 = 0; d1 < 3; ++d1 )
251  result[d0][d1] +=
252  mg * view( sd.s[Dim::I][i], sd.s[Dim::J][j],
253  sd.s[Dim::K][k], d1 );
254  }
255  }
256 }
