129 template<class ExecutionPolicy, class FunctorType, class NeighborListType>
130 inline void neighbor_parallel_for( const ExecutionPolicy& exec_policy,
131  const FunctorType& functor,
132  const NeighborListType& list,
133  const SerialNeighborOpTag& tag,
134  const std::string& str = "" )
135 {
136  std::ignore = tag;
137 
138  // Serial neighbor operation.
139  auto functor_wrapper =
140  KOKKOS_LAMBDA( const int i )
141  {
142  for ( int n = 0;
143  n < NeighborList<NeighborListType>::numNeighbor(list,i);
144  ++n )
145  {
146  functor(
147  i, NeighborList<NeighborListType>::getNeighbor(list,i,n) );
148  }
149  };
150 
151  // Create the kokkos execution policy
152  using kokkos_policy =
153  Kokkos::RangePolicy<typename ExecutionPolicy::execution_space>;
154  kokkos_policy k_policy( exec_policy.begin(), exec_policy.end() );
155 
156  // Execute the functor.
157  Kokkos::parallel_for( str, k_policy, functor_wrapper );
158 
159  // Fence.
160  Kokkos::fence();
161 }

206 template<class ExecutionPolicy, class FunctorType, class NeighborListType>
207 inline void neighbor_parallel_for( const ExecutionPolicy& exec_policy,
208  const FunctorType& functor,
209  const NeighborListType& list,
210  const TeamNeighborOpTag& tag,
211  const std::string& str = "" )
212 {
213  std::ignore = tag;
214 
215  // Create the kokkos execution policy
216  using kokkos_policy =
217  Kokkos::TeamPolicy<typename ExecutionPolicy::execution_space,
218  Kokkos::IndexType<int>,
219  Kokkos::Schedule<Kokkos::Dynamic> >;
220  kokkos_policy k_policy( exec_policy.end() - exec_policy.begin(),
221  Kokkos::AUTO );
222 
223  // Create a team operator.
224  auto functor_wrapper =
225  KOKKOS_LAMBDA( const typename kokkos_policy::member_type& team )
226  {
227  auto i = team.league_rank();
228  auto num_n = NeighborList<NeighborListType>::numNeighbor( list, i );
229  Kokkos::parallel_for(
230  Kokkos::TeamThreadRange(team,num_n),
231  [&]( const int n ) {
232  functor(
233  i, NeighborList<NeighborListType>::getNeighbor(list,i,n) );
234  });
235 
236  };
237 
238  // Execute the functor.
239  Kokkos::parallel_for( str, k_policy, functor_wrapper );
240 
241  // Fence.
242  Kokkos::fence();
243 }
