95template <class ListType>
96KOKKOS_INLINE_FUNCTION std::size_t
97totalNeighbor( const ListType& list, const std::size_t num_particles )
98{
99 std::size_t total_n = 0;
100 // Sum neighbors across all particles.
101 for ( std::size_t p = 0; p < num_particles; p++ )
102 total_n += NeighborList<ListType>::numNeighbor( list, p );
103 return total_n;
104}
105
107template <class ListType>
108KOKKOS_INLINE_FUNCTION std::size_t
109maxNeighbor( const ListType& list, const std::size_t num_particles )
110{
111 std::size_t max_n = 0;
112 for ( std::size_t p = 0; p < num_particles; p++ )
113 if ( NeighborList<ListType>::numNeighbor( list, p ) > max_n )
114 max_n = NeighborList<ListType>::numNeighbor( list, p );
115 return max_n;
116}
117} // namespace Impl

127template <class ExecutionSpace, class ListType>
128Kokkos::View<int* [2], typename ListType::memory_space>
129neighborHistogram( ExecutionSpace exec_space, const std::size_t num_particles,
130 const ListType& list, const int num_bin )
131{
132 // Allocate View of neighbors per particle
133 auto num_neigh = Kokkos::View<int*, typename ListType::memory_space>(
134 "particle_count", num_particles );
135
136 // Extract from neighbor list interface.
137 auto extract_functor = KOKKOS_LAMBDA( const int p )
138 {
139 num_neigh( p ) = NeighborList<ListType>::numNeighbor( list, p );
140 };
141 Kokkos::RangePolicy<ExecutionSpace> particle_policy( exec_space, 0,
142 num_particles );
143 Kokkos::parallel_for( particle_policy, extract_functor );
144 Kokkos::fence();
145
146 auto bin_data = Cabana::binByKey( num_neigh, num_bin );
147
148 auto histogram = Kokkos::View<int* [2], typename ListType::memory_space>(
149 "particle_count", num_bin, 2 );
150 auto histogram_functor = KOKKOS_LAMBDA( const int b )
151 {
152 int max_neigh = NeighborList<ListType>::maxNeighbor( list );
153 double bin_width =
154 static_cast<double>( max_neigh ) / static_cast<double>( num_bin );
155 if ( num_bin > max_neigh )
156 bin_width = 1;
157 // Wait to cast back to int to get the actual bin edge.
158 histogram( b, 0 ) = static_cast<int>( ( b + 1 ) * bin_width );
159 histogram( b, 1 ) = bin_data.binSize( b );
160 };
161 Kokkos::RangePolicy<ExecutionSpace> bin_policy( exec_space, 0, num_bin );
162 Kokkos::parallel_for( bin_policy, histogram_functor );
163 Kokkos::fence();
164
165 return histogram;
166}
