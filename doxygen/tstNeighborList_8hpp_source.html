117 Cabana::AoSoA<Cabana::MemberTypes<double[3]>,TEST_MEMSPACE>
118 createParticles( const int num_particle,
119  const double neighborhood_radius,
120  const double box_min,
121  const double box_max )
122 {
123  // Create the AoSoA with (100x100x100) particles.
124  using DataTypes = Cabana::MemberTypes<double[3]>;
125  using AoSoA_t = Cabana::AoSoA<DataTypes,TEST_MEMSPACE>;
126  AoSoA_t aosoa( num_particle );
127 
128  // Create particles within a region sized (10x10x10) neighborhood radii.
129  auto position = aosoa.slice<0>();
130  using PoolType = Kokkos::Random_XorShift64_Pool<TEST_EXECSPACE>;
131  using RandomType = Kokkos::Random_XorShift64<TEST_EXECSPACE>;
132  PoolType pool( 342343901 );
133  auto random_coord_op =
134  KOKKOS_LAMBDA( const int p )
135  {
136  auto gen = pool.get_state();
137  for ( int d = 0; d < 3; ++d )
138  position( p, d ) =
139  Kokkos::rand<RandomType,double>::draw(gen,box_min,box_max);
140  pool.free_state( gen );
141  };
142  Kokkos::RangePolicy<TEST_EXECSPACE> exec_policy( 0, num_particle );
143  Kokkos::parallel_for( exec_policy, random_coord_op );
144  Kokkos::fence();
145  return aosoa;
146 }

149 template<class PositionSlice>
150 TestNeighborList<typename PositionSlice::kokkos_memory_space>
151 computeFullNeighborList( const PositionSlice& position,
152  const double neighborhood_radius )
153 {
154  // Build a neighbor list with a brute force n^2 implementation. Count
155  // first.
156  TestNeighborList<typename PositionSlice::kokkos_memory_space> list;
157  int num_particle = position.size();
158  double rsqr = neighborhood_radius * neighborhood_radius;
159  list.counts = Kokkos::View<int*,typename PositionSlice::kokkos_memory_space>(
160  "test_neighbor_count", num_particle );
161  Kokkos::deep_copy( list.counts, 0 );
162  auto count_op =
163  KOKKOS_LAMBDA( const int i )
164  {
165  for ( int j = 0; j < num_particle; ++j )
166  {
167  if ( i != j )
168  {
169  double dsqr = 0.0;
170  for ( int d = 0; d < 3; ++d )
171  dsqr += (position(i,d)-position(j,d)) *
172  (position(i,d)-position(j,d));
173  if ( dsqr <= rsqr )
174  list.counts( i ) += 1;
175  }
176  }
177  };
178  Kokkos::RangePolicy<TEST_EXECSPACE> exec_policy( 0, num_particle );
179  Kokkos::parallel_for( exec_policy, count_op );
180  Kokkos::fence();
181 
182  // Allocate.
183  auto max_op =
184  KOKKOS_LAMBDA( const int i, int& max_val )
185  {
186  if ( max_val < list.counts(i) )
187  max_val = list.counts(i);
188  };
189  int max_n;
190  Kokkos::parallel_reduce( exec_policy, max_op, Kokkos::Max<int>(max_n) );
191  Kokkos::fence();
192  list.neighbors = Kokkos::View<int**,typename PositionSlice::kokkos_memory_space>(
193  "test_neighbors", num_particle, max_n );
194 
195  // Fill.
196  auto fill_op =
197  KOKKOS_LAMBDA( const int i )
198  {
199  int n_count = 0;
200  for ( int j = 0; j < num_particle; ++j )
201  {
202  if ( i != j )
203  {
204  double dsqr = 0.0;
205  for ( int d = 0; d < 3; ++d )
206  dsqr += (position(i,d)-position(j,d)) *
207  (position(i,d)-position(j,d));
208  if ( dsqr <= rsqr )
209  {
210  list.neighbors( i, n_count ) = j;
211  ++n_count;
212  }
213  }
214  }
215  };
216  Kokkos::parallel_for( exec_policy, fill_op );
217  Kokkos::fence();
218 
219  return list;
220 }

223 template<class ListType, class PositionSlice>
224 void checkFullNeighborList( const ListType& list,
225  const PositionSlice& position,
226  const double neighborhood_radius )
227 {
228  auto test_list = computeFullNeighborList( position, neighborhood_radius );
229 
230  // Check the results.
231  int num_particle = position.size();
232  for ( int p = 0; p < num_particle; ++p )
233  {
234  // First check that the number of neighbors are the same.
235  EXPECT_EQ(
236  Cabana::NeighborList<ListType>::numNeighbor(list,p),
237  test_list.counts(p) );
238 
239  // Now extract the neighbors.
240  std::vector<int> computed_neighbors( test_list.counts(p) );
241  std::vector<int> actual_neighbors( test_list.counts(p) );
242  for ( int n = 0; n < test_list.counts(p); ++n )
243  {
244  computed_neighbors[n] =
245  Cabana::NeighborList<ListType>::getNeighbor(list,p,n);
246  actual_neighbors[n] = test_list.neighbors(p,n);
247  }
248 
249  // Sort them because we have no guarantee of the order we will find
250  // them in.
251  std::sort( computed_neighbors.begin(), computed_neighbors.end() );
252  std::sort( actual_neighbors.begin(), actual_neighbors.end() );
253 
254  // Now compare directly.
255  for ( int n = 0; n < test_list.counts(p); ++n )
256  EXPECT_EQ( computed_neighbors[n], actual_neighbors[n] );
257  }
258 }

261 template<class ListType, class PositionSlice>
262 void checkHalfNeighborList( const ListType& list,
263  const PositionSlice& position,
264  const double neighborhood_radius )
265 {
266  // First build a full list.
267  auto full_list = computeFullNeighborList( position, neighborhood_radius );
268 
269  // Check that the full list is twice the size of the half list.
270  int num_particle = position.size();
271  int half_size = 0;
272  int full_size = 0;
273  for ( int p = 0; p < num_particle; ++p )
274  {
275  half_size += Cabana::NeighborList<ListType>::numNeighbor(list,p);
276  full_size += full_list.counts(p);
277  }
278  EXPECT_EQ( full_size, 2*half_size );
279 
280  // Check the half list by ensuring that a particle does not show up in the
281  // neighbor list of its neighbors.
282  for ( int p = 0; p < num_particle; ++p )
283  {
284  // Check each neighbor of p
285  for ( int n = 0;
286  n < Cabana::NeighborList<ListType>::numNeighbor(list,p);
287  ++n )
288  {
289  // Get the id of the nth neighbor of p.
290  auto p_n = Cabana::NeighborList<ListType>::getNeighbor(list,p,n);
291 
292  // Check that p is not in the neighbor list of the nth neighbor of
293  // p.
294  for ( int m = 0;
295  m < Cabana::NeighborList<ListType>::numNeighbor(list,p_n);
296  ++m )
297  {
298  auto n_m =
299  Cabana::NeighborList<ListType>::getNeighbor(list,p_n,m);
300  EXPECT_NE( n_m, p );
301  }
302  }
303  }

354 //---------------------------------------------------------------------------//
355 void testNeighborParallelFor()
356 {
357  // Create the AoSoA and fill with random particle positions.
358  int num_particle = 1e3;
359  double test_radius = 2.32;
360  double cell_size_ratio = 0.5;
361  double box_min = -5.3 * test_radius;
362  double box_max = 4.7 * test_radius;
363  auto aosoa =
364  createParticles( num_particle, test_radius, box_min, box_max );
365  using aosoa_t = decltype(aosoa);
366 
367  // Create the neighbor list.
368  double grid_min[3] = { box_min, box_min, box_min };
369  double grid_max[3] = { box_max, box_max, box_max };
370  Cabana::VerletList<TEST_MEMSPACE,Cabana::FullNeighborTag>
371  nlist( aosoa.slice<0>(), 0, aosoa.size(),
372  test_radius, cell_size_ratio, grid_min, grid_max );
373 
374  // Create Kokkos views for the write operation.
375  using kokkos_memory_space = typename TEST_MEMSPACE::kokkos_memory_space;
376  Kokkos::View<int*,kokkos_memory_space> test_result( "test_result", num_particle );
377  Kokkos::View<int*,kokkos_memory_space> serial_result( "serial_result", num_particle );
378  Kokkos::View<int*,kokkos_memory_space> team_result( "team_result", num_particle );
379 
380  // Test the list parallel operation by adding a value from each neighbor
381  // to the particle and compare to counts.
382  auto serial_count_op = KOKKOS_LAMBDA( const int i, const int n )
383  { Kokkos::atomic_add( &serial_result(i), n ); };
384  auto team_count_op = KOKKOS_LAMBDA( const int i, const int n )
385  { Kokkos::atomic_add( &team_result(i), n ); };
386  Cabana::RangePolicy<aosoa_t::vector_length,TEST_EXECSPACE> policy( aosoa );
387  Cabana::neighbor_parallel_for(
388  policy, serial_count_op, nlist, Cabana::SerialNeighborOpTag() );
389  Cabana::neighbor_parallel_for(
390  policy, team_count_op, nlist, Cabana::TeamNeighborOpTag() );
391 
392  // Get the expected result in serial
393  for ( int p = 0; p < num_particle; ++p )
394  for ( int n = 0; n < nlist._counts(p); ++n )
395  test_result(p) += nlist._neighbors( nlist._offsets(p) + n );
396 
397  // Check the result.
398  for ( int p = 0; p < num_particle; ++p )
399  {
400  EXPECT_EQ( test_result(p), serial_result(p) );
401  EXPECT_EQ( test_result(p), team_result(p) );
402  }
403 }
