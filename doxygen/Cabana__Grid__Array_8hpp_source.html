228template <class Scalar, class EntityType, class MeshType, class... Params>
229class Array
230{
231 public:
233 using value_type = Scalar;
234
236 using entity_type = EntityType;
237
239 using mesh_type = MeshType;
240
242 static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;
243
245 using array_layout = ArrayLayout<entity_type, mesh_type>;
246
248 using view_type = std::conditional_t<
249 3 == num_space_dim, Kokkos::View<value_type****, Params...>,
250 std::conditional_t<2 == num_space_dim,
251 Kokkos::View<value_type***, Params...>, void>>;
252
254 using memory_space = typename view_type::memory_space;
256 using device_type [[deprecated]] = typename memory_space::device_type;
258 using execution_space = typename memory_space::execution_space;
259
266 Array( const std::string& label,
267 const std::shared_ptr<array_layout>& layout )
268 : _layout( layout )
269 , _data( createView<value_type, Params...>(
270 label, layout->indexSpace( Ghost(), Local() ) ) )
271 {
272 }
273
280 Array( const std::shared_ptr<array_layout>& layout, const view_type& view )
281 : _layout( layout )
282 , _data( view )
283 {
284 for ( std::size_t d = 0; d < num_space_dim + 1; ++d )
285 if ( (long)view.extent( d ) !=
286 layout->indexSpace( Ghost(), Local() ).extent( d ) )
287 throw std::runtime_error(
288 "Layout and view dimensions do not match" );
289 }
290
292 std::shared_ptr<array_layout> layout() const { return _layout; }
293
295 view_type view() const { return _data; }
296
298 std::string label() const { return _data.label(); }
299
300 private:
301 std::shared_ptr<array_layout> _layout;
302 view_type _data;
303
304 public:
306 using subview_type = decltype( createSubview(
307 _data, _layout->indexSpace( Ghost(), Local() ) ) );
309 using subview_layout = typename subview_type::array_layout;
311 using subview_memory_traits = typename subview_type::memory_traits;
313 using subarray_type = Array<Scalar, EntityType, MeshType, subview_layout,
314 memory_space, subview_memory_traits>;
315};

357//---------------------------------------------------------------------------//
358// Subarray creation.
359//---------------------------------------------------------------------------//
368template <class Scalar, class EntityType, class MeshType, class... Params>
369std::shared_ptr<Array<
370 Scalar, EntityType, MeshType,
371 typename Array<Scalar, EntityType, MeshType, Params...>::subview_layout,
372 typename Array<Scalar, EntityType, MeshType, Params...>::memory_space,
373 typename Array<Scalar, EntityType, MeshType,
374 Params...>::subview_memory_traits>>
375createSubarray( const Array<Scalar, EntityType, MeshType, Params...>& array,
376 const int dof_min, const int dof_max )
377{
378 if ( dof_min < 0 || dof_max > array.layout()->dofsPerEntity() )
379 throw std::logic_error( "Subarray dimensions out of bounds" );
380
381 auto space = array.layout()->indexSpace( Ghost(), Local() );
382 std::array<long, MeshType::num_space_dim + 1> min;
383 std::array<long, MeshType::num_space_dim + 1> max;
384 for ( std::size_t d = 0; d < MeshType::num_space_dim; ++d )
385 {
386 min[d] = space.min( d );
387 max[d] = space.max( d );
388 }
389 min.back() = dof_min;
390 max.back() = dof_max;
391 IndexSpace<MeshType::num_space_dim + 1> sub_space( min, max );
392 auto sub_view = createSubview( array.view(), sub_space );
393 auto sub_layout = createArrayLayout( array.layout()->localGrid(),
394 dof_max - dof_min, EntityType() );
395 return std::make_shared<Array<
396 Scalar, EntityType, MeshType,
397 typename Array<Scalar, EntityType, MeshType, Params...>::subview_layout,
398 typename Array<Scalar, EntityType, MeshType, Params...>::memory_space,
399 typename Array<Scalar, EntityType, MeshType,
400 Params...>::subview_memory_traits>>( sub_layout,
401 sub_view );
402}

721

780template <class Array_t>
781void dot( const Array_t& a, const Array_t& b,
782 std::vector<typename Array_t::value_type>& products )
783{
784 static_assert( is_array<Array_t>::value, "Cabana::Grid::Array required" );
785 if ( products.size() !=
786 static_cast<unsigned>( a.layout()->dofsPerEntity() ) )
787 throw std::runtime_error( "Incorrect vector size" );
788
789 for ( auto& p : products )
790 p = 0.0;
791
792 DotFunctor<typename Array_t::view_type, Array_t::num_space_dim> functor(
793 a.view(), b.view() );
794 typename Array_t::execution_space exec_space;
795 Kokkos::parallel_reduce(
796 "ArrayOp::dot",
797 createExecutionPolicy( a.layout()->indexSpace( Own(), Local() ),
798 exec_space ),
799 functor,
800 Kokkos::View<typename Array_t::value_type*, Kokkos::HostSpace>(
801 products.data(), products.size() ) );
802 exec_space.fence( "ArrayOp::dot before MPI_Allreduce" );
803
804 MPI_Allreduce( MPI_IN_PLACE, products.data(), products.size(),
805 MpiTraits<typename Array_t::value_type>::type(), MPI_SUM,
806 a.layout()->localGrid()->globalGrid().comm() );
807}
