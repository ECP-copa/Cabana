121//---------------------------------------------------------------------------//
122// Neighborhood discriminator.
123template <class Tag>
124class NeighborDiscriminator;
125
126// Full list specialization.
127template <>
128class NeighborDiscriminator<FullNeighborTag>
129{
130 public:
131 // Full neighbor lists count and store the neighbors of all
132 // particles. The only criteria for a potentially valid neighbor is
133 // that the particle does not neighbor itself (i.e. the particle index
134 // "p" is not the same as the neighbor index "n").
135 KOKKOS_INLINE_FUNCTION
136 static bool isValid( const std::size_t p, const double, const double,
137 const double, const std::size_t n, const double,
138 const double, const double )
139 {
140 return ( p != n );
141 }
142};
143
144// Half list specialization.
145template <>
146class NeighborDiscriminator<HalfNeighborTag>
147{
148 public:
149 // Half neighbor lists only store half of the neighbors be eliminating
150 // duplicate pairs such that the fact that particle "p" neighbors
151 // particle "n" is stored in the list but "n" neighboring "p" is not
152 // stored but rather implied. We discriminate by only storing neighbors
153 // who's coordinates are greater in the x direction. If they are the same
154 // then the y direction is checked next and finally the z direction if the
155 // y coordinates are the same.
156 KOKKOS_INLINE_FUNCTION
157 static bool isValid( const std::size_t p, const double xp, const double yp,
158 const double zp, const std::size_t n, const double xn,
159 const double yn, const double zn )
160 {
161 return ( ( p != n ) &&
162 ( ( xn > xp ) ||
163 ( ( xn == xp ) &&
164 ( ( yn > yp ) || ( ( yn == yp ) && ( zn > zp ) ) ) ) ) );
165 }
166};

168//---------------------------------------------------------------------------//
169// Cell stencil.
170template <class Scalar>
171struct LinkedCellStencil
172{
173 Scalar rsqr;
174 CartesianGrid<double> grid;
175 int max_cells_dir;
176 int max_cells;
177 int cell_range;
178
179 LinkedCellStencil( const Scalar neighborhood_radius,
180 const Scalar cell_size_ratio, const Scalar grid_min[3],
181 const Scalar grid_max[3] )
182 : rsqr( neighborhood_radius * neighborhood_radius )
183 {
184 Scalar dx = neighborhood_radius * cell_size_ratio;
185 grid = CartesianGrid<double>( grid_min[0], grid_min[1], grid_min[2],
186 grid_max[0], grid_max[1], grid_max[2], dx,
187 dx, dx );
188 cell_range = std::ceil( 1 / cell_size_ratio );
189 max_cells_dir = 2 * cell_range + 1;
190 max_cells = max_cells_dir * max_cells_dir * max_cells_dir;
191 }
192
193 // Given a cell, get the index bounds of the cell stencil.
194 KOKKOS_INLINE_FUNCTION
195 void getCells( const int cell, int& imin, int& imax, int& jmin, int& jmax,
196 int& kmin, int& kmax ) const
197 {
198 int i, j, k;
199 grid.ijkBinIndex( cell, i, j, k );
200
201 kmin = ( k - cell_range > 0 ) ? k - cell_range : 0;
202 kmax =
203 ( k + cell_range + 1 < grid._nz ) ? k + cell_range + 1 : grid._nz;
204
205 jmin = ( j - cell_range > 0 ) ? j - cell_range : 0;
206 jmax =
207 ( j + cell_range + 1 < grid._ny ) ? j + cell_range + 1 : grid._ny;
208
209 imin = ( i - cell_range > 0 ) ? i - cell_range : 0;
210 imax =
211 ( i + cell_range + 1 < grid._nx ) ? i + cell_range + 1 : grid._nx;
212 }
213};

216// Verlet List Builder
217//---------------------------------------------------------------------------//
218template <class DeviceType, class PositionSlice, class AlgorithmTag,
219 class LayoutTag, class BuildOpTag>
220struct VerletListBuilder
221{
222 // Types.
223 using device = DeviceType;
224 using PositionValueType = typename PositionSlice::value_type;
225 using RandomAccessPositionSlice =
226 typename PositionSlice::random_access_slice;
227 using memory_space = typename device::memory_space;
228 using execution_space = typename device::execution_space;
229
230 // List data.
231 VerletListData<memory_space, LayoutTag> _data;
232
233 // Neighbor cutoff.
234 PositionValueType rsqr;
235
236 // Positions.
237 RandomAccessPositionSlice position;
238 std::size_t pid_begin, pid_end;
239
240 // Binning Data.
241 BinningData<memory_space> bin_data_1d;
242 LinkedCellList<memory_space> linked_cell_list;
243
244 // Cell stencil.
245 LinkedCellStencil<PositionValueType> cell_stencil;
246
247 // Check to count or refill.
248 bool refill;
249 bool count;
250
251 // Maximum allocated neighbors per particle
252 std::size_t alloc_n;
253
254 // Constructor.
255 VerletListBuilder( PositionSlice slice, const std::size_t begin,
256 const std::size_t end,
257 const PositionValueType neighborhood_radius,
258 const PositionValueType cell_size_ratio,
259 const PositionValueType grid_min[3],
260 const PositionValueType grid_max[3],
261 const std::size_t max_neigh )
262 : pid_begin( begin )
263 , pid_end( end )
264 , cell_stencil( neighborhood_radius, cell_size_ratio, grid_min,
265 grid_max )
266 , alloc_n( max_neigh )
267 {
268 count = true;
269 refill = false;
270
271 // Create the count view.
272 _data.counts =
273 Kokkos::View<int*, memory_space>( "num_neighbors", slice.size() );
274
275 // Make a guess for the number of neighbors per particle for 2D lists.
276 initCounts( LayoutTag() );
277
278 // Get the positions with random access read-only memory.
279 position = slice;
280
281 // Bin the particles in the grid. Don't actually sort them but make a
282 // permutation vector. Note that we are binning all particles here and
283 // not just the requested range. This is because all particles are
284 // treated as candidates for neighbors.
285 double grid_size = cell_size_ratio * neighborhood_radius;
286 PositionValueType grid_delta[3] = { grid_size, grid_size, grid_size };
287 linked_cell_list = LinkedCellList<memory_space>( position, grid_delta,
288 grid_min, grid_max );
289 bin_data_1d = linked_cell_list.binningData();
290
291 // We will use the square of the distance for neighbor determination.
292 rsqr = neighborhood_radius * neighborhood_radius;
293 }

295 // Neighbor count team operator (only used for CSR lists).
296 struct CountNeighborsTag
297 {
298 };
299 using CountNeighborsPolicy =
300 Kokkos::TeamPolicy<execution_space, CountNeighborsTag,
301 Kokkos::IndexType<int>,
302 Kokkos::Schedule<Kokkos::Dynamic>>;
303
304 KOKKOS_INLINE_FUNCTION
305 void
306 operator()( const CountNeighborsTag&,
307 const typename CountNeighborsPolicy::member_type& team ) const
308 {
309 // The league rank of the team is the cardinal cell index we are
310 // working on.
311 int cell = team.league_rank();
312
313 // Get the stencil for this cell.
314 int imin, imax, jmin, jmax, kmin, kmax;
315 cell_stencil.getCells( cell, imin, imax, jmin, jmax, kmin, kmax );
316
317 // Operate on the particles in the bin.
318 std::size_t b_offset = bin_data_1d.binOffset( cell );
319 Kokkos::parallel_for(
320 Kokkos::TeamThreadRange( team, 0, bin_data_1d.binSize( cell ) ),
321 [&]( const int bi )
322 {
323 // Get the true particle id. The binned particle index is the
324 // league rank of the team.
325 std::size_t pid = linked_cell_list.permutation( bi + b_offset );
326
327 if ( ( pid >= pid_begin ) && ( pid < pid_end ) )
328 {
329 // Cache the particle coordinates.
330 double x_p = position( pid, 0 );
331 double y_p = position( pid, 1 );
332 double z_p = position( pid, 2 );
333
334 // Loop over the cell stencil.
335 int stencil_count = 0;
336 for ( int i = imin; i < imax; ++i )
337 for ( int j = jmin; j < jmax; ++j )
338 for ( int k = kmin; k < kmax; ++k )
339 {
340 // See if we should actually check this box for
341 // neighbors.
342 if ( cell_stencil.grid.minDistanceToPoint(
343 x_p, y_p, z_p, i, j, k ) <= rsqr )
344 {
345 std::size_t n_offset =
346 linked_cell_list.binOffset( i, j, k );
347 std::size_t num_n =
348 linked_cell_list.binSize( i, j, k );
349
350 // Check the particles in this bin to see if
351 // they are neighbors. If they are add to
352 // the count for this bin.
353 int cell_count = 0;
354 neighbor_reduce( team, pid, x_p, y_p, z_p,
355 n_offset, num_n,
356 cell_count, BuildOpTag() );
357 stencil_count += cell_count;
358 }
359 }
360 Kokkos::single( Kokkos::PerThread( team ), [&]()
361 { _data.counts( pid ) = stencil_count; } );
362 }
363 } );
364 }
365
366 // Neighbor count team vector loop (only used for CSR lists).
367 KOKKOS_INLINE_FUNCTION void
368 neighbor_reduce( const typename CountNeighborsPolicy::member_type& team,
369 const std::size_t pid, const double x_p, const double y_p,
370 const double z_p, const int n_offset, const int num_n,
371 int& cell_count, TeamVectorOpTag ) const
372 {
373 Kokkos::parallel_reduce(
374 Kokkos::ThreadVectorRange( team, num_n ),
375 [&]( const int n, int& local_count ) {
376 neighbor_kernel( pid, x_p, y_p, z_p, n_offset, n, local_count );
377 },
378 cell_count );
379 }
380
381 // Neighbor count serial loop (only used for CSR lists).
382 KOKKOS_INLINE_FUNCTION
383 void neighbor_reduce( const typename CountNeighborsPolicy::member_type,
384 const std::size_t pid, const double x_p,
385 const double y_p, const double z_p,
386 const int n_offset, const int num_n, int& cell_count,
387 TeamOpTag ) const
388 {
389 for ( int n = 0; n < num_n; n++ )
390 neighbor_kernel( pid, x_p, y_p, z_p, n_offset, n, cell_count );
391 }

393 // Neighbor count kernel
394 KOKKOS_INLINE_FUNCTION
395 void neighbor_kernel( const int pid, const double x_p, const double y_p,
396 const double z_p, const int n_offset, const int n,
397 int& local_count ) const
398 {
399 // Get the true id of the candidate neighbor.
400 std::size_t nid = linked_cell_list.permutation( n_offset + n );
401
402 // Cache the candidate neighbor particle coordinates.
403 double x_n = position( nid, 0 );
404 double y_n = position( nid, 1 );
405 double z_n = position( nid, 2 );
406
407 // If this could be a valid neighbor, continue.
408 if ( NeighborDiscriminator<AlgorithmTag>::isValid(
409 pid, x_p, y_p, z_p, nid, x_n, y_n, z_n ) )
410 {
411 // Calculate the distance between the particle and its candidate
412 // neighbor.
413 PositionValueType dx = x_p - x_n;
414 PositionValueType dy = y_p - y_n;
415 PositionValueType dz = z_p - z_n;
416 PositionValueType dist_sqr = dx * dx + dy * dy + dz * dz;
417
418 // If within the cutoff add to the count.
419 if ( dist_sqr <= rsqr )
420 local_count += 1;
421 }
422 }

424 // Process the CSR counts by computing offsets and allocating the neighbor
425 // list.
426 template <class KokkosMemorySpace>
427 struct OffsetScanOp
428 {
429 using kokkos_mem_space = KokkosMemorySpace;
430 Kokkos::View<int*, kokkos_mem_space> counts;
431 Kokkos::View<int*, kokkos_mem_space> offsets;
432 KOKKOS_INLINE_FUNCTION
433 void operator()( const int i, int& update, const bool final_pass ) const
434 {
435 if ( final_pass )
436 offsets( i ) = update;
437 update += counts( i );
438 }
439 };
440
441 void initCounts( VerletLayoutCSR ) {}
442
443 void initCounts( VerletLayout2D )
444 {
445 if ( alloc_n > 0 )
446 {
447 count = false;
448
449 _data.neighbors = Kokkos::View<int**, memory_space>(
450 Kokkos::ViewAllocateWithoutInitializing( "neighbors" ),
451 _data.counts.size(), alloc_n );
452 }
453 }
454
455 void processCounts( VerletLayoutCSR )
456 {
457 // Allocate offsets.
458 _data.offsets = Kokkos::View<int*, memory_space>(
459 Kokkos::ViewAllocateWithoutInitializing( "neighbor_offsets" ),
460 _data.counts.size() );
461
462 // Calculate offsets from counts and the total number of counts.
463 OffsetScanOp<memory_space> offset_op;
464 offset_op.counts = _data.counts;
465 offset_op.offsets = _data.offsets;
466 int total_num_neighbor;
467 Kokkos::RangePolicy<execution_space> range_policy(
468 0, _data.counts.extent( 0 ) );
469 Kokkos::parallel_scan( "Cabana::VerletListBuilder::offset_scan",
470 range_policy, offset_op, total_num_neighbor );
471 Kokkos::fence();
472
473 // Allocate the neighbor list.
474 _data.neighbors = Kokkos::View<int*, memory_space>(
475 Kokkos::ViewAllocateWithoutInitializing( "neighbors" ),
476 total_num_neighbor );
477
478 // Reset the counts. We count again when we fill.
479 Kokkos::deep_copy( _data.counts, 0 );
480 }
481
482 // Process 2D counts by computing the maximum number of neighbors and
483 // reallocating the 2D data structure if needed.
484 void processCounts( VerletLayout2D )
485 {
486 // Calculate the maximum number of neighbors.
487 auto counts = _data.counts;
488 int max;
489 Kokkos::Max<int> max_reduce( max );
490 Kokkos::parallel_reduce(
491 "Cabana::VerletListBuilder::reduce_max",
492 Kokkos::RangePolicy<execution_space>( 0, _data.counts.size() ),
493 KOKKOS_LAMBDA( const int i, int& value ) {
494 if ( counts( i ) > value )
495 value = counts( i );
496 },
497 max_reduce );
498 Kokkos::fence();
499 _data.max_n = static_cast<std::size_t>( max );
500
501 // Reallocate the neighbor list if previous size is exceeded.
502 if ( count or _data.max_n > _data.neighbors.extent( 1 ) )
503 {
504 refill = true;
505 Kokkos::deep_copy( _data.counts, 0 );
506 _data.neighbors = Kokkos::View<int**, memory_space>(
507 Kokkos::ViewAllocateWithoutInitializing( "neighbors" ),
508 _data.counts.size(), _data.max_n );
509 }
510 }

577 neighbor_for( const typename FillNeighborsPolicy::member_type& team,

588 KOKKOS_INLINE_FUNCTION
589 void neighbor_for( const typename FillNeighborsPolicy::member_type team,

600 // Neighbor fill kernel.
601 KOKKOS_INLINE_FUNCTION
602 void neighbor_kernel( const int pid, const double x_p, const double y_p,
603 const double z_p, const int n_offset,
604 const int n ) const
605 {
606 // Get the true id of the candidate neighbor.
607 std::size_t nid = linked_cell_list.permutation( n_offset + n );
608
609 // Cache the candidate neighbor particle coordinates.
610 double x_n = position( nid, 0 );
611 double y_n = position( nid, 1 );
612 double z_n = position( nid, 2 );
613
614 // If this could be a valid neighbor, continue.
615 if ( NeighborDiscriminator<AlgorithmTag>::isValid(
616 pid, x_p, y_p, z_p, nid, x_n, y_n, z_n ) )
617 {
618 // Calculate the distance between the particle and its candidate
619 // neighbor.
620 PositionValueType dx = x_p - x_n;
621 PositionValueType dy = y_p - y_n;
622 PositionValueType dz = z_p - z_n;
623 PositionValueType dist_sqr = dx * dx + dy * dy + dz * dz;
624
625 // If within the cutoff increment the neighbor count and add as a
626 // neighbor at that index.
627 if ( dist_sqr <= rsqr )
628 {
629 _data.addNeighbor( pid, nid );
630 }
631 }
632 }

658template <class MemorySpace, class AlgorithmTag, class LayoutTag,

660class VerletList
661{
662 public:
663 static_assert( Kokkos::is_memory_space<MemorySpace>::value, "" );
664
666 using memory_space = MemorySpace;
667
669 using execution_space = typename memory_space::execution_space;

672 VerletListData<memory_space, LayoutTag> _data;

713 template <class PositionSlice>
714 VerletList( PositionSlice x, const std::size_t begin, const std::size_t end,
715 const typename PositionSlice::value_type neighborhood_radius,
716 const typename PositionSlice::value_type cell_size_ratio,
717 const typename PositionSlice::value_type grid_min[3],
718 const typename PositionSlice::value_type grid_max[3],
719 const std::size_t max_neigh = 0,
720 typename std::enable_if<( is_slice<PositionSlice>::value ),
721 int>::type* = 0 )
722 {
723 build( x, begin, end, neighborhood_radius, cell_size_ratio, grid_min,
724 grid_max, max_neigh );
725 }
726
731 template <class PositionSlice>
732 void build( PositionSlice x, const std::size_t begin, const std::size_t end,
733 const typename PositionSlice::value_type neighborhood_radius,
734 const typename PositionSlice::value_type cell_size_ratio,
735 const typename PositionSlice::value_type grid_min[3],
736 const typename PositionSlice::value_type grid_max[3],
737 const std::size_t max_neigh = 0 )
738 {
739 // Use the default execution space.
740 build( execution_space{}, x, begin, end, neighborhood_radius,
741 cell_size_ratio, grid_min, grid_max, max_neigh );
742 }
747 template <class PositionSlice, class ExecutionSpace>
748 void build( ExecutionSpace, PositionSlice x, const std::size_t begin,
749 const std::size_t end,
750 const typename PositionSlice::value_type neighborhood_radius,
751 const typename PositionSlice::value_type cell_size_ratio,
752 const typename PositionSlice::value_type grid_min[3],
753 const typename PositionSlice::value_type grid_max[3],
754 const std::size_t max_neigh = 0 )
755 {
756 Kokkos::Profiling::pushRegion( "Cabana::VerletList::build" );
757
758 static_assert( is_accessible_from<memory_space, ExecutionSpace>{}, "" );
759
760 assert( end >= begin );
761 assert( end <= x.size() );
762
763 using device_type = Kokkos::Device<ExecutionSpace, memory_space>;
764
765 // Create a builder functor.
766 using builder_type =
767 Impl::VerletListBuilder<device_type, PositionSlice, AlgorithmTag,
768 LayoutTag, BuildTag>;
769 builder_type builder( x, begin, end, neighborhood_radius,
770 cell_size_ratio, grid_min, grid_max, max_neigh );
771
772 // For each particle in the range check each neighboring bin for
773 // neighbor particles. Bins are at least the size of the neighborhood
774 // radius so the bin in which the particle resides and any surrounding
775 // bins are guaranteed to contain the neighboring particles.
776 // For CSR lists, we count, then fill neighbors. For 2D lists, we
777 // count and fill at the same time, unless the array size is exceeded,
778 // at which point only counting is continued to reallocate and refill.
779 typename builder_type::FillNeighborsPolicy fill_policy(
780 builder.bin_data_1d.numBin(), Kokkos::AUTO, 4 );
781 if ( builder.count )
782 {
783 typename builder_type::CountNeighborsPolicy count_policy(
784 builder.bin_data_1d.numBin(), Kokkos::AUTO, 4 );
785 Kokkos::parallel_for( "Cabana::VerletList::count_neighbors",
786 count_policy, builder );
787 }
788 else
789 {
790 builder.processCounts( LayoutTag() );
791 Kokkos::parallel_for( "Cabana::VerletList::fill_neighbors",
792 fill_policy, builder );
793 }
794 Kokkos::fence();
795
796 // Process the counts by computing offsets and allocating the neighbor
797 // list, if needed.
798 builder.processCounts( LayoutTag() );
799
800 // For each particle in the range fill (or refill) its part of the
801 // neighbor list.
802 if ( builder.count or builder.refill )
803 {
804 Kokkos::parallel_for( "Cabana::VerletList::fill_neighbors",
805 fill_policy, builder );
806 Kokkos::fence();
807 }
808
809 // Get the data from the builder.
810 _data = builder._data;
811
812 Kokkos::Profiling::popRegion();
813 }
