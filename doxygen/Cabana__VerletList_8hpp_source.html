119 template <class Tag>

122 // Full list specialization.
123 template <>
124 class NeighborDiscriminator<FullNeighborTag>
125 {
126  public:
127  // Full neighbor lists count and store the neighbors of all
128  // particles. The only criteria for a potentially valid neighbor is
129  // that the particle does not neighbor itself (i.e. the particle index
130  // "p" is not the same as the neighbor index "n").
131  KOKKOS_INLINE_FUNCTION
132  static bool isValid( const std::size_t p, const double, const double,
133  const double, const std::size_t n, const double,
134  const double, const double )
135  {
136  return ( p != n );
137  }
138 };
139 
140 // Half list specialization.
141 template <>
142 class NeighborDiscriminator<HalfNeighborTag>
143 {
144  public:
145  // Half neighbor lists only store half of the neighbors be eliminating
146  // duplicate pairs such that the fact that particle "p" neighbors
147  // particle "n" is stored in the list but "n" neighboring "p" is not
148  // stored but rather implied. We discriminate by only storing neighbors
149  // who's coordinates are greater in the x direction. If they are the same
150  // then the y direction is checked next and finally the z direction if the
151  // y coordinates are the same.
152  KOKKOS_INLINE_FUNCTION
153  static bool isValid( const std::size_t p, const double xp, const double yp,
154  const double zp, const std::size_t n, const double xn,
155  const double yn, const double zn )
156  {
157  return ( ( p != n ) &&
158  ( ( xn > xp ) ||
159  ( ( xn == xp ) &&
160  ( ( yn > yp ) || ( ( yn == yp ) && ( zn > zp ) ) ) ) ) );
161  }
162 };

164 //---------------------------------------------------------------------------//
165 // Cell stencil.
166 template <class Scalar>
167 struct LinkedCellStencil
168 {
169  Scalar rsqr;
170  CartesianGrid<double> grid;
171  int max_cells_dir;
172  int max_cells;
173  int cell_range;
174 
175  LinkedCellStencil( const Scalar neighborhood_radius,
176  const Scalar cell_size_ratio, const Scalar grid_min[3],
177  const Scalar grid_max[3] )
178  : rsqr( neighborhood_radius * neighborhood_radius )
179  {
180  Scalar dx = neighborhood_radius * cell_size_ratio;
181  grid = CartesianGrid<double>( grid_min[0], grid_min[1], grid_min[2],
182  grid_max[0], grid_max[1], grid_max[2], dx,
183  dx, dx );
184  cell_range = std::ceil( 1 / cell_size_ratio );
185  max_cells_dir = 2 * cell_range + 1;
186  max_cells = max_cells_dir * max_cells_dir * max_cells_dir;
187  }
188 
189  // Given a cell, get the index bounds of the cell stencil.
190  KOKKOS_INLINE_FUNCTION
191  void getCells( const int cell, int& imin, int& imax, int& jmin, int& jmax,
192  int& kmin, int& kmax ) const
193  {
194  int i, j, k;
195  grid.ijkBinIndex( cell, i, j, k );
196 
197  kmin = ( k - cell_range > 0 ) ? k - cell_range : 0;
198  kmax =
199  ( k + cell_range + 1 < grid._nz ) ? k + cell_range + 1 : grid._nz;
200 
201  jmin = ( j - cell_range > 0 ) ? j - cell_range : 0;
202  jmax =
203  ( j + cell_range + 1 < grid._ny ) ? j + cell_range + 1 : grid._ny;
204 
205  imin = ( i - cell_range > 0 ) ? i - cell_range : 0;
206  imax =
207  ( i + cell_range + 1 < grid._nx ) ? i + cell_range + 1 : grid._nx;
208  }
209 };

214 template <class DeviceType, class PositionSlice, class AlgorithmTag,
215  class LayoutTag, class BuildOpTag>
216 struct VerletListBuilder
217 {
218  // Types.
219  using device = DeviceType;
220  using PositionValueType = typename PositionSlice::value_type;
221  using RandomAccessPositionSlice =
222  typename PositionSlice::random_access_slice;
223  using memory_space = typename device::memory_space;
224  using execution_space = typename device::execution_space;
225 
226  // List data.
227  VerletListData<memory_space, LayoutTag> _data;
228 
229  // Neighbor cutoff.
230  PositionValueType rsqr;
231 
232  // Positions.
233  RandomAccessPositionSlice position;
234  std::size_t pid_begin, pid_end;
235 
236  // Binning Data.
237  BinningData<device> bin_data_1d;
238  LinkedCellList<device> linked_cell_list;
239 
240  // Cell stencil.
241  LinkedCellStencil<PositionValueType> cell_stencil;
242 
243  // Check to count or refill.
244  bool refill;
245  bool count;
246 
247  // Maximum neighbors per particle
248  std::size_t max_n;
249 
250  // Constructor.
251  VerletListBuilder( PositionSlice slice, const std::size_t begin,
252  const std::size_t end,
253  const PositionValueType neighborhood_radius,
254  const PositionValueType cell_size_ratio,
255  const PositionValueType grid_min[3],
256  const PositionValueType grid_max[3],
257  const std::size_t max_neigh )
258  : pid_begin( begin )
259  , pid_end( end )
260  , cell_stencil( neighborhood_radius, cell_size_ratio, grid_min,
261  grid_max )
262  , max_n( max_neigh )
263  {
264  count = true;
265  refill = false;
266 
267  // Create the count view.
268  _data.counts =
269  Kokkos::View<int*, memory_space>( "num_neighbors", slice.size() );
270 
271  // Make a guess for the number of neighbors per particle for 2D lists.
272  initCounts( LayoutTag() );
273 
274  // Get the positions with random access read-only memory.
275  position = slice;
276 
277  // Bin the particles in the grid. Don't actually sort them but make a
278  // permutation vector. Note that we are binning all particles here and
279  // not just the requested range. This is because all particles are
280  // treated as candidates for neighbors.
281  double grid_size = cell_size_ratio * neighborhood_radius;
282  PositionValueType grid_delta[3] = { grid_size, grid_size, grid_size };
283  linked_cell_list =
284  LinkedCellList<device>( position, grid_delta, grid_min, grid_max );
285  bin_data_1d = linked_cell_list.binningData();
286 
287  // We will use the square of the distance for neighbor determination.
288  rsqr = neighborhood_radius * neighborhood_radius;
289  }

291  // Neighbor count team operator (only used for CSR lists).
292  struct CountNeighborsTag
293  {
294  };
295  using CountNeighborsPolicy =
296  Kokkos::TeamPolicy<execution_space, CountNeighborsTag,
297  Kokkos::IndexType<int>,
298  Kokkos::Schedule<Kokkos::Dynamic>>;
299 
300  KOKKOS_INLINE_FUNCTION
301  void
302  operator()( const CountNeighborsTag&,
303  const typename CountNeighborsPolicy::member_type& team ) const
304  {
305  // The league rank of the team is the cardinal cell index we are
306  // working on.
307  int cell = team.league_rank();
308 
309  // Get the stencil for this cell.
310  int imin, imax, jmin, jmax, kmin, kmax;
311  cell_stencil.getCells( cell, imin, imax, jmin, jmax, kmin, kmax );
312 
313  // Operate on the particles in the bin.
314  std::size_t b_offset = bin_data_1d.binOffset( cell );
315  Kokkos::parallel_for(
316  Kokkos::TeamThreadRange( team, 0, bin_data_1d.binSize( cell ) ),
317  [&]( const int bi )
318  {
319  // Get the true particle id. The binned particle index is the
320  // league rank of the team.
321  std::size_t pid = linked_cell_list.permutation( bi + b_offset );
322 
323  if ( ( pid >= pid_begin ) && ( pid < pid_end ) )
324  {
325  // Cache the particle coordinates.
326  double x_p = position( pid, 0 );
327  double y_p = position( pid, 1 );
328  double z_p = position( pid, 2 );
329 
330  // Loop over the cell stencil.
331  int stencil_count = 0;
332  for ( int i = imin; i < imax; ++i )
333  for ( int j = jmin; j < jmax; ++j )
334  for ( int k = kmin; k < kmax; ++k )
335  {
336  // See if we should actually check this box for
337  // neighbors.
338  if ( cell_stencil.grid.minDistanceToPoint(
339  x_p, y_p, z_p, i, j, k ) <= rsqr )
340  {
341  std::size_t n_offset =
342  linked_cell_list.binOffset( i, j, k );
343  std::size_t num_n =
344  linked_cell_list.binSize( i, j, k );
345 
346  // Check the particles in this bin to see if
347  // they are neighbors. If they are add to
348  // the count for this bin.
349  int cell_count = 0;
350  neighbor_reduce( team, pid, x_p, y_p, z_p,
351  n_offset, num_n,
352  cell_count, BuildOpTag() );
353  stencil_count += cell_count;
354  }
355  }
356  Kokkos::single( Kokkos::PerThread( team ), [&]()
357  { _data.counts( pid ) = stencil_count; } );
358  }
359  } );
360  }

389  // Neighbor count kernel
390  KOKKOS_INLINE_FUNCTION
391  void neighbor_kernel( const int pid, const double x_p, const double y_p,
392  const double z_p, const int n_offset, const int n,
393  int& local_count ) const
394  {
395  // Get the true id of the candidate neighbor.
396  std::size_t nid = linked_cell_list.permutation( n_offset + n );
397 
398  // Cache the candidate neighbor particle coordinates.
399  double x_n = position( nid, 0 );
400  double y_n = position( nid, 1 );
401  double z_n = position( nid, 2 );
402 
403  // If this could be a valid neighbor, continue.
404  if ( NeighborDiscriminator<AlgorithmTag>::isValid(
405  pid, x_p, y_p, z_p, nid, x_n, y_n, z_n ) )
406  {
407  // Calculate the distance between the particle and its candidate
408  // neighbor.
409  PositionValueType dx = x_p - x_n;
410  PositionValueType dy = y_p - y_n;
411  PositionValueType dz = z_p - z_n;
412  PositionValueType dist_sqr = dx * dx + dy * dy + dz * dz;
413 
414  // If within the cutoff add to the count.
415  if ( dist_sqr <= rsqr )
416  local_count += 1;
417  }
418  }

422  template <class KokkosMemorySpace>
423  struct OffsetScanOp

429  void operator()( const int i, int& update, const bool final_pass ) const

432  offsets( i ) = update;

451  void processCounts( VerletLayoutCSR )
452  {
453  // Allocate offsets.
454  _data.offsets = Kokkos::View<int*, memory_space>(
455  Kokkos::ViewAllocateWithoutInitializing( "neighbor_offsets" ),
456  _data.counts.size() );
457 
458  // Calculate offsets from counts and the total number of counts.
459  OffsetScanOp<memory_space> offset_op;
460  offset_op.counts = _data.counts;
461  offset_op.offsets = _data.offsets;
462  int total_num_neighbor;
463  Kokkos::RangePolicy<execution_space> range_policy(
464  0, _data.counts.extent( 0 ) );
465  Kokkos::parallel_scan( "Cabana::VerletListBuilder::offset_scan",
466  range_policy, offset_op, total_num_neighbor );
467  Kokkos::fence();
468 
469  // Allocate the neighbor list.
470  _data.neighbors = Kokkos::View<int*, memory_space>(
471  Kokkos::ViewAllocateWithoutInitializing( "neighbors" ),
472  total_num_neighbor );
473 
474  // Reset the counts. We count again when we fill.
475  Kokkos::deep_copy( _data.counts, 0 );
476  }
477 
478  // Process 2D counts by computing the maximum number of neighbors and
479  // reallocating the 2D data structure if needed.
480  void processCounts( VerletLayout2D )
481  {
482  // Calculate the maximum number of neighbors.
483  auto counts = _data.counts;
484  int max_num_neighbor = 0;
485  Kokkos::Max<int> max_reduce( max_num_neighbor );
486  Kokkos::parallel_reduce(
487  "Cabana::VerletListBuilder::reduce_max",
488  Kokkos::RangePolicy<execution_space>( 0, _data.counts.size() ),
489  KOKKOS_LAMBDA( const int i, int& value ) {
490  if ( counts( i ) > value )
491  value = counts( i );
492  },
493  max_reduce );
494  Kokkos::fence();
495 
496  // Reallocate the neighbor list if previous size is exceeded.
497  if ( count or ( std::size_t )
498  max_num_neighbor > _data.neighbors.extent( 1 ) )
499  {
500  refill = true;
501  Kokkos::deep_copy( _data.counts, 0 );
502  _data.neighbors = Kokkos::View<int**, memory_space>(
503  Kokkos::ViewAllocateWithoutInitializing( "neighbors" ),
504  _data.counts.size(), max_num_neighbor );
505  }
506  }

743  template <class PositionSlice, class ExecutionSpace>
744  void build( ExecutionSpace, PositionSlice x, const std::size_t begin,
745  const std::size_t end,
746  const typename PositionSlice::value_type neighborhood_radius,
747  const typename PositionSlice::value_type cell_size_ratio,
748  const typename PositionSlice::value_type grid_min[3],
749  const typename PositionSlice::value_type grid_max[3],
750  const std::size_t max_neigh = 0 )
751  {
752  static_assert( is_accessible_from<memory_space, ExecutionSpace>{}, "" );
753 
754  assert( end >= begin );
755  assert( end <= x.size() );
756 
757  using device_type = Kokkos::Device<ExecutionSpace, memory_space>;
758 
759  // Create a builder functor.
760  using builder_type =
761  Impl::VerletListBuilder<device_type, PositionSlice, AlgorithmTag,
762  LayoutTag, BuildTag>;
763  builder_type builder( x, begin, end, neighborhood_radius,
764  cell_size_ratio, grid_min, grid_max, max_neigh );
765 
766  // For each particle in the range check each neighboring bin for
767  // neighbor particles. Bins are at least the size of the neighborhood
768  // radius so the bin in which the particle resides and any surrounding
769  // bins are guaranteed to contain the neighboring particles.
770  // For CSR lists, we count, then fill neighbors. For 2D lists, we
771  // count and fill at the same time, unless the array size is exceeded,
772  // at which point only counting is continued to reallocate and refill.
773  typename builder_type::FillNeighborsPolicy fill_policy(
774  builder.bin_data_1d.numBin(), Kokkos::AUTO, 4 );
775  if ( builder.count )
776  {
777  typename builder_type::CountNeighborsPolicy count_policy(
778  builder.bin_data_1d.numBin(), Kokkos::AUTO, 4 );
779  Kokkos::parallel_for( "Cabana::VerletList::count_neighbors",
780  count_policy, builder );
781  }
782  else
783  {
784  builder.processCounts( LayoutTag() );
785  Kokkos::parallel_for( "Cabana::VerletList::fill_neighbors",
786  fill_policy, builder );
787  }
788  Kokkos::fence();
789 
790  // Process the counts by computing offsets and allocating the neighbor
791  // list, if needed.
792  builder.processCounts( LayoutTag() );
793 
794  // For each particle in the range fill (or refill) its part of the
795  // neighbor list.
796  if ( builder.count or builder.refill )
797  {
798  Kokkos::parallel_for( "Cabana::VerletList::fill_neighbors",
799  fill_policy, builder );
800  Kokkos::fence();
801  }
802 
803  // Get the data from the builder.
804  _data = builder._data;
805  }

809  void setNeighbor( const std::size_t particle_index,
810  const std::size_t neighbor_index,
811  const int new_index ) const
812  {
813  _data.setNeighbor( particle_index, neighbor_index, new_index );
