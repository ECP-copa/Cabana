94 // Neighborhood discriminator.
95 template <class Tag>
96 class NeighborDiscriminator;
97 
98 // Full list specialization.
99 template <>
100 class NeighborDiscriminator<FullNeighborTag>
101 {
102  public:
103  // Full neighbor lists count and store the neighbors of all
104  // particles. The only criteria for a potentially valid neighbor is
105  // that the particle does not neighbor itself (i.e. the particle index
106  // "p" is not the same as the neighbor index "n").
107  KOKKOS_INLINE_FUNCTION
108  static bool isValid( const std::size_t p, const double, const double,
109  const double, const std::size_t n, const double,
110  const double, const double )
111  {
112  return ( p != n );
113  }
114 };
115 
116 // Half list specialization.
117 template <>
118 class NeighborDiscriminator<HalfNeighborTag>
119 {
120  public:
121  // Half neighbor lists only store half of the neighbors be eliminating
122  // duplicate pairs such that the fact that particle "p" neighbors
123  // particle "n" is stored in the list but "n" neighboring "p" is not
124  // stored but rather implied. We discriminate by only storing neighbors
125  // who's coordinates are greater in the x direction. If they are the same
126  // then the y direction is checked next and finally the z direction if the
127  // y coordinates are the same.
128  KOKKOS_INLINE_FUNCTION
129  static bool isValid( const std::size_t p, const double xp, const double yp,
130  const double zp, const std::size_t n, const double xn,
131  const double yn, const double zn )
132  {
133  return ( ( p != n ) &&
134  ( ( xn > xp ) ||
135  ( ( xn == xp ) &&
136  ( ( yn > yp ) || ( ( yn == yp ) && ( zn > zp ) ) ) ) ) );
137  }
138 };

140 //---------------------------------------------------------------------------//
141 // Cell stencil.
142 template <class Scalar>
143 struct LinkedCellStencil
144 {
145  Scalar rsqr;
146  CartesianGrid<double> grid;
147  int max_cells_dir;
148  int max_cells;
149  int cell_range;
150 
151  LinkedCellStencil( const Scalar neighborhood_radius,
152  const Scalar cell_size_ratio, const Scalar grid_min[3],
153  const Scalar grid_max[3] )
154  : rsqr( neighborhood_radius * neighborhood_radius )
155  {
156  Scalar dx = neighborhood_radius * cell_size_ratio;
157  grid = CartesianGrid<double>( grid_min[0], grid_min[1], grid_min[2],
158  grid_max[0], grid_max[1], grid_max[2], dx,
159  dx, dx );
160  cell_range = std::ceil( 1 / cell_size_ratio );
161  max_cells_dir = 2 * cell_range + 1;
162  max_cells = max_cells_dir * max_cells_dir * max_cells_dir;
163  }
164 
165  // Given a cell, get the index bounds of the cell stencil.
166  KOKKOS_INLINE_FUNCTION
167  void getCells( const int cell, int& imin, int& imax, int& jmin, int& jmax,
168  int& kmin, int& kmax ) const
169  {
170  int i, j, k;
171  grid.ijkBinIndex( cell, i, j, k );
172 
173  kmin = ( k - cell_range > 0 ) ? k - cell_range : 0;
174  kmax =
175  ( k + cell_range + 1 < grid._nz ) ? k + cell_range + 1 : grid._nz;
176 
177  jmin = ( j - cell_range > 0 ) ? j - cell_range : 0;
178  jmax =
179  ( j + cell_range + 1 < grid._ny ) ? j + cell_range + 1 : grid._ny;
180 
181  imin = ( i - cell_range > 0 ) ? i - cell_range : 0;
182  imax =
183  ( i + cell_range + 1 < grid._nx ) ? i + cell_range + 1 : grid._nx;
184  }
185 };

187 //---------------------------------------------------------------------------//
188 // Verlet List Builder
189 //---------------------------------------------------------------------------//
190 template <class DeviceType, class PositionSlice, class AlgorithmTag,
191  class LayoutTag, class BuildOpTag>
192 struct VerletListBuilder
193 {
194  // Types.
195  using device = DeviceType;
196  using PositionValueType = typename PositionSlice::value_type;
197  using RandomAccessPositionSlice =
198  typename PositionSlice::random_access_slice;
199  using memory_space = typename device::memory_space;
200  using execution_space = typename device::execution_space;
201 
202  // List data.
203  VerletListData<device, LayoutTag> _data;
204 
205  // Neighbor cutoff.
206  PositionValueType rsqr;
207 
208  // Positions.
209  RandomAccessPositionSlice position;
210  std::size_t pid_begin, pid_end;
211 
212  // Binning Data.
213  BinningData<device> bin_data_1d;
214  LinkedCellList<device> linked_cell_list;
215 
216  // Cell stencil.
217  LinkedCellStencil<PositionValueType> cell_stencil;
218 
219  // Check to count or refill.
220  bool refill;
221  bool count;
222 
223  // Maximum neighbors per particle
224  std::size_t max_n;
225 
226  // Constructor.
227  VerletListBuilder( PositionSlice slice, const std::size_t begin,
228  const std::size_t end,
229  const PositionValueType neighborhood_radius,
230  const PositionValueType cell_size_ratio,
231  const PositionValueType grid_min[3],
232  const PositionValueType grid_max[3],
233  const std::size_t max_neigh )
234  : pid_begin( begin )
235  , pid_end( end )
236  , cell_stencil( neighborhood_radius, cell_size_ratio, grid_min,
237  grid_max )
238  , max_n( max_neigh )
239  {
240  count = true;
241  refill = false;
242 
243  // Create the count view.
244  _data.counts =
245  Kokkos::View<int*, memory_space>( "num_neighbors", slice.size() );
246 
247  // Make a guess for the number of neighbors per particle for 2D lists.
248  initCounts( LayoutTag() );
249 
250  // Get the positions with random access read-only memory.
251  position = slice;
252 
253  // Bin the particles in the grid. Don't actually sort them but make a
254  // permutation vector. Note that we are binning all particles here and
255  // not just the requested range. This is because all particles are
256  // treated as candidates for neighbors.
257  double grid_size = cell_size_ratio * neighborhood_radius;
258  PositionValueType grid_delta[3] = { grid_size, grid_size, grid_size };
259  linked_cell_list =
260  LinkedCellList<device>( position, grid_delta, grid_min, grid_max );
261  bin_data_1d = linked_cell_list.binningData();
262 
263  // We will use the square of the distance for neighbor determination.
264  rsqr = neighborhood_radius * neighborhood_radius;
265  }

267  // Neighbor count team operator (only used for CSR lists).
268  struct CountNeighborsTag
269  {
270  };
271  using CountNeighborsPolicy =
272  Kokkos::TeamPolicy<execution_space, CountNeighborsTag,
273  Kokkos::IndexType<int>,
274  Kokkos::Schedule<Kokkos::Dynamic>>;
275 
276  KOKKOS_INLINE_FUNCTION
277  void
278  operator()( const CountNeighborsTag&,
279  const typename CountNeighborsPolicy::member_type& team ) const
280  {
281  // The league rank of the team is the cardinal cell index we are
282  // working on.
283  int cell = team.league_rank();
284 
285  // Get the stencil for this cell.
286  int imin, imax, jmin, jmax, kmin, kmax;
287  cell_stencil.getCells( cell, imin, imax, jmin, jmax, kmin, kmax );
288 
289  // Operate on the particles in the bin.
290  std::size_t b_offset = bin_data_1d.binOffset( cell );
291  Kokkos::parallel_for(
292  Kokkos::TeamThreadRange( team, 0, bin_data_1d.binSize( cell ) ),
293  [&]( const int bi ) {
294  // Get the true particle id. The binned particle index is the
295  // league rank of the team.
296  std::size_t pid = linked_cell_list.permutation( bi + b_offset );
297 
298  if ( ( pid >= pid_begin ) && ( pid < pid_end ) )
299  {
300  // Cache the particle coordinates.
301  double x_p = position( pid, 0 );
302  double y_p = position( pid, 1 );
303  double z_p = position( pid, 2 );
304 
305  // Loop over the cell stencil.
306  int stencil_count = 0;
307  for ( int i = imin; i < imax; ++i )
308  for ( int j = jmin; j < jmax; ++j )
309  for ( int k = kmin; k < kmax; ++k )
310  {
311  // See if we should actually check this box for
312  // neighbors.
313  if ( cell_stencil.grid.minDistanceToPoint(
314  x_p, y_p, z_p, i, j, k ) <= rsqr )
315  {
316  std::size_t n_offset =
317  linked_cell_list.binOffset( i, j, k );
318  std::size_t num_n =
319  linked_cell_list.binSize( i, j, k );
320 
321  // Check the particles in this bin to see if
322  // they are neighbors. If they are add to
323  // the count for this bin.
324  int cell_count = 0;
325  neighbor_reduce( team, pid, x_p, y_p, z_p,
326  n_offset, num_n,
327  cell_count, BuildOpTag() );
328  stencil_count += cell_count;
329  }
330  }
331  Kokkos::single( Kokkos::PerThread( team ), [&]() {
332  _data.counts( pid ) = stencil_count;
333  } );
334  }
335  } );
336  }

338  // Neighbor count team vector loop (only used for CSR lists).
339  KOKKOS_INLINE_FUNCTION void
340  neighbor_reduce( const typename CountNeighborsPolicy::member_type& team,
341  const std::size_t pid, const double x_p, const double y_p,
342  const double z_p, const int n_offset, const int num_n,
343  int& cell_count, TeamVectorOpTag ) const
344  {
345  Kokkos::parallel_reduce(
346  Kokkos::ThreadVectorRange( team, num_n ),
347  [&]( const int n, int& local_count ) {
348  neighbor_kernel( pid, x_p, y_p, z_p, n_offset, n, local_count );
349  },
350  cell_count );
351  }

353  // Neighbor count serial loop (only used for CSR lists).
354  KOKKOS_INLINE_FUNCTION
355  void neighbor_reduce( const typename CountNeighborsPolicy::member_type,
356  const std::size_t pid, const double x_p,
357  const double y_p, const double z_p,
358  const int n_offset, const int num_n, int& cell_count,
359  TeamOpTag ) const
360  {
361  for ( int n = 0; n < num_n; n++ )
362  neighbor_kernel( pid, x_p, y_p, z_p, n_offset, n, cell_count );
363  }

365  // Neighbor count kernel
366  KOKKOS_INLINE_FUNCTION
367  void neighbor_kernel( const int pid, const double x_p, const double y_p,
368  const double z_p, const int n_offset, const int n,
369  int& local_count ) const
370  {
371  // Get the true id of the candidate neighbor.
372  std::size_t nid = linked_cell_list.permutation( n_offset + n );
373 
374  // Cache the candidate neighbor particle coordinates.
375  double x_n = position( nid, 0 );
376  double y_n = position( nid, 1 );
377  double z_n = position( nid, 2 );
378 
379  // If this could be a valid neighbor, continue.
380  if ( NeighborDiscriminator<AlgorithmTag>::isValid(
381  pid, x_p, y_p, z_p, nid, x_n, y_n, z_n ) )
382  {
383  // Calculate the distance between the particle and its candidate
384  // neighbor.
385  PositionValueType dx = x_p - x_n;
386  PositionValueType dy = y_p - y_n;
387  PositionValueType dz = z_p - z_n;
388  PositionValueType dist_sqr = dx * dx + dy * dy + dz * dz;
389 
390  // If within the cutoff add to the count.
391  if ( dist_sqr <= rsqr )
392  local_count += 1;
393  }
394  }

396  // Process the CSR counts by computing offsets and allocating the neighbor
397  // list.
398  template <class KokkosMemorySpace>
399  struct OffsetScanOp
400  {
401  using kokkos_mem_space = KokkosMemorySpace;
402  Kokkos::View<int*, kokkos_mem_space> counts;
403  Kokkos::View<int*, kokkos_mem_space> offsets;
404  KOKKOS_INLINE_FUNCTION
405  void operator()( const int i, int& update, const bool final_pass ) const
406  {
407  if ( final_pass )
408  offsets( i ) = update;
409  update += counts( i );
410  }
411  };

413  void initCounts( VerletLayoutCSR ) {}
414 
415  void initCounts( VerletLayout2D )
416  {
417  if ( max_n > 0 )
418  {
419  count = false;
420 
421  _data.neighbors = Kokkos::View<int**, memory_space>(
422  Kokkos::ViewAllocateWithoutInitializing( "neighbors" ),
423  _data.counts.size(), max_n );
424  }
425  }
426 
427  void processCounts( VerletLayoutCSR )
428  {
429  // Allocate offsets.
430  _data.offsets = Kokkos::View<int*, memory_space>(
431  Kokkos::ViewAllocateWithoutInitializing( "neighbor_offsets" ),
432  _data.counts.size() );
433 
434  // Calculate offsets from counts and the total number of counts.
435  OffsetScanOp<memory_space> offset_op;
436  offset_op.counts = _data.counts;
437  offset_op.offsets = _data.offsets;
438  int total_num_neighbor;
439  Kokkos::RangePolicy<execution_space> range_policy(
440  0, _data.counts.extent( 0 ) );
441  Kokkos::parallel_scan( "Cabana::VerletListBuilder::offset_scan",
442  range_policy, offset_op, total_num_neighbor );
443  Kokkos::fence();
444 
445  // Allocate the neighbor list.
446  _data.neighbors = Kokkos::View<int*, memory_space>(
447  Kokkos::ViewAllocateWithoutInitializing( "neighbors" ),
448  total_num_neighbor );
449 
450  // Reset the counts. We count again when we fill.
451  Kokkos::deep_copy( _data.counts, 0 );
452  }
453 
454  // Process 2D counts by computing the maximum number of neighbors and
455  // reallocating the 2D data structure if needed.
456  void processCounts( VerletLayout2D )
457  {
458  // Calculate the maximum number of neighbors.
459  auto counts = _data.counts;
460  int max_num_neighbor = 0;
461  Kokkos::Max<int> max_reduce( max_num_neighbor );
462  Kokkos::parallel_reduce(
463  "Cabana::VerletListBuilder::reduce_max",
464  Kokkos::RangePolicy<execution_space>( 0, _data.counts.size() ),
465  KOKKOS_LAMBDA( const int i, int& value ) {
466  if ( counts( i ) > value )
467  value = counts( i );
468  },
469  max_reduce );
470  Kokkos::fence();
471 
472  // Reallocate the neighbor list if previous size is exceeded.
473  if ( count or ( std::size_t )
474  max_num_neighbor > _data.neighbors.extent( 1 ) )
475  {
476  refill = true;
477  Kokkos::deep_copy( _data.counts, 0 );
478  _data.neighbors = Kokkos::View<int**, memory_space>(
479  Kokkos::ViewAllocateWithoutInitializing( "neighbors" ),
480  _data.counts.size(), max_num_neighbor );
481  }
482  }

572  // Neighbor fill kernel.
573  KOKKOS_INLINE_FUNCTION
574  void neighbor_kernel( const int pid, const double x_p, const double y_p,
575  const double z_p, const int n_offset,
576  const int n ) const
577  {
578  // Get the true id of the candidate neighbor.
579  std::size_t nid = linked_cell_list.permutation( n_offset + n );
580 
581  // Cache the candidate neighbor particle coordinates.
582  double x_n = position( nid, 0 );
583  double y_n = position( nid, 1 );
584  double z_n = position( nid, 2 );
585 
586  // If this could be a valid neighbor, continue.
587  if ( NeighborDiscriminator<AlgorithmTag>::isValid(
588  pid, x_p, y_p, z_p, nid, x_n, y_n, z_n ) )
589  {
590  // Calculate the distance between the particle and its candidate
591  // neighbor.
592  PositionValueType dx = x_p - x_n;
593  PositionValueType dy = y_p - y_n;
594  PositionValueType dz = z_p - z_n;
595  PositionValueType dist_sqr = dx * dx + dy * dy + dz * dz;
596 
597  // If within the cutoff increment the neighbor count and add as a
598  // neighbor at that index.
599  if ( dist_sqr <= rsqr )
600  {
601  _data.addNeighbor( pid, nid );
602  }
603  }
604  }
605 };

632  class BuildTag = TeamVectorOpTag>
633 class VerletList
634 {
635  public:
636  // Device type.
637  using device_type = DeviceType;
638 
639  // The memory space in which the neighbor list data resides.
640  using memory_space = typename device_type::memory_space;
641 
642  // Execution space.
643  using execution_space = typename device_type::execution_space;

645  // Verlet list data.
646  VerletListData<device_type, LayoutTag> _data;

687  template <class PositionSlice>
688  VerletList( PositionSlice x, const std::size_t begin, const std::size_t end,
689  const typename PositionSlice::value_type neighborhood_radius,
690  const typename PositionSlice::value_type cell_size_ratio,
691  const typename PositionSlice::value_type grid_min[3],
692  const typename PositionSlice::value_type grid_max[3],
693  const std::size_t max_neigh = 0,
694  typename std::enable_if<( is_slice<PositionSlice>::value ),
695  int>::type* = 0 )
696  {
697  // Create a builder functor.
698  using builder_type =
699  Impl::VerletListBuilder<DeviceType, PositionSlice, AlgorithmTag,
700  LayoutTag, BuildTag>;
701  builder_type builder( x, begin, end, neighborhood_radius,
702  cell_size_ratio, grid_min, grid_max, max_neigh );
703 
704  // For each particle in the range check each neighboring bin for
705  // neighbor particles. Bins are at least the size of the neighborhood
706  // radius so the bin in which the particle resides and any surrounding
707  // bins are guaranteed to contain the neighboring particles.
708  // For CSR lists, we count, then fill neighbors. For 2D lists, we
709  // count and fill at the same time, unless the array size is exceeded,
710  // at which point only counting is continued to reallocate and refill.
711  typename builder_type::FillNeighborsPolicy fill_policy(
712  builder.bin_data_1d.numBin(), Kokkos::AUTO, 4 );
713  if ( builder.count )

716  builder.bin_data_1d.numBin(), Kokkos::AUTO, 4 );
717  Kokkos::parallel_for( "Cabana::VerletList::count_neighbors",
718  count_policy, builder );
719  }
720  else
721  {
722  builder.processCounts( LayoutTag() );
723  Kokkos::parallel_for( "Cabana::VerletList::fill_neighbors",
724  fill_policy, builder );
725  }
726  Kokkos::fence();
727 
728  // Process the counts by computing offsets and allocating the neighbor
729  // list, if needed.
730  builder.processCounts( LayoutTag() );
731

733  // neighbor list.
734  if ( builder.count or builder.refill )

736  Kokkos::parallel_for( "Cabana::VerletList::fill_neighbors",
737  fill_policy, builder );
738  Kokkos::fence();
739  }
740 
741  // Get the data from the builder.
742  _data = builder._data;
743  }
