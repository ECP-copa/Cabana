28 template<class Tag>

31 // Full list specialization.
32 template<>
33 class NeighborDiscriminator<FullNeighborTag>
34 {
35  public:
36  // Full neighbor lists count and store the neighbors of all
37  // particles. The only criteria for a potentially valid neighbor is
38  // that the particle does not neighbor itself (i.e. the particle index
39  // "p" is not the same as the neighbor index "n").
40  KOKKOS_INLINE_FUNCTION
41  static bool isValid( const std::size_t p,
42  const double xp, const double yp, const double zp,
43  const std::size_t n,
44  const double xn, const double yn, const double zn )
45  {
46  return ( p != n );
47  }
48 };
49 
50 // Half list specialization.
51 template<>
52 class NeighborDiscriminator<HalfNeighborTag>
53 {
54  public:
55  // Half neighbor lists only store half of the neighbors be eliminating
56  // duplicate pairs such that the fact that particle "p" neighbors
57  // particle "n" is stored in the list but "n" neighboring "p" is not
58  // stored but rather implied. We discriminate by only storing neighbors
59  // who's coordinates are greater in the x direction. If they are the same
60  // then the y direction is checked next and finally the z direction if the
61  // y coordinates are the same.
62  KOKKOS_INLINE_FUNCTION
63  static bool isValid( const std::size_t p,
64  const double xp, const double yp, const double zp,
65  const std::size_t n,
66  const double xn, const double yn, const double zn )
67  {
68  return ( (p != n) &&
69  ( (xn>xp) ||
70  ( (xn==xp) && ( (yn>yp) || ((yn==yp) && (zn>zp) )))) );
71  }
72 };

75 // Cell stencil.
76 template<class Scalar>
77 struct LinkedCellStencil
78 {
79  Scalar rsqr;
80  CartesianGrid<double> grid;
81  int max_cells_dir;
82  int max_cells;
83  int cell_range;
84 
85  LinkedCellStencil( const Scalar neighborhood_radius,
86  const Scalar cell_size_ratio,
87  const Scalar grid_min[3],
88  const Scalar grid_max[3] )
89  : rsqr( neighborhood_radius * neighborhood_radius )
90  {
91  Scalar dx = neighborhood_radius * cell_size_ratio;
92  grid = CartesianGrid<double>( grid_min[0], grid_min[1], grid_min[2],
93  grid_max[0], grid_max[1], grid_max[2],
94  dx, dx, dx );
95  cell_range = std::ceil( 1 / cell_size_ratio );
96  max_cells_dir = 2 * cell_range + 1;
97  max_cells = max_cells_dir * max_cells_dir * max_cells_dir;
98  }
99 
100  // Given a cell, get the index bounds of the cell stencil.
101  KOKKOS_INLINE_FUNCTION
102  void getCells( const int cell,
103  int& imin,
104  int& imax,
105  int& jmin,
106  int& jmax,
107  int& kmin,
108  int& kmax ) const
109  {
110  int i, j, k;
111  grid.ijkBinIndex( cell, i, j, k );
112 
113  kmin = (k - cell_range > 0) ? k - cell_range : 0;
114  kmax = (k + cell_range + 1 < grid._nz)
115  ? k + cell_range + 1 : grid._nz;
116 
117  jmin = (j - cell_range > 0) ? j - cell_range : 0;
118  jmax = (j + cell_range + 1 < grid._ny)
119  ? j + cell_range + 1 : grid._ny;
120 
121  imin = (i - cell_range > 0) ? i - cell_range : 0;
122  imax = (i + cell_range + 1 < grid._nx)
123  ? i + cell_range + 1 : grid._nx;
124  }
125 };

128 template<class PositionSlice, class AlgorithmTag>
129 struct VerletListBuilder
130 {
131  // Types.
132  using PositionValueType = typename PositionSlice::value_type;
133  using RandomAccessPositionSlice =
134  typename PositionSlice::random_access_slice;
135  using memory_space = typename PositionSlice::memory_space;
136  using kokkos_memory_space = typename PositionSlice::kokkos_memory_space;
137  using kokkos_execution_space = typename PositionSlice::kokkos_execution_space;
138 
139  // Number of neighbors per particle.
140  Kokkos::View<int*,kokkos_memory_space> counts;
141 
142  // Offsets into the neighbor list.
143  Kokkos::View<int*,kokkos_memory_space> offsets;
144 
145  // Neighbor list.
146  Kokkos::View<int*,kokkos_memory_space> neighbors;
147 
148  // Neighbor cutoff.
149  PositionValueType rsqr;
150 
151  // Positions.
152  RandomAccessPositionSlice position;
153 
154  // Binning Data.
155  BinningData<memory_space> bin_data_1d;
156  LinkedCellList<memory_space> linked_cell_list;
157 
158  // Cell stencil.
159  LinkedCellStencil<PositionValueType> cell_stencil;
160 
161  // Constructor.
162  VerletListBuilder(
163  PositionSlice slice,
164  const std::size_t begin,
165  const std::size_t end,
166  const PositionValueType neighborhood_radius,
167  const PositionValueType cell_size_ratio,
168  const PositionValueType grid_min[3],
169  const PositionValueType grid_max[3])
170  : counts( "num_neighbors", slice.size() )
171  , offsets( "neighbor_offsets", slice.size() )
172  , cell_stencil( neighborhood_radius, cell_size_ratio, grid_min, grid_max )
173  {
174  // Get the positions with random access read-only memory.
175  position = slice;
176 
177  // Bin the particles in the grid. Don't actually sort them but make a
178  // permutation vector. Note that we are binning all particles here and
179  // not just the requested range. This is because all particles are
180  // treated as candidates for neighbors.
181  double grid_size = cell_size_ratio * neighborhood_radius;
182  PositionValueType grid_delta[3] = { grid_size, grid_size, grid_size };
183  linked_cell_list = LinkedCellList<memory_space>(
184  position, grid_delta, grid_min, grid_max );
185  bin_data_1d = linked_cell_list.binningData();
186 
187  // We will use the square of the distance for neighbor determination.
188  rsqr = neighborhood_radius * neighborhood_radius;
189  }

191  // Neighbor count team operator.
192  struct CountNeighborsTag {};
193  using CountNeighborsPolicy =
194  Kokkos::TeamPolicy<kokkos_execution_space,
195  CountNeighborsTag,
196  Kokkos::IndexType<int>,
197  Kokkos::Schedule<Kokkos::Dynamic> >;
198  KOKKOS_INLINE_FUNCTION
199  void operator()(
200  const CountNeighborsTag&,
201  const typename CountNeighborsPolicy::member_type& team ) const
202  {
203  // The league rank of the team is the cardinal cell index we are
204  // working on.
205  int cell = team.league_rank();
206 
207  // Get the stencil for this cell.
208  int imin, imax, jmin, jmax, kmin, kmax;
209  cell_stencil.getCells( cell, imin, imax, jmin, jmax, kmin, kmax );
210 
211  // Operate on the particles in the bin.
212  std::size_t b_offset = bin_data_1d.binOffset(cell);
213  Kokkos::parallel_for(
214  Kokkos::TeamThreadRange(team,0,bin_data_1d.binSize(cell)),
215  [&] ( const int bi )
216  {
217  // Get the true particle id. The binned particle index is the
218  // league rank of the team.
219  std::size_t pid = linked_cell_list.permutation( bi + b_offset );
220 
221  // Cache the particle coordinates.
222  double x_p = position(pid,0);
223  double y_p = position(pid,1);
224  double z_p = position(pid,2);
225 
226  // Loop over the cell stencil.
227  int stencil_count = 0;
228  for ( int i = imin; i < imax; ++i )
229  for ( int j = jmin; j < jmax; ++j )
230  for ( int k = kmin; k < kmax; ++k )
231  {
232  // See if we should actually check this box for
233  // neighbors.
234  if ( cell_stencil.grid.minDistanceToPoint(x_p,y_p,z_p,i,j,k)
235  <= rsqr )
236  {
237  // Check the particles in this bin to see if they are
238  // neighbors. If they are add to the count for this bin.
239  int cell_count = 0;
240  std::size_t a_offset = linked_cell_list.binOffset(i,j,k);
241  Kokkos::parallel_reduce(
242  Kokkos::ThreadVectorRange(
243  team,linked_cell_list.binSize(i,j,k)),
244  [&] ( const int n, int& local_count ) {
245 
246  // Get the true id of the candidate neighbor.
247  std::size_t nid =
248  linked_cell_list.permutation( a_offset + n );
249 
250  // Cache the candidate neighbor particle
251  // coordinates.
252  double x_n = position(nid,0);
253  double y_n = position(nid,1);
254  double z_n = position(nid,2);
255 
256  // If this could be a valid neighbor, continue.
257  if ( NeighborDiscriminator<AlgorithmTag>::isValid(
258  pid,x_p,y_p,z_p,nid,x_n,y_n,z_n) )
259  {
260  // Calculate the distance between the particle
261  // and its candidate neighbor.
262  PositionValueType dx = x_p - x_n;
263  PositionValueType dy = y_p - y_n;
264  PositionValueType dz = z_p - z_n;
265  PositionValueType dist_sqr = dx*dx + dy*dy + dz*dz;
266 
267  // If within the cutoff add to the count.
268  if ( dist_sqr <= rsqr )
269  local_count += 1;
270  }
271  },
272  cell_count );
273  stencil_count += cell_count;
274  }
275  }
276  Kokkos::single(Kokkos::PerThread(team), [&] () {
277  counts(pid) = stencil_count;
278  });
279  });
280  }

282  // Process the counts by computing offsets and allocating the neighbor
283  // list.
284  template<class KokkosMemorySpace>
285  struct OffsetScanOp
286  {
287  using kokkos_mem_space = KokkosMemorySpace;
288  Kokkos::View<int*,kokkos_mem_space> counts;
289  Kokkos::View<int*,kokkos_mem_space> offsets;
290  KOKKOS_INLINE_FUNCTION
291  void operator()( const int i, int& update, const bool final_pass ) const
292  {
293  if ( final_pass ) offsets(i) = update;
294  update += counts(i);
295  }
296  };
297  void processCounts()
298  {
299  // Calculate offsets from counts and the total number of counts.
300  OffsetScanOp<kokkos_memory_space> offset_op;
301  offset_op.counts = counts;
302  offset_op.offsets = offsets;
303  int total_num_neighbor;
304  Kokkos::RangePolicy<kokkos_execution_space> range_policy(
305  0, counts.extent(0) );
306  Kokkos::parallel_scan(
307  "Cabana::VerletListBuilder::offset_scan",
308  range_policy, offset_op, total_num_neighbor );
309  Kokkos::fence();
310 
311  // Allocate the neighbor list.
312  neighbors = Kokkos::View<int*,kokkos_memory_space>(
313  "neighbors", total_num_neighbor );
314 
315  // Reset the counts. We count again when we fill.
316  Kokkos::deep_copy( counts, 0 );
317  }

366  std::size_t a_offset = linked_cell_list.binOffset(i,j,k);
367  Kokkos::parallel_for(
368  Kokkos::ThreadVectorRange(
369  team,linked_cell_list.binSize(i,j,k)),
370  [&] ( const int n ) {
371 
372  // Get the true id of the candidate neighbor.
373  std::size_t nid =
374  linked_cell_list.permutation( a_offset + n );
375 
376  // Cache the candidate neighbor particle coordinates.
377  double x_n = position(nid,0);
378  double y_n = position(nid,1);
379  double z_n = position(nid,2);
380 
381  // If this could be a valid neighbor, continue.
382  if ( NeighborDiscriminator<AlgorithmTag>::isValid(
383  pid,x_p,y_p,z_p,nid,x_n,y_n,z_n) )
384  {
385  // Calculate the distance between the particle
386  // and its candidate neighbor.
387  PositionValueType dx = x_p - x_n;
388  PositionValueType dy = y_p - y_n;
389  PositionValueType dz = z_p - z_n;
390  PositionValueType dist_sqr = dx*dx + dy*dy + dz*dz;
391 
392  // If within the cutoff increment the neighbor
393  // count and add as a neighbor at that index.
394  if ( dist_sqr <= rsqr )
395  {
396  neighbors(
397  offsets(pid) +
398  Kokkos::atomic_fetch_add(&counts(pid),1) )
399  = nid;
400  }
401  }
402  });
403  }
404  }
405  });

422 template<class MemorySpace, class AlgorithmTag>
423 class VerletList
424 {
425  public:
426 
427  // The memory space in which the neighbor list data resides.
428  using memory_space = MemorySpace;
429  using kokkos_memory_space = typename memory_space::kokkos_memory_space;
430 
431  // Number of neighbors per particle.
432  Kokkos::View<int*,kokkos_memory_space> _counts;
433 
434  // Offsets into the neighbor list.
435  Kokkos::View<int*,kokkos_memory_space> _offsets;
436 
437  // Neighbor list.
438  Kokkos::View<int*,kokkos_memory_space> _neighbors;
439 
470  template<class PositionSlice>
471  VerletList(
472  PositionSlice x,
473  const std::size_t begin,
474  const std::size_t end,
475  const typename PositionSlice::value_type neighborhood_radius,
476  const typename PositionSlice::value_type cell_size_ratio,
477  const typename PositionSlice::value_type grid_min[3],
478  const typename PositionSlice::value_type grid_max[3],
479  typename std::enable_if<(is_slice<PositionSlice>::value),int>::type * = 0 )
480  {
481  // Create a builder functor.
482  using builder_type =
483  Impl::VerletListBuilder<PositionSlice,AlgorithmTag>;
484  builder_type builder( x, begin, end,
485  neighborhood_radius, cell_size_ratio,
486  grid_min, grid_max );
487 
488  // For each particle in the range check each neighboring bin for
489  // neighbor particles. Bins are at least the size of the neighborhood
490  // radius so the bin in which the particle resides and any surrounding
491  // bins are guaranteed to contain the neighboring particles.
492  typename builder_type::CountNeighborsPolicy
493  count_policy( builder.bin_data_1d.numBin(), Kokkos::AUTO, 4 );
494  Kokkos::parallel_for(
495  "Cabana::VerletList::count_neighbors",
496  count_policy, builder );
497  Kokkos::fence();
498 
499  // Process the counts by computing offsets and allocating the neighbor
500  // list.
501  builder.processCounts();
502 
503  // For each particle in the range fill its part of the neighbor list.
504  typename builder_type::FillNeighborsPolicy
505  fill_policy( builder.bin_data_1d.numBin(), Kokkos::AUTO, 4 );
506  Kokkos::parallel_for(
507  "Cabana::VerletList::fill_neighbors",
508  fill_policy, builder );
509  Kokkos::fence();
510 
511  // Get the data from the builder.
512  _counts = builder.counts;
513  _offsets = builder.offsets;
514  _neighbors = builder.neighbors;
515  }
516 };
