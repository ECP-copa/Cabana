275  template <typename Scalar>
276  KOKKOS_INLINE_FUNCTION static void value( const Scalar x0,
277  Scalar values[num_knot] )
278  {
279  // Constants
280  Scalar nine_eights = 9.0 / 8.0;
281 
282  // Knot at i - 1
283  Scalar xn = x0 - static_cast<int>( x0 ) + 0.5;
284  values[0] = 0.5 * xn * xn - 1.5 * xn + nine_eights;
285 
286  // Knot at i
287  xn -= 1.0;
288  values[1] = -xn * xn + 0.75;
289 
290  // Knot at i + 1
291  xn -= 1.0;
292  values[2] = 0.5 * xn * xn + 1.5 * xn + nine_eights;
293  }
294 
304  template <typename Scalar>
305  KOKKOS_INLINE_FUNCTION static void
306  gradient( const Scalar x0, const Scalar rdx, Scalar gradients[num_knot] )
307  {
308  // Knot at i - 1
309  Scalar xn = x0 - static_cast<int>( x0 ) + 0.5;
310  gradients[0] = ( xn - 1.5 ) * rdx;
311 
312  // Knot at i
313  xn -= 1.0;
314  gradients[1] = ( -2.0 * xn ) * rdx;
315 
316  // Knot at i + 1
317  xn -= 1.0;
318  gradients[2] = ( xn + 1.5 ) * rdx;
319  }

371  template <class Scalar>
372  KOKKOS_INLINE_FUNCTION static void stencil( const Scalar x0,
373  int indices[num_knot] )
374  {
375  indices[0] = static_cast<int>( x0 ) - 1;
376  indices[1] = indices[0] + 1;
377  indices[2] = indices[1] + 1;
378  indices[3] = indices[2] + 1;
379  }

388  template <typename Scalar>
389  KOKKOS_INLINE_FUNCTION static void value( const Scalar x0,
390  Scalar values[num_knot] )
391  {
392  // Constants
393  Scalar one_sixth = 1.0 / 6.0;
394  Scalar two_thirds = one_sixth * 4.0;
395  Scalar four_thirds = two_thirds * 2.0;
396 
397  // Knot at i - 1
398  Scalar xn = x0 - static_cast<int>( x0 ) + 1.0;
399  Scalar xn2 = xn * xn;
400  values[0] = -xn * xn2 * one_sixth + xn2 - 2.0 * xn + four_thirds;
401 
402  // Knot at i
403  xn -= 1.0;
404  xn2 = xn * xn;
405  values[1] = 0.5 * xn * xn2 - xn2 + two_thirds;
406 
407  // Knot at i + 1
408  xn -= 1.0;
409  xn2 = xn * xn;
410  values[2] = -0.5 * xn * xn2 - xn2 + two_thirds;
411 
412  // Knot at i + 2
413  xn -= 1.0;
414  xn2 = xn * xn;
415  values[3] = xn * xn2 * one_sixth + xn2 + 2.0 * xn + four_thirds;
416  }
417 
427  template <typename Scalar>
428  KOKKOS_INLINE_FUNCTION static void
429  gradient( const Scalar x0, const Scalar rdx, Scalar gradients[num_knot] )
430  {
431  // Knot at i - 1
432  Scalar xn = x0 - static_cast<int>( x0 ) + 1.0;
433  gradients[0] = ( -0.5 * xn * xn + 2.0 * xn - 2.0 ) * rdx;
434 
435  // Knot at i
436  xn -= 1.0;
437  gradients[1] = ( 1.5 * xn * xn - 2.0 * xn ) * rdx;
438 
439  // Knot at i + 1
440  xn -= 1.0;
441  gradients[2] = ( -1.5 * xn * xn - 2.0 * xn ) * rdx;
442 
443  // Knot at i + 2
444  xn -= 1.0;
445  gradients[3] = ( 0.5 * xn * xn + 2.0 * xn + 2.0 ) * rdx;
446  }

750 //---------------------------------------------------------------------------//
751 // Evaluate spline data at a point in a uniform mesh.
752 template <typename Scalar, int Order, std::size_t NumSpaceDim, class Device,
753  class EntityType, class DataTags>
754 KOKKOS_INLINE_FUNCTION void evaluateSpline(
755  const LocalMesh<Device, UniformMesh<Scalar, NumSpaceDim>>& local_mesh,
756  const Scalar p[NumSpaceDim],
757  SplineData<Scalar, Order, NumSpaceDim, EntityType, DataTags>& data )
758 {
759  // data type
760  using sd_type =
761  SplineData<Scalar, Order, NumSpaceDim, EntityType, DataTags>;
762 
763  // spline type.
764 
765  // Get the low corner of the mesh.
766  Scalar low_x[NumSpaceDim];
767  Scalar low_x_p1[NumSpaceDim];
768  int low_id[NumSpaceDim];
769  int low_id_p1[NumSpaceDim];
770  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
771  {
772  low_id[d] = 0;
773  low_id_p1[d] = 1;
774  }
775  local_mesh.coordinates( EntityType(), low_id, low_x );
776  local_mesh.coordinates( EntityType(), low_id_p1, low_x_p1 );
777 
778  // Compute the physical cell size.
779  Scalar dx[NumSpaceDim];
780  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
781  {
782  dx[d] = low_x_p1[d] - low_x[d];
783  setSplineData( SplinePhysicalCellSize(), data, d, dx[d] );
784  }
785 
786  // Compute the inverse physicall cell size.
787  Scalar rdx[NumSpaceDim];
788  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
789  rdx[d] = 1.0 / dx[d];
790 
791  // Compute the reference coordinates.
792  Scalar x[NumSpaceDim];
