278 template <typename Scalar>
279 KOKKOS_INLINE_FUNCTION static void value( const Scalar x0,
280 Scalar values[num_knot] )
281 {
282 // Constants
283 Scalar nine_eights = 9.0 / 8.0;
284
285 // Knot at i - 1
286 Scalar xn = x0 - static_cast<int>( x0 ) + 0.5;
287 values[0] = 0.5 * xn * xn - 1.5 * xn + nine_eights;
288
289 // Knot at i
290 xn -= 1.0;
291 values[1] = -xn * xn + 0.75;
292
293 // Knot at i + 1
294 xn -= 1.0;
295 values[2] = 0.5 * xn * xn + 1.5 * xn + nine_eights;
296 }

310 {
311 // Knot at i - 1
312 Scalar xn = x0 - static_cast<int>( x0 ) + 0.5;
313 gradients[0] = ( xn - 1.5 ) * rdx;
314
315 // Knot at i
316 xn -= 1.0;

391 template <typename Scalar>
392 KOKKOS_INLINE_FUNCTION static void value( const Scalar x0,
393 Scalar values[num_knot] )
394 {
395 // Constants
396 Scalar one_sixth = 1.0 / 6.0;
397 Scalar two_thirds = one_sixth * 4.0;
398 Scalar four_thirds = two_thirds * 2.0;
399
400 // Knot at i - 1
401 Scalar xn = x0 - static_cast<int>( x0 ) + 1.0;
402 Scalar xn2 = xn * xn;
403 values[0] = -xn * xn2 * one_sixth + xn2 - 2.0 * xn + four_thirds;
404
405 // Knot at i
406 xn -= 1.0;
407 xn2 = xn * xn;
408 values[1] = 0.5 * xn * xn2 - xn2 + two_thirds;
409
410 // Knot at i + 1
411 xn -= 1.0;
412 xn2 = xn * xn;
413 values[2] = -0.5 * xn * xn2 - xn2 + two_thirds;
414
415 // Knot at i + 2
416 xn -= 1.0;
417 xn2 = xn * xn;
418 values[3] = xn * xn2 * one_sixth + xn2 + 2.0 * xn + four_thirds;
419 }

430 template <typename Scalar>
431 KOKKOS_INLINE_FUNCTION static void
432 gradient( const Scalar x0, const Scalar rdx, Scalar gradients[num_knot] )
433 {
434 // Knot at i - 1
435 Scalar xn = x0 - static_cast<int>( x0 ) + 1.0;
436 gradients[0] = ( -0.5 * xn * xn + 2.0 * xn - 2.0 ) * rdx;
437
438 // Knot at i
439 xn -= 1.0;
440 gradients[1] = ( 1.5 * xn * xn - 2.0 * xn ) * rdx;
441
442 // Knot at i + 1
443 xn -= 1.0;
444 gradients[2] = ( -1.5 * xn * xn - 2.0 * xn ) * rdx;
445
446 // Knot at i + 2
447 xn -= 1.0;
448 gradients[3] = ( 0.5 * xn * xn + 2.0 * xn + 2.0 ) * rdx;
449 }

794template <typename Scalar, int Order, std::size_t NumSpaceDim, class Device,
795 class EntityType, class DataTags>
796KOKKOS_INLINE_FUNCTION void evaluateSpline(
797 const LocalMesh<Device, UniformMesh<Scalar, NumSpaceDim>>& local_mesh,
798 const Scalar p[NumSpaceDim],
799 SplineData<Scalar, Order, NumSpaceDim, EntityType, DataTags>& data )
800{
801 // data type
802 using sd_type =
803 SplineData<Scalar, Order, NumSpaceDim, EntityType, DataTags>;
804
805 // spline type.
806
807 // Get the low corner of the mesh.
808 Scalar low_x[NumSpaceDim];
809 Scalar low_x_p1[NumSpaceDim];
810 int low_id[NumSpaceDim];
811 int low_id_p1[NumSpaceDim];
812 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
813 {
814 low_id[d] = 0;
815 low_id_p1[d] = 1;
816 }
817 local_mesh.coordinates( EntityType(), low_id, low_x );
818 local_mesh.coordinates( EntityType(), low_id_p1, low_x_p1 );
819
820 // Compute the physical cell size.
821 Scalar dx[NumSpaceDim];
822 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
823 {
824 dx[d] = low_x_p1[d] - low_x[d];
825 setSplineData( SplinePhysicalCellSize(), data, d, dx[d] );
826 }
827
828 // Compute the inverse physicall cell size.
829 Scalar rdx[NumSpaceDim];
830 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
831 rdx[d] = 1.0 / dx[d];
832
833 // Compute the reference coordinates.
834 Scalar x[NumSpaceDim];
835 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
836 {
837 x[d] = sd_type::spline_type::mapToLogicalGrid( p[d], rdx[d], low_x[d] );
838 setSplineData( SplineLogicalPosition(), data, d, x[d] );
839 }
840
841 // Compute the stencil.
842 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
843 {
844 sd_type::spline_type::stencil( x[d], data.s[d] );
845 }
846
847 // Compute the weight values.
848 setSplineData( SplineWeightValues(), data, x );
849
850 // Compute the weight gradients.
851 setSplineData( SplineWeightPhysicalGradients(), data, x, rdx );
852
853 // Compute the physical distance.
854 setSplineData( SplinePhysicalDistance(), data, low_x, p, dx );
855}
