794 template <typename Scalar, int Order, std::size_t NumSpaceDim, class Device,
795  class EntityType, class DataTags>
796 KOKKOS_INLINE_FUNCTION void evaluateSpline(
797  const LocalMesh<Device, UniformMesh<Scalar, NumSpaceDim>>& local_mesh,
798  const Scalar p[NumSpaceDim],
799  SplineData<Scalar, Order, NumSpaceDim, EntityType, DataTags>& data )
800 {
801  // data type
802  using sd_type =
803  SplineData<Scalar, Order, NumSpaceDim, EntityType, DataTags>;
804 
805  // spline type.
806 
807  // Get the low corner of the mesh.
808  Scalar low_x[NumSpaceDim];
809  Scalar low_x_p1[NumSpaceDim];
810  int low_id[NumSpaceDim];
811  int low_id_p1[NumSpaceDim];
812  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
813  {
814  low_id[d] = 0;
815  low_id_p1[d] = 1;
816  }
817  local_mesh.coordinates( EntityType(), low_id, low_x );
818  local_mesh.coordinates( EntityType(), low_id_p1, low_x_p1 );
819 
820  // Compute the physical cell size.
821  Scalar dx[NumSpaceDim];
822  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
823  {
824  dx[d] = low_x_p1[d] - low_x[d];
825  setSplineData( SplinePhysicalCellSize(), data, d, dx[d] );
826  }
827 
828  // Compute the inverse physicall cell size.
829  Scalar rdx[NumSpaceDim];
830  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
831  rdx[d] = 1.0 / dx[d];
832 
833  // Compute the reference coordinates.
834  Scalar x[NumSpaceDim];
835  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
836  {
837  x[d] = sd_type::spline_type::mapToLogicalGrid( p[d], rdx[d], low_x[d] );
838  setSplineData( SplineLogicalPosition(), data, d, x[d] );
839  }
840 
841  // Compute the stencil.
842  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
843  {
844  sd_type::spline_type::stencil( x[d], data.s[d] );
845  }
846 
847  // Compute the weight values.
848  setSplineData( SplineWeightValues(), data, x );
849 
850  // Compute the weight gradients.
851  setSplineData( SplineWeightPhysicalGradients(), data, x, rdx );
852 
853  // Compute the physical distance.
854  setSplineData( SplinePhysicalDistance(), data, low_x, p, dx );
855 }
