74 struct ViewOffset<Dimension, Kokkos::LayoutCabanaSlice<LayoutDims...>, void>

76  public:

78  using is_regular = std::true_type;
79

81  typedef Dimension dimension_type;

84  static constexpr int Stride = array_layout::Stride;
85  static constexpr int VectorLength = array_layout::VectorLength;
86  static constexpr int D0 = array_layout::D0;
87  static constexpr int D1 = array_layout::D1;
88  static constexpr int D2 = array_layout::D2;
89  static constexpr int D3 = array_layout::D3;
90  static constexpr int D4 = array_layout::D4;
91  static constexpr int D5 = array_layout::D5;

95  //----------------------------------------
96 
97  // rank 1
98  template <typename S>
99  KOKKOS_INLINE_FUNCTION constexpr size_type operator()( S const& s ) const
100  {
101  return Stride * s;
102  }
103 
104  // rank 2
105  template <typename S, typename A>
106  KOKKOS_INLINE_FUNCTION constexpr size_type operator()( S const& s,
107  A const& a ) const
108  {
109  return Stride * s + a;
110  }
111 
112  // rank 3
113  template <typename S, typename A, typename I0>
114  KOKKOS_INLINE_FUNCTION constexpr size_type
115  operator()( S const& s, A const& a, I0 const& i0 ) const
116  {
117  return Stride * s + a + VectorLength * i0;
118  }
119 
120  // rank 4
121  template <typename S, typename A, typename I0, typename I1>
122  KOKKOS_INLINE_FUNCTION constexpr size_type
123  operator()( S const& s, A const& a, I0 const& i0, I1 const& i1 ) const
124  {
125  return Stride * s + a + VectorLength * ( i1 + D1 * i0 );
126  }
127 
128  // rank 5
129  template <typename S, typename A, typename I0, typename I1, typename I2>
130  KOKKOS_INLINE_FUNCTION constexpr size_type
131  operator()( S const& s, A const& a, I0 const& i0, I1 const& i1,
132  I2 const& i2 ) const
133  {
134  return Stride * s + a + VectorLength * ( i2 + D2 * ( i1 + D1 * i0 ) );
135  }
136 
137  // rank 6
138  template <typename S, typename A, typename I0, typename I1, typename I2,
139  typename I3>
140  KOKKOS_INLINE_FUNCTION constexpr size_type
141  operator()( S const& s, A const& a, I0 const& i0, I1 const& i1,
142  I2 const& i2, I3 const& i3 ) const
143  {
144  return Stride * s + a +
145  VectorLength * ( i3 + D3 * i2 + D2 * ( i1 + D1 * i0 ) );
146  }
147 
148  // rank 7
149  template <typename S, typename A, typename I0, typename I1, typename I2,
150  typename I3, typename I4>
151  KOKKOS_INLINE_FUNCTION constexpr size_type
152  operator()( S const& s, A const& a, I0 const& i0, I1 const& i1,
153  I2 const& i2, I3 const& i3, I4 const& i4 ) const
154  {
155  return Stride * s + a +
156  VectorLength *
157  ( i4 + D4 * ( i3 + D3 * i2 + D2 * ( i1 + D1 * i0 ) ) );
158  }
159 
160  // rank 8
161  template <typename S, typename A, typename I0, typename I1, typename I2,
162  typename I3, typename I4, typename I5>
163  KOKKOS_INLINE_FUNCTION constexpr size_type
164  operator()( S const& s, A const& a, I0 const& i0, I1 const& i1,
165  I2 const& i2, I3 const& i3, I4 const& i4, I5 const& i5 ) const
166  {
167  return Stride * s + a +
168  VectorLength *
169  ( i5 + D5 * ( i4 + D4 * ( i3 + D3 * i2 +
170  D2 * ( i1 + D1 * i0 ) ) ) );
171  }

211  /* Cardinality of the domain index space */
212  KOKKOS_INLINE_FUNCTION
213  constexpr size_type size() const
214  {
215  return m_dim.N0 * m_dim.N1 * m_dim.N2 * m_dim.N3 * m_dim.N4 * m_dim.N5 *
216  m_dim.N6 * m_dim.N7;
217  }
218 
219  /* Span of the range space, largest stride * dimension */
220  KOKKOS_INLINE_FUNCTION
221  constexpr size_type span() const { return m_dim.N0 * Stride; }
222 
223  KOKKOS_INLINE_FUNCTION constexpr bool span_is_contiguous() const
224  {
225  return span() == size();
226  }
227 
228  /* Strides of dimensions */
229  KOKKOS_INLINE_FUNCTION constexpr size_type stride_0() const
230  {
231  return Stride;
232  }
233 
234  KOKKOS_INLINE_FUNCTION constexpr size_type stride_1() const { return 1; }
235 
236  KOKKOS_INLINE_FUNCTION constexpr size_type stride_2() const
237  {
238  return m_dim.N7 * m_dim.N6 * m_dim.N5 * m_dim.N4 * m_dim.N3 *
239  VectorLength;
240  }
241 
242  KOKKOS_INLINE_FUNCTION constexpr size_type stride_3() const
243  {
244  return m_dim.N7 * m_dim.N6 * m_dim.N5 * m_dim.N4 * VectorLength;
245  }
246 
247  KOKKOS_INLINE_FUNCTION constexpr size_type stride_4() const
248  {
249  return m_dim.N7 * m_dim.N6 * m_dim.N5 * VectorLength;
250  }
251 
252  KOKKOS_INLINE_FUNCTION constexpr size_type stride_5() const
253  {
254  return m_dim.N7 * m_dim.N6 * VectorLength;
255  }
256 
257  KOKKOS_INLINE_FUNCTION constexpr size_type stride_6() const
258  {
259  return m_dim.N7 * VectorLength;
260  }
261 
262  KOKKOS_INLINE_FUNCTION constexpr size_type stride_7() const
263  {
264  return VectorLength;
265  }
266 
267  // Stride with [ rank ] value is the total length
268  template <typename iType>
269  KOKKOS_INLINE_FUNCTION void stride( iType* const s ) const
270  {
271  if ( 0 < dimension_type::rank )
272  {
273  s[0] = stride_0();
274  }
275  if ( 1 < dimension_type::rank )
276  {
277  s[1] = stride_1();
278  }
279  if ( 2 < dimension_type::rank )
280  {
281  s[2] = stride_2();
282  }
283  if ( 3 < dimension_type::rank )
284  {
285  s[3] = stride_3();
286  }
287  if ( 4 < dimension_type::rank )
288  {
289  s[4] = stride_4();
290  }
291  if ( 5 < dimension_type::rank )
292  {
293  s[5] = stride_5();
294  }
295  if ( 6 < dimension_type::rank )
296  {
297  s[6] = stride_6();
298  }
299  if ( 7 < dimension_type::rank )
300  {
301  s[7] = stride_7();
302  }
303  s[dimension_type::rank] = span();
304  }

358 // Given a tuple member type T of the given rank get the Kokkos view
359 // data layout parameters. The tuple index effectively introduces 2 new
360 // dimensions to the problem on top of the member dimensions - one for the
361 // struct index and one for the vector index.
362 template <typename T, std::size_t Rank, int VectorLength, int Stride>
363 struct KokkosDataTypeImpl;
364 
365 // Rank-0
366 template <typename T, int VectorLength, int Stride>
367 struct KokkosDataTypeImpl<T, 0, VectorLength, Stride>
368 {
369  using value_type = typename std::remove_all_extents<T>::type;
370  using data_type = value_type* [VectorLength];
371  using cabana_layout = Kokkos::LayoutCabanaSlice<Stride, VectorLength>;
372 
373  inline static cabana_layout createLayout( const std::size_t num_soa )
374  {
375  return cabana_layout( num_soa );
376  }
377 };
378 
379 // Rank-1
380 template <typename T, int VectorLength, int Stride>
381 struct KokkosDataTypeImpl<T, 1, VectorLength, Stride>
382 {
383  using value_type = typename std::remove_all_extents<T>::type;
384  static constexpr std::size_t D0 = std::extent<T, 0>::value;
385  using data_type = value_type* [VectorLength][D0];
386  using cabana_layout = Kokkos::LayoutCabanaSlice<Stride, VectorLength, D0>;
387 
388  inline static cabana_layout createLayout( const std::size_t num_soa )
389  {
390  return cabana_layout( num_soa );
391  }
392 };
393 
394 // Rank-2
395 template <typename T, int VectorLength, int Stride>
396 struct KokkosDataTypeImpl<T, 2, VectorLength, Stride>
397 {
398  using value_type = typename std::remove_all_extents<T>::type;
399  static constexpr std::size_t D0 = std::extent<T, 0>::value;
400  static constexpr std::size_t D1 = std::extent<T, 1>::value;
401  using data_type = value_type* [VectorLength][D0][D1];
402  using cabana_layout =
403  Kokkos::LayoutCabanaSlice<Stride, VectorLength, D0, D1>;
404 
405  inline static cabana_layout createLayout( const std::size_t num_soa )
406  {
407  return cabana_layout( num_soa );
408  }
409 };

475 template <typename DataType, typename DeviceType, typename MemoryAccessType,
476  int VectorLength, int Stride>
477 class Slice
478 {
479  public:
480  // Ensure the vector length is valid.
481  static_assert( Impl::IsVectorLengthValid<VectorLength>::value,
482  "Vector length must be valid" );
483 
484  // Slice type.
485  using slice_type =
486  Slice<DataType, DeviceType, MemoryAccessType, VectorLength, Stride>;
487 
488  // Device type
489  using device_type = DeviceType;
490 
491  // Memory space.
492  using memory_space = typename device_type::memory_space;
493 
494  // Execution space.
495  using execution_space = typename device_type::execution_space;
496 
497  // Memory access type.
498  static_assert( is_memory_access_tag<MemoryAccessType>::value,
499  "Slice memory access type must be a Cabana access type" );
500  using memory_access_type = MemoryAccessType;
501 
502  // Vector length.
503  static constexpr int vector_length = VectorLength;
504 
505  // SoA stride.
506  static constexpr int soa_stride = Stride;
507 
508  // Size type.
509  using size_type = typename memory_space::size_type;
510 
511  // Index type.
512  using index_type = Impl::Index<vector_length>;
513 
514  // Maximum supported rank.
515  static constexpr std::size_t max_supported_rank = 3;
516 
517  // Maximum label length.
518  static constexpr std::size_t max_label_length = 128;
519 
520  // Kokkos view wrapper.
521  using view_wrapper =
522  Impl::KokkosViewWrapper<DataType, vector_length, soa_stride>;
523 
524  // Kokkos view type.
525  using kokkos_view =
526  Kokkos::View<typename view_wrapper::data_type,
527  typename view_wrapper::cabana_layout, DeviceType,
528  typename MemoryAccessType::kokkos_memory_traits>;
529 
530  // View type aliases.
531  using reference_type = typename kokkos_view::reference_type;
532  using value_type = typename kokkos_view::value_type;
533  using pointer_type = typename kokkos_view::pointer_type;
534  using view_layout = typename kokkos_view::array_layout;
535 
536  // Compatible memory access slice types.
537  using default_access_slice =
538  Slice<DataType, DeviceType, DefaultAccessMemory, VectorLength, Stride>;
539  using atomic_access_slice =
540  Slice<DataType, DeviceType, AtomicAccessMemory, VectorLength, Stride>;
541  using random_access_slice =
542  Slice<DataType, DeviceType, RandomAccessMemory, VectorLength, Stride>;
543 
544  // Declare slices of different memory access types to be friends.
545  friend class Slice<DataType, DeviceType, DefaultAccessMemory, VectorLength,
546  Stride>;
547  friend class Slice<DataType, DeviceType, AtomicAccessMemory, VectorLength,
548  Stride>;
549  friend class Slice<DataType, DeviceType, RandomAccessMemory, VectorLength,
550  Stride>;
551 
552  // Equivalent Kokkos view rank. This rank assumes that the struct and
553  // array dimension are merged. For the true rank of the raw AoSoA data use
554  // the rank() function below which will account for the extra dimension
555  // from separate struct and array indices. This enumeration is for
556  // compatibility with Kokkos views.
557  enum
558  {
559  Rank = std::rank<DataType>::value + 1
560  };
561 
562  public:
566  Slice()
567  : _size( 0 )
568  {
569  }
570 
578  Slice( const pointer_type data, const size_type size,
579  const size_type num_soa, const std::string& label = "" )
580  : _view( data, view_wrapper::createLayout( num_soa ) )
581  , _size( size )
582  {
583  std::strcpy( _label, label.c_str() );
584  }
585 
593  template <class MAT>
594  Slice( const Slice<DataType, DeviceType, MAT, VectorLength, Stride>& rhs )
595  : _view( rhs._view )
596  , _size( rhs._size )
597  {
598  std::strcpy( _label, rhs._label );
599  }
600 
610  template <class MAT>
611  Slice& operator=(
612  const Slice<DataType, DeviceType, MAT, VectorLength, Stride>& rhs )
613  {
614  _view = rhs._view;
615  _size = rhs._size;
616  std::strcpy( _label, rhs._label );
617  return *this;
618  }
619 
625  std::string label() const { return std::string( _label ); }
626 
631  KOKKOS_INLINE_FUNCTION
632  size_type size() const { return _size; }
633 
638  KOKKOS_INLINE_FUNCTION
639  size_type numSoA() const { return _view.extent( 0 ); }
640 
646  KOKKOS_INLINE_FUNCTION
647  size_type arraySize( const size_type s ) const
648  {
649  return ( static_cast<size_type>( s ) < _view.extent( 0 ) - 1 )
650  ? vector_length
651  : ( _size % vector_length );
652  }

660  typename std::enable_if<( 0 == std::rank<U>::value &&

696  reference_type>::type
697  access( const size_type s, const size_type a, const size_type d0,
698  const size_type d1, const size_type d2 ) const
699  {
700  return _view( s, a, d0, d1, d2 );
701  }

705

709  typename std::enable_if<( 0 == std::rank<U>::value &&
710  std::is_same<U, DataType>::value ),
711  reference_type>::type
712  operator()( const size_type i ) const
713  {
714  return access( index_type::s( i ), index_type::a( i ) );
715  }

717  // Rank 1
718  template <typename U = DataType>
719  KOKKOS_FORCEINLINE_FUNCTION
720  typename std::enable_if<( 1 == std::rank<U>::value &&
