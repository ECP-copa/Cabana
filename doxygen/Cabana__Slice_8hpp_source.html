234  // Stride with [ rank ] value is the total length
235  template< typename iType >
236  KOKKOS_INLINE_FUNCTION
237  void stride( iType * const s ) const
238  {
239  if ( 0 < dimension_type::rank ) { s[0] = stride_0(); }
240  if ( 1 < dimension_type::rank ) { s[1] = stride_1(); }
241  if ( 2 < dimension_type::rank ) { s[2] = stride_2(); }
242  if ( 3 < dimension_type::rank ) { s[3] = stride_3(); }
243  if ( 4 < dimension_type::rank ) { s[4] = stride_4(); }
244  if ( 5 < dimension_type::rank ) { s[5] = stride_5(); }
245  if ( 6 < dimension_type::rank ) { s[6] = stride_6(); }
246  if ( 7 < dimension_type::rank ) { s[7] = stride_7(); }
247  s[dimension_type::rank] = span();
248  }

306 // Given a tuple member type T of the given rank get the Kokkos view
307 // data layout parameters. The tuple index effectively introduces 2 new
308 // dimensions to the problem on top of the member dimensions - one for the
309 // struct index and one for the vector index.
310 template<typename T, std::size_t Rank, int VectorLength, int Stride>
311 struct KokkosDataTypeImpl;

313 // Rank-0
314 template<typename T, int VectorLength, int Stride>
315 struct KokkosDataTypeImpl<T,0,VectorLength,Stride>
316 {
317  using value_type = typename std::remove_all_extents<T>::type;
318  using data_type = value_type*[VectorLength];
319  using cabana_layout = Kokkos::LayoutCabanaSlice<Stride,VectorLength>;
320 
321  inline static cabana_layout createLayout( const std::size_t num_soa )
322  {
323  return cabana_layout( num_soa );
324  }
325 };
