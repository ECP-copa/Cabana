116 //----------------------------------------
117
118 // rank 1
119 template <typename S>
120 KOKKOS_INLINE_FUNCTION constexpr size_type operator()( S const& s ) const
121 {
122 return Stride * s;
123 }
124
125 // rank 2
126 template <typename S, typename A>
127 KOKKOS_INLINE_FUNCTION constexpr size_type operator()( S const& s,
128 A const& a ) const
129 {
130 return Stride * s + a;
131 }
132
133 // rank 3
134 template <typename S, typename A, typename I0>
135 KOKKOS_INLINE_FUNCTION constexpr size_type
136 operator()( S const& s, A const& a, I0 const& i0 ) const
137 {
138 return Stride * s + a + VectorLength * i0;
139 }
140
141 // rank 4
142 template <typename S, typename A, typename I0, typename I1>
143 KOKKOS_INLINE_FUNCTION constexpr size_type
144 operator()( S const& s, A const& a, I0 const& i0, I1 const& i1 ) const
145 {
146 return Stride * s + a + VectorLength * ( i1 + D1 * i0 );
147 }
148
149 // rank 5
150 template <typename S, typename A, typename I0, typename I1, typename I2>
151 KOKKOS_INLINE_FUNCTION constexpr size_type
152 operator()( S const& s, A const& a, I0 const& i0, I1 const& i1,
153 I2 const& i2 ) const
154 {
155 return Stride * s + a + VectorLength * ( i2 + D2 * ( i1 + D1 * i0 ) );
156 }
157
158 // rank 6
159 template <typename S, typename A, typename I0, typename I1, typename I2,
160 typename I3>
161 KOKKOS_INLINE_FUNCTION constexpr size_type
162 operator()( S const& s, A const& a, I0 const& i0, I1 const& i1,
163 I2 const& i2, I3 const& i3 ) const
164 {
165 return Stride * s + a +
166 VectorLength * ( i3 + D3 * i2 + D2 * ( i1 + D1 * i0 ) );
167 }
168
169 // rank 7
170 template <typename S, typename A, typename I0, typename I1, typename I2,
171 typename I3, typename I4>
172 KOKKOS_INLINE_FUNCTION constexpr size_type
173 operator()( S const& s, A const& a, I0 const& i0, I1 const& i1,
174 I2 const& i2, I3 const& i3, I4 const& i4 ) const
175 {
176 return Stride * s + a +
177 VectorLength *
178 ( i4 + D4 * ( i3 + D3 * i2 + D2 * ( i1 + D1 * i0 ) ) );
179 }
180
181 // rank 8
182 template <typename S, typename A, typename I0, typename I1, typename I2,
183 typename I3, typename I4, typename I5>
184 KOKKOS_INLINE_FUNCTION constexpr size_type
185 operator()( S const& s, A const& a, I0 const& i0, I1 const& i1,
186 I2 const& i2, I3 const& i3, I4 const& i4, I5 const& i5 ) const
187 {
188 return Stride * s + a +
189 VectorLength *
190 ( i5 + D5 * ( i4 + D4 * ( i3 + D3 * i2 +
191 D2 * ( i1 + D1 * i0 ) ) ) );
192 }

288 // Stride with [ rank ] value is the total length
289 template <typename iType>
290 KOKKOS_INLINE_FUNCTION void stride( iType* const s ) const
291 {
292 if ( 0 < dimension_type::rank )
293 {
294 s[0] = stride_0();
295 }
296 if ( 1 < dimension_type::rank )
297 {
298 s[1] = stride_1();
299 }
300 if ( 2 < dimension_type::rank )
301 {
302 s[2] = stride_2();
303 }
304 if ( 3 < dimension_type::rank )
305 {
306 s[3] = stride_3();
307 }
308 if ( 4 < dimension_type::rank )
309 {
310 s[4] = stride_4();
311 }
312 if ( 5 < dimension_type::rank )
313 {
314 s[5] = stride_5();
315 }
316 if ( 6 < dimension_type::rank )
317 {
318 s[6] = stride_6();
319 }
320 if ( 7 < dimension_type::rank )
321 {
322 s[7] = stride_7();
323 }
324 s[dimension_type::rank] = span();
325 }

382// Given a tuple member type T of the given rank get the Kokkos view
383// data layout parameters. The tuple index effectively introduces 2 new
384// dimensions to the problem on top of the member dimensions - one for the
385// struct index and one for the vector index.
386template <typename T, std::size_t Rank, int VectorLength, int Stride>
387struct KokkosDataTypeImpl;
388
389// Rank-0
390template <typename T, int VectorLength, int Stride>
391struct KokkosDataTypeImpl<T, 0, VectorLength, Stride>
392{
393 using value_type = typename std::remove_all_extents<T>::type;
394 using data_type = value_type* [VectorLength];
395 using cabana_layout = Kokkos::LayoutCabanaSlice<Stride, VectorLength>;
396
397 inline static cabana_layout createLayout( const std::size_t num_soa )
398 {
399 return cabana_layout( num_soa );
400 }
401};
402
403// Rank-1
404template <typename T, int VectorLength, int Stride>
405struct KokkosDataTypeImpl<T, 1, VectorLength, Stride>
406{
407 using value_type = typename std::remove_all_extents<T>::type;
408 static constexpr std::size_t D0 = std::extent<T, 0>::value;
409 using data_type = value_type* [VectorLength][D0];
410 using cabana_layout = Kokkos::LayoutCabanaSlice<Stride, VectorLength, D0>;
411
412 inline static cabana_layout createLayout( const std::size_t num_soa )
413 {
414 return cabana_layout( num_soa );
415 }
416};
417
418// Rank-2
419template <typename T, int VectorLength, int Stride>
420struct KokkosDataTypeImpl<T, 2, VectorLength, Stride>
421{
422 using value_type = typename std::remove_all_extents<T>::type;
423 static constexpr std::size_t D0 = std::extent<T, 0>::value;
424 static constexpr std::size_t D1 = std::extent<T, 1>::value;
425 using data_type = value_type* [VectorLength][D0][D1];
426 using cabana_layout =
427 Kokkos::LayoutCabanaSlice<Stride, VectorLength, D0, D1>;

441 static constexpr std::size_t D1 = std::extent<T, 1>::value;
442 static constexpr std::size_t D2 = std::extent<T, 2>::value;
