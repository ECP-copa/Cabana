29 //---------------------------------------------------------------------------//
30 // Given a tuple member type T of the given rank get the Kokkos view
31 // data layout parameters. The tuple index effectively introduces 2 new
32 // dimensions to the problem on top of the member dimensions - one for the
33 // struct index and one for the vector index.
34 template<typename T, std::size_t Rank, int VectorLength>
35 struct KokkosDataTypeImpl;
36 
37 // Rank-0
38 template<typename T, int VectorLength>
39 struct KokkosDataTypeImpl<T,0,VectorLength>
40 {
41  using value_type = typename std::remove_all_extents<T>::type;
42  using data_type = value_type*[VectorLength];
43 
44  inline static Kokkos::LayoutStride createLayout( const std::size_t num_soa,
45  const std::size_t stride )
46  {
47  return Kokkos::LayoutStride( num_soa, stride,
48  VectorLength, 1 );
49  }
50 };
51 
52 // Rank-1
53 template<typename T, int VectorLength>
54 struct KokkosDataTypeImpl<T,1,VectorLength>
55 {
56  using value_type = typename std::remove_all_extents<T>::type;
57  static constexpr std::size_t D0 = std::extent<T,0>::value;
58  using data_type = value_type*[VectorLength][D0];
59 
60  inline static Kokkos::LayoutStride createLayout( const std::size_t num_soa,
61  const std::size_t stride )
62  {
63  return Kokkos::LayoutStride( num_soa, stride,
64  VectorLength, 1,
65  D0, VectorLength );
66  }
67 };
68 
69 // Rank-2
70 template<typename T, int VectorLength>
71 struct KokkosDataTypeImpl<T,2,VectorLength>
72 {
73  using value_type = typename std::remove_all_extents<T>::type;
74  static constexpr std::size_t D0 = std::extent<T,0>::value;
75  static constexpr std::size_t D1 = std::extent<T,1>::value;
76  using data_type = value_type*[VectorLength][D0][D1];
77 
78  inline static Kokkos::LayoutStride createLayout( const std::size_t num_soa,
79  const std::size_t stride )
80  {
81  return Kokkos::LayoutStride( num_soa, stride,
82  VectorLength, 1,
83  D0, D1*VectorLength,
84  D1, VectorLength );
85  }
86 };

88 // Rank-3
89 template<typename T, int VectorLength>
90 struct KokkosDataTypeImpl<T,3,VectorLength>
91 {
92  using value_type = typename std::remove_all_extents<T>::type;
93  static constexpr std::size_t D0 = std::extent<T,0>::value;
94  static constexpr std::size_t D1 = std::extent<T,1>::value;
95  static constexpr std::size_t D2 = std::extent<T,2>::value;
96  using data_type = value_type*[VectorLength][D0][D1][D2];
97 
98  inline static Kokkos::LayoutStride createLayout( const std::size_t num_soa,
99  const std::size_t stride )
100  {
101  return Kokkos::LayoutStride( num_soa, stride,
102  VectorLength, 1,
103  D0, D1*D2*VectorLength,
104  D1, D2*VectorLength,
105  D2, VectorLength );
106  }
107 };

109 // Data type specialization.
110 template<typename T, int VectorLength>
111 struct KokkosDataType
112 {
113  using kokkos_data_type =
114  KokkosDataTypeImpl<T,std::rank<T>::value,VectorLength>;
115  using data_type = typename kokkos_data_type::data_type;
116 
117  inline static Kokkos::LayoutStride createLayout( const std::size_t num_soa,
118  const std::size_t stride )
119  {
120  return kokkos_data_type::createLayout( num_soa, stride );
121  }
122 };
