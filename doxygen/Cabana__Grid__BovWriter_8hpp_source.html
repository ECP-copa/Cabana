109template <class Array_t, std::size_t N>
110MPI_Datatype createSubarray( const Array_t& array,
111 const std::array<long, N>& owned_extents,
112 const std::array<long, N>& global_extents )
113{
114 using value_type = typename Array_t::value_type;
115 const auto& global_grid = array.layout()->localGrid()->globalGrid();
116
117 std::array<int, N> local_start;
118 std::array<int, N> local_size;
119 std::array<int, N> global_size;
120 for ( std::size_t i = 0; i < N - 1; ++i )
121 {
122 local_start[i] =
123 static_cast<int>( global_grid.globalOffset( N - i - 2 ) );
124 local_size[i] = static_cast<int>( owned_extents[N - i - 2] );
125 global_size[i] = static_cast<int>( global_extents[N - i - 2] );
126 }
127 local_start.back() = 0;
128 local_size.back() = owned_extents.back();
129 global_size.back() = global_extents.back();
130
131 MPI_Datatype subarray;
132 MPI_Type_create_subarray( N, global_size.data(), local_size.data(),
133 local_start.data(), MPI_ORDER_C,
134 MpiTraits<value_type>::type(), &subarray );
135
136 return subarray;
137}

141template <class TargetView, class SourceView, class Indices, class ExecSpace>
142std::enable_if_t<4 == TargetView::rank, void>
143reorderView( TargetView& target, const SourceView& source,
144 const Indices& index_space, const ExecSpace& exec_space )
145{
146 Kokkos::parallel_for(
147 "bov_reorder", createExecutionPolicy( index_space, exec_space ),
148 KOKKOS_LAMBDA( const int k, const int j, const int i, const int l ) {
149 target( k, j, i, l ) = source( i, j, k, l );
150 } );
151 exec_space.fence();
152}

182template <class ExecutionSpace, class Array_t>
183void writeTimeStep( ExecutionSpace, const std::string& prefix,
184 const int time_step_index, const double time,
185 const Array_t& array, const bool gather_array = true )
186{
187 static_assert( isUniformMesh<typename Array_t::mesh_type>::value,
188 "ViSIT BOV writer can only be used with uniform mesh" );
189
190 // Types
191 using entity_type = typename Array_t::entity_type;
192 using value_type = typename Array_t::value_type;
193 using memory_space = typename Array_t::memory_space;
194 const std::size_t num_space_dim = Array_t::num_space_dim;
195
196 // Get the global grid.
197 const auto& global_grid = array.layout()->localGrid()->globalGrid();
198
199 // Get the global mesh.
200 const auto& global_mesh = global_grid.globalMesh();
201
202 // If this is a node field, determine periodicity so we can add the last
203 // node back to the visualization if needed.
204 std::array<long, num_space_dim + 1> global_extents;
205 for ( std::size_t i = 0; i < num_space_dim + 1; ++i )
206 {
207 global_extents[i] = -1;
208 }
209 for ( std::size_t d = 0; d < num_space_dim; ++d )
210 {
211 if ( std::is_same<entity_type, Cell>::value )
212 global_extents[d] = global_grid.globalNumEntity( Cell(), d );
213 else if ( std::is_same<entity_type, Node>::value )
214 global_extents[d] = global_grid.globalNumEntity( Cell(), d ) + 1;
215 }
216 global_extents[num_space_dim] = array.layout()->dofsPerEntity();
217
218 auto owned_index_space = array.layout()->indexSpace( Own(), Local() );
219 std::array<long, num_space_dim + 1> owned_extents;
220 for ( std::size_t i = 0; i < num_space_dim + 1; ++i )
221 {
222 owned_extents[i] = -1;
223 }
224 for ( std::size_t d = 0; d < num_space_dim; ++d )
225 {
226 if ( std::is_same<entity_type, Cell>::value )
227 {
228 owned_extents[d] = owned_index_space.extent( d );
229 }
230 else if ( std::is_same<entity_type, Node>::value )
231 {
232 if ( !global_grid.isPeriodic( d ) ||
233 global_grid.dimBlockId( d ) <
234 global_grid.dimNumBlock( d ) - 1 )
235 owned_extents[d] = owned_index_space.extent( d );
236 else
237 owned_extents[d] = owned_index_space.extent( d ) + 1;
238 }
239 }
240 owned_extents[num_space_dim] = array.layout()->dofsPerEntity();
241
242 // Gather halo data if any dimensions are periodic.
243 if ( gather_array )
244 {
245 for ( std::size_t d = 0; d < num_space_dim; ++d )
246 {
247 if ( global_grid.isPeriodic( d ) )
248 {
249 auto halo =
250 createHalo( NodeHaloPattern<num_space_dim>(), 0, array );
251 halo->gather( ExecutionSpace(), array );
252 break;
253 }
254 }
255 }
256
257 // Create a contiguous array of the owned array values. Note that we
258 // reorder to KJI grid ordering to conform to the BOV format.
259 std::array<long, num_space_dim + 1> local_space_min;
260 std::array<long, num_space_dim + 1> local_space_max;
261 for ( std::size_t d = 0; d < num_space_dim; ++d )
262 {
263 local_space_min[d] = owned_index_space.min( d );
264 local_space_max[d] = owned_index_space.min( d ) + owned_extents[d];
265 }
266 local_space_min.back() = 0;
267 local_space_max.back() = owned_extents.back();
268 IndexSpace<num_space_dim + 1> local_space( local_space_min,
269 local_space_max );
270 auto owned_subview = createSubview( array.view(), local_space );
271
272 std::array<long, num_space_dim + 1> reorder_space_size;
273 for ( std::size_t d = 0; d < num_space_dim; ++d )
274 {
275 reorder_space_size[d] = owned_extents[num_space_dim - d - 1];
276 }
277 reorder_space_size.back() = owned_extents.back();
278 IndexSpace<num_space_dim + 1> reorder_space( reorder_space_size );
279 auto owned_view = createView<value_type, Kokkos::LayoutRight, memory_space>(
280 array.label(), reorder_space );
281 reorderView( owned_view, owned_subview, reorder_space, ExecutionSpace() );
282
283 // Compose a data file name prefix.
284 std::stringstream file_name;
285 file_name << prefix << "_" << std::setfill( '0' ) << std::setw( 6 )
286 << time_step_index;
287
288 // Open a binary data file.
289 std::string data_file_name = file_name.str() + ".dat";
290 MPI_File data_file;
291 MPI_File_open( global_grid.comm(), data_file_name.c_str(),
292 MPI_MODE_WRONLY | MPI_MODE_CREATE, MPI_INFO_NULL,
293 &data_file );
294
295 // Create the global subarray in which we are writing the local data.
296 auto subarray = createSubarray( array, owned_extents, global_extents );
297 MPI_Type_commit( &subarray );
298
299 // Set the data in the file this process is going to write to.
300 MPI_File_set_view( data_file, 0, MpiTraits<value_type>::type(), subarray,
301 "native", MPI_INFO_NULL );
302
303 // Write the view to binary.
304 MPI_Status status;
305 MPI_File_write_all( data_file, owned_view.data(), owned_view.size(),
306 MpiTraits<value_type>::type(), &status );
307
308 // Clean up.
309 MPI_File_close( &data_file );
310 MPI_Type_free( &subarray );
311
312 // Create a VisIt BOV header with global data. Only create the header
313 // on rank 0.
314 int rank;
315 MPI_Comm_rank( global_grid.comm(), &rank );
316 if ( 0 == rank )
317 {
318 // Open a file for writing.
319 std::string header_file_name = file_name.str() + ".bov";
320 std::fstream header;
321 header.open( header_file_name, std::fstream::out );
322
323 // Write the current time.
324 header << "TIME: " << time << std::endl;
325
326 // Data file name.
327 header << "DATA_FILE: " << data_file_name << std::endl;
328
329 // Global data size.
330 header << "DATA_SIZE: ";
331 for ( std::size_t d = 0; d < num_space_dim; ++d )
332 {
333 header << global_extents[d] << " ";
334 }
335 for ( std::size_t d = num_space_dim; d < 3; ++d )
336 {
337 header << 1;
338 }
339 header << std::endl;
340
341 // Data format.
342 header << "DATA_FORMAT: " << BovFormat<value_type>::value()
343 << std::endl;
344
345 // Variable name.
346 header << "VARIABLE: " << array.label() << std::endl;
347
348 // Endian order
349 header << "DATA_ENDIAN: LITTLE" << std::endl;
350
351 // Data location.
352 header << "CENTERING: " << BovCentering<entity_type>::value()
353 << std::endl;
354
355 // Mesh low corner.
356 header << "BRICK_ORIGIN: ";
357 for ( std::size_t d = 0; d < num_space_dim; ++d )
358 {
359 header << global_mesh.lowCorner( d ) << " ";
360 }
361 for ( std::size_t d = num_space_dim; d < 3; ++d )
362 {
363 header << 0.0;
364 }
365 header << std::endl;
366
367 // Mesh global width
368 header << "BRICK_SIZE: ";
369 for ( std::size_t d = 0; d < num_space_dim; ++d )
370 {
371 header << global_grid.globalNumEntity( Cell(), d ) *
372 global_mesh.cellSize( d )
373 << " ";
374 }
375 for ( std::size_t d = num_space_dim; d < 3; ++d )
376 {
377 header << 0.0;
378 }
379 header << std::endl;
380
381 // Number of data components. Scalar and vector types are
382 // supported.
383 header << "DATA_COMPONENTS: " << global_extents[num_space_dim]
384 << std::endl;
385
386 // Close the header.
387 header.close();
388 }
389}
