86 // Create the MPI subarray for the given array.
87 template <class Array_t, std::size_t N>
88 MPI_Datatype createSubarray( const Array_t& array,
89  const std::array<long, N>& owned_extents,
90  const std::array<long, N>& global_extents )
91 {
92  using value_type = typename Array_t::value_type;
93  const auto& global_grid = array.layout()->localGrid()->globalGrid();
94 
95  std::array<int, N> local_start;
96  std::array<int, N> local_size;
97  std::array<int, N> global_size;
98  for ( std::size_t i = 0; i < N - 1; ++i )
99  {
100  local_start[i] =
101  static_cast<int>( global_grid.globalOffset( N - i - 2 ) );
102  local_size[i] = static_cast<int>( owned_extents[N - i - 2] );
103  global_size[i] = static_cast<int>( global_extents[N - i - 2] );
104  }
105  local_start.back() = 0;
106  local_size.back() = owned_extents.back();
107  global_size.back() = global_extents.back();
108 
109  MPI_Datatype subarray;
110  MPI_Type_create_subarray( N, global_size.data(), local_size.data(),
111  local_start.data(), MPI_ORDER_C,
112  MpiTraits<value_type>::type(), &subarray );
113 
114  return subarray;
115 }

118 // Reorder a view to the required ordering for I/O
119 template <class TargetView, class SourceView, class Indices, class ExecSpace>
120 std::enable_if_t<4 == TargetView::rank, void>
121 reorderView( TargetView& target, const SourceView& source,
122  const Indices& index_space, const ExecSpace& exec_space )
123 {
124  Kokkos::parallel_for(
125  "bov_reorder", createExecutionPolicy( index_space, exec_space ),
126  KOKKOS_LAMBDA( const int k, const int j, const int i, const int l ) {
127  target( k, j, i, l ) = source( i, j, k, l );
128  } );
129  exec_space.fence();
130 }

158 template <class Array_t>
159 void writeTimeStep( const int time_step_index, const double time,
160  const Array_t& array, const bool gather_array = true )
161 {
162  static_assert( isUniformMesh<typename Array_t::mesh_type>::value,
163  "ViSIT BOV writer can only be used with uniform mesh" );
164 
165  // Types
166  using entity_type = typename Array_t::entity_type;
167  using value_type = typename Array_t::value_type;
168  using device_type = typename Array_t::device_type;
169  using execution_space = typename device_type::execution_space;
170  const std::size_t num_space_dim = Array_t::num_space_dim;
171 
172  // Get the global grid.
173  const auto& global_grid = array.layout()->localGrid()->globalGrid();
174 
175  // Get the global mesh.
176  const auto& global_mesh = global_grid.globalMesh();
177 
178  // If this is a node field, determine periodicity so we can add the last
179  // node back to the visualization if needed.
180  std::array<long, num_space_dim + 1> global_extents;
181  for ( std::size_t i = 0; i < num_space_dim + 1; ++i )
182  {
183  global_extents[i] = -1;
184  }
185  for ( std::size_t d = 0; d < num_space_dim; ++d )
186  {
187  if ( std::is_same<entity_type, Cell>::value )
188  global_extents[d] = global_grid.globalNumEntity( Cell(), d );
189  else if ( std::is_same<entity_type, Node>::value )
190  global_extents[d] = global_grid.globalNumEntity( Cell(), d ) + 1;
191  }
192  global_extents[num_space_dim] = array.layout()->dofsPerEntity();
193 
194  auto owned_index_space = array.layout()->indexSpace( Own(), Local() );
195  std::array<long, num_space_dim + 1> owned_extents;
196  for ( std::size_t i = 0; i < num_space_dim + 1; ++i )
197  {
198  owned_extents[i] = -1;
199  }
200  for ( std::size_t d = 0; d < num_space_dim; ++d )
201  {
202  if ( std::is_same<entity_type, Cell>::value )
203  {
204  owned_extents[d] = owned_index_space.extent( d );
205  }
206  else if ( std::is_same<entity_type, Node>::value )
207  {
208  if ( !global_grid.isPeriodic( d ) ||
209  global_grid.dimBlockId( d ) <
210  global_grid.dimNumBlock( d ) - 1 )
211  owned_extents[d] = owned_index_space.extent( d );
212  else
213  owned_extents[d] = owned_index_space.extent( d ) + 1;
214  }
215  }
216  owned_extents[num_space_dim] = array.layout()->dofsPerEntity();
217 
218  // Gather halo data if any dimensions are periodic.
219  if ( gather_array )
220  {
221  for ( std::size_t d = 0; d < num_space_dim; ++d )
222  {
223  if ( global_grid.isPeriodic( d ) )
224  {
225  auto halo =
226  createHalo( NodeHaloPattern<num_space_dim>(), 0, array );
227  halo->gather( execution_space(), array );
228  break;
229  }
230  }
231  }
232 
233  // Create a contiguous array of the owned array values. Note that we
234  // reorder to KJI grid ordering to conform to the BOV format.
235  std::array<long, num_space_dim + 1> local_space_min;
236  std::array<long, num_space_dim + 1> local_space_max;
237  for ( std::size_t d = 0; d < num_space_dim; ++d )
238  {
239  local_space_min[d] = owned_index_space.min( d );
240  local_space_max[d] = owned_index_space.min( d ) + owned_extents[d];
241  }
242  local_space_min.back() = 0;
243  local_space_max.back() = owned_extents.back();
244  IndexSpace<num_space_dim + 1> local_space( local_space_min,
245  local_space_max );
246  auto owned_subview = createSubview( array.view(), local_space );
247 
248  std::array<long, num_space_dim + 1> reorder_space_size;
249  for ( std::size_t d = 0; d < num_space_dim; ++d )
250  {
251  reorder_space_size[d] = owned_extents[num_space_dim - d - 1];
252  }
253  reorder_space_size.back() = owned_extents.back();
254  IndexSpace<num_space_dim + 1> reorder_space( reorder_space_size );
255  auto owned_view = createView<value_type, Kokkos::LayoutRight, device_type>(
256  array.label(), reorder_space );
257  reorderView( owned_view, owned_subview, reorder_space, execution_space() );
258 
259  // Compose a data file name prefix.
260  std::stringstream file_name;
261  file_name << "grid_" << array.label() << "_" << std::setfill( '0' )
262  << std::setw( 6 ) << time_step_index;
263 
264  // Open a binary data file.
265  std::string data_file_name = file_name.str() + ".dat";
266  MPI_File data_file;
267  MPI_File_open( global_grid.comm(), data_file_name.c_str(),
268  MPI_MODE_WRONLY | MPI_MODE_CREATE, MPI_INFO_NULL,
269  &data_file );
270 
271  // Create the global subarray in which we are writing the local data.
272  auto subarray = createSubarray( array, owned_extents, global_extents );
273  MPI_Type_commit( &subarray );
274 
275  // Set the data in the file this process is going to write to.
276  MPI_File_set_view( data_file, 0, MpiTraits<value_type>::type(), subarray,
277  "native", MPI_INFO_NULL );
278 
279  // Write the view to binary.
280  MPI_Status status;
281  MPI_File_write_all( data_file, owned_view.data(), owned_view.size(),
282  MpiTraits<value_type>::type(), &status );
283 
284  // Clean up.
285  MPI_File_close( &data_file );
286  MPI_Type_free( &subarray );
287 
288  // Create a VisIt BOV header with global data. Only create the header
289  // on rank 0.
290  int rank;
291  MPI_Comm_rank( global_grid.comm(), &rank );
292  if ( 0 == rank )
293  {
294  // Open a file for writing.
295  std::string header_file_name = file_name.str() + ".bov";
296  std::fstream header;
297  header.open( header_file_name, std::fstream::out );
298 
299  // Write the current time.
300  header << "TIME: " << time << std::endl;
301 
302  // Data file name.
303  header << "DATA_FILE: " << data_file_name << std::endl;
304 
305  // Global data size.
306  header << "DATA_SIZE: ";
307  for ( std::size_t d = 0; d < num_space_dim; ++d )
308  {
309  header << global_extents[d] << " ";
310  }
311  for ( std::size_t d = num_space_dim; d < 3; ++d )
312  {
313  header << 1;
314  }
315  header << std::endl;
316 
317  // Data format.
318  header << "DATA_FORMAT: " << BovFormat<value_type>::value()
319  << std::endl;
320 
321  // Variable name.
322  header << "VARIABLE: " << array.label() << std::endl;
323 
324  // Endian order
325  header << "DATA_ENDIAN: LITTLE" << std::endl;
326 
327  // Data location.
328  header << "CENTERING: " << BovCentering<entity_type>::value()
329  << std::endl;
330 
331  // Mesh low corner.
332  header << "BRICK_ORIGIN: ";
333  for ( std::size_t d = 0; d < num_space_dim; ++d )
334  {
335  header << global_mesh.lowCorner( d ) << " ";
336  }
337  for ( std::size_t d = num_space_dim; d < 3; ++d )
338  {
339  header << 0.0;
340  }
341  header << std::endl;
342 
343  // Mesh global width
344  header << "BRICK_SIZE: ";
345  for ( std::size_t d = 0; d < num_space_dim; ++d )
346  {
347  header << global_grid.globalNumEntity( Cell(), d ) *
348  global_mesh.cellSize( d )
349  << " ";
350  }
351  for ( std::size_t d = num_space_dim; d < 3; ++d )
352  {
353  header << 0.0;
354  }
355  header << std::endl;
356 
357  // Number of data components. Scalar and vector types are
358  // supported.
359  header << "DATA_COMPONENTS: " << global_extents[num_space_dim]
360  << std::endl;
361 
362  // Close the header.
363  header.close();
364  }
365 }
