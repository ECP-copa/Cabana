106template <class Array_t, std::size_t N>
107MPI_Datatype createSubarray( const Array_t& array,
108 const std::array<long, N>& owned_extents,
109 const std::array<long, N>& global_extents )
110{
111 using value_type = typename Array_t::value_type;
112 const auto& global_grid = array.layout()->localGrid()->globalGrid();
113
114 std::array<int, N> local_start;
115 std::array<int, N> local_size;
116 std::array<int, N> global_size;
117 for ( std::size_t i = 0; i < N - 1; ++i )
118 {
119 local_start[i] =
120 static_cast<int>( global_grid.globalOffset( N - i - 2 ) );
121 local_size[i] = static_cast<int>( owned_extents[N - i - 2] );
122 global_size[i] = static_cast<int>( global_extents[N - i - 2] );
123 }
124 local_start.back() = 0;
125 local_size.back() = owned_extents.back();
126 global_size.back() = global_extents.back();
127
128 MPI_Datatype subarray;
129 MPI_Type_create_subarray( N, global_size.data(), local_size.data(),
130 local_start.data(), MPI_ORDER_C,
131 MpiTraits<value_type>::type(), &subarray );
132
133 return subarray;
134}

138template <class TargetView, class SourceView, class Indices, class ExecSpace>
139std::enable_if_t<4 == TargetView::rank, void>
140reorderView( TargetView& target, const SourceView& source,
141 const Indices& index_space, const ExecSpace& exec_space )
142{
143 Kokkos::parallel_for(
144 "bov_reorder", createExecutionPolicy( index_space, exec_space ),
145 KOKKOS_LAMBDA( const int k, const int j, const int i, const int l ) {
146 target( k, j, i, l ) = source( i, j, k, l );
147 } );
148 exec_space.fence();
149}

178template <class Array_t>
179void writeTimeStep( const int time_step_index, const double time,
180 const Array_t& array, const bool gather_array = true )
181{
182 static_assert( isUniformMesh<typename Array_t::mesh_type>::value,
183 "ViSIT BOV writer can only be used with uniform mesh" );
184
185 // Types
186 using entity_type = typename Array_t::entity_type;
187 using value_type = typename Array_t::value_type;
188 using device_type = typename Array_t::device_type;
189 using execution_space = typename device_type::execution_space;
190 const std::size_t num_space_dim = Array_t::num_space_dim;
191
192 // Get the global grid.
193 const auto& global_grid = array.layout()->localGrid()->globalGrid();
194
195 // Get the global mesh.
196 const auto& global_mesh = global_grid.globalMesh();
197
198 // If this is a node field, determine periodicity so we can add the last
199 // node back to the visualization if needed.
200 std::array<long, num_space_dim + 1> global_extents;
201 for ( std::size_t i = 0; i < num_space_dim + 1; ++i )
202 {
203 global_extents[i] = -1;
204 }
205 for ( std::size_t d = 0; d < num_space_dim; ++d )
206 {
207 if ( std::is_same<entity_type, Cell>::value )
208 global_extents[d] = global_grid.globalNumEntity( Cell(), d );
209 else if ( std::is_same<entity_type, Node>::value )
210 global_extents[d] = global_grid.globalNumEntity( Cell(), d ) + 1;
211 }
212 global_extents[num_space_dim] = array.layout()->dofsPerEntity();
213
214 auto owned_index_space = array.layout()->indexSpace( Own(), Local() );
215 std::array<long, num_space_dim + 1> owned_extents;
216 for ( std::size_t i = 0; i < num_space_dim + 1; ++i )
217 {
218 owned_extents[i] = -1;
219 }
220 for ( std::size_t d = 0; d < num_space_dim; ++d )
221 {
222 if ( std::is_same<entity_type, Cell>::value )
223 {
224 owned_extents[d] = owned_index_space.extent( d );
225 }
226 else if ( std::is_same<entity_type, Node>::value )
227 {
228 if ( !global_grid.isPeriodic( d ) ||
229 global_grid.dimBlockId( d ) <
230 global_grid.dimNumBlock( d ) - 1 )
231 owned_extents[d] = owned_index_space.extent( d );
232 else
233 owned_extents[d] = owned_index_space.extent( d ) + 1;
234 }
235 }
236 owned_extents[num_space_dim] = array.layout()->dofsPerEntity();
237
238 // Gather halo data if any dimensions are periodic.
239 if ( gather_array )
240 {
241 for ( std::size_t d = 0; d < num_space_dim; ++d )
242 {
243 if ( global_grid.isPeriodic( d ) )
244 {
245 auto halo =
246 createHalo( NodeHaloPattern<num_space_dim>(), 0, array );
247 halo->gather( execution_space(), array );
248 break;
249 }
250 }
251 }
252
253 // Create a contiguous array of the owned array values. Note that we
254 // reorder to KJI grid ordering to conform to the BOV format.
255 std::array<long, num_space_dim + 1> local_space_min;
256 std::array<long, num_space_dim + 1> local_space_max;
257 for ( std::size_t d = 0; d < num_space_dim; ++d )
258 {
259 local_space_min[d] = owned_index_space.min( d );
260 local_space_max[d] = owned_index_space.min( d ) + owned_extents[d];
261 }
262 local_space_min.back() = 0;
263 local_space_max.back() = owned_extents.back();
264 IndexSpace<num_space_dim + 1> local_space( local_space_min,
265 local_space_max );
266 auto owned_subview = createSubview( array.view(), local_space );
267
268 std::array<long, num_space_dim + 1> reorder_space_size;
269 for ( std::size_t d = 0; d < num_space_dim; ++d )
270 {
271 reorder_space_size[d] = owned_extents[num_space_dim - d - 1];
272 }
273 reorder_space_size.back() = owned_extents.back();
274 IndexSpace<num_space_dim + 1> reorder_space( reorder_space_size );
275 auto owned_view = createView<value_type, Kokkos::LayoutRight, device_type>(
276 array.label(), reorder_space );
277 reorderView( owned_view, owned_subview, reorder_space, execution_space() );
278
279 // Compose a data file name prefix.
280 std::stringstream file_name;
281 file_name << "grid_" << array.label() << "_" << std::setfill( '0' )
282 << std::setw( 6 ) << time_step_index;
283
284 // Open a binary data file.
285 std::string data_file_name = file_name.str() + ".dat";
286 MPI_File data_file;
287 MPI_File_open( global_grid.comm(), data_file_name.c_str(),
288 MPI_MODE_WRONLY | MPI_MODE_CREATE, MPI_INFO_NULL,
289 &data_file );
290
291 // Create the global subarray in which we are writing the local data.
292 auto subarray = createSubarray( array, owned_extents, global_extents );
293 MPI_Type_commit( &subarray );
294
295 // Set the data in the file this process is going to write to.
296 MPI_File_set_view( data_file, 0, MpiTraits<value_type>::type(), subarray,
297 "native", MPI_INFO_NULL );
298
299 // Write the view to binary.
300 MPI_Status status;
301 MPI_File_write_all( data_file, owned_view.data(), owned_view.size(),
302 MpiTraits<value_type>::type(), &status );
303
304 // Clean up.
305 MPI_File_close( &data_file );
306 MPI_Type_free( &subarray );
307
308 // Create a VisIt BOV header with global data. Only create the header
309 // on rank 0.
310 int rank;
311 MPI_Comm_rank( global_grid.comm(), &rank );
312 if ( 0 == rank )
313 {
314 // Open a file for writing.
315 std::string header_file_name = file_name.str() + ".bov";
316 std::fstream header;
317 header.open( header_file_name, std::fstream::out );
318
319 // Write the current time.
320 header << "TIME: " << time << std::endl;
321
322 // Data file name.
323 header << "DATA_FILE: " << data_file_name << std::endl;
324
325 // Global data size.
326 header << "DATA_SIZE: ";
327 for ( std::size_t d = 0; d < num_space_dim; ++d )
328 {
329 header << global_extents[d] << " ";
330 }
331 for ( std::size_t d = num_space_dim; d < 3; ++d )
332 {
333 header << 1;
334 }
335 header << std::endl;
336
337 // Data format.
338 header << "DATA_FORMAT: " << BovFormat<value_type>::value()
339 << std::endl;
340
341 // Variable name.
342 header << "VARIABLE: " << array.label() << std::endl;
343
344 // Endian order
345 header << "DATA_ENDIAN: LITTLE" << std::endl;
346
347 // Data location.
348 header << "CENTERING: " << BovCentering<entity_type>::value()
349 << std::endl;
350
351 // Mesh low corner.
352 header << "BRICK_ORIGIN: ";
353 for ( std::size_t d = 0; d < num_space_dim; ++d )
354 {
355 header << global_mesh.lowCorner( d ) << " ";
356 }
357 for ( std::size_t d = num_space_dim; d < 3; ++d )
358 {
359 header << 0.0;
360 }
361 header << std::endl;
362
363 // Mesh global width
364 header << "BRICK_SIZE: ";
365 for ( std::size_t d = 0; d < num_space_dim; ++d )
366 {
367 header << global_grid.globalNumEntity( Cell(), d ) *
368 global_mesh.cellSize( d )
369 << " ";
370 }
371 for ( std::size_t d = num_space_dim; d < 3; ++d )
372 {
373 header << 0.0;
374 }
375 header << std::endl;
376
377 // Number of data components. Scalar and vector types are
378 // supported.
379 header << "DATA_COMPONENTS: " << global_extents[num_space_dim]
380 << std::endl;
381
382 // Close the header.
383 header.close();
384 }
385}
