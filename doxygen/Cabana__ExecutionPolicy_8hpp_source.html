34 template <int VectorLength, class IndexType>
35 class StructRange
36 {
37  public:
38  KOKKOS_INLINE_FUNCTION
39  static constexpr IndexType structBegin( const IndexType begin )
40  {
41  return Index<VectorLength>::s( begin );
42  }
43 
44  KOKKOS_INLINE_FUNCTION
45  static constexpr IndexType structEnd( const IndexType end )
46  {
47  // If the end is also at the front of an array that means the struct
48  // index of end is also the ending struct index. If not, we are not
49  // iterating all the way through the arrays of the last struct. In
50  // this case we add 1 to ensure that the loop over structs loops
51  // through all structs with data.
52  return ( 0 == Index<VectorLength>::a( end ) )
53  ? Index<VectorLength>::s( end )
54  : Index<VectorLength>::s( end ) + 1;
55  }
56 
57  KOKKOS_INLINE_FUNCTION
58  static constexpr IndexType size( const IndexType begin,
59  const IndexType end )
60  {
61  return structEnd( end ) - structBegin( begin );
62  }
63 };

75 template <int VectorLength, class... Properties>
76 class SimdPolicy
77  : public Kokkos::TeamPolicy<
78  typename Kokkos::Impl::PolicyTraits<Properties...>::execution_space,
79  Kokkos::Schedule<Kokkos::Dynamic>>
80 {
81  private:
82  typedef Kokkos::Impl::PolicyTraits<Properties...> traits;
83 
84  public:
85  using work_tag = typename traits::work_tag;
86  using execution_space = typename traits::execution_space;
87  using base_type =
88  Kokkos::TeamPolicy<execution_space, Kokkos::Schedule<Kokkos::Dynamic>>;
89  using execution_policy = SimdPolicy<VectorLength, Properties...>;
90  using index_type = typename traits::index_type;
91 
99  SimdPolicy( const index_type begin, const index_type end )
100  : base_type(
101  Impl::StructRange<VectorLength, index_type>::size( begin, end ),
102  1, VectorLength )
103  , _struct_begin(
104  Impl::StructRange<VectorLength, index_type>::structBegin(
105  begin ) )
106  , _struct_end(
107  Impl::StructRange<VectorLength, index_type>::structEnd( end ) )
108  , _array_begin( Impl::Index<VectorLength>::a( begin ) )
109  , _array_end( Impl::Index<VectorLength>::a( end ) )
110  {
111  }
112 
114  KOKKOS_INLINE_FUNCTION index_type structBegin() const
115  {
116  return _struct_begin;
117  }
118 
120  KOKKOS_INLINE_FUNCTION index_type structEnd() const { return _struct_end; }
121 
123  KOKKOS_INLINE_FUNCTION index_type arrayBegin( const index_type s ) const
124  {
125  // If the given struct index is also the index of the struct index in
126  // begin, use the starting array index. If not, that means we have
127  // passed the first struct and all subsequent structs start at array
128  // index 0.
129  return ( s == _struct_begin ) ? _array_begin : 0;
130  }
131 
132  // Given a struct id get the ending array index.
133  KOKKOS_INLINE_FUNCTION index_type arrayEnd( const index_type s ) const
134  {
135  // If we are in the last unfilled struct then use the array
136  // index of end. If not, we are looping through the current array all
137  // the way to the end so use the vector length.
138  return ( ( s == _struct_end - 1 ) && ( _array_end != 0 ) )
139  ? _array_end
140  : VectorLength;
141  }
142 
143  private:
144  index_type _struct_begin;
145  index_type _struct_end;
146  index_type _array_begin;
147  index_type _array_end;
148 };
