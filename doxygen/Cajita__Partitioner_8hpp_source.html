57 template <std::size_t NumSpaceDim>
58 class ManualBlockPartitioner : public BlockPartitioner<NumSpaceDim>
59 {
60  public:
62  static constexpr std::size_t num_space_dim = NumSpaceDim;
63 
68  ManualBlockPartitioner( const std::array<int, NumSpaceDim>& ranks_per_dim )
69  : _ranks_per_dim( ranks_per_dim )
70  {
71  }
72 
77  std::array<int, NumSpaceDim>
78  ranksPerDimension( MPI_Comm comm,
79  const std::array<int, NumSpaceDim>& ) const override
80  {
81  int comm_size;
82  MPI_Comm_size( comm, &comm_size );
83  int nrank = 1;
84  for ( std::size_t d = 0; d < num_space_dim; ++d )
85  nrank *= _ranks_per_dim[d];
86  if ( comm_size != nrank )
87  throw std::runtime_error(
88  "ManualPartitioner ranks do not match comm size" );
89  return _ranks_per_dim;
90  }
91 
92  private:
93  std::array<int, NumSpaceDim> _ranks_per_dim;
94 };
