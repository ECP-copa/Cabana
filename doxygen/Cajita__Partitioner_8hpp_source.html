81  std::array<int, NumSpaceDim>

83  const std::array<int, NumSpaceDim>& ) const override
84  {
85  int comm_size;
86  MPI_Comm_size( comm, &comm_size );
87  int nrank = 1;
88  for ( std::size_t d = 0; d < num_space_dim; ++d )
89  nrank *= _ranks_per_dim[d];

94  }
