75template <std::size_t NumSpaceDim>
76class ManualBlockPartitioner : public BlockPartitioner<NumSpaceDim>
77{
78 public:
80 static constexpr std::size_t num_space_dim = NumSpaceDim;
81
86 ManualBlockPartitioner( const std::array<int, NumSpaceDim>& ranks_per_dim )
87 : _ranks_per_dim( ranks_per_dim )
88 {
89 }

95 std::array<int, NumSpaceDim>
96 ranksPerDimension( MPI_Comm comm,
97 const std::array<int, NumSpaceDim>& ) const override
98 {
99 int comm_size;
100 MPI_Comm_size( comm, &comm_size );
101 int nrank = 1;
102 for ( std::size_t d = 0; d < num_space_dim; ++d )
103 nrank *= _ranks_per_dim[d];
104 if ( comm_size != nrank )
105 throw std::runtime_error(
106 "ManualPartitioner ranks do not match comm size" );
107 return _ranks_per_dim;
108 }
109
120 void ownedCellInfo(
121 MPI_Comm cart_comm,
122 const std::array<int, num_space_dim>& global_cells_per_dim,
123 std::array<int, num_space_dim>& owned_num_cell,
124 std::array<int, num_space_dim>& global_cell_offset ) const override
125 {
126 // Get the cells per dimension and the remainder.
127 std::array<int, num_space_dim> cells_per_dim;
128 std::array<int, num_space_dim> dim_remainder;
129 std::array<int, num_space_dim> cart_rank;
130 averageCellInfo( cart_comm, global_cells_per_dim, cart_rank,
131 cells_per_dim, dim_remainder );
132
133 // Compute the global cell offset and the local low corner on this rank
134 // by computing the starting global cell index via exclusive scan.
135 global_cell_offset =
136 globalCellOffsetHelper( cart_rank, cells_per_dim, dim_remainder );
137
138 // Compute the number of local cells in this rank in each dimension.
139 owned_num_cell =
140 ownedCellsHelper( cart_rank, cells_per_dim, dim_remainder );
141 }

151 std::array<int, num_space_dim> ownedCellsPerDimension(
152 MPI_Comm cart_comm,
153 const std::array<int, num_space_dim>& global_cells_per_dim ) const
154 {
155 // Get the cells per dimension and the remainder.
156 std::array<int, num_space_dim> cells_per_dim;
157 std::array<int, num_space_dim> dim_remainder;
158 std::array<int, num_space_dim> cart_rank;
159 averageCellInfo( cart_comm, global_cells_per_dim, cart_rank,
160 cells_per_dim, dim_remainder );
161
162 // Compute the number of local cells in this rank in each dimension.
163 return ownedCellsHelper( cart_rank, cells_per_dim, dim_remainder );
164 }
165
166 private:
177 inline void
178 averageCellInfo( MPI_Comm cart_comm,
179 const std::array<int, num_space_dim>& global_cells_per_dim,
180 std::array<int, num_space_dim>& cart_rank,
181 std::array<int, num_space_dim>& cells_per_dim,
182 std::array<int, num_space_dim>& dim_remainder ) const
183 {
184 // Get the Cartesian size and topology index of this rank.
185 int linear_rank;
186 MPI_Comm_rank( cart_comm, &linear_rank );
187 MPI_Cart_coords( cart_comm, linear_rank, num_space_dim,
188 cart_rank.data() );
189 // Get the cells per dimension and the remainder.
190 for ( std::size_t d = 0; d < num_space_dim; ++d )
191 {
192 cells_per_dim[d] = global_cells_per_dim[d] / _ranks_per_dim[d];
193 dim_remainder[d] = global_cells_per_dim[d] % _ranks_per_dim[d];
194 }
195 }
196
206 inline std::array<int, num_space_dim> ownedCellsHelper(
207 const std::array<int, num_space_dim>& cart_rank,
208 const std::array<int, num_space_dim>& cells_per_dim,
209 const std::array<int, num_space_dim>& dim_remainder ) const
210 {
211 std::array<int, num_space_dim> owned_num_cell;
212 // Compute the number of local cells in this rank in each dimension.
213 for ( std::size_t d = 0; d < num_space_dim; ++d )
214 {
215 owned_num_cell[d] = cells_per_dim[d];
216 if ( dim_remainder[d] > cart_rank[d] )
217 ++owned_num_cell[d];
218 }
219 return owned_num_cell;
220 }
221
231 inline std::array<int, num_space_dim> globalCellOffsetHelper(
232 const std::array<int, num_space_dim>& cart_rank,
233 const std::array<int, num_space_dim>& cells_per_dim,
234 const std::array<int, num_space_dim>& dim_remainder ) const
235 {
236 std::array<int, num_space_dim> global_cell_offset;
237 // Compute the global cell offset and the local low corner on this rank
238 // by computing the starting global cell index via exclusive scan.
239 for ( std::size_t d = 0; d < num_space_dim; ++d )
240 {
241 global_cell_offset[d] = 0;
242 for ( int r = 0; r < cart_rank[d]; ++r )
243 {
244 global_cell_offset[d] += cells_per_dim[d];
245 if ( dim_remainder[d] > r )
246 ++global_cell_offset[d];
247 }
248 }
249 return global_cell_offset;
250 }
251
252 private:
253 std::array<int, NumSpaceDim> _ranks_per_dim;
254};

365 std::array<int, num_space_dim>& cells_per_dim,
