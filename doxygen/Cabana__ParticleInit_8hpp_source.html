86template <class ExecutionSpace, class InitFunctor, class ParticleListType,
87 class ArrayType>
88int createParticles(
89 InitRandom, ExecutionSpace exec_space, const InitFunctor& create_functor,
90 ParticleListType& particle_list, const std::size_t num_particles,
91 const ArrayType box_min, const ArrayType box_max,
92 const std::size_t previous_num_particles = 0,
93 const bool shrink_to_fit = true, const uint64_t seed = 342343901,
94 typename std::enable_if<is_particle_list<ParticleListType>::value,
95 int>::type* = 0 )
96{
97 // Memory space.
98 using memory_space = typename ParticleListType::memory_space;
99
100 using PoolType = Kokkos::Random_XorShift64_Pool<ExecutionSpace>;
101 using RandomType = Kokkos::Random_XorShift64<ExecutionSpace>;
102 PoolType pool( seed );
103
104 // Resize the aosoa prior to lambda capture.
105 auto& aosoa = particle_list.aosoa();
106 aosoa.resize( previous_num_particles + num_particles );
107
108 // Creation count.
109 auto count = Kokkos::View<int*, memory_space>( "particle_count", 1 );
110 Kokkos::deep_copy( count, previous_num_particles );
111
112 // Copy corners to device accessible arrays.
113 auto kokkos_min = Impl::copyArray( box_min );
114 auto kokkos_max = Impl::copyArray( box_max );
115
116 auto random_coord_op = KOKKOS_LAMBDA( const int p )
117 {
118 // Particle coordinate.
119 double px[3];
120
121 auto gen = pool.get_state();
122 auto particle = particle_list.getParticle( p );
123 for ( int d = 0; d < 3; ++d )
124 px[d] = Kokkos::rand<RandomType, double>::draw( gen, kokkos_min[d],
125 kokkos_max[d] );
126 pool.free_state( gen );
127
128 // No volume information, so pass zero.
129 int create = create_functor( count( 0 ), px, 0.0, particle );
130
131 // If we created a new particle insert it into the list.
132 if ( create )
133 {
134 auto c = Kokkos::atomic_fetch_add( &count( 0 ), 1 );
135 particle_list.setParticle( particle, c );
136 }
137 };
138
139 Kokkos::RangePolicy<ExecutionSpace> exec_policy(
140 exec_space, previous_num_particles,
141 previous_num_particles + num_particles );
142 Kokkos::parallel_for( exec_policy, random_coord_op );
143 Kokkos::fence();
144
145 auto count_host =
146 Kokkos::create_mirror_view_and_copy( Kokkos::HostSpace(), count );
147 aosoa.resize( count_host( 0 ) );
148 if ( shrink_to_fit )
149 aosoa.shrinkToFit();
150 return count_host( 0 );
151}

203template <class ExecutionSpace, class PositionType, class ArrayType>
204void createParticles(
205 InitRandom, ExecutionSpace exec_space, PositionType& positions,
206 const std::size_t num_particles, const ArrayType box_min,
207 const ArrayType box_max, const std::size_t previous_num_particles = 0,
208 const uint64_t seed = 342343901,
209 typename std::enable_if<( is_slice<PositionType>::value ||
210 Kokkos::is_view<PositionType>::value ),
211 int>::type* = 0 )
212{
213 // Ensure correct space for the particles (View or Slice).
214 checkSize( positions, num_particles + previous_num_particles );
215
216 // Copy corners to device accessible arrays.
217 auto kokkos_min = Impl::copyArray( box_min );
218 auto kokkos_max = Impl::copyArray( box_max );
219
220 using PoolType = Kokkos::Random_XorShift64_Pool<ExecutionSpace>;
221 using RandomType = Kokkos::Random_XorShift64<ExecutionSpace>;
222 PoolType pool( seed );
223 auto random_coord_op = KOKKOS_LAMBDA( const int p )
224 {
225 auto gen = pool.get_state();
226 for ( int d = 0; d < 3; ++d )
227 positions( p, d ) = Kokkos::rand<RandomType, double>::draw(
228 gen, kokkos_min[d], kokkos_max[d] );
229 pool.free_state( gen );
230 };

232 Kokkos::RangePolicy<ExecutionSpace> exec_policy(
233 exec_space, previous_num_particles,

324template <class ExecutionSpace, class InitFunctor, class ParticleListType,
325 class PositionTag, class ArrayType>
326int createParticles(
327 InitRandom tag, ExecutionSpace exec_space,
328 const InitFunctor& create_functor, ParticleListType& particle_list,
329 PositionTag position_tag, const std::size_t num_particles,
330 const double min_dist, const ArrayType box_min, const ArrayType box_max,
331 const std::size_t previous_num_particles = 0,
332 const bool shrink_to_fit = true, const uint64_t seed = 342343901,
333 typename std::enable_if<is_particle_list<ParticleListType>::value,
334 int>::type* = 0 )
335{
336 double min_dist_sqr = min_dist * min_dist;
337
338 // Resize the aosoa prior to lambda capture.
339 auto& aosoa = particle_list.aosoa();
340 aosoa.resize( previous_num_particles + num_particles );
341
342 auto positions = particle_list.slice( position_tag );
343
344 // Create the functor which ignores particles within the radius of another.
345 auto min_distance_op =
346 KOKKOS_LAMBDA( const int id, const double px[3], const double,
347 typename ParticleListType::particle_type& particle )
348 {
349 // Ensure this particle is not within the minimum distance of any other
350 // existing particle.
351 for ( int n = 0; n < id; n++ )
352 {
353 double dx = positions( n, 0 ) - px[0];
354 double dy = positions( n, 1 ) - px[1];
355 double dz = positions( n, 2 ) - px[2];
356 double dist = dx * dx + dy * dy + dz * dz;
357 if ( dist < min_dist_sqr )
358 return false;
359 }
360
361 bool create = create_functor( id, px, 0.0, particle );
362 return create;
363 };
364
365 // Pass the functor to the general case.
366 return createParticles( tag, exec_space, min_distance_op, particle_list,
367 num_particles, box_min, box_max,
368 previous_num_particles, shrink_to_fit, seed );
369}

443template <class ExecutionSpace, class PositionType>
444[[deprecated( "Use createParticles instead." )]] void
445createRandomParticlesMinDistance( ExecutionSpace exec_space,
446 PositionType& positions,
447 const std::size_t num_particles,
448 const double box_min, const double box_max,
449 const double min_dist )
450{
451 double min_dist_sqr = min_dist * min_dist;
452
453 using PoolType = Kokkos::Random_XorShift64_Pool<ExecutionSpace>;
454 using RandomType = Kokkos::Random_XorShift64<ExecutionSpace>;
455 PoolType pool( 342343901 );
456
457 auto random_coord_op = KOKKOS_LAMBDA( const int p )
458 {
459 auto gen = pool.get_state();
460
461 // Create particles. Only add particles that are outside a minimum
462 // distance from other particles.
463 bool found_neighbor = true;
464
465 // Keep trying new random coordinates until we insert one that is
466 // not within the minimum distance of any other particle.
467 while ( found_neighbor )
468 {
469 found_neighbor = false;
470
471 for ( int d = 0; d < 3; ++d )
472 positions( p, d ) = Kokkos::rand<RandomType, double>::draw(
473 gen, box_min, box_max );
474
475 for ( int n = 0; n < p; n++ )
476 {
477 double dx = positions( n, 0 ) - positions( p, 0 );
478 double dy = positions( n, 1 ) - positions( p, 1 );
479 double dz = positions( n, 2 ) - positions( p, 2 );
480 double dist = dx * dx + dy * dy + dz * dz;
481 if ( dist < min_dist_sqr )
482 {
483 found_neighbor = true;
484 break;
485 }
486 }
487 }
488 pool.free_state( gen );
489 };
490
491 Kokkos::RangePolicy<ExecutionSpace> exec_policy( exec_space, 0,
492 num_particles );
493 Kokkos::parallel_for( exec_policy, random_coord_op );
494 Kokkos::fence();
495}
