32 template <class ExecutionSpace, class PositionType>
33 void createRandomParticlesMinDistance( ExecutionSpace, PositionType& positions,
34  const std::size_t num_particles,
35  const double box_min,
36  const double box_max,
37  const double min_dist )
38 {
39  double min_dist_sqr = min_dist * min_dist;
40 
41  using PoolType = Kokkos::Random_XorShift64_Pool<ExecutionSpace>;
42  using RandomType = Kokkos::Random_XorShift64<ExecutionSpace>;
43  PoolType pool( 342343901 );
44  auto random_coord_op = KOKKOS_LAMBDA( const int p )
45  {
46  auto gen = pool.get_state();
47 
48  // Create particles. Only add particles that are outside a minimum
49  // distance from other particles.
50  bool found_neighbor = true;
51 
52  // Keep trying new random coordinates until we insert one that is
53  // not within the minimum distance of any other particle.
54  while ( found_neighbor )
55  {
56  found_neighbor = false;
57 
58  for ( int d = 0; d < 3; ++d )
59  positions( p, d ) = Kokkos::rand<RandomType, double>::draw(
60  gen, box_min, box_max );
61 
62  for ( int n = 0; n < p; n++ )
63  {
64  double dx = positions( n, 0 ) - positions( p, 0 );
65  double dy = positions( n, 1 ) - positions( p, 1 );
66  double dz = positions( n, 2 ) - positions( p, 2 );
67  double dist = dx * dx + dy * dy + dz * dz;
68  if ( dist < min_dist_sqr )
69  {
70  found_neighbor = true;
71  break;
72  }
73  }
74  pool.free_state( gen );
75  }
76  };
77  Kokkos::RangePolicy<ExecutionSpace> exec_policy( 0, num_particles );
78  Kokkos::parallel_for( exec_policy, random_coord_op );
79  Kokkos::fence();
80 }
