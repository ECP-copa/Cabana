48template <class MeshType>
49class GlobalMesh
50{
51 public:
53 using mesh_type = MeshType;
54
56 using scalar_type = typename mesh_type::scalar_type;
57
59 static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;
60
62 GlobalMesh(
63 const std::array<scalar_type, num_space_dim>& global_low_corner,
64 const std::array<scalar_type, num_space_dim>& global_high_corner,
65 const scalar_type cell_size )
66 : _global_low_corner( global_low_corner )
67 , _global_high_corner( global_high_corner )
68 {
69 // Check that the domain is evenly divisible by the cell size in each
70 // dimension within round-off error.
71 for ( std::size_t d = 0; d < num_space_dim; ++d )
72 {
73 _cell_size[d] = cell_size;
74 scalar_type ext = globalNumCell( d ) * _cell_size[d];
75 if ( std::abs( ext - extent( d ) ) >
76 scalar_type( 100.0 ) *
77 std::numeric_limits<scalar_type>::epsilon() )
78 throw std::logic_error(
79 "Extent not evenly divisible by uniform cell size" );
80 }
81 }
82
84 GlobalMesh(
85 const std::array<scalar_type, num_space_dim>& global_low_corner,
86 const std::array<scalar_type, num_space_dim>& global_high_corner,
87 const std::array<scalar_type, num_space_dim>& cell_size )
88 : _global_low_corner( global_low_corner )
89 , _global_high_corner( global_high_corner )
90 , _cell_size( cell_size )
91 {
92 // Check that the domain is evenly divisible by the cell size in each
93 // dimension within round-off error.
94 for ( std::size_t d = 0; d < num_space_dim; ++d )
95 {
96 scalar_type ext = globalNumCell( d ) * _cell_size[d];
97 if ( std::abs( ext - extent( d ) ) >
98 scalar_type( 100.0 ) *
99 std::numeric_limits<scalar_type>::epsilon() )
100 throw std::logic_error(
101 "Extent not evenly divisible by uniform cell size" );
102 }
103 }
104
106 GlobalMesh(
107 const std::array<scalar_type, num_space_dim>& global_low_corner,
108 const std::array<scalar_type, num_space_dim>& global_high_corner,
109 const std::array<int, num_space_dim>& global_num_cell )
110 : _global_low_corner( global_low_corner )
111 , _global_high_corner( global_high_corner )
112 {
113 // Compute the cell size in each dimension.
114 for ( std::size_t d = 0; d < num_space_dim; ++d )
115 _cell_size[d] = ( _global_high_corner[d] - _global_low_corner[d] ) /
116 global_num_cell[d];
117
118 // Check that the domain is evenly divisible by the cell size in each
119 // dimension within round-off error and that we got the expected
120 // number of cells.
121 for ( std::size_t d = 0; d < num_space_dim; ++d )
122 {
123 scalar_type ext = globalNumCell( d ) * _cell_size[d];
124 if ( std::abs( ext - extent( d ) ) >
125 scalar_type( 100.0 ) *
126 std::numeric_limits<scalar_type>::epsilon() )
127 throw std::logic_error(
128 "Extent not evenly divisible by uniform cell size" );
129 if ( globalNumCell( d ) != global_num_cell[d] )
130 throw std::logic_error( "Global number of cells mismatch" );
131 }
132 }
133
134 // GLOBAL MESH INTERFACE
135
138 scalar_type lowCorner( const std::size_t dim ) const
139 {
140 return _global_low_corner[dim];
141 }
142
145 scalar_type highCorner( const std::size_t dim ) const
146 {
147 return _global_high_corner[dim];
148 }
149
152 scalar_type extent( const std::size_t dim ) const
153 {
154 return highCorner( dim ) - lowCorner( dim );
155 }
156
159 int globalNumCell( const std::size_t dim ) const
160 {
161 return std::rint( extent( dim ) / _cell_size[dim] );
162 }
163
164 // UNIFORM MESH SPECIFIC
165
168 scalar_type cellSize( const std::size_t dim ) const
169 {
170 return _cell_size[dim];
171 }
172
173 private:
174 std::array<scalar_type, num_space_dim> _global_low_corner;
175 std::array<scalar_type, num_space_dim> _global_high_corner;
176 std::array<scalar_type, num_space_dim> _cell_size;
177};
