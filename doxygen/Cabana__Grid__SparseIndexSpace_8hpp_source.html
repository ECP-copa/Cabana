68 if ( input_int )
69 return bitLength( input_int >> 1 ) + static_cast<Integer>( 1 );

71 return 0;

86//---------------------------------------------------------------------------//
93template <typename Integer>
94KOKKOS_INLINE_FUNCTION constexpr Integer
95binaryReverse( Integer input_int, char loc = sizeof( Integer ) * 8 - 1 )
96{
97 if ( input_int == 0 )
98 return 0;
99 return ( ( input_int & 1 ) << loc ) |
100 binaryReverse( input_int >> 1, loc - 1 );
101}
102
103//---------------------------------------------------------------------------//
109template <typename Integer>
110KOKKOS_INLINE_FUNCTION constexpr unsigned countLeadingZeros( Integer input_int )
111{
112 unsigned res{ 0 };
113 input_int = binaryReverse( input_int );
114 if ( input_int == 0 )
115 return sizeof( Integer ) * 8;
116 while ( ( input_int & 1 ) == 0 )
117 res++, input_int >>= 1;
118 return res;
119}

121//---------------------------------------------------------------------------//
128KOKKOS_INLINE_FUNCTION
129constexpr int bitPack( const uint64_t mask, const uint64_t data )
130{
131 uint64_t slresult = 0;
132 uint64_t& ulresult{ slresult };
133 uint64_t uldata = data;
134 int count = 0;
135 ulresult = 0;
136
137 uint64_t rmask = binaryReverse( mask );
138 unsigned char lz{ 0 };
139
140 while ( rmask )
141 {
142 lz = countLeadingZeros( rmask );
143 uldata >>= lz;
144 ulresult <<= 1;
145 count++;
146 ulresult |= ( uldata & 1 );
147 uldata >>= 1;
148 rmask <<= lz + 1;
149 }
150 ulresult <<= 64 - count; // 64 bits (maybe not use a constant 64 ...?)
151 ulresult = binaryReverse( ulresult );
152 return (int)slresult;
153}
154
155//---------------------------------------------------------------------------//
162KOKKOS_INLINE_FUNCTION
163constexpr uint64_t bitSpread( const uint64_t mask, const int data )
164{
165 uint64_t rmask = binaryReverse( mask );
166 int dat = data;
167 uint64_t result = 0;
168 unsigned char lz{ 0 };
169 while ( rmask )
170 {
171 lz = countLeadingZeros( rmask ) + 1;
172 result = result << lz | ( dat & 1 );
173 dat >>= 1, rmask <<= lz;
174 }
175 result = binaryReverse( result ) >> countLeadingZeros( mask );
176 return result;
177}

223 Key operator()( int tile_i, int tile_j, int tile_k ) const
224 {
225 return tile_i * _tile_num_j_mul_k + tile_j * _tile_num[2] + tile_k;
226 }

228 private:
229 // The index bounds of the tiles on the current MPI rank
230 Kokkos::Array<int, 3> _tile_num;

875 KOKKOS_INLINE_FUNCTION
876 void insert( int tile_i, int tile_j, int tile_k ) const
877 {
878 // Note: reallocation will be performed after insert
879 // tile ijk => tile key
880 key_type tile_key = ijk2key( tile_i, tile_j, tile_k );
881 // Try to insert the tile key into the map. Give it a dummy value for
882 // now.

884 // If the tile key was actually inserted, atomically increment the

888 _tile_table.value_at( insert_result.index() ) =
889 Kokkos::atomic_fetch_add( &( _tile_table_info( 0 ) ), 1 );
890 }
