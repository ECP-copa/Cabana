35template <class MeshType, class EntityType>
36struct L2G
37{
39 using mesh_type = MeshType;
40
42 static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;
43
45 using entity_type = EntityType;

48 Kokkos::Array<int, num_space_dim> local_own_min;
49
51 Kokkos::Array<int, num_space_dim> local_own_max;
52

55
57 Kokkos::Array<int, num_space_dim> global_num_entity;
58
60 Kokkos::Array<bool, num_space_dim> periodic;
61
63 Kokkos::Array<bool, num_space_dim> boundary_lo;
64
66 Kokkos::Array<bool, num_space_dim> boundary_hi;

69 L2G( const LocalGrid<MeshType>& local_grid )
70 {
71 // Local index set of owned entities.
72 auto local_own_space =
73 local_grid.indexSpace( Own(), EntityType(), Local() );
74
75 // Get the local owned min.
76 for ( std::size_t d = 0; d < num_space_dim; ++d )
77 local_own_min[d] = local_own_space.min( d );
78
79 // Get the local owned max.
80 for ( std::size_t d = 0; d < num_space_dim; ++d )
81 local_own_max[d] = local_own_space.max( d );
82
83 // Global index set of owned entities.
84 auto global_own_space =
85 local_grid.indexSpace( Own(), EntityType(), Global() );
86
87 // Get the global owned min.
88 for ( std::size_t d = 0; d < num_space_dim; ++d )
89 global_own_min[d] = global_own_space.min( d );
90
91 // Get the global grid.
92 const auto& global_grid = local_grid.globalGrid();
93
94 // Global number of entities.
95 for ( std::size_t d = 0; d < num_space_dim; ++d )
96 global_num_entity[d] =
97 global_grid.globalNumEntity( EntityType(), d );
98
99 // Periodicity
100 for ( std::size_t d = 0; d < num_space_dim; ++d )
101 periodic[d] = global_grid.isPeriodic( d );
102
103 // Determine if a block is on the low or high boundaries.
104 for ( std::size_t d = 0; d < num_space_dim; ++d )
105 {
106 boundary_lo[d] = global_grid.onLowBoundary( d );
107 boundary_hi[d] = global_grid.onHighBoundary( d );
108 }
109 }
110
112 KOKKOS_INLINE_FUNCTION
113 void operator()( const int lijk[num_space_dim],
114 int gijk[num_space_dim] ) const
115 {
116 for ( std::size_t d = 0; d < num_space_dim; ++d )
117 {
118 // Compute periodic wrap-around on low boundary.
119 if ( periodic[d] && lijk[d] < local_own_min[d] && boundary_lo[d] )
120 {
121 gijk[d] = global_num_entity[d] - local_own_min[d] + lijk[d];
122 }
123
124 // Compute periodic wrap-around on high boundary.
125 else if ( periodic[d] && local_own_max[d] <= lijk[d] &&
126 boundary_hi[d] )
127 {
128 gijk[d] = lijk[d] - local_own_max[d];
129 }
130
131 // Otherwise compute I indices as normal.
132 else
133 {
134 gijk[d] = lijk[d] - local_own_min[d] + global_own_min[d];
135 }
136 }
137 }

140 template <std::size_t NSD = num_space_dim>
141 KOKKOS_INLINE_FUNCTION std::enable_if_t<3 == NSD, void>
142 operator()( const int li, const int lj, const int lk, int& gi, int& gj,
143 int& gk ) const
144 {
145 int lijk[num_space_dim] = { li, lj, lk };
146 int gijk[num_space_dim];
147 this->operator()( lijk, gijk );
148 gi = gijk[0];
149 gj = gijk[1];
150 gk = gijk[2];
151 }
152
154 template <std::size_t NSD = num_space_dim>
155 KOKKOS_INLINE_FUNCTION std::enable_if_t<2 == NSD, void>
156 operator()( const int li, const int lj, int& gi, int& gj ) const
157 {
158 int lijk[num_space_dim] = { li, lj };
159 int gijk[num_space_dim];
160 this->operator()( lijk, gijk );
161 gi = gijk[0];
162 gj = gijk[1];
163 }
164};
