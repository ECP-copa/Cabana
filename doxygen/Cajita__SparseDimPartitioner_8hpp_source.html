175  std::array<int, num_space_dim>
176  ownedTilesPerDimension( MPI_Comm cart_comm ) const
177  {
178  // Get the Cartesian topology index of this rank.
179  std::array<int, num_space_dim> cart_rank;
180  int linear_rank;
181  MPI_Comm_rank( cart_comm, &linear_rank );
182  MPI_Cart_coords( cart_comm, linear_rank, num_space_dim,
183  cart_rank.data() );
184

187  auto rec_mirror = Kokkos::create_mirror_view_and_copy(
188  Kokkos::HostSpace(), _rectangle_partition_dev );
189  for ( std::size_t d = 0; d < num_space_dim; ++d )
190  tiles_per_dim[d] = rec_mirror( cart_rank[d] + 1, d ) -
191  rec_mirror( cart_rank[d], d );

340  template <class ParticlePosViewType, typename ArrayType, typename CellUnit>
341  void computeLocalWorkLoad( const ParticlePosViewType& view,
342  int particle_num,
343  const ArrayType& global_lower_corner,
344  const CellUnit dx )
345  {
346  resetWorkload();
347  // make a local copy
348  auto workload = _workload_per_tile;
349  Kokkos::Array<CellUnit, num_space_dim> lower_corner;
350  for ( std::size_t d = 0; d < num_space_dim; ++d )
351  {
352  lower_corner[d] = global_lower_corner[d];
353  }
354 
355  Kokkos::parallel_for(
356  "compute_local_workload_parpos",
357  Kokkos::RangePolicy<execution_space>( 0, particle_num ),
358  KOKKOS_LAMBDA( const int i ) {
359  int ti = static_cast<int>(
360  ( view( i, 0 ) - lower_corner[0] ) / dx - 0.5 ) >>
361  cell_bits_per_tile_dim;
362  int tj = static_cast<int>(
363  ( view( i, 1 ) - lower_corner[1] ) / dx - 0.5 ) >>
364  cell_bits_per_tile_dim;
365  int tz = static_cast<int>(
366  ( view( i, 2 ) - lower_corner[2] ) / dx - 0.5 ) >>
367  cell_bits_per_tile_dim;
368  Kokkos::atomic_increment( &workload( ti + 1, tj + 1, tz + 1 ) );
369  } );
370  }

403  void computeFullPrefixSum( MPI_Comm comm )
404  {
405  // local copy
406  auto workload = _workload_per_tile;
407  auto prefix_sum = _workload_prefix_sum;
408  int total_size = _workload_per_tile.size();
409 
410  // MPI all reduce: compute workload in all MPI ranks from the local
411  // workload matrix, save the results in _workload_prefix_sum
412  MPI_Allreduce( workload.data(), prefix_sum.data(), total_size, MPI_INT,
413  MPI_SUM, comm );
414 
415  // compute the prefix sum (in three dimensions)
416  // prefix sum in the dimension 0
417  for ( int j = 0;
418  j < static_cast<int>( _workload_prefix_sum.extent( 1 ) ); ++j )
419  for ( int k = 0;
420  k < static_cast<int>( _workload_prefix_sum.extent( 2 ) );
421  ++k )
422  Kokkos::parallel_scan(
423  "scan_prefix_sum_dim0",
424  Kokkos::RangePolicy<execution_space>(
425  0, _workload_prefix_sum.extent( 0 ) ),
426  KOKKOS_LAMBDA( const int i, int& update,
427  const bool final ) {
428  const float val_i = prefix_sum( i, j, k );
429  update += val_i;
430  if ( final )
431  {
432  prefix_sum( i, j, k ) = update;
433  }
434  } );
435 
436  // prefix sum in the dimension 1
437  for ( int i = 0;
438  i < static_cast<int>( _workload_prefix_sum.extent( 0 ) ); ++i )
439  for ( int k = 0;
440  k < static_cast<int>( _workload_prefix_sum.extent( 2 ) );
441  ++k )
442  Kokkos::parallel_scan(
443  "scan_prefix_sum_dim1",
444  Kokkos::RangePolicy<execution_space>(
445  0, _workload_prefix_sum.extent( 1 ) ),
446  KOKKOS_LAMBDA( const int j, int& update,
447  const bool final ) {
448  const float val_i = prefix_sum( i, j, k );
449  update += val_i;
450  if ( final )
451  {
452  prefix_sum( i, j, k ) = update;
453  }
454  } );
455 
456  // prefix sum in the dimension 2
457  for ( int i = 0;
458  i < static_cast<int>( _workload_prefix_sum.extent( 0 ) ); ++i )
459  for ( int j = 0;
460  j < static_cast<int>( _workload_prefix_sum.extent( 1 ) );
461  ++j )
462  Kokkos::parallel_scan(
463  "scan_prefix_sum_dim2",
464  Kokkos::RangePolicy<execution_space>(
465  0, _workload_prefix_sum.extent( 2 ) ),
466  KOKKOS_LAMBDA( const int k, int& update,
467  const bool final ) {
468  const float val_i = prefix_sum( i, j, k );
469  update += val_i;
470  if ( final )
471  {
472  prefix_sum( i, j, k ) = update;
473  }
474  } );
475  }

531  void optimizePartition( bool& is_changed, int iter_seed )
532  {
533  is_changed = false;
534  // loop over three dimensions, optimize the partition in dimension di
535  for ( int iter_id = iter_seed;
536  iter_id < iter_seed + static_cast<int>( num_space_dim );
537  ++iter_id )
538  {
539  int di = iter_id % num_space_dim;
540  // compute the dimensions that should be fixed (dj and dk)
541  int dj = ( di + 1 ) % num_space_dim;
542  int dk = ( di + 2 ) % num_space_dim;
543  auto rank_k = _ranks_per_dim[dk];
544 
545  auto rank = _ranks_per_dim[di];
546  auto rec_mirror = Kokkos::create_mirror_view_and_copy(
547  Kokkos::HostSpace(), _rectangle_partition_dev );
548  auto rec_partition = _rectangle_partition_dev;
549 
550  SubWorkloadFunctor<partition_view, workload_view>
551  compute_sub_workload( _rectangle_partition_dev,
552  _workload_prefix_sum );
553 
554  // compute average workload in the dimension di
555  Kokkos::View<int*, memory_space> ave_workload(
556  "ave_workload", _ranks_per_dim[dj] * _ranks_per_dim[dk] );
557  Kokkos::parallel_for(
558  "compute_average_workload",
559  Kokkos::RangePolicy<execution_space>(
560  0, _ranks_per_dim[dj] * _ranks_per_dim[dk] ),
561  KOKKOS_LAMBDA( uint32_t jnk ) {
562  // compute rank_id in the fixed dimensions
563  int j = static_cast<int>( jnk / rank_k );
564  int k = static_cast<int>( jnk % rank_k );
565  // compute the average workload with the partition of the
566  // two fixed dimensions
567  ave_workload( jnk ) =
568  compute_sub_workload( di, 0, rec_partition( rank, di ),
569  dj, j, dk, k ) /
570  rank;
571  } );
572 
573  // point_i: current partition position
574  int point_i = 1;
575  // last_point: the opimized position for the lask partition
576  int last_point = 0;
577  // current_workload: the workload between [last_point, point_i)
578  Kokkos::View<int*, memory_space> current_workload(
579  "current_workload", _ranks_per_dim[dj] * _ranks_per_dim[dk] );
580  for ( int current_rank = 1; current_rank < rank; current_rank++ )
581  {
582  int last_diff = __INT_MAX__;
583  while ( true )
584  {
585  // compute current workload between [last_point, point_i)
586  Kokkos::parallel_for(
587  "compute_current_workload",
588  Kokkos::RangePolicy<execution_space>(
589  0, _ranks_per_dim[dj] * _ranks_per_dim[dk] ),
590  KOKKOS_LAMBDA( uint32_t jnk ) {
591  int j = static_cast<int>( jnk / rank_k );
592  int k = static_cast<int>( jnk % rank_k );
593  current_workload( jnk ) = compute_sub_workload(
594  di, last_point, point_i, dj, j, dk, k );
595  } );
596  // compute the (w_jk^ave - w_jk^{last_point:point_i})
597  Kokkos::parallel_for(
598  "compute_diff",
599  Kokkos::RangePolicy<execution_space>(
600  0, _ranks_per_dim[dj] * _ranks_per_dim[dk] ),
601  KOKKOS_LAMBDA( uint32_t jnk ) {
602  auto wl =
603  current_workload( jnk ) - ave_workload( jnk );
604  wl *= wl;
605  current_workload( jnk ) = wl;
606  } );
607  // compute the sum of the difference in all rank_j*rank_k
608  // regions
609  int diff;
610  Kokkos::parallel_reduce(
611  "diff_reduce",
612  Kokkos::RangePolicy<execution_space>(
613  0, _ranks_per_dim[dj] * _ranks_per_dim[dk] ),
614  KOKKOS_LAMBDA( const int idx, int& update ) {
615  update += current_workload( idx );
616  },
617  diff );
618  // record the new optimal position
619  if ( diff <= last_diff )
620  {
621  // check if point_i reach the total_tile_num
622  if ( point_i == rec_mirror( rank, di ) )
623  {
624  rec_mirror( current_rank, di ) = point_i;
625  break;
626  }
627  last_diff = diff;
628  point_i++;
629  }
630  else
631  {
632  // final optimal position
633  if ( rec_mirror( current_rank, di ) != point_i - 1 )
634  {
635  rec_mirror( current_rank, di ) = point_i - 1;
636  is_changed = true;
637  }
638  last_point = point_i - 1;
639  break;
640  }
641  } // end while (optimization for the current rank)
642  } // end for (all partition/rank in the optimized dimension)
643  Kokkos::deep_copy( _rectangle_partition_dev, rec_mirror );
644  } // end for (3 dimensions)
645  }

764  {
765  int end[num_space_dim], start[num_space_dim];
766  end[dim_i] = i_end;
767  end[dim_j] = rec_partition( j + 1, dim_j );
768  end[dim_k] = rec_partition( k + 1, dim_k );
769 
770  start[dim_i] = i_start;
771  start[dim_j] = rec_partition( j, dim_j );
772  start[dim_k] = rec_partition( k, dim_k );
773 
774  // S[i][j][k] = S[i-1][j][k] + S[i][j-1][k] + S[i][j][k-1] -
775  // S[i-1][j-1][k]
776  // - S[i][j-1][k-1] - S[i-1][j][k-1] + S[i-1][j-1][k-1] + a[i][j][k]
777  return workload_prefix_sum( end[0], end[1], end[2] ) // S[i][j][k]
778  - workload_prefix_sum( start[0], end[1],
779  end[2] ) // S[i-1][j][k]
780  - workload_prefix_sum( end[0], start[1],
781  end[2] ) // S[i][j-1][k]
782  - workload_prefix_sum( end[0], end[1],
783  start[2] ) // S[i][j][k-1]
784  + workload_prefix_sum( start[0], start[1],
785  end[2] ) // S[i-1][j-1][k]
786  + workload_prefix_sum( end[0], start[1],
787  start[2] ) // S[i][j-1][k-1]
788  + workload_prefix_sum( start[0], end[1],
789  start[2] ) // S[i-1][j][k-1]
790  - workload_prefix_sum( start[0], start[1],
791  start[2] ); // S[i-1][j-1][k-1]
792  }
