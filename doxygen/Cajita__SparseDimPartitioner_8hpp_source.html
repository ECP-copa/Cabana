113  std::array<int, 3> ranksPerDimension( MPI_Comm comm )
114  {
115  int comm_size;
116  MPI_Comm_size( comm, &comm_size );
117 
118  std::array<int, 3> ranks_per_dim;
119  for ( std::size_t d = 0; d < 3; ++d )
120  ranks_per_dim[d] = 0;
121  MPI_Dims_create( comm_size, 3, ranks_per_dim.data() );
122 
123  std::copy( ranks_per_dim.begin(), ranks_per_dim.end(),
124  _ranks_per_dim.data() );
125 
126  return ranks_per_dim;
127  }

134  std::array<int, 3>
135  ranksPerDimension( MPI_Comm comm, const std::array<int, 3>& ) const override
136  {
137  std::array<int, 3> ranks_per_dim = {
138  _ranks_per_dim[0], _ranks_per_dim[1], _ranks_per_dim[2] };
139  int comm_size;
140  MPI_Comm_size( comm, &comm_size );
141  int nrank = 1;
142  for ( std::size_t d = 0; d < num_space_dim; ++d )
143  nrank *= _ranks_per_dim[d];
144  if ( comm_size != nrank )
145  throw std::runtime_error(
146  "SparsePartitioner ranks do not match comm size" );
147  return ranks_per_dim;
148  }

154  std::array<int, 3> ownedTilesPerDimension( MPI_Comm cart_comm ) const
155  {
156  // Get the Cartesian topology index of this rank.
157  std::array<int, 3> cart_rank;
158  int linear_rank;
159  MPI_Comm_rank( cart_comm, &linear_rank );
160  MPI_Cart_coords( cart_comm, linear_rank, 3, cart_rank.data() );
161 
162  // Get the tiles per dimension and the remainder.
163  std::array<int, 3> tiles_per_dim;
164  auto rec_mirror = Kokkos::create_mirror_view_and_copy(
165  Kokkos::HostSpace(), _rectangle_partition_dev );
166  for ( int d = 0; d < 3; ++d )
167  tiles_per_dim[d] = rec_mirror( cart_rank[d] + 1, d ) -
168  rec_mirror( cart_rank[d], d );
169  return tiles_per_dim;
170  }

176  std::array<int, 3> ownedCellsPerDimension( MPI_Comm cart_comm ) const
177  {
178  auto tiles_per_dim = ownedTilesPerDimension( cart_comm );
179  for ( int i = 0; i < 3; ++i )

181  // compute cells_per_dim from tiles_per_dim
182  tiles_per_dim[i] <<= cell_bits_per_tile_dim * 3;

261  template <class ParticlePosViewType, typename ArrayType, typename CellUnit>
262  void computeLocalWorkLoad( const ParticlePosViewType& view,
263  int particle_num,
264  const ArrayType& global_lower_corner,
265  const CellUnit dx )
266  {
267  resetWorkload();
268  // make a local copy
269  auto workload = _workload_per_tile;
270  Kokkos::Array<CellUnit, 3> lower_corner;
271  for ( int d = 0; d < 3; ++d )
272  {
273  lower_corner[d] = global_lower_corner[d];
274  }
275 
276  Kokkos::parallel_for(
277  "compute_local_workload_parpos",
278  Kokkos::RangePolicy<execution_space>( 0, particle_num ),
279  KOKKOS_LAMBDA( const int i ) {
280  int ti = static_cast<int>(
281  ( view( i, 0 ) - lower_corner[0] ) / dx - 0.5 ) >>
282  cell_bits_per_tile_dim;
283  int tj = static_cast<int>(
284  ( view( i, 1 ) - lower_corner[1] ) / dx - 0.5 ) >>
285  cell_bits_per_tile_dim;
286  int tz = static_cast<int>(
287  ( view( i, 2 ) - lower_corner[2] ) / dx - 0.5 ) >>
288  cell_bits_per_tile_dim;
289  Kokkos::atomic_increment( &workload( ti + 1, tj + 1, tz + 1 ) );
290  } );
291  }

298  template <class SparseMapType>
299  void computeLocalWorkLoad( const SparseMapType& sparseMap )
300  {
301  resetWorkload();
302  // make a local copy
303  auto workload = _workload_per_tile;
304  Kokkos::parallel_for(
305  "compute_local_workload_sparsmap",
306  Kokkos::RangePolicy<execution_space>( 0, sparseMap.capacity() ),
307  KOKKOS_LAMBDA( uint32_t i ) {
308  if ( sparseMap.valid_at( i ) )
309  {
310  auto key = sparseMap.key_at( i );
311  int ti, tj, tk;
312  sparseMap.key2ijk( key, ti, tj, tk );
313  Kokkos::atomic_increment(
314  &workload( ti + 1, tj + 1, tk + 1 ) );
315  }
316  } );
317  }

324  void computeFullPrefixSum( MPI_Comm comm )
325  {
326  // local copy
327  auto workload = _workload_per_tile;
328  auto prefix_sum = _workload_prefix_sum;
329  int total_size = _workload_per_tile.size();
330 
331  // MPI all reduce: compute workload in all MPI ranks from the local
332  // workload matrix, save the results in _workload_prefix_sum
333  MPI_Allreduce( workload.data(), prefix_sum.data(), total_size, MPI_INT,
334  MPI_SUM, comm );
335 
336  // compute the prefix sum (in three dimensions)
337  // prefix sum in the dimension 0
338  for ( int j = 0;
339  j < static_cast<int>( _workload_prefix_sum.extent( 1 ) ); ++j )
340  for ( int k = 0;
341  k < static_cast<int>( _workload_prefix_sum.extent( 2 ) );
342  ++k )
343  Kokkos::parallel_scan(
344  "scan_prefix_sum_dim0",
345  Kokkos::RangePolicy<execution_space>(
346  0, _workload_prefix_sum.extent( 0 ) ),
347  KOKKOS_LAMBDA( const int i, int& update,
348  const bool final ) {
349  const float val_i = prefix_sum( i, j, k );
350  update += val_i;
351  if ( final )
352  {
353  prefix_sum( i, j, k ) = update;
354  }
355  } );
356 
357  // prefix sum in the dimension 1
358  for ( int i = 0;
359  i < static_cast<int>( _workload_prefix_sum.extent( 0 ) ); ++i )
360  for ( int k = 0;
361  k < static_cast<int>( _workload_prefix_sum.extent( 2 ) );
362  ++k )
363  Kokkos::parallel_scan(
364  "scan_prefix_sum_dim1",
365  Kokkos::RangePolicy<execution_space>(
366  0, _workload_prefix_sum.extent( 1 ) ),
367  KOKKOS_LAMBDA( const int j, int& update,
368  const bool final ) {
369  const float val_i = prefix_sum( i, j, k );
370  update += val_i;
371  if ( final )
372  {
373  prefix_sum( i, j, k ) = update;
374  }
375  } );
376 
377  // prefix sum in the dimension 2
378  for ( int i = 0;
379  i < static_cast<int>( _workload_prefix_sum.extent( 0 ) ); ++i )
380  for ( int j = 0;
381  j < static_cast<int>( _workload_prefix_sum.extent( 1 ) );
382  ++j )
383  Kokkos::parallel_scan(
384  "scan_prefix_sum_dim2",
385  Kokkos::RangePolicy<execution_space>(
386  0, _workload_prefix_sum.extent( 2 ) ),
387  KOKKOS_LAMBDA( const int k, int& update,
388  const bool final ) {
389  const float val_i = prefix_sum( i, j, k );
390  update += val_i;
391  if ( final )
392  {
393  prefix_sum( i, j, k ) = update;
394  }
395  } );
396  }

452  void optimizePartition( bool& is_changed, int iter_seed )
453  {
454  is_changed = false;
455  // loop over three dimensions, optimize the partition in dimension di
456  for ( int iter_id = iter_seed; iter_id < iter_seed + 3; ++iter_id )
457  {
458  int di = iter_id % 3;
459  // compute the dimensions that should be fixed (dj and dk)
460  int dj = ( di + 1 ) % 3;
461  int dk = ( di + 2 ) % 3;
462  auto rank_k = _ranks_per_dim[dk];
463 
464  auto rank = _ranks_per_dim[di];
465  auto rec_mirror = Kokkos::create_mirror_view_and_copy(
466  Kokkos::HostSpace(), _rectangle_partition_dev );
467  auto rec_partition = _rectangle_partition_dev;
468 
469  SubWorkloadFunctor<partition_view, workload_view>
470  compute_sub_workload( _rectangle_partition_dev,
471  _workload_prefix_sum );
472 
473  // compute average workload in the dimension di
474  Kokkos::View<int*, memory_space> ave_workload(
475  "ave_workload", _ranks_per_dim[dj] * _ranks_per_dim[dk] );
476  Kokkos::parallel_for(
477  "compute_average_workload",
478  Kokkos::RangePolicy<execution_space>(
479  0, _ranks_per_dim[dj] * _ranks_per_dim[dk] ),
480  KOKKOS_LAMBDA( uint32_t jnk ) {
481  // compute rank_id in the fixed dimensions
482  int j = static_cast<int>( jnk / rank_k );
483  int k = static_cast<int>( jnk % rank_k );
484  // compute the average workload with the partition of the
485  // two fixed dimensions
486  ave_workload( jnk ) =
487  compute_sub_workload( di, 0, rec_partition( rank, di ),
488  dj, j, dk, k ) /
489  rank;
490  } );
491 
492  // point_i: current partition position
493  int point_i = 1;
494  // last_point: the opimized position for the lask partition
495  int last_point = 0;
496  // current_workload: the workload between [last_point, point_i)
497  Kokkos::View<int*, memory_space> current_workload(
498  "current_workload", _ranks_per_dim[dj] * _ranks_per_dim[dk] );
499  for ( int current_rank = 1; current_rank < rank; current_rank++ )
500  {
501  int last_diff = __INT_MAX__;
502  while ( true )
503  {
504  // compute current workload between [last_point, point_i)
505  Kokkos::parallel_for(
506  "compute_current_workload",
507  Kokkos::RangePolicy<execution_space>(
508  0, _ranks_per_dim[dj] * _ranks_per_dim[dk] ),
509  KOKKOS_LAMBDA( uint32_t jnk ) {
510  int j = static_cast<int>( jnk / rank_k );
511  int k = static_cast<int>( jnk % rank_k );
512  current_workload( jnk ) = compute_sub_workload(
513  di, last_point, point_i, dj, j, dk, k );
514  } );
515  // compute the (w_jk^ave - w_jk^{last_point:point_i})
516  Kokkos::parallel_for(
517  "compute_diff",
518  Kokkos::RangePolicy<execution_space>(
519  0, _ranks_per_dim[dj] * _ranks_per_dim[dk] ),
520  KOKKOS_LAMBDA( uint32_t jnk ) {
521  auto wl =
522  current_workload( jnk ) - ave_workload( jnk );
523  wl *= wl;
524  current_workload( jnk ) = wl;
525  } );
526  // compute the sum of the difference in all rank_j*rank_k
527  // regions
528  int diff;
529  Kokkos::parallel_reduce(
530  "diff_reduce",
531  Kokkos::RangePolicy<execution_space>(
532  0, _ranks_per_dim[dj] * _ranks_per_dim[dk] ),
533  KOKKOS_LAMBDA( const int idx, int& update ) {
534  update += current_workload( idx );
535  },
536  diff );
537  // record the new optimal position
538  if ( diff <= last_diff )
539  {
540  // check if point_i reach the total_tile_num
541  if ( point_i == rec_mirror( rank, di ) )
542  {
543  rec_mirror( current_rank, di ) = point_i;
544  break;
545  }
546  last_diff = diff;
547  point_i++;
548  }
549  else
550  {
551  // final optimal position
552  if ( rec_mirror( current_rank, di ) != point_i - 1 )
553  {
554  rec_mirror( current_rank, di ) = point_i - 1;
555  is_changed = true;
556  }
557  last_point = point_i - 1;
558  break;
559  }
560  } // end while (optimization for the current rank)
561  } // end for (all partition/rank in the optimized dimension)
562  Kokkos::deep_copy( _rectangle_partition_dev, rec_mirror );
563  } // end for (3 dimensions)
564  }

588  template <typename PartitionViewHost, typename WorkloadViewHost>
589  int currentRankWorkload( MPI_Comm cart_comm, PartitionViewHost& rec_view,
590  WorkloadViewHost& prefix_sum_view )
591  {
592  SubWorkloadFunctor<PartitionViewHost, WorkloadViewHost>
593  compute_sub_workload_host( rec_view, prefix_sum_view );
594 
595  // Get the Cartesian topology index of this rank.
596  Kokkos::Array<int, 3> cart_rank;
597  int linear_rank;
598  MPI_Comm_rank( cart_comm, &linear_rank );
599  MPI_Cart_coords( cart_comm, linear_rank, 3, cart_rank.data() );
600 
601  // compute total workload of the current rank
602  int workload_current_rank = compute_sub_workload_host(
603  0, rec_view( cart_rank[0], 0 ), rec_view( cart_rank[0] + 1, 0 ), 1,
604  cart_rank[1], 2, cart_rank[2] );
605 
606  return workload_current_rank;

631  prefix_sum_view.extent( 1 ) - 1,

641  float computeImbalanceFactor( MPI_Comm cart_comm )
642  {
643  auto rec_mirror = Kokkos::create_mirror_view_and_copy(
644  Kokkos::HostSpace(), _rectangle_partition_dev );
645  auto prefix_sum_mirror = Kokkos::create_mirror_view_and_copy(

662  template <typename PartitionView, typename WorkloadView>
663  struct SubWorkloadFunctor
664  {
665  PartitionView rec_partition;
666  WorkloadView workload_prefix_sum;
667 
668  SubWorkloadFunctor( PartitionView rec_par, WorkloadView pre_sum )
669  : rec_partition( rec_par )
670  , workload_prefix_sum( pre_sum )
671  {
672  }
673 
674  // compute the workload in region rounded by:
675  // [i_start, i_end) in dim_i
676  // [partition[j], partition[j+1]) in dim_j
677  // [partition[k], partition[k+1]) in dim_k
678  KOKKOS_INLINE_FUNCTION int operator()( int dim_i, int i_start,
679  int i_end, int dim_j, int j,
680  int dim_k, int k ) const
681  {
682  int end[3], start[3];
683  end[dim_i] = i_end;
684  end[dim_j] = rec_partition( j + 1, dim_j );
685  end[dim_k] = rec_partition( k + 1, dim_k );
686 
687  start[dim_i] = i_start;
688  start[dim_j] = rec_partition( j, dim_j );
689  start[dim_k] = rec_partition( k, dim_k );
690 
691  // S[i][j][k] = S[i-1][j][k] + S[i][j-1][k] + S[i][j][k-1] -
692  // S[i-1][j-1][k]
693  // - S[i][j-1][k-1] - S[i-1][j][k-1] + S[i-1][j-1][k-1] + a[i][j][k]
694  return workload_prefix_sum( end[0], end[1], end[2] ) // S[i][j][k]
695  - workload_prefix_sum( start[0], end[1],
696  end[2] ) // S[i-1][j][k]
697  - workload_prefix_sum( end[0], start[1],
698  end[2] ) // S[i][j-1][k]
699  - workload_prefix_sum( end[0], end[1],
700  start[2] ) // S[i][j][k-1]
701  + workload_prefix_sum( start[0], start[1],
702  end[2] ) // S[i-1][j-1][k]
703  + workload_prefix_sum( end[0], start[1],
704  start[2] ) // S[i][j-1][k-1]
705  + workload_prefix_sum( start[0], end[1],
706  start[2] ) // S[i-1][j][k-1]
707  - workload_prefix_sum( start[0], start[1],
708  start[2] ); // S[i-1][j-1][k-1]
709  }
710  };
