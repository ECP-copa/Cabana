169  {

275  template <class ParticlePosViewType, typename ArrayType, typename CellUnit>
276  void computeLocalWorkLoad( const ParticlePosViewType& view,
277  int particle_num,
278  const ArrayType& global_lower_corner,
279  const CellUnit dx )
280  {
281  resetWorkload();
282  // make a local copy
283  auto workload = _workload_per_tile;
284  Kokkos::Array<CellUnit, 3> lower_corner;
285  for ( int d = 0; d < 3; ++d )
286  {
287  lower_corner[d] = global_lower_corner[d];
288  }
289 
290  Kokkos::parallel_for(
291  "compute_local_workload_parpos",
292  Kokkos::RangePolicy<execution_space>( 0, particle_num ),
293  KOKKOS_LAMBDA( const int i ) {
294  int ti = static_cast<int>(
295  ( view( i, 0 ) - lower_corner[0] ) / dx - 0.5 ) >>
296  cell_bits_per_tile_dim;
297  int tj = static_cast<int>(
298  ( view( i, 1 ) - lower_corner[1] ) / dx - 0.5 ) >>
299  cell_bits_per_tile_dim;
300  int tz = static_cast<int>(
301  ( view( i, 2 ) - lower_corner[2] ) / dx - 0.5 ) >>
302  cell_bits_per_tile_dim;
303  Kokkos::atomic_increment( &workload( ti + 1, tj + 1, tz + 1 ) );
304  } );
305  }

338  void computeFullPrefixSum( MPI_Comm comm )
339  {
340  // local copy
341  auto workload = _workload_per_tile;
342  auto prefix_sum = _workload_prefix_sum;
343  int total_size = _workload_per_tile.size();
344 
345  // MPI all reduce: compute workload in all MPI ranks from the local
346  // workload matrix, save the results in _workload_prefix_sum
347  MPI_Allreduce( workload.data(), prefix_sum.data(), total_size, MPI_INT,
348  MPI_SUM, comm );
349 
350  // compute the prefix sum (in three dimensions)
351  // prefix sum in the dimension 0
352  for ( int j = 0;
353  j < static_cast<int>( _workload_prefix_sum.extent( 1 ) ); ++j )
354  for ( int k = 0;
355  k < static_cast<int>( _workload_prefix_sum.extent( 2 ) );
356  ++k )
357  Kokkos::parallel_scan(
358  "scan_prefix_sum_dim0",
359  Kokkos::RangePolicy<execution_space>(
360  0, _workload_prefix_sum.extent( 0 ) ),
361  KOKKOS_LAMBDA( const int i, int& update,
362  const bool final ) {
363  const float val_i = prefix_sum( i, j, k );
364  update += val_i;
365  if ( final )
366  {
367  prefix_sum( i, j, k ) = update;
368  }
369  } );
370 
371  // prefix sum in the dimension 1
372  for ( int i = 0;
373  i < static_cast<int>( _workload_prefix_sum.extent( 0 ) ); ++i )
374  for ( int k = 0;
375  k < static_cast<int>( _workload_prefix_sum.extent( 2 ) );
376  ++k )
377  Kokkos::parallel_scan(
378  "scan_prefix_sum_dim1",
379  Kokkos::RangePolicy<execution_space>(
380  0, _workload_prefix_sum.extent( 1 ) ),
381  KOKKOS_LAMBDA( const int j, int& update,
382  const bool final ) {
383  const float val_i = prefix_sum( i, j, k );
384  update += val_i;
385  if ( final )
386  {
387  prefix_sum( i, j, k ) = update;
388  }
389  } );
390 
391  // prefix sum in the dimension 2
392  for ( int i = 0;
393  i < static_cast<int>( _workload_prefix_sum.extent( 0 ) ); ++i )
394  for ( int j = 0;
395  j < static_cast<int>( _workload_prefix_sum.extent( 1 ) );
396  ++j )
397  Kokkos::parallel_scan(
398  "scan_prefix_sum_dim2",
399  Kokkos::RangePolicy<execution_space>(
400  0, _workload_prefix_sum.extent( 2 ) ),
401  KOKKOS_LAMBDA( const int k, int& update,
402  const bool final ) {
403  const float val_i = prefix_sum( i, j, k );
404  update += val_i;
405  if ( final )
406  {
407  prefix_sum( i, j, k ) = update;
408  }
409  } );
410  }

466  void optimizePartition( bool& is_changed, int iter_seed )
467  {
468  is_changed = false;
469  // loop over three dimensions, optimize the partition in dimension di
470  for ( int iter_id = iter_seed; iter_id < iter_seed + 3; ++iter_id )
471  {
472  int di = iter_id % 3;
473  // compute the dimensions that should be fixed (dj and dk)
474  int dj = ( di + 1 ) % 3;
475  int dk = ( di + 2 ) % 3;
476  auto rank_k = _ranks_per_dim[dk];
477 
478  auto rank = _ranks_per_dim[di];
479  auto rec_mirror = Kokkos::create_mirror_view_and_copy(
480  Kokkos::HostSpace(), _rectangle_partition_dev );
481  auto rec_partition = _rectangle_partition_dev;
482 
483  SubWorkloadFunctor<partition_view, workload_view>
484  compute_sub_workload( _rectangle_partition_dev,
485  _workload_prefix_sum );
486 
487  // compute average workload in the dimension di
488  Kokkos::View<int*, memory_space> ave_workload(
489  "ave_workload", _ranks_per_dim[dj] * _ranks_per_dim[dk] );
490  Kokkos::parallel_for(
491  "compute_average_workload",
492  Kokkos::RangePolicy<execution_space>(
493  0, _ranks_per_dim[dj] * _ranks_per_dim[dk] ),
494  KOKKOS_LAMBDA( uint32_t jnk ) {
495  // compute rank_id in the fixed dimensions
496  int j = static_cast<int>( jnk / rank_k );
497  int k = static_cast<int>( jnk % rank_k );
498  // compute the average workload with the partition of the
499  // two fixed dimensions
500  ave_workload( jnk ) =
501  compute_sub_workload( di, 0, rec_partition( rank, di ),
502  dj, j, dk, k ) /
503  rank;
504  } );
505 
506  // point_i: current partition position
507  int point_i = 1;
508  // last_point: the opimized position for the lask partition
509  int last_point = 0;
510  // current_workload: the workload between [last_point, point_i)
511  Kokkos::View<int*, memory_space> current_workload(
512  "current_workload", _ranks_per_dim[dj] * _ranks_per_dim[dk] );
513  for ( int current_rank = 1; current_rank < rank; current_rank++ )
514  {
515  int last_diff = __INT_MAX__;
516  while ( true )
517  {
518  // compute current workload between [last_point, point_i)
519  Kokkos::parallel_for(
520  "compute_current_workload",
521  Kokkos::RangePolicy<execution_space>(
522  0, _ranks_per_dim[dj] * _ranks_per_dim[dk] ),
523  KOKKOS_LAMBDA( uint32_t jnk ) {
524  int j = static_cast<int>( jnk / rank_k );
525  int k = static_cast<int>( jnk % rank_k );
526  current_workload( jnk ) = compute_sub_workload(
527  di, last_point, point_i, dj, j, dk, k );
528  } );
529  // compute the (w_jk^ave - w_jk^{last_point:point_i})
530  Kokkos::parallel_for(
531  "compute_diff",
532  Kokkos::RangePolicy<execution_space>(
533  0, _ranks_per_dim[dj] * _ranks_per_dim[dk] ),
534  KOKKOS_LAMBDA( uint32_t jnk ) {
535  auto wl =
536  current_workload( jnk ) - ave_workload( jnk );
537  wl *= wl;
538  current_workload( jnk ) = wl;
539  } );
540  // compute the sum of the difference in all rank_j*rank_k
541  // regions
542  int diff;
543  Kokkos::parallel_reduce(
544  "diff_reduce",
545  Kokkos::RangePolicy<execution_space>(
546  0, _ranks_per_dim[dj] * _ranks_per_dim[dk] ),
547  KOKKOS_LAMBDA( const int idx, int& update ) {
548  update += current_workload( idx );
549  },
550  diff );
551  // record the new optimal position
552  if ( diff <= last_diff )
553  {
554  // check if point_i reach the total_tile_num
555  if ( point_i == rec_mirror( rank, di ) )
556  {
557  rec_mirror( current_rank, di ) = point_i;
558  break;
559  }
560  last_diff = diff;
561  point_i++;
562  }
563  else
564  {
565  // final optimal position
566  if ( rec_mirror( current_rank, di ) != point_i - 1 )
567  {
568  rec_mirror( current_rank, di ) = point_i - 1;
569  is_changed = true;
570  }
571  last_point = point_i - 1;
572  break;
573  }
574  } // end while (optimization for the current rank)
575  } // end for (all partition/rank in the optimized dimension)
576  Kokkos::deep_copy( _rectangle_partition_dev, rec_mirror );
577  } // end for (3 dimensions)
578  }

693  KOKKOS_INLINE_FUNCTION int operator()( int dim_i, int i_start,
694  int i_end, int dim_j, int j,
695  int dim_k, int k ) const
696  {
697  int end[3], start[3];
698  end[dim_i] = i_end;
699  end[dim_j] = rec_partition( j + 1, dim_j );
700  end[dim_k] = rec_partition( k + 1, dim_k );
701 
702  start[dim_i] = i_start;
703  start[dim_j] = rec_partition( j, dim_j );
704  start[dim_k] = rec_partition( k, dim_k );
705 
706  // S[i][j][k] = S[i-1][j][k] + S[i][j-1][k] + S[i][j][k-1] -
707  // S[i-1][j-1][k]
708  // - S[i][j-1][k-1] - S[i-1][j][k-1] + S[i-1][j-1][k-1] + a[i][j][k]
709  return workload_prefix_sum( end[0], end[1], end[2] ) // S[i][j][k]
710  - workload_prefix_sum( start[0], end[1],
711  end[2] ) // S[i-1][j][k]
712  - workload_prefix_sum( end[0], start[1],
713  end[2] ) // S[i][j-1][k]
714  - workload_prefix_sum( end[0], end[1],
715  start[2] ) // S[i][j][k-1]
716  + workload_prefix_sum( start[0], start[1],
717  end[2] ) // S[i-1][j-1][k]
718  + workload_prefix_sum( end[0], start[1],
719  start[2] ) // S[i][j-1][k-1]
720  + workload_prefix_sum( start[0], end[1],
721  start[2] ) // S[i-1][j][k-1]
722  - workload_prefix_sum( start[0], start[1],
723  start[2] ); // S[i-1][j-1][k-1]
724  }
725  };

735  // represent the rectangle partition in each dimension
736  // with form [0, p_1, ..., p_n, cell_num], n = rank num in current
