115 MPI_Comm comm, float max_workload_coeff, int workload_num,
116 int num_step_rebalance,
117 const std::array<int, num_space_dim>& ranks_per_dim,

140 std::array<int, num_space_dim> ranksPerDimension( MPI_Comm comm )
141 {
142 int comm_size;
143 MPI_Comm_size( comm, &comm_size );
144
145 std::array<int, num_space_dim> ranks_per_dim;
146 for ( std::size_t d = 0; d < num_space_dim; ++d )
147 ranks_per_dim[d] = 0;
148 MPI_Dims_create( comm_size, num_space_dim, ranks_per_dim.data() );
149
150 std::copy( ranks_per_dim.begin(), ranks_per_dim.end(),
151 _ranks_per_dim.data() );
152
153 return ranks_per_dim;
154 }

161 std::array<int, num_space_dim>
162 ranksPerDimension( MPI_Comm comm,
163 const std::array<int, num_space_dim>& ) const override
164 {
165 std::array<int, num_space_dim> ranks_per_dim = {
166 _ranks_per_dim[0], _ranks_per_dim[1], _ranks_per_dim[2] };
167 int comm_size;
168 MPI_Comm_size( comm, &comm_size );
169 int nrank = 1;
170 for ( std::size_t d = 0; d < num_space_dim; ++d )
171 nrank *= _ranks_per_dim[d];
172 if ( comm_size != nrank )
173 throw std::runtime_error(
174 "SparsePartitioner ranks do not match comm size" );
175 return ranks_per_dim;
176 }

182 std::array<int, num_space_dim>
183 ownedTilesPerDimension( MPI_Comm cart_comm ) const
184 {
185 // Get the Cartesian topology index of this rank.
186 std::array<int, num_space_dim> cart_rank;
187 int linear_rank;
188 MPI_Comm_rank( cart_comm, &linear_rank );
189 MPI_Cart_coords( cart_comm, linear_rank, num_space_dim,
190 cart_rank.data() );
191
192 // Get the tiles per dimension and the remainder.
193 std::array<int, num_space_dim> tiles_per_dim;
194 auto rec_mirror = Kokkos::create_mirror_view_and_copy(
195 Kokkos::HostSpace(), _rectangle_partition_dev );
196 for ( std::size_t d = 0; d < num_space_dim; ++d )
197 tiles_per_dim[d] = rec_mirror( cart_rank[d] + 1, d ) -
198 rec_mirror( cart_rank[d], d );
199 return tiles_per_dim;

281 void initializeRecPartition( std::vector<int>& rec_partition_i,
282 std::vector<int>& rec_partition_j,
283 std::vector<int>& rec_partition_k )
284 {
285
286 int max_size = 0;
287 for ( std::size_t d = 0; d < num_space_dim; ++d )
288 max_size =
289 max_size < _ranks_per_dim[d] ? _ranks_per_dim[d] : max_size;
290
291 typedef typename execution_space::array_layout layout;
292 Kokkos::View<int* [num_space_dim], layout, Kokkos::HostSpace>
293 rectangle_partition( "rectangle_partition_host", max_size + 1 );
294
295 for ( int id = 0; id < _ranks_per_dim[0] + 1; ++id )
296 rectangle_partition( id, 0 ) = rec_partition_i[id];
297
298 for ( int id = 0; id < _ranks_per_dim[1] + 1; ++id )
299 rectangle_partition( id, 1 ) = rec_partition_j[id];
300
301 for ( int id = 0; id < _ranks_per_dim[2] + 1; ++id )
302 rectangle_partition( id, 2 ) = rec_partition_k[id];
303
304 _rectangle_partition_dev = Kokkos::create_mirror_view_and_copy(
305 memory_space(), rectangle_partition );

347 template <class ParticlePosViewType, typename ArrayType, typename CellUnit>
348 void computeLocalWorkLoad( const ParticlePosViewType& view,
349 int particle_num,
350 const ArrayType& global_lower_corner,
351 const CellUnit dx )
352 {
353 resetWorkload();
354 // make a local copy
355 auto workload = _workload_per_tile;
356 Kokkos::Array<CellUnit, num_space_dim> lower_corner;
357 for ( std::size_t d = 0; d < num_space_dim; ++d )
358 {
359 lower_corner[d] = global_lower_corner[d];
360 }
361
362 Kokkos::parallel_for(
363 "compute_local_workload_parpos",
364 Kokkos::RangePolicy<execution_space>( 0, particle_num ),
365 KOKKOS_LAMBDA( const int i ) {
366 int ti = static_cast<int>(
367 ( view( i, 0 ) - lower_corner[0] ) / dx - 0.5 ) >>
368 cell_bits_per_tile_dim;
369 int tj = static_cast<int>(
370 ( view( i, 1 ) - lower_corner[1] ) / dx - 0.5 ) >>
371 cell_bits_per_tile_dim;
372 int tz = static_cast<int>(
373 ( view( i, 2 ) - lower_corner[2] ) / dx - 0.5 ) >>
374 cell_bits_per_tile_dim;
375 Kokkos::atomic_increment( &workload( ti + 1, tj + 1, tz + 1 ) );
376 } );
377 Kokkos::fence();
378 }

385 template <class SparseMapType>
386 void computeLocalWorkLoad( const SparseMapType& sparseMap )
387 {
388 resetWorkload();
389 // make a local copy
390 auto workload = _workload_per_tile;
391 Kokkos::parallel_for(
392 "compute_local_workload_sparsmap",
393 Kokkos::RangePolicy<execution_space>( 0, sparseMap.capacity() ),
394 KOKKOS_LAMBDA( uint32_t i ) {
395 if ( sparseMap.valid_at( i ) )
396 {
397 auto key = sparseMap.key_at( i );
398 int ti, tj, tk;
399 sparseMap.key2ijk( key, ti, tj, tk );
400 Kokkos::atomic_increment(
401 &workload( ti + 1, tj + 1, tk + 1 ) );
402 }
403 } );
404 Kokkos::fence();
405 }

412 void computeFullPrefixSum( MPI_Comm comm )
413 {
414 // local copy
415 auto workload = _workload_per_tile;
416 auto prefix_sum = _workload_prefix_sum;
417 int total_size = _workload_per_tile.size();
418
419 // MPI all reduce: compute workload in all MPI ranks from the local
420 // workload matrix, save the results in _workload_prefix_sum
421 MPI_Allreduce( workload.data(), prefix_sum.data(), total_size, MPI_INT,
422 MPI_SUM, comm );
423 MPI_Barrier( comm );
424
425 // compute the prefix sum (in three dimensions)
426 // prefix sum in the dimension 0
427 for ( int j = 0;
428 j < static_cast<int>( _workload_prefix_sum.extent( 1 ) ); ++j )
429 for ( int k = 0;
430 k < static_cast<int>( _workload_prefix_sum.extent( 2 ) );
431 ++k )
432 Kokkos::parallel_scan(
433 "scan_prefix_sum_dim0",
434 Kokkos::RangePolicy<execution_space>(
435 0, _workload_prefix_sum.extent( 0 ) ),
436 KOKKOS_LAMBDA( const int i, int& update,
437 const bool final ) {
438 const float val_i = prefix_sum( i, j, k );
439 update += val_i;
440 if ( final )
441 {
442 prefix_sum( i, j, k ) = update;
443 }
444 } );
445 Kokkos::fence();
446
447 // prefix sum in the dimension 1
448 for ( int i = 0;
449 i < static_cast<int>( _workload_prefix_sum.extent( 0 ) ); ++i )
450 for ( int k = 0;
451 k < static_cast<int>( _workload_prefix_sum.extent( 2 ) );
452 ++k )
453 Kokkos::parallel_scan(
454 "scan_prefix_sum_dim1",
455 Kokkos::RangePolicy<execution_space>(
456 0, _workload_prefix_sum.extent( 1 ) ),
457 KOKKOS_LAMBDA( const int j, int& update,
458 const bool final ) {
459 const float val_i = prefix_sum( i, j, k );
460 update += val_i;
461 if ( final )
462 {
463 prefix_sum( i, j, k ) = update;
464 }
465 } );
466 Kokkos::fence();
467
468 // prefix sum in the dimension 2
469 for ( int i = 0;
470 i < static_cast<int>( _workload_prefix_sum.extent( 0 ) ); ++i )
471 for ( int j = 0;
472 j < static_cast<int>( _workload_prefix_sum.extent( 1 ) );
473 ++j )
474 Kokkos::parallel_scan(
475 "scan_prefix_sum_dim2",
476 Kokkos::RangePolicy<execution_space>(
477 0, _workload_prefix_sum.extent( 2 ) ),
478 KOKKOS_LAMBDA( const int k, int& update,
479 const bool final ) {
480 const float val_i = prefix_sum( i, j, k );
481 update += val_i;
482 if ( final )
483 {
484 prefix_sum( i, j, k ) = update;
485 }
486 } );
487 Kokkos::fence();
488 }

500 template <class ParticlePosViewType, typename ArrayType, typename CellUnit>
501 int optimizePartition( const ParticlePosViewType& view, int particle_num,
502 const ArrayType& global_lower_corner,
503 const CellUnit dx, MPI_Comm comm )
504 {
505 computeLocalWorkLoad( view, particle_num, global_lower_corner, dx );
506 MPI_Barrier( comm );
507
508 computeFullPrefixSum( comm );
509 MPI_Barrier( comm );
510
511 // each iteration covers partitioner optization in all three dimensions
512 // (with a random dim sequence)
513 for ( int i = 0; i < _max_optimize_iteration; ++i )
514 {
515 bool is_changed = false; // record changes in current iteration
516 bool dim_covered[3] = { false, false, false };
517 for ( int d = 0; d < 3; ++d )
518 {
519 int random_dim_id = std::rand() % num_space_dim;
520 while ( dim_covered[random_dim_id] )
521 random_dim_id = std::rand() % num_space_dim;
522
523 bool is_dim_changed = false; // record changes in current dim
524 optimizePartition( is_dim_changed, random_dim_id );
525
526 // update control info
527 is_changed = is_changed || is_dim_changed;
528 dim_covered[random_dim_id] = true;
529 }
530 // return if the current partition is optimal
531 if ( !is_changed )
532 return i;
533 }
534 return _max_optimize_iteration;
535 }

582 void optimizePartition( bool& is_changed, int iter_seed )
583 {
584 is_changed = false;
585 // loop over three dimensions, optimize the partition in dimension di
586 for ( int iter_id = iter_seed;
587 iter_id < iter_seed + static_cast<int>( num_space_dim );
588 ++iter_id )
589 {
590 int di = iter_id % num_space_dim;
591 // compute the dimensions that should be fixed (dj and dk)
592 int dj = ( di + 1 ) % num_space_dim;
593 int dk = ( di + 2 ) % num_space_dim;
594 auto rank_k = _ranks_per_dim[dk];
595
596 auto rank = _ranks_per_dim[di];
597 auto rec_mirror = Kokkos::create_mirror_view_and_copy(
598 Kokkos::HostSpace(), _rectangle_partition_dev );
599 auto rec_partition = _rectangle_partition_dev;
600
601 SubWorkloadFunctor<partition_view, workload_view>
602 compute_sub_workload( _rectangle_partition_dev,
603 _workload_prefix_sum );
604
605 // compute average workload in the dimension di
606 Kokkos::View<int*, memory_space> ave_workload(
607 "ave_workload", _ranks_per_dim[dj] * _ranks_per_dim[dk] );
608 Kokkos::parallel_for(
609 "compute_average_workload",
610 Kokkos::RangePolicy<execution_space>(
611 0, _ranks_per_dim[dj] * _ranks_per_dim[dk] ),
612 KOKKOS_LAMBDA( uint32_t jnk ) {
613 // compute rank_id in the fixed dimensions
614 int j = static_cast<int>( jnk / rank_k );
615 int k = static_cast<int>( jnk % rank_k );
616 // compute the average workload with the partition of the
617 // two fixed dimensions
618 ave_workload( jnk ) =
619 compute_sub_workload( di, 0, rec_partition( rank, di ),
620 dj, j, dk, k ) /
621 rank;
622 } );
623 Kokkos::fence();
624
625 // point_i: current partition position
626 int point_i = 1;
627 // equal_start_point: register the beginning pos of potentially
628 // equivalent partitions
629 int equal_start_point = 1;
630 // last_point: the opimized position for the lask partition
631 int last_point = 0;
632 // current_workload: the workload between [last_point, point_i)
633 Kokkos::View<int*, memory_space> current_workload(
634 "current_workload", _ranks_per_dim[dj] * _ranks_per_dim[dk] );
635 for ( int current_rank = 1; current_rank < rank; current_rank++ )
636 {
637 int last_diff = __INT_MAX__;
638 while ( true )
639 {
640 // compute current workload between [last_point, point_i)
641 Kokkos::parallel_for(
642 "compute_current_workload",
643 Kokkos::RangePolicy<execution_space>(
644 0, _ranks_per_dim[dj] * _ranks_per_dim[dk] ),
645 KOKKOS_LAMBDA( uint32_t jnk ) {
646 int j = static_cast<int>( jnk / rank_k );
647 int k = static_cast<int>( jnk % rank_k );
648 current_workload( jnk ) = compute_sub_workload(
649 di, last_point, point_i, dj, j, dk, k );
650 } );
651 Kokkos::fence();
652
653 // compute the (w_jk^ave - w_jk^{last_point:point_i})
654 Kokkos::parallel_for(
655 "compute_diff",
656 Kokkos::RangePolicy<execution_space>(
657 0, _ranks_per_dim[dj] * _ranks_per_dim[dk] ),
658 KOKKOS_LAMBDA( uint32_t jnk ) {
659 auto wl =
660 current_workload( jnk ) - ave_workload( jnk );
661 // compute absolute diff (rather than squares to
662 // avoid potential overflow)
663 // TODO: update when Kokkos::abs() available
664 wl = wl > 0 ? wl : -wl;
665 current_workload( jnk ) = wl;
666 } );
667 Kokkos::fence();
668
669 // compute the sum of the difference in all rank_j*rank_k
670 // regions
671 int diff;
672 Kokkos::parallel_reduce(
673 "diff_reduce",
674 Kokkos::RangePolicy<execution_space>(
675 0, _ranks_per_dim[dj] * _ranks_per_dim[dk] ),
676 KOKKOS_LAMBDA( const int idx, int& update ) {
677 update += current_workload( idx );
678 },
679 diff );
680 Kokkos::fence();
681
682 // record the new optimal position
683 if ( diff <= last_diff )
684 {
685 // register starting points of potentially equivalent
686 // partitions
687 if ( diff != last_diff )
688 equal_start_point = point_i;
689
690 // check if point_i reach the total_tile_num
691 if ( point_i == rec_mirror( rank, di ) )
692 {
693 rec_mirror( current_rank, di ) = point_i;
694 break;
695 }
696
697 last_diff = diff;
698 point_i++;
699 }
700 else
701 {
702 // final optimal position - middle position of all
703 // potentially equivalent partitions
704 if ( rec_mirror( current_rank, di ) !=
705 ( point_i - 1 + equal_start_point ) / 2 )
706 {
707 rec_mirror( current_rank, di ) =
708 ( point_i - 1 + equal_start_point ) / 2;
709 is_changed = true;
710 }
711 last_point = point_i - 1;
712 break;
713 }
714 } // end while (optimization for the current rank)
715 } // end for (all partition/rank in the optimized dimension)
716 Kokkos::deep_copy( _rectangle_partition_dev, rec_mirror );
717 } // end for (3 dimensions)
718 }

796 float computeImbalanceFactor( MPI_Comm cart_comm )
797 {
798 auto rec_mirror = Kokkos::create_mirror_view_and_copy(
799 Kokkos::HostSpace(), _rectangle_partition_dev );
800 auto prefix_sum_mirror = Kokkos::create_mirror_view_and_copy(
801 Kokkos::HostSpace(), _workload_prefix_sum );
802

804 currentRankWorkload( cart_comm, rec_mirror, prefix_sum_mirror );
805 int workload_ave_rank = averageRankWorkload( prefix_sum_mirror );
806
807 return static_cast<float>( workload_current_rank ) /
808 static_cast<float>( workload_ave_rank );

835 int i_end, int dim_j, int j,

837 {
838 int end[num_space_dim], start[num_space_dim];
839 end[dim_i] = i_end;
840 end[dim_j] = rec_partition( j + 1, dim_j );
841 end[dim_k] = rec_partition( k + 1, dim_k );
842
843 start[dim_i] = i_start;
844 start[dim_j] = rec_partition( j, dim_j );
845 start[dim_k] = rec_partition( k, dim_k );
846
847 // S[i][j][k] = S[i-1][j][k] + S[i][j-1][k] + S[i][j][k-1] -
848 // S[i-1][j-1][k]
849 // - S[i][j-1][k-1] - S[i-1][j][k-1] + S[i-1][j-1][k-1] + a[i][j][k]
850 return workload_prefix_sum( end[0], end[1], end[2] ) // S[i][j][k]
851 - workload_prefix_sum( start[0], end[1],
852 end[2] ) // S[i-1][j][k]
853 - workload_prefix_sum( end[0], start[1],
854 end[2] ) // S[i][j-1][k]
855 - workload_prefix_sum( end[0], end[1],
856 start[2] ) // S[i][j][k-1]
857 + workload_prefix_sum( start[0], start[1],
858 end[2] ) // S[i-1][j-1][k]
859 + workload_prefix_sum( end[0], start[1],
860 start[2] ) // S[i][j-1][k-1]
861 + workload_prefix_sum( start[0], end[1],
862 start[2] ) // S[i-1][j][k-1]
863 - workload_prefix_sum( start[0], start[1],
864 start[2] ); // S[i-1][j-1][k-1]
865 }
866 };
