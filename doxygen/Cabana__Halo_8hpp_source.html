286 void apply() override
287 {
288 Kokkos::Profiling::pushRegion( "Cabana::gather" );
289
290 // Get the buffers and particle data (local copies for lambdas below).
291 auto send_buffer = this->getSendBuffer();
292 auto recv_buffer = this->getReceiveBuffer();
293 auto aosoa = this->getData();
294
295 // Get the steering vector for the sends.
296 auto steering = _halo.getExportSteering();
297 // Gather from the local data into a tuple-contiguous send buffer.
298 auto gather_send_buffer_func = KOKKOS_LAMBDA( const std::size_t i )
299 {
300 send_buffer( i ) = aosoa.getTuple( steering( i ) );
301 };
302 Kokkos::parallel_for( "Cabana::gather::gather_send_buffer",
303 _send_policy, gather_send_buffer_func );
304 Kokkos::fence();
305
306 // The halo has it's own communication space so choose any mpi tag.
307 const int mpi_tag = 2345;
308
309 // Post non-blocking receives.
310 int num_n = _halo.numNeighbor();
311 std::vector<MPI_Request> requests( num_n );
312 std::pair<std::size_t, std::size_t> recv_range = { 0, 0 };
313 for ( int n = 0; n < num_n; ++n )
314 {
315 recv_range.second = recv_range.first + _halo.numImport( n );
316
317 auto recv_subview = Kokkos::subview( recv_buffer, recv_range );
318
319 MPI_Irecv( recv_subview.data(),
320 recv_subview.size() * sizeof( data_type ), MPI_BYTE,
321 _halo.neighborRank( n ), mpi_tag, _halo.comm(),
322 &( requests[n] ) );
323
324 recv_range.first = recv_range.second;
325 }
326
327 // Do blocking sends.
328 std::pair<std::size_t, std::size_t> send_range = { 0, 0 };
329 for ( int n = 0; n < num_n; ++n )
330 {
331 send_range.second = send_range.first + _halo.numExport( n );
332
333 auto send_subview = Kokkos::subview( send_buffer, send_range );
334
335 MPI_Send( send_subview.data(),
336 send_subview.size() * sizeof( data_type ), MPI_BYTE,
337 _halo.neighborRank( n ), mpi_tag, _halo.comm() );
338
339 send_range.first = send_range.second;
340 }
341
342 // Wait on non-blocking receives.
343 std::vector<MPI_Status> status( num_n );
344 const int ec =
345 MPI_Waitall( requests.size(), requests.data(), status.data() );
346 if ( MPI_SUCCESS != ec )
347 throw std::logic_error( "Failed MPI Communication" );
348
349 // Extract the receive buffer into the ghosted elements.
350 std::size_t num_local = _halo.numLocal();
351 auto extract_recv_buffer_func = KOKKOS_LAMBDA( const std::size_t i )
352 {
353 std::size_t ghost_idx = i + num_local;
354 aosoa.setTuple( ghost_idx, recv_buffer( i ) );
355 };
356 Kokkos::parallel_for( "Cabana::gather::extract_recv_buffer",
357 _recv_policy, extract_recv_buffer_func );
358 Kokkos::fence();
359
360 // Barrier before completing to ensure synchronization.
361 MPI_Barrier( _halo.comm() );
362
363 Kokkos::Profiling::popRegion();
364 }
