230 void gather( const Halo_t& halo, AoSoA_t& aosoa,
231  typename std::enable_if<( is_halo<Halo_t>::value &&
232  is_aosoa<AoSoA_t>::value ),
233  int>::type* = 0 )
234 {
235  // Check that the AoSoA is the right size.
236  if ( aosoa.size() != halo.numLocal() + halo.numGhost() )
237  throw std::runtime_error( "AoSoA is the wrong size for gather!" );
238 
239  // Allocate a send buffer.
240  Kokkos::View<typename AoSoA_t::tuple_type*, typename Halo_t::memory_space>
241  send_buffer(
242  Kokkos::ViewAllocateWithoutInitializing( "halo_send_buffer" ),
243  halo.totalNumExport() );
244 
245  // Get the steering vector for the sends.
246  auto steering = halo.getExportSteering();
247 
248  // Gather from the local data into a tuple-contiguous send buffer.
249  auto gather_send_buffer_func = KOKKOS_LAMBDA( const std::size_t i )
250  {
251  send_buffer( i ) = aosoa.getTuple( steering( i ) );
252  };
253  Kokkos::RangePolicy<typename Halo_t::execution_space>
254  gather_send_buffer_policy( 0, halo.totalNumExport() );
255  Kokkos::parallel_for( "Cabana::gather::gather_send_buffer",
256  gather_send_buffer_policy, gather_send_buffer_func );
257  Kokkos::fence();
258 
259  // Allocate a receive buffer.
260  Kokkos::View<typename AoSoA_t::tuple_type*, typename Halo_t::memory_space>
261  recv_buffer(
262  Kokkos::ViewAllocateWithoutInitializing( "halo_recv_buffer" ),
263  halo.totalNumImport() );
264 
265  // The halo has it's own communication space so choose any mpi tag.
266  const int mpi_tag = 2345;
267 
268  // Post non-blocking receives.
269  int num_n = halo.numNeighbor();
270  std::vector<MPI_Request> requests( num_n );
271  std::pair<std::size_t, std::size_t> recv_range = { 0, 0 };
272  for ( int n = 0; n < num_n; ++n )
273  {
274  recv_range.second = recv_range.first + halo.numImport( n );
275 
276  auto recv_subview = Kokkos::subview( recv_buffer, recv_range );
277 
278  MPI_Irecv( recv_subview.data(),
279  recv_subview.size() * sizeof( typename AoSoA_t::tuple_type ),
280  MPI_BYTE, halo.neighborRank( n ), mpi_tag, halo.comm(),
281  &( requests[n] ) );
282 
283  recv_range.first = recv_range.second;
284  }
285 
286  // Do blocking sends.
287  std::pair<std::size_t, std::size_t> send_range = { 0, 0 };
288  for ( int n = 0; n < num_n; ++n )
289  {
290  send_range.second = send_range.first + halo.numExport( n );
291 
292  auto send_subview = Kokkos::subview( send_buffer, send_range );
293 
294  MPI_Send( send_subview.data(),
295  send_subview.size() * sizeof( typename AoSoA_t::tuple_type ),
296  MPI_BYTE, halo.neighborRank( n ), mpi_tag, halo.comm() );
297 
298  send_range.first = send_range.second;
299  }
300 
301  // Wait on non-blocking receives.
302  std::vector<MPI_Status> status( num_n );
303  const int ec =
304  MPI_Waitall( requests.size(), requests.data(), status.data() );
305  if ( MPI_SUCCESS != ec )
306  throw std::logic_error( "Failed MPI Communication" );
307 
308  // Extract the receive buffer into the ghosted elements.
309  std::size_t num_local = halo.numLocal();
310  auto extract_recv_buffer_func = KOKKOS_LAMBDA( const std::size_t i )
311  {
312  std::size_t ghost_idx = i + num_local;
313  aosoa.setTuple( ghost_idx, recv_buffer( i ) );
314  };
315  Kokkos::RangePolicy<typename Halo_t::execution_space>
316  extract_recv_buffer_policy( 0, halo.totalNumImport() );
317  Kokkos::parallel_for( "Cabana::gather::extract_recv_buffer",
318  extract_recv_buffer_policy,
319  extract_recv_buffer_func );
320  Kokkos::fence();
321 
322  // Barrier before completing to ensure synchronization.
323  MPI_Barrier( halo.comm() );
324 }
