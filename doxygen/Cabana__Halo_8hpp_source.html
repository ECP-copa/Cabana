286 template <class ExecutionSpace>
287 void apply( ExecutionSpace )
288 {
289 Kokkos::Profiling::pushRegion( "Cabana::gather" );
290
291 // Get the buffers and particle data (local copies for lambdas below).
292 auto send_buffer = this->getSendBuffer();
293 auto recv_buffer = this->getReceiveBuffer();
294 auto aosoa = this->getData();
295
296 // Get the steering vector for the sends.
297 auto steering = _halo.getExportSteering();
298 // Gather from the local data into a tuple-contiguous send buffer.
299 auto gather_send_buffer_func = KOKKOS_LAMBDA( const std::size_t i )
300 {
301 send_buffer( i ) = aosoa.getTuple( steering( i ) );
302 };
303 Kokkos::RangePolicy<ExecutionSpace> send_policy( 0, _send_size );
304 Kokkos::parallel_for( "Cabana::gather::gather_send_buffer", send_policy,
305 gather_send_buffer_func );
306 Kokkos::fence();
307
308 // The halo has it's own communication space so choose any mpi tag.
309 const int mpi_tag = 2345;
310
311 // Post non-blocking receives.
312 int num_n = _halo.numNeighbor();
313 std::vector<MPI_Request> requests( num_n );
314 std::pair<std::size_t, std::size_t> recv_range = { 0, 0 };
315 for ( int n = 0; n < num_n; ++n )
316 {
317 recv_range.second = recv_range.first + _halo.numImport( n );
318
319 auto recv_subview = Kokkos::subview( recv_buffer, recv_range );
320
321 MPI_Irecv( recv_subview.data(),
322 recv_subview.size() * sizeof( data_type ), MPI_BYTE,
323 _halo.neighborRank( n ), mpi_tag, _halo.comm(),
324 &( requests[n] ) );
325
326 recv_range.first = recv_range.second;
327 }
328
329 // Do blocking sends.
330 std::pair<std::size_t, std::size_t> send_range = { 0, 0 };
331 for ( int n = 0; n < num_n; ++n )
332 {
333 send_range.second = send_range.first + _halo.numExport( n );
334
335 auto send_subview = Kokkos::subview( send_buffer, send_range );
336
337 MPI_Send( send_subview.data(),
338 send_subview.size() * sizeof( data_type ), MPI_BYTE,
339 _halo.neighborRank( n ), mpi_tag, _halo.comm() );
340
341 send_range.first = send_range.second;
342 }
343
344 // Wait on non-blocking receives.
345 std::vector<MPI_Status> status( num_n );
346 const int ec =
347 MPI_Waitall( requests.size(), requests.data(), status.data() );
348 if ( MPI_SUCCESS != ec )
349 throw std::logic_error( "Failed MPI Communication" );
350
351 // Extract the receive buffer into the ghosted elements.
352 std::size_t num_local = _halo.numLocal();
353 auto extract_recv_buffer_func = KOKKOS_LAMBDA( const std::size_t i )
354 {
355 std::size_t ghost_idx = i + num_local;
356 aosoa.setTuple( ghost_idx, recv_buffer( i ) );
357 };
358 Kokkos::RangePolicy<ExecutionSpace> recv_policy( 0, _recv_size );
359 Kokkos::parallel_for( "Cabana::gather::extract_recv_buffer",
360 recv_policy, extract_recv_buffer_func );
361 Kokkos::fence();
362
363 // Barrier before completing to ensure synchronization.
364 MPI_Barrier( _halo.comm() );
365
366 Kokkos::Profiling::popRegion();
367 }

369 void apply() override { apply( execution_space{} ); }
