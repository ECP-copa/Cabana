286  void apply() override
287  {
288  // Get the buffers and particle data (local copies for lambdas below).
289  auto send_buffer = this->getSendBuffer();
290  auto recv_buffer = this->getReceiveBuffer();
291  auto aosoa = this->getData();
292 
293  // Get the steering vector for the sends.
294  auto steering = _halo.getExportSteering();
295  // Gather from the local data into a tuple-contiguous send buffer.
296  auto gather_send_buffer_func = KOKKOS_LAMBDA( const std::size_t i )
297  {
298  send_buffer( i ) = aosoa.getTuple( steering( i ) );
299  };
300  Kokkos::parallel_for( "Cabana::gather::gather_send_buffer",
301  _send_policy, gather_send_buffer_func );
302  Kokkos::fence();
303 
304  // The halo has it's own communication space so choose any mpi tag.
305  const int mpi_tag = 2345;
306 
307  // Post non-blocking receives.
308  int num_n = _halo.numNeighbor();
309  std::vector<MPI_Request> requests( num_n );
310  std::pair<std::size_t, std::size_t> recv_range = { 0, 0 };
311  for ( int n = 0; n < num_n; ++n )
312  {
313  recv_range.second = recv_range.first + _halo.numImport( n );
314 
315  auto recv_subview = Kokkos::subview( recv_buffer, recv_range );
316 
317  MPI_Irecv( recv_subview.data(),
318  recv_subview.size() * sizeof( data_type ), MPI_BYTE,
319  _halo.neighborRank( n ), mpi_tag, _halo.comm(),
320  &( requests[n] ) );
321 
322  recv_range.first = recv_range.second;
323  }
324 
325  // Do blocking sends.
326  std::pair<std::size_t, std::size_t> send_range = { 0, 0 };
327  for ( int n = 0; n < num_n; ++n )
328  {
329  send_range.second = send_range.first + _halo.numExport( n );
330 
331  auto send_subview = Kokkos::subview( send_buffer, send_range );
332 
333  MPI_Send( send_subview.data(),
334  send_subview.size() * sizeof( data_type ), MPI_BYTE,
335  _halo.neighborRank( n ), mpi_tag, _halo.comm() );
336 
337  send_range.first = send_range.second;
338  }
339 
340  // Wait on non-blocking receives.
341  std::vector<MPI_Status> status( num_n );
342  const int ec =
343  MPI_Waitall( requests.size(), requests.data(), status.data() );
344  if ( MPI_SUCCESS != ec )
345  throw std::logic_error( "Failed MPI Communication" );
346 
347  // Extract the receive buffer into the ghosted elements.
348  std::size_t num_local = _halo.numLocal();
349  auto extract_recv_buffer_func = KOKKOS_LAMBDA( const std::size_t i )
350  {
351  std::size_t ghost_idx = i + num_local;
352  aosoa.setTuple( ghost_idx, recv_buffer( i ) );
353  };
354  Kokkos::parallel_for( "Cabana::gather::extract_recv_buffer",
355  _recv_policy, extract_recv_buffer_func );
356  Kokkos::fence();
357 
358  // Barrier before completing to ensure synchronization.
359  MPI_Barrier( _halo.comm() );
360  }
