226 template <class Halo_t, class AoSoA_t>
227 void gather( const Halo_t& halo, AoSoA_t& aosoa,
228  typename std::enable_if<( is_halo<Halo_t>::value &&
229  is_aosoa<AoSoA_t>::value ),
230  int>::type* = 0 )
231 {
232  // Check that the AoSoA is the right size.
233  if ( aosoa.size() != halo.numLocal() + halo.numGhost() )
234  throw std::runtime_error( "AoSoA is the wrong size for gather!" );
235 
236  // Allocate a send buffer.
237  Kokkos::View<typename AoSoA_t::tuple_type*, typename Halo_t::memory_space>
238  send_buffer(
239  Kokkos::ViewAllocateWithoutInitializing( "halo_send_buffer" ),
240  halo.totalNumExport() );
241 
242  // Get the steering vector for the sends.
243  auto steering = halo.getExportSteering();
244 
245  // Gather from the local data into a tuple-contiguous send buffer.
246  auto gather_send_buffer_func = KOKKOS_LAMBDA( const std::size_t i )
247  {
248  send_buffer( i ) = aosoa.getTuple( steering( i ) );
249  };
250  Kokkos::RangePolicy<typename Halo_t::execution_space>
251  gather_send_buffer_policy( 0, halo.totalNumExport() );
252  Kokkos::parallel_for( "Cabana::gather::gather_send_buffer",
253  gather_send_buffer_policy, gather_send_buffer_func );
254  Kokkos::fence();
255 
256  // Allocate a receive buffer.
257  Kokkos::View<typename AoSoA_t::tuple_type*, typename Halo_t::memory_space>
258  recv_buffer(
259  Kokkos::ViewAllocateWithoutInitializing( "halo_recv_buffer" ),
260  halo.totalNumImport() );
261 
262  // The halo has it's own communication space so choose any mpi tag.
263  const int mpi_tag = 2345;
264 
265  // Post non-blocking receives.
266  int num_n = halo.numNeighbor();
267  std::vector<MPI_Request> requests( num_n );
268  std::pair<std::size_t, std::size_t> recv_range = { 0, 0 };
269  for ( int n = 0; n < num_n; ++n )
270  {
271  recv_range.second = recv_range.first + halo.numImport( n );
272 
273  auto recv_subview = Kokkos::subview( recv_buffer, recv_range );
274 
275  MPI_Irecv( recv_subview.data(),
276  recv_subview.size() * sizeof( typename AoSoA_t::tuple_type ),
277  MPI_BYTE, halo.neighborRank( n ), mpi_tag, halo.comm(),
278  &( requests[n] ) );
279 
280  recv_range.first = recv_range.second;
281  }
282 
283  // Do blocking sends.
284  std::pair<std::size_t, std::size_t> send_range = { 0, 0 };
285  for ( int n = 0; n < num_n; ++n )
286  {
287  send_range.second = send_range.first + halo.numExport( n );
288 
289  auto send_subview = Kokkos::subview( send_buffer, send_range );
290 
291  MPI_Send( send_subview.data(),
292  send_subview.size() * sizeof( typename AoSoA_t::tuple_type ),
293  MPI_BYTE, halo.neighborRank( n ), mpi_tag, halo.comm() );
294 
295  send_range.first = send_range.second;
296  }
297 
298  // Wait on non-blocking receives.
299  std::vector<MPI_Status> status( num_n );
300  const int ec =
301  MPI_Waitall( requests.size(), requests.data(), status.data() );
302  if ( MPI_SUCCESS != ec )
303  throw std::logic_error( "Failed MPI Communication" );
304 
305  // Extract the receive buffer into the ghosted elements.
306  std::size_t num_local = halo.numLocal();
307  auto extract_recv_buffer_func = KOKKOS_LAMBDA( const std::size_t i )
308  {
309  std::size_t ghost_idx = i + num_local;
310  aosoa.setTuple( ghost_idx, recv_buffer( i ) );
311  };
312  Kokkos::RangePolicy<typename Halo_t::execution_space>
313  extract_recv_buffer_policy( 0, halo.totalNumImport() );
314  Kokkos::parallel_for( "Cabana::gather::extract_recv_buffer",
315  extract_recv_buffer_policy,
316  extract_recv_buffer_func );
317  Kokkos::fence();
318 
319  // Barrier before completing to ensure synchronization.
320  MPI_Barrier( halo.comm() );
321 }

362  // Get the raw slice data.

365  // Allocate a send buffer. Note this one is layout right so the components
366  // are consecutive.

368  typename Halo_t::memory_space>
