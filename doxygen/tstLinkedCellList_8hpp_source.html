13 #include <Cabana_LinkedCellList.hpp>

15 #include <gtest/gtest.h>

17 namespace Test
18 {
19 //---------------------------------------------------------------------------//
20 void testLinkedList()
21 {
22  // Make an AoSoA with positions and ijk cell ids.
23  enum MyFields { Position = 0, CellId = 1 };
24  using DataTypes = Cabana::MemberTypes<double[3],int[3]>;
25  using AoSoA_t = Cabana::AoSoA<DataTypes,TEST_MEMSPACE>;
26  using size_type = typename AoSoA_t::memory_space::size_type;
27  int num_p = 1000;
28  AoSoA_t aosoa( num_p );
29 
30  // Set the problem so each particle lives in the center of a cell on a
31  // regular grid of cell size 1 and total size 10x10x10. We are making them
32  // in the reverse order we expect the sort to happen. The sort binary
33  // operator should order by i first and k last.
34  int nx = 10;
35  double dx = 1.0;
36  double x_min = 0.0;
37  double x_max = x_min + nx * dx;
38  auto pos = aosoa.slice<Position>();
39  auto cell_id = aosoa.slice<CellId>();
40  std::size_t particle_id = 0;
41  for ( int k = 0; k < nx; ++k )
42  {
43  for ( int j = 0; j < nx; ++j )
44  {
45  for ( int i = 0; i < nx; ++i, ++particle_id )
46  {
47  cell_id( particle_id, 0 ) = i;
48  cell_id( particle_id, 1 ) = j;
49  cell_id( particle_id, 2 ) = k;
50 
51  pos( particle_id, 0 ) = x_min + (i + 0.5) * dx;
52  pos( particle_id, 1 ) = x_min + (j + 0.5) * dx;
53  pos( particle_id, 2 ) = x_min + (k + 0.5) * dx;
54  }
55  }
56  }
57 
58  // Create a grid.
59  double grid_delta[3] = {dx,dx,dx};
60  double grid_min[3] = {x_min,x_min,x_min};
61  double grid_max[3] = {x_max,x_max,x_max};
62 
63  // Bin and permute the particles in the grid. First do this by only
64  // operating on a subset of the particles.
65  {
66  std::size_t begin = 250;
67  std::size_t end = 750;
68  Cabana::LinkedCellList<typename AoSoA_t::memory_space>
69  cell_list( aosoa.slice<Position>(), begin, end,
70  grid_delta, grid_min, grid_max );
71  Cabana::permute( cell_list, aosoa );
72 
73  // Checking the binning.
74  EXPECT_EQ( cell_list.totalBins(), nx*nx*nx );
75  EXPECT_EQ( cell_list.numBin(0), nx );
76  EXPECT_EQ( cell_list.numBin(1), nx );
77  EXPECT_EQ( cell_list.numBin(2), nx );
78 
79  // The order should be reversed with the i index moving the slowest
80  // for those that are actually in the binning range. Do this pass
81  // first. We do this by looping through in the sorted order and check
82  // those that had original indices in the sorting range.
83  particle_id = 0;
84  for ( int i = 0; i < nx; ++i )
85  {
86  for ( int j = 0; j < nx; ++j )
87  {
88  for ( int k = 0; k < nx; ++k )
89  {
90  std::size_t original_id = i + j * nx + k * nx * nx;
91  if ( begin <= original_id && original_id < end )
92  {
93  // Get what should be the local id of the particle in
94  // the newly sorted decomposition. We are looping
95  // through this in k-fastest order (the indexing of
96  // the grid cells) and therefore should get the
97  // particles in their sorted order.
98  int sort_id = begin + particle_id;
99 
100  // Back calculate what we think the ijk indices of
101  // the particle are based on k-fastest ordering.
102  int grid_id = i * nx * nx + j * nx + k;
103  int grid_i = grid_id / (nx*nx);
104  int grid_j = (grid_id / nx) % nx;
105  int grid_k = grid_id % nx;
106 
107  // Check the indices of the particle
108  EXPECT_EQ( cell_id( sort_id, 0 ), grid_i );
109  EXPECT_EQ( cell_id( sort_id, 1 ), grid_j );
110  EXPECT_EQ( cell_id( sort_id, 2 ), grid_k );
111 
112  // Check that we binned the particle and got the right
113  // offset.
114  EXPECT_EQ( cell_list.binSize(i,j,k), 1 );
115  EXPECT_EQ( cell_list.binOffset(i,j,k),
116  size_type(particle_id) );
117 
118  // Increment the particle id.
119  ++particle_id;
120  }
121  }
122  }
123  }
124 
125  // For those that are outside the binned range things should be
126  // unchanged and the bins should be unchanged.
127  particle_id = 0;
128  for ( int k = 0; k < nx; ++k )
129  {
130  for ( int j = 0; j < nx; ++j )
131  {
132  for ( int i = 0; i < nx; ++i, ++particle_id )
133  {
134  if ( begin > particle_id || particle_id >= end )
135  {
136  EXPECT_EQ( cell_id( particle_id, 0 ), i );
137  EXPECT_EQ( cell_id( particle_id, 1 ), j );
138  EXPECT_EQ( cell_id( particle_id, 2 ), k );
139  EXPECT_EQ( cell_list.binSize(i,j,k), 0 );
140  }
141  }
142  }
143  }
144  }
145 
146  // Now bin and permute all of the particles.
147  {
148  Cabana::LinkedCellList<typename AoSoA_t::memory_space>
149  cell_list( aosoa.slice<Position>(), grid_delta, grid_min, grid_max );
150  Cabana::permute( cell_list, aosoa );
151 
152  // Checking the binning. The order should be reversed with the i index
153  // moving the slowest.
154  EXPECT_EQ( cell_list.totalBins(), nx*nx*nx );
155  EXPECT_EQ( cell_list.numBin(0), nx );
156  EXPECT_EQ( cell_list.numBin(1), nx );
157  EXPECT_EQ( cell_list.numBin(2), nx );
158  particle_id = 0;
159  for ( int i = 0; i < nx; ++i )
160  {
161  for ( int j = 0; j < nx; ++j )
162  {
163  for ( int k = 0; k < nx; ++k, ++particle_id )
164  {
165  EXPECT_EQ( cell_id( particle_id, 0 ), i );
166  EXPECT_EQ( cell_id( particle_id, 1 ), j );
167  EXPECT_EQ( cell_id( particle_id, 2 ), k );
168  EXPECT_EQ( cell_list.cardinalBinIndex(i,j,k), particle_id );
169  EXPECT_EQ( cell_list.binSize(i,j,k), 1 );
170  EXPECT_EQ( cell_list.binOffset(i,j,k),
171  size_type(particle_id) );
172  }
173  }
174  }
175  }
176 }
