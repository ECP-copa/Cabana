124 SparseDimPartitioner(
125 MPI_Comm comm, float max_workload_coeff, int workload_num,
126 int num_step_rebalance,
127 const std::array<int, num_space_dim>& ranks_per_dim,
128 const std::array<int, num_space_dim>& global_cells_per_dim,
129 int max_optimize_iteration = 10 )
130 : _workload_threshold(
131 static_cast<int>( max_workload_coeff * workload_num ) )
132 , _num_step_rebalance( num_step_rebalance )
133 , _max_optimize_iteration( max_optimize_iteration )

136 std::copy( ranks_per_dim.begin(), ranks_per_dim.end(),
137 _ranks_per_dim.data() );

150 std::array<int, num_space_dim> ranksPerDimension( MPI_Comm comm )
151 {
152 int comm_size;
153 MPI_Comm_size( comm, &comm_size );
154
155 std::array<int, num_space_dim> ranks_per_dim;
156 for ( std::size_t d = 0; d < num_space_dim; ++d )
157 ranks_per_dim[d] = 0;
158 MPI_Dims_create( comm_size, num_space_dim, ranks_per_dim.data() );
159
160 std::copy( ranks_per_dim.begin(), ranks_per_dim.end(),
161 _ranks_per_dim.data() );
162
163 return ranks_per_dim;
164 }

171 std::array<int, num_space_dim>
172 ranksPerDimension( MPI_Comm comm,
173 const std::array<int, num_space_dim>& ) const override
174 {
175 std::array<int, num_space_dim> ranks_per_dim = {
176 _ranks_per_dim[0], _ranks_per_dim[1], _ranks_per_dim[2] };
177 int comm_size;
178 MPI_Comm_size( comm, &comm_size );
179 int nrank = 1;
180 for ( std::size_t d = 0; d < num_space_dim; ++d )
181 nrank *= _ranks_per_dim[d];
182 if ( comm_size != nrank )
183 throw std::runtime_error(
184 "SparsePartitioner ranks do not match comm size" );
185 return ranks_per_dim;
186 }

192 std::array<int, num_space_dim>
193 ownedTilesPerDimension( MPI_Comm cart_comm ) const
194 {
195 // Get the Cartesian topology index of this rank.
196 std::array<int, num_space_dim> cart_rank;
197 int linear_rank;
198 MPI_Comm_rank( cart_comm, &linear_rank );
199 MPI_Cart_coords( cart_comm, linear_rank, num_space_dim,
200 cart_rank.data() );
201
202 // Get the tiles per dimension and the remainder.
203 std::array<int, num_space_dim> tiles_per_dim;
204 auto rec_mirror = Kokkos::create_mirror_view_and_copy(
205 Kokkos::HostSpace(), _rectangle_partition_dev );
206 for ( std::size_t d = 0; d < num_space_dim; ++d )
207 tiles_per_dim[d] = rec_mirror( cart_rank[d] + 1, d ) -
208 rec_mirror( cart_rank[d], d );
209 return tiles_per_dim;
210 }

291 void initializeRecPartition( std::vector<int>& rec_partition_i,
292 std::vector<int>& rec_partition_j,
293 std::vector<int>& rec_partition_k )
294 {
295 int max_size = 0;
296 for ( std::size_t d = 0; d < num_space_dim; ++d )
297 max_size =
298 max_size < _ranks_per_dim[d] ? _ranks_per_dim[d] : max_size;
299
300 typedef typename execution_space::array_layout layout;
301 Kokkos::View<int* [num_space_dim], layout, Kokkos::HostSpace>
302 rectangle_partition( "rectangle_partition_host", max_size + 1 );
303
304 for ( int id = 0; id < _ranks_per_dim[0] + 1; ++id )
305 rectangle_partition( id, 0 ) = rec_partition_i[id];
306
307 for ( int id = 0; id < _ranks_per_dim[1] + 1; ++id )
308 rectangle_partition( id, 1 ) = rec_partition_j[id];
309
310 for ( int id = 0; id < _ranks_per_dim[2] + 1; ++id )
311 rectangle_partition( id, 2 ) = rec_partition_k[id];
312
313 _rectangle_partition_dev = Kokkos::create_mirror_view_and_copy(
314 memory_space(), rectangle_partition );
315 }

356 template <class ParticlePosViewType, typename ArrayType, typename CellUnit>
357 void computeLocalWorkLoad( const ParticlePosViewType& view,
358 int particle_num,
359 const ArrayType& global_lower_corner,
360 const CellUnit dx )
361 {
362 resetWorkload();
363 // make a local copy
364 auto workload = _workload_per_tile;
365 Kokkos::Array<CellUnit, num_space_dim> lower_corner;
366 for ( std::size_t d = 0; d < num_space_dim; ++d )
367 {
368 lower_corner[d] = global_lower_corner[d];
369 }
370
371 Kokkos::parallel_for(
372 "compute_local_workload_parpos",
373 Kokkos::RangePolicy<execution_space>( 0, particle_num ),
374 KOKKOS_LAMBDA( const int i ) {
375 int ti = static_cast<int>(
376 ( view( i, 0 ) - lower_corner[0] ) / dx - 0.5 ) >>
377 cell_bits_per_tile_dim;
378 int tj = static_cast<int>(
379 ( view( i, 1 ) - lower_corner[1] ) / dx - 0.5 ) >>
380 cell_bits_per_tile_dim;
381 int tz = static_cast<int>(
382 ( view( i, 2 ) - lower_corner[2] ) / dx - 0.5 ) >>
383 cell_bits_per_tile_dim;
384 Kokkos::atomic_increment( &workload( ti + 1, tj + 1, tz + 1 ) );
385 } );
386 Kokkos::fence();
387 }

394 template <class SparseMapType>
395 void computeLocalWorkLoad( const SparseMapType& sparseMap )
396 {
397 resetWorkload();
398 // make a local copy
399 auto workload = _workload_per_tile;
400 Kokkos::parallel_for(
401 "compute_local_workload_sparsmap",
402 Kokkos::RangePolicy<execution_space>( 0, sparseMap.capacity() ),
403 KOKKOS_LAMBDA( uint32_t i ) {
404 if ( sparseMap.valid_at( i ) )
405 {
406 auto key = sparseMap.key_at( i );
407 int ti, tj, tk;
408 sparseMap.key2ijk( key, ti, tj, tk );
409 Kokkos::atomic_increment(
410 &workload( ti + 1, tj + 1, tk + 1 ) );
411 }
412 } );
413 Kokkos::fence();
414 }

421 void computeFullPrefixSum( MPI_Comm comm )
422 {
423 // local copy
424 auto workload = _workload_per_tile;
425 auto prefix_sum = _workload_prefix_sum;
426 int total_size = _workload_per_tile.size();
427
428 // MPI all reduce: compute workload in all MPI ranks from the local
429 // workload matrix, save the results in _workload_prefix_sum
430 MPI_Allreduce( workload.data(), prefix_sum.data(), total_size, MPI_INT,
431 MPI_SUM, comm );
432 MPI_Barrier( comm );
433
434 // compute the prefix sum (in three dimensions)
435 // prefix sum in the dimension 0
436 for ( int j = 0;
437 j < static_cast<int>( _workload_prefix_sum.extent( 1 ) ); ++j )
438 for ( int k = 0;
439 k < static_cast<int>( _workload_prefix_sum.extent( 2 ) );
440 ++k )
441 Kokkos::parallel_scan(
442 "scan_prefix_sum_dim0",
443 Kokkos::RangePolicy<execution_space>(
444 0, _workload_prefix_sum.extent( 0 ) ),
445 KOKKOS_LAMBDA( const int i, int& update,
446 const bool final ) {
447 const float val_i = prefix_sum( i, j, k );
448 update += val_i;
449 if ( final )
450 {
451 prefix_sum( i, j, k ) = update;
452 }
453 } );
454 Kokkos::fence();
455
456 // prefix sum in the dimension 1
457 for ( int i = 0;
458 i < static_cast<int>( _workload_prefix_sum.extent( 0 ) ); ++i )
459 for ( int k = 0;
460 k < static_cast<int>( _workload_prefix_sum.extent( 2 ) );
461 ++k )
462 Kokkos::parallel_scan(
463 "scan_prefix_sum_dim1",
464 Kokkos::RangePolicy<execution_space>(
465 0, _workload_prefix_sum.extent( 1 ) ),
466 KOKKOS_LAMBDA( const int j, int& update,
467 const bool final ) {
468 const float val_i = prefix_sum( i, j, k );
469 update += val_i;
470 if ( final )
471 {
472 prefix_sum( i, j, k ) = update;
473 }
474 } );
475 Kokkos::fence();
476
477 // prefix sum in the dimension 2
478 for ( int i = 0;
479 i < static_cast<int>( _workload_prefix_sum.extent( 0 ) ); ++i )
480 for ( int j = 0;
481 j < static_cast<int>( _workload_prefix_sum.extent( 1 ) );
482 ++j )
483 Kokkos::parallel_scan(
484 "scan_prefix_sum_dim2",
485 Kokkos::RangePolicy<execution_space>(
486 0, _workload_prefix_sum.extent( 2 ) ),
487 KOKKOS_LAMBDA( const int k, int& update,
488 const bool final ) {
489 const float val_i = prefix_sum( i, j, k );
490 update += val_i;
491 if ( final )
492 {
493 prefix_sum( i, j, k ) = update;
494 }
495 } );
496 Kokkos::fence();
497 }

509 template <class ParticlePosViewType, typename ArrayType, typename CellUnit>
510 int optimizePartition( const ParticlePosViewType& view, int particle_num,
511 const ArrayType& global_lower_corner,
512 const CellUnit dx, MPI_Comm comm )
513 {
514 computeLocalWorkLoad( view, particle_num, global_lower_corner, dx );
515 MPI_Barrier( comm );
516
517 computeFullPrefixSum( comm );
518 MPI_Barrier( comm );
519
520 // each iteration covers partitioner optization in all three dimensions
521 // (with a random dim sequence)
522 for ( int i = 0; i < _max_optimize_iteration; ++i )
523 {
524 bool is_changed = false; // record changes in current iteration
525 bool dim_covered[3] = { false, false, false };
526 for ( int d = 0; d < 3; ++d )
527 {
528 int random_dim_id = std::rand() % num_space_dim;
529 while ( dim_covered[random_dim_id] )
530 random_dim_id = std::rand() % num_space_dim;
531
532 bool is_dim_changed = false; // record changes in current dim
533 optimizePartition( is_dim_changed, random_dim_id );
534
535 // update control info
536 is_changed = is_changed || is_dim_changed;
537 dim_covered[random_dim_id] = true;
538 }
539 // return if the current partition is optimal
540 if ( !is_changed )
541 return i;
542 }
543 return _max_optimize_iteration;
544 }

591 void optimizePartition( bool& is_changed, int iter_seed )
592 {
593 is_changed = false;
594 // loop over three dimensions, optimize the partition in dimension di
595 for ( int iter_id = iter_seed;
596 iter_id < iter_seed + static_cast<int>( num_space_dim );
597 ++iter_id )
598 {
599 int di = iter_id % num_space_dim;
600 // compute the dimensions that should be fixed (dj and dk)
601 int dj = ( di + 1 ) % num_space_dim;
602 int dk = ( di + 2 ) % num_space_dim;
603 auto rank_k = _ranks_per_dim[dk];
604
605 auto rank = _ranks_per_dim[di];
606 auto rec_mirror = Kokkos::create_mirror_view_and_copy(
607 Kokkos::HostSpace(), _rectangle_partition_dev );
608 auto rec_partition = _rectangle_partition_dev;
609
610 SubWorkloadFunctor<partition_view, workload_view>
611 compute_sub_workload( _rectangle_partition_dev,
612 _workload_prefix_sum );
613
614 // compute average workload in the dimension di
615 Kokkos::View<int*, memory_space> ave_workload(
616 "ave_workload", _ranks_per_dim[dj] * _ranks_per_dim[dk] );
617 Kokkos::parallel_for(
618 "compute_average_workload",
619 Kokkos::RangePolicy<execution_space>(
620 0, _ranks_per_dim[dj] * _ranks_per_dim[dk] ),
621 KOKKOS_LAMBDA( uint32_t jnk ) {
622 // compute rank_id in the fixed dimensions
623 int j = static_cast<int>( jnk / rank_k );
624 int k = static_cast<int>( jnk % rank_k );
625 // compute the average workload with the partition of the
626 // two fixed dimensions
627 ave_workload( jnk ) =
628 compute_sub_workload( di, 0, rec_partition( rank, di ),
629 dj, j, dk, k ) /
630 rank;
631 } );
632 Kokkos::fence();
633
634 // point_i: current partition position
635 int point_i = 1;
636 // equal_start_point: register the beginning pos of potentially
637 // equivalent partitions
638 int equal_start_point = 1;
639 // last_point: the opimized position for the lask partition
640 int last_point = 0;
641 // current_workload: the workload between [last_point, point_i)
642 Kokkos::View<int*, memory_space> current_workload(
643 "current_workload", _ranks_per_dim[dj] * _ranks_per_dim[dk] );
644 for ( int current_rank = 1; current_rank < rank; current_rank++ )
645 {
646 int last_diff = __INT_MAX__;
647 while ( true )
648 {
649 // compute current workload between [last_point, point_i)
650 Kokkos::parallel_for(
651 "compute_current_workload",
652 Kokkos::RangePolicy<execution_space>(
653 0, _ranks_per_dim[dj] * _ranks_per_dim[dk] ),
654 KOKKOS_LAMBDA( uint32_t jnk ) {
655 int j = static_cast<int>( jnk / rank_k );
656 int k = static_cast<int>( jnk % rank_k );
657 current_workload( jnk ) = compute_sub_workload(
658 di, last_point, point_i, dj, j, dk, k );
659 } );
660 Kokkos::fence();
661
662 // compute the (w_jk^ave - w_jk^{last_point:point_i})
663 Kokkos::parallel_for(
664 "compute_diff",
665 Kokkos::RangePolicy<execution_space>(
666 0, _ranks_per_dim[dj] * _ranks_per_dim[dk] ),
667 KOKKOS_LAMBDA( uint32_t jnk ) {
668 auto wl =
669 current_workload( jnk ) - ave_workload( jnk );
670 // compute absolute diff (rather than squares to
671 // avoid potential overflow)
672 // TODO: update when Kokkos::abs() available
673 wl = wl > 0 ? wl : -wl;
674 current_workload( jnk ) = wl;
675 } );
676 Kokkos::fence();
677
678 // compute the sum of the difference in all rank_j*rank_k
679 // regions
680 int diff;
681 Kokkos::parallel_reduce(
682 "diff_reduce",
683 Kokkos::RangePolicy<execution_space>(
684 0, _ranks_per_dim[dj] * _ranks_per_dim[dk] ),
685 KOKKOS_LAMBDA( const int idx, int& update ) {
686 update += current_workload( idx );
687 },
688 diff );
689 Kokkos::fence();
690
691 // record the new optimal position
692 if ( diff <= last_diff )
693 {
694 // register starting points of potentially equivalent
695 // partitions
696 if ( diff != last_diff )
697 equal_start_point = point_i;
698
699 // check if point_i reach the total_tile_num
700 if ( point_i == rec_mirror( rank, di ) )
701 {
702 rec_mirror( current_rank, di ) = point_i;
703 break;
704 }
705
706 last_diff = diff;
707 point_i++;
708 }
709 else
710 {
711 // final optimal position - middle position of all
712 // potentially equivalent partitions
713 if ( rec_mirror( current_rank, di ) !=
714 ( point_i - 1 + equal_start_point ) / 2 )
715 {
716 rec_mirror( current_rank, di ) =
717 ( point_i - 1 + equal_start_point ) / 2;
718 is_changed = true;
719 }
720 last_point = point_i - 1;
721 break;
722 }
723 } // end while (optimization for the current rank)
724 } // end for (all partition/rank in the optimized dimension)
725 Kokkos::deep_copy( _rectangle_partition_dev, rec_mirror );
726 } // end for (3 dimensions)
727 }

791 int averageRankWorkload( WorkloadViewHost& prefix_sum_view )
792 {
793 // compute total workload of the current rank
794 return prefix_sum_view( prefix_sum_view.extent( 0 ) - 1,
795 prefix_sum_view.extent( 1 ) - 1,
796 prefix_sum_view.extent( 2 ) - 1 ) /
797 ( _ranks_per_dim[0] * _ranks_per_dim[1] * _ranks_per_dim[2] );
798 }
799
805 float computeImbalanceFactor( MPI_Comm cart_comm )
806 {
807 auto rec_mirror = Kokkos::create_mirror_view_and_copy(
808 Kokkos::HostSpace(), _rectangle_partition_dev );
809 auto prefix_sum_mirror = Kokkos::create_mirror_view_and_copy(
810 Kokkos::HostSpace(), _workload_prefix_sum );
811
812 int workload_current_rank =
813 currentRankWorkload( cart_comm, rec_mirror, prefix_sum_mirror );
814 int workload_ave_rank = averageRankWorkload( prefix_sum_mirror );
815
816 return static_cast<float>( workload_current_rank ) /
817 static_cast<float>( workload_ave_rank );
818 }

824 template <typename PartitionView, typename WorkloadView>
825 struct SubWorkloadFunctor
826 {
828 PartitionView rec_partition;
830 WorkloadView workload_prefix_sum;
831
833 SubWorkloadFunctor( PartitionView rec_par, WorkloadView pre_sum )
834 : rec_partition( rec_par )
835 , workload_prefix_sum( pre_sum )
836 {
837 }
838
843 KOKKOS_INLINE_FUNCTION int operator()( int dim_i, int i_start,
844 int i_end, int dim_j, int j,
845 int dim_k, int k ) const
846 {
847 int end[num_space_dim], start[num_space_dim];
848 end[dim_i] = i_end;
849 end[dim_j] = rec_partition( j + 1, dim_j );
850 end[dim_k] = rec_partition( k + 1, dim_k );
851
852 start[dim_i] = i_start;
853 start[dim_j] = rec_partition( j, dim_j );
854 start[dim_k] = rec_partition( k, dim_k );
855
856 // S[i][j][k] = S[i-1][j][k] + S[i][j-1][k] + S[i][j][k-1] -
857 // S[i-1][j-1][k]
858 // - S[i][j-1][k-1] - S[i-1][j][k-1] + S[i-1][j-1][k-1] + a[i][j][k]
859 return workload_prefix_sum( end[0], end[1], end[2] ) // S[i][j][k]
860 - workload_prefix_sum( start[0], end[1],
861 end[2] ) // S[i-1][j][k]
862 - workload_prefix_sum( end[0], start[1],
863 end[2] ) // S[i][j-1][k]
864 - workload_prefix_sum( end[0], end[1],
865 start[2] ) // S[i][j][k-1]
866 + workload_prefix_sum( start[0], start[1],
867 end[2] ) // S[i-1][j-1][k]
868 + workload_prefix_sum( end[0], start[1],
869 start[2] ) // S[i][j-1][k-1]
870 + workload_prefix_sum( start[0], end[1],
871 start[2] ) // S[i-1][j][k-1]
872 - workload_prefix_sum( start[0], start[1],
873 start[2] ); // S[i-1][j-1][k-1]
874 }
875 };
876
877 private:
878 // workload_threshold
879 int _workload_threshold;
880 // default check point for re-balance
881 int _num_step_rebalance;
882 // max_optimize iterations
883 int _max_optimize_iteration;
884
885 // represent the rectangle partition in each dimension
886 // with form [0, p_1, ..., p_n, cell_num], n = rank num in current
887 // dimension, partition in this dimension would be [0, p_1), [p_1, p_2) ...
888 // [p_n, total_tile_num] (unit: tile)
889 partition_view _rectangle_partition_dev;
890 // the workload of each tile on current
891 workload_view _workload_per_tile;
892 // 3d prefix sum of the workload of each tile on current
893 workload_view _workload_prefix_sum;
894 // ranks per dimension
895 Kokkos::Array<int, num_space_dim> _ranks_per_dim;
896
897 void allocate( const std::array<int, num_space_dim>& global_cells_per_dim )
898 {

900 Kokkos::view_alloc( Kokkos::WithoutInitializing,
901 "workload_per_tile" ),
902 ( global_cells_per_dim[0] >> cell_bits_per_tile_dim ) + 1,
903 ( global_cells_per_dim[1] >> cell_bits_per_tile_dim ) + 1,
904 ( global_cells_per_dim[2] >> cell_bits_per_tile_dim ) + 1 );
905
906 _workload_prefix_sum = workload_view(
907 Kokkos::view_alloc( Kokkos::WithoutInitializing,

909 ( global_cells_per_dim[0] >> cell_bits_per_tile_dim ) + 1,
910 ( global_cells_per_dim[1] >> cell_bits_per_tile_dim ) + 1,

912 }
