279template <class MeshType>
280std::shared_ptr<LocalGrid<MeshType>>
281createSparseLocalGrid( const std::shared_ptr<GlobalGrid<MeshType>>& global_grid,
282 const int halo_cell_width,
283 const int cell_num_per_tile_dim )
284{
285 static_assert( isSparseMesh<MeshType>::value,
286 "createSparseLocalGrid supports only SparseMesh" );
287 return std::make_shared<LocalGrid<MeshType>>( global_grid, halo_cell_width,
288 cell_num_per_tile_dim );
289}
