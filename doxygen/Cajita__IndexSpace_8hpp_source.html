28 template <long N>
29 class IndexSpace
30 {
31  public:
33  static constexpr long Rank = N;
34 
36  IndexSpace()
37  {
38  std::fill( _min.data(), _min.data() + Rank, -1 );
39  std::fill( _max.data(), _max.data() + Rank, -1 );
40  }
41 
45  IndexSpace( const std::initializer_list<long>& size )
46  {
47  std::fill( _min.data(), _min.data() + Rank, 0 );
48  std::copy( size.begin(), size.end(), _max.data() );
49  }
50 
54  IndexSpace( const std::initializer_list<long>& min,
55  const std::initializer_list<long>& max )
56  {

58  std::copy( max.begin(), max.end(), _max.data() );
59  }

64  IndexSpace( const std::array<long, N>& size )
65  {

80  KOKKOS_INLINE_FUNCTION

82  {
83  for ( long i = 0; i < N; ++i )
84  {

86  return false;

115  KOKKOS_INLINE_FUNCTION

117  {
118  return Kokkos::tie( _min[dim], _max[dim] );
119  }

122  KOKKOS_INLINE_FUNCTION
123  long rank() const { return Rank; }
124 
126  KOKKOS_INLINE_FUNCTION
127  long extent( const long dim ) const { return _max[dim] - _min[dim]; }
128 
130  KOKKOS_INLINE_FUNCTION
131  long size() const
132  {
133  long size = 1;
134  for ( long d = 0; d < Rank; ++d )
135  size *= extent( d );
136  return size;
137  }
138 
139  private:
140  // Minimum index bounds.
141  Kokkos::Array<long, Rank> _min;
142 
143  // Maximum index bounds.
144  Kokkos::Array<long, Rank> _max;
145 };

403 // Given an N-dimensional index space append an additional dimension with the
404 // given size.
405 template <long N>
406 IndexSpace<N + 1> appendDimension( const IndexSpace<N>& index_space,
407  const long size )
408 {
409  std::array<long, N + 1> min;
410  for ( int d = 0; d < N; ++d )
411  min[d] = index_space.min( d );
412  min[N] = 0;
413 
414  std::array<long, N + 1> max;
415  for ( int d = 0; d < N; ++d )
416  max[d] = index_space.max( d );
417  max[N] = size;
418 
419  return IndexSpace<N + 1>( min, max );
420 }
421 
422 //---------------------------------------------------------------------------//
423 // Given an N-dimensional index space append an additional dimension with the
424 // given range.
425 template <long N>
426 IndexSpace<N + 1> appendDimension( const IndexSpace<N>& index_space,
427  const long min, const long max )
428 {
429  std::array<long, N + 1> range_min;
430  for ( int d = 0; d < N; ++d )
431  range_min[d] = index_space.min( d );
432  range_min[N] = min;
433 
434  std::array<long, N + 1> range_max;
435  for ( int d = 0; d < N; ++d )
436  range_max[d] = index_space.max( d );
437  range_max[N] = max;
438 
439  return IndexSpace<N + 1>( range_min, range_max );
440 }
