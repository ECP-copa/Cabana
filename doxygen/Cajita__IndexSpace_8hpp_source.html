143  KOKKOS_INLINE_FUNCTION
144  bool inRange( const long index[N] ) const
145  {
146  bool result = true;
147  for ( long i = 0; i < N; ++i )
148  result =
149  result && ( _min[i] <= index[i] ) && ( index[i] < _max[i] );
150  return result;
151  }

421 template <long N>
422 IndexSpace<N + 1> appendDimension( const IndexSpace<N>& index_space,
423  const long size )
424 {
425  std::array<long, N + 1> min;
426  for ( int d = 0; d < N; ++d )
427  min[d] = index_space.min( d );
428  min[N] = 0;
429 
430  std::array<long, N + 1> max;
431  for ( int d = 0; d < N; ++d )
432  max[d] = index_space.max( d );
433  max[N] = size;
434 
435  return IndexSpace<N + 1>( min, max );
436 }
