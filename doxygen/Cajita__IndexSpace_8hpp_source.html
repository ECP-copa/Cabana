125  KOKKOS_INLINE_FUNCTION
126  long extent( const long dim ) const { return _max[dim] - _min[dim]; }
127 
129  KOKKOS_INLINE_FUNCTION
130  long size() const
131  {
132  long size = 1;
133  for ( long d = 0; d < Rank; ++d )
134  size *= extent( d );
135  return size;
136  }

139  KOKKOS_INLINE_FUNCTION
140  bool inRange( const long index[N] ) const
141  {
142  bool result = true;
143  for ( long i = 0; i < N; ++i )
144  result =
145  result && ( _min[i] <= index[i] ) && ( index[i] < _max[i] );
146  return result;
147  }
148 
149  private:
150  // Minimum index bounds.
151  Kokkos::Array<long, Rank> _min;
152 
153  // Maximum index bounds.
154  Kokkos::Array<long, Rank> _max;
155 };

417 template <long N>
418 IndexSpace<N + 1> appendDimension( const IndexSpace<N>& index_space,
419  const long size )
420 {
421  std::array<long, N + 1> min;
422  for ( int d = 0; d < N; ++d )
423  min[d] = index_space.min( d );
424  min[N] = 0;
425 
426  std::array<long, N + 1> max;
427  for ( int d = 0; d < N; ++d )
428  max[d] = index_space.max( d );
429  max[N] = size;
430 
431  return IndexSpace<N + 1>( min, max );
432 }
