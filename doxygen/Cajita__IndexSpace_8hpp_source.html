31template <long N>
32class IndexSpace
33{
34 public:
36 static constexpr long Rank = N;
37
39 IndexSpace()
40 {
41 std::fill( _min.data(), _min.data() + Rank, -1 );
42 std::fill( _max.data(), _max.data() + Rank, -1 );
43 }
44
48 template <typename Scalar>
49 IndexSpace( const std::initializer_list<Scalar>& size )
50 {
51 std::fill( _min.data(), _min.data() + Rank, 0 );
52 std::copy( size.begin(), size.end(), _max.data() );
53 }
54
58 template <typename Scalar>
59 IndexSpace( const std::initializer_list<Scalar>& min,
60 const std::initializer_list<Scalar>& max )
61 {
62 std::copy( min.begin(), min.end(), _min.data() );
63 std::copy( max.begin(), max.end(), _max.data() );
64 }
65
69 IndexSpace( const std::array<long, N>& size )
70 {
71 std::fill( _min.data(), _min.data() + Rank, 0 );
72 std::copy( size.begin(), size.end(), _max.data() );
73 }
74
78 IndexSpace( const std::array<long, N>& min, const std::array<long, N>& max )
79 {
80 std::copy( min.begin(), min.end(), _min.data() );
81 std::copy( max.begin(), max.end(), _max.data() );
82 }
83
85 KOKKOS_INLINE_FUNCTION
86 bool operator==( const IndexSpace<N>& rhs ) const
87 {
88 for ( long i = 0; i < N; ++i )
89 {
90 if ( min( i ) != rhs.min( i ) || max( i ) != rhs.max( i ) )
91 return false;
92 }
93 return true;
94 }
95
97 KOKKOS_INLINE_FUNCTION
98 bool operator!=( const IndexSpace<N>& rhs ) const
99 {
100 return !( operator==( rhs ) );
101 }
102
104 KOKKOS_INLINE_FUNCTION
105 long min( const long dim ) const { return _min[dim]; }
106
108 KOKKOS_INLINE_FUNCTION
109 Kokkos::Array<long, Rank> min() const { return _min; }
110
112 KOKKOS_INLINE_FUNCTION
113 long max( const long dim ) const { return _max[dim]; }
114
116 KOKKOS_INLINE_FUNCTION
117 Kokkos::Array<long, Rank> max() const { return _max; }
118
120 KOKKOS_INLINE_FUNCTION
121 Kokkos::pair<long, long> range( const long dim ) const
122 {
123 return Kokkos::tie( _min[dim], _max[dim] );
124 }
125
127 KOKKOS_INLINE_FUNCTION
128 long rank() const { return Rank; }
129
131 KOKKOS_INLINE_FUNCTION
132 long extent( const long dim ) const { return _max[dim] - _min[dim]; }
133
135 KOKKOS_INLINE_FUNCTION
136 long size() const
137 {
138 long size = 1;
139 for ( long d = 0; d < Rank; ++d )
140 size *= extent( d );
141 return size;
142 }
143
145 KOKKOS_INLINE_FUNCTION
146 bool inRange( const long index[N] ) const
147 {
148 bool result = true;
149 for ( long i = 0; i < N; ++i )
150 result =
151 result && ( _min[i] <= index[i] ) && ( index[i] < _max[i] );
152 return result;
153 }
154
155 protected:
157 Kokkos::Array<long, Rank> _min;
158
160 Kokkos::Array<long, Rank> _max;
161};

437template <long N>
438IndexSpace<N + 1> appendDimension( const IndexSpace<N>& index_space,
439 const long size )
440{
441 std::array<long, N + 1> min;
442 for ( int d = 0; d < N; ++d )
443 min[d] = index_space.min( d );
444 min[N] = 0;
445
446 std::array<long, N + 1> max;
447 for ( int d = 0; d < N; ++d )
448 max[d] = index_space.max( d );
449 max[N] = size;
450
451 return IndexSpace<N + 1>( min, max );
452}

454//---------------------------------------------------------------------------//
460template <long N>
461IndexSpace<N + 1> appendDimension( const IndexSpace<N>& index_space,
462 const long min, const long max )
463{
464 std::array<long, N + 1> range_min;
465 for ( int d = 0; d < N; ++d )
466 range_min[d] = index_space.min( d );
467 range_min[N] = min;
468
469 std::array<long, N + 1> range_max;
470 for ( int d = 0; d < N; ++d )
471 range_max[d] = index_space.max( d );
472 range_max[N] = max;
473
474 return IndexSpace<N + 1>( range_min, range_max );
475}
