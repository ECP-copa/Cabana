135  size *= extent( d );

139  // Determine if a set of indices is within the range of the index space.
140  KOKKOS_INLINE_FUNCTION
141  bool inRange( const long index[N] ) const
142  {
143  bool result = true;
144  for ( long i = 0; i < N; ++i )
145  result =
146  result && ( _min[i] <= index[i] ) && ( index[i] < _max[i] );
147  return result;
148  }
149 
150  private:
151  // Minimum index bounds.
152  Kokkos::Array<long, Rank> _min;
153 
154  // Maximum index bounds.
155  Kokkos::Array<long, Rank> _max;
156 };

414 // Given an N-dimensional index space append an additional dimension with the
415 // given size.
416 template <long N>
417 IndexSpace<N + 1> appendDimension( const IndexSpace<N>& index_space,
418  const long size )
419 {
420  std::array<long, N + 1> min;
421  for ( int d = 0; d < N; ++d )
422  min[d] = index_space.min( d );
423  min[N] = 0;
424 
425  std::array<long, N + 1> max;
426  for ( int d = 0; d < N; ++d )
427  max[d] = index_space.max( d );
428  max[N] = size;
429 
430  return IndexSpace<N + 1>( min, max );
431 }

433 //---------------------------------------------------------------------------//
434 // Given an N-dimensional index space append an additional dimension with the
435 // given range.
436 template <long N>
437 IndexSpace<N + 1> appendDimension( const IndexSpace<N>& index_space,
438  const long min, const long max )
439 {
440  std::array<long, N + 1> range_min;
441  for ( int d = 0; d < N; ++d )
442  range_min[d] = index_space.min( d );
443  range_min[N] = min;
444 
445  std::array<long, N + 1> range_max;
446  for ( int d = 0; d < N; ++d )
447  range_max[d] = index_space.max( d );
448  range_max[N] = max;
449 
450  return IndexSpace<N + 1>( range_min, range_max );
