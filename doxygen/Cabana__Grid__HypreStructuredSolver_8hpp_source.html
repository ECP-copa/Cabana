49class HypreStructuredSolver
50{
51 public:
53 using entity_type = EntityType;

68 template <class ArrayLayout_t>
69 HypreStructuredSolver( const ArrayLayout_t& layout,
70 const bool is_preconditioner = false )
71 : _comm( layout.localGrid()->globalGrid().comm() )
72 , _is_preconditioner( is_preconditioner )
73 {
74 static_assert( is_array_layout<ArrayLayout_t>::value,
75 "Must use an array layout" );
76 static_assert(
77 std::is_same<typename ArrayLayout_t::entity_type,
78 entity_type>::value,
79 "Array layout entity type mush match solver entity type" );
80
81 // Spatial dimension.
82 const std::size_t num_space_dim = ArrayLayout_t::num_space_dim;
83
84 // Only create data structures if this is not a preconditioner.
85 if ( !_is_preconditioner )
86 {
87 // Create the grid.
88 auto error = HYPRE_StructGridCreate( _comm, num_space_dim, &_grid );
89 checkHypreError( error );
90
91 // Get the global index space spanned by the local grid on this
92 // rank. Note that the upper bound is not a bound but rather the
93 // last index as this is what Hypre wants. Note that we reordered
94 // this to KJI from IJK to be consistent with HYPRE ordering. By
95 // setting up the grid like this, HYPRE will then want layout-right
96 // data indexed as (i,j,k) or (i,j,k,l) which will allow us to
97 // directly use Kokkos::deep_copy to move data between Cajita arrays
98 // and HYPRE data structures.
99 auto global_space = layout.indexSpace( Own(), Global() );
100 _lower.resize( num_space_dim );
101 _upper.resize( num_space_dim );
102 for ( std::size_t d = 0; d < num_space_dim; ++d )
103 {
104 _lower[d] = static_cast<HYPRE_Int>(
105 global_space.min( num_space_dim - d - 1 ) );
106 _upper[d] = static_cast<HYPRE_Int>(
107 global_space.max( num_space_dim - d - 1 ) - 1 );
108 }
109 error = HYPRE_StructGridSetExtents( _grid, _lower.data(),
110 _upper.data() );
111 checkHypreError( error );
112
113 // Get periodicity. Note we invert the order of this to KJI as well.
114 const auto& global_grid = layout.localGrid()->globalGrid();
115 HYPRE_Int periodic[num_space_dim];
116 for ( std::size_t d = 0; d < num_space_dim; ++d )
117 periodic[num_space_dim - 1 - d] =
118 global_grid.isPeriodic( d )
119 ? layout.localGrid()->globalGrid().globalNumEntity(
120 EntityType(), d )
121 : 0;
122 error = HYPRE_StructGridSetPeriodic( _grid, periodic );
123 checkHypreError( error );
124
125 // Assemble the grid.
126 error = HYPRE_StructGridAssemble( _grid );
127 checkHypreError( error );
128
129 // Allocate LHS and RHS vectors and initialize to zero. Note that we
130 // are fixing the views under these vectors to layout-right.
131 std::array<long, num_space_dim> reorder_size;
132 for ( std::size_t d = 0; d < num_space_dim; ++d )
133 {
134 reorder_size[d] = global_space.extent( d );
135 }
136 IndexSpace<num_space_dim> reorder_space( reorder_size );
137 auto vector_values =
138 createView<HYPRE_Complex, Kokkos::LayoutRight, memory_space>(
139 "vector_values", reorder_space );
140 Kokkos::deep_copy( vector_values, 0.0 );
141
142 error = HYPRE_StructVectorCreate( _comm, _grid, &_b );
143 checkHypreError( error );
144 error = HYPRE_StructVectorInitialize( _b );
145 checkHypreError( error );
146 error = HYPRE_StructVectorSetBoxValues(
147 _b, _lower.data(), _upper.data(), vector_values.data() );
148 checkHypreError( error );
149 error = HYPRE_StructVectorAssemble( _b );
150 checkHypreError( error );
151
152 error = HYPRE_StructVectorCreate( _comm, _grid, &_x );
153 checkHypreError( error );
154 error = HYPRE_StructVectorInitialize( _x );
155 checkHypreError( error );
156 error = HYPRE_StructVectorSetBoxValues(
157 _x, _lower.data(), _upper.data(), vector_values.data() );
158 checkHypreError( error );
159 error = HYPRE_StructVectorAssemble( _x );
160 checkHypreError( error );
161 }
162 }

189 template <std::size_t NumSpaceDim>
190 void
191 setMatrixStencil( const std::vector<std::array<int, NumSpaceDim>>& stencil,
192 const bool is_symmetric = false )
193 {
194 // This function is only valid for non-preconditioners.
195 if ( _is_preconditioner )
196 throw std::logic_error(
197 "Cannot call setMatrixStencil() on preconditioners" );
198
199 // Create the stencil.
200 _stencil_size = stencil.size();
201 auto error =
202 HYPRE_StructStencilCreate( NumSpaceDim, _stencil_size, &_stencil );
203 checkHypreError( error );
204 std::array<HYPRE_Int, NumSpaceDim> offset;
205 for ( unsigned n = 0; n < stencil.size(); ++n )
206 {
207 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
208 offset[d] = stencil[n][d];
209 error = HYPRE_StructStencilSetElement( _stencil, n, offset.data() );
210 checkHypreError( error );
211 }
212
213 // Create the matrix object. Must be done after the stencil is setup
214 error = HYPRE_StructMatrixCreate( _comm, _grid, _stencil, &_A );
215 checkHypreError( error );
216 error = HYPRE_StructMatrixSetSymmetric( _A, is_symmetric );
217 checkHypreError( error );
218 error = HYPRE_StructMatrixInitialize( _A );
219 checkHypreError( error );
220 }

230 template <class Array_t>
231 void setMatrixValues( const Array_t& values )
232 {
233 static_assert( is_array<Array_t>::value, "Must use an array" );
234 static_assert(
235 std::is_same<typename Array_t::entity_type, entity_type>::value,
236 "Array entity type mush match solver entity type" );
237 static_assert(
238 std::is_same<typename Array_t::memory_space, MemorySpace>::value,
239 "Array memory space and solver memory space are different." );
240
241 static_assert(
242 std::is_same<typename Array_t::value_type, value_type>::value,
243 "Array value type and solver value type are different." );
244
245 // This function is only valid for non-preconditioners.
246 if ( _is_preconditioner )
247 throw std::logic_error(
248 "Cannot call setMatrixValues() on preconditioners" );
249
250 if ( values.layout()->dofsPerEntity() !=
251 static_cast<int>( _stencil_size ) )
252 throw std::runtime_error(
253 "Number of matrix values does not match stencil size" );
254
255 // Spatial dimension.
256 const std::size_t num_space_dim = Array_t::num_space_dim;
257
258 // Copy the matrix entries into HYPRE. The HYPRE layout is fixed as
259 // layout-right.
260 auto owned_space = values.layout()->indexSpace( Own(), Local() );
261 std::array<long, num_space_dim + 1> reorder_size;
262 for ( std::size_t d = 0; d < num_space_dim; ++d )
263 {
264 reorder_size[d] = owned_space.extent( d );
265 }
266 reorder_size.back() = _stencil_size;
267 IndexSpace<num_space_dim + 1> reorder_space( reorder_size );
268 auto a_values =
269 createView<HYPRE_Complex, Kokkos::LayoutRight, memory_space>(
270 "a_values", reorder_space );
271 auto values_subv = createSubview( values.view(), owned_space );
272 Kokkos::deep_copy( a_values, values_subv );
273
274 // Insert values into the HYPRE matrix.
275 std::vector<HYPRE_Int> indices( _stencil_size );
276 std::iota( indices.begin(), indices.end(), 0 );
277 auto error = HYPRE_StructMatrixSetBoxValues(
278 _A, _lower.data(), _upper.data(), indices.size(), indices.data(),
279 a_values.data() );
280 checkHypreError( error );
281 error = HYPRE_StructMatrixAssemble( _A );
282 checkHypreError( error );
283 }

358 template <class Array_t>
359 void solve( const Array_t& b, Array_t& x )
360 {
361 Kokkos::Profiling::ScopedRegion region(
362 "Cabana::Grid::HypreStructuredSolver::solve" );
363
364 static_assert( is_array<Array_t>::value, "Must use an array" );
365 static_assert(
366 std::is_same<typename Array_t::entity_type, entity_type>::value,
367 "Array entity type mush match solver entity type" );
368 static_assert(
369 std::is_same<typename Array_t::memory_space, MemorySpace>::value,
370 "Array memory space and solver memory space are different." );
371
372 static_assert(
373 std::is_same<typename Array_t::value_type, value_type>::value,
374 "Array value type and solver value type are different." );
375
376 // This function is only valid for non-preconditioners.
377 if ( _is_preconditioner )
378 throw std::logic_error( "Cannot call solve() on preconditioners" );
379
380 if ( b.layout()->dofsPerEntity() != 1 ||
381 x.layout()->dofsPerEntity() != 1 )
382 throw std::runtime_error(
383 "Structured solver only for scalar fields" );
384
385 // Spatial dimension.
386 const std::size_t num_space_dim = Array_t::num_space_dim;
387
388 // Copy the RHS into HYPRE. The HYPRE layout is fixed as layout-right.
389 auto owned_space = b.layout()->indexSpace( Own(), Local() );
390 std::array<long, num_space_dim + 1> reorder_size;
391 for ( std::size_t d = 0; d < num_space_dim; ++d )
392 {
393 reorder_size[d] = owned_space.extent( d );
394 }
395 reorder_size.back() = 1;
396 IndexSpace<num_space_dim + 1> reorder_space( reorder_size );
397 auto vector_values =
398 createView<HYPRE_Complex, Kokkos::LayoutRight, memory_space>(
399 "vector_values", reorder_space );
400 auto b_subv = createSubview( b.view(), owned_space );
401 Kokkos::deep_copy( vector_values, b_subv );
402
403 // Insert b values into the HYPRE vector.
404 auto error = HYPRE_StructVectorSetBoxValues(
405 _b, _lower.data(), _upper.data(), vector_values.data() );
406 checkHypreError( error );
407 error = HYPRE_StructVectorAssemble( _b );
408 checkHypreError( error );
409
410 // Solve the problem
411 this->solveImpl();
412
413 // Extract the solution from the LHS
414 error = HYPRE_StructVectorGetBoxValues(
415 _x, _lower.data(), _upper.data(), vector_values.data() );
416 checkHypreError( error );
417
418 // Copy the HYPRE solution to the LHS.
419 auto x_subv = createSubview( x.view(), owned_space );
420 Kokkos::deep_copy( x_subv, vector_values );
421 }

1489 Scalar, typename ArrayLayout_t::entity_type, MemorySpace>>

1492 const bool is_preconditioner = false )

1498 return createHypreStructPCG<Scalar, MemorySpace>( layout,
1499 is_preconditioner );
1500 else if ( "GMRES" == solver_type )
1501 return createHypreStructGMRES<Scalar, MemorySpace>( layout,
1502 is_preconditioner );
1503 else if ( "BiCGSTAB" == solver_type )
1504 return createHypreStructBiCGSTAB<Scalar, MemorySpace>(
1505 layout, is_preconditioner );
1506 else if ( "PFMG" == solver_type )
1507 return createHypreStructPFMG<Scalar, MemorySpace>( layout,
1508 is_preconditioner );
1509 else if ( "SMG" == solver_type )
1510 return createHypreStructSMG<Scalar, MemorySpace>( layout,
1511 is_preconditioner );
1512 else if ( "Jacobi" == solver_type )
1513 return createHypreStructJacobi<Scalar, MemorySpace>(
1514 layout, is_preconditioner );
1515 else if ( "Diagonal" == solver_type )
1516 return createHypreStructDiagonal<Scalar, MemorySpace>(
1517 layout, is_preconditioner );
1518 else
1519 throw std::runtime_error( "Invalid solver type" );
1520}
