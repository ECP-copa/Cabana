45 template <class MeshType>
46 class GlobalMesh
47 {
48  public:
50  using mesh_type = MeshType;
51 
53  using scalar_type = typename mesh_type::scalar_type;
54 
56  static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;
57 
59  GlobalMesh(
60  const std::array<scalar_type, num_space_dim>& global_low_corner,
61  const std::array<scalar_type, num_space_dim>& global_high_corner,
62  const scalar_type cell_size )
63  : _global_low_corner( global_low_corner )
64  , _global_high_corner( global_high_corner )
65  {
66  // Check that the domain is evenly divisible by the cell size in each
67  // dimension within round-off error.
68  for ( std::size_t d = 0; d < num_space_dim; ++d )
69  {
70  _cell_size[d] = cell_size;
71  scalar_type ext = globalNumCell( d ) * _cell_size[d];
72  if ( std::abs( ext - extent( d ) ) >
73  scalar_type( 100.0 ) *
74  std::numeric_limits<scalar_type>::epsilon() )
75  throw std::logic_error(
76  "Extent not evenly divisible by uniform cell size" );
77  }
78  }

81  GlobalMesh(
82  const std::array<scalar_type, num_space_dim>& global_low_corner,
83  const std::array<scalar_type, num_space_dim>& global_high_corner,
84  const std::array<scalar_type, num_space_dim>& cell_size )
85  : _global_low_corner( global_low_corner )
86  , _global_high_corner( global_high_corner )
87  , _cell_size( cell_size )
88  {
89  // Check that the domain is evenly divisible by the cell size in each
90  // dimension within round-off error.
91  for ( std::size_t d = 0; d < num_space_dim; ++d )
92  {

94  if ( std::abs( ext - extent( d ) ) >
95  scalar_type( 100.0 ) *
96  std::numeric_limits<scalar_type>::epsilon() )

103  GlobalMesh(
104  const std::array<scalar_type, num_space_dim>& global_low_corner,
105  const std::array<scalar_type, num_space_dim>& global_high_corner,
106  const std::array<int, num_space_dim>& global_num_cell )
107  : _global_low_corner( global_low_corner )
108  , _global_high_corner( global_high_corner )
109  {
110  // Compute the cell size in each dimension.
111  for ( std::size_t d = 0; d < num_space_dim; ++d )
112  _cell_size[d] = ( _global_high_corner[d] - _global_low_corner[d] ) /
113  global_num_cell[d];
114 
115  // Check that the domain is evenly divisible by the cell size in each
116  // dimension within round-off error and that we got the expected
117  // number of cells.
118  for ( std::size_t d = 0; d < num_space_dim; ++d )
119  {
120  scalar_type ext = globalNumCell( d ) * _cell_size[d];
121  if ( std::abs( ext - extent( d ) ) >
122  scalar_type( 100.0 ) *
123  std::numeric_limits<scalar_type>::epsilon() )
124  throw std::logic_error(
125  "Extent not evenly divisible by uniform cell size" );
126  if ( globalNumCell( d ) != global_num_cell[d] )
127  throw std::logic_error( "Global number of cells mismatch" );
128  }
129  }
130 
131  // GLOBAL MESH INTERFACE
132 
135  scalar_type lowCorner( const std::size_t dim ) const
136  {
137  return _global_low_corner[dim];
138  }
139 
142  scalar_type highCorner( const std::size_t dim ) const
143  {
144  return _global_high_corner[dim];
145  }
146 
149  scalar_type extent( const std::size_t dim ) const
150  {
151  return highCorner( dim ) - lowCorner( dim );
152  }
153 
156  int globalNumCell( const std::size_t dim ) const
157  {
158  return std::rint( extent( dim ) / _cell_size[dim] );
159  }
160 
161  // UNIFORM MESH SPECIFIC
162 
165  scalar_type cellSize( const std::size_t dim ) const
166  {
167  return _cell_size[dim];
168  }
169 
170  private:
171  std::array<scalar_type, num_space_dim> _global_low_corner;
172  std::array<scalar_type, num_space_dim> _global_high_corner;
173  std::array<scalar_type, num_space_dim> _cell_size;
174 };
