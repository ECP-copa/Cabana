62template <class ExecutionSpace, class InitFunctor, class ParticleListType,
63 class LocalGridType>
64int createParticles(
65 Cabana::InitRandom, const ExecutionSpace& exec_space,
66 const InitFunctor& create_functor, ParticleListType& particle_list,
67 const int particles_per_cell, LocalGridType& local_grid,
68 const bool shrink_to_fit = true, const uint64_t seed = 123456,
69 typename std::enable_if<Cajita::is_particle_list<ParticleListType>::value,
70 int>::type* = 0 )
71{
72 // Memory space.
73 using memory_space = typename ParticleListType::memory_space;
74
75 // Create a local mesh.
76 auto local_mesh = Cajita::createLocalMesh<ExecutionSpace>( local_grid );
77
78 // Get the global grid.
79 const auto& global_grid = local_grid.globalGrid();
80
81 // Get the local set of owned cell indices.
82 auto owned_cells =
83 local_grid.indexSpace( Cajita::Own(), Cajita::Cell(), Cajita::Local() );
84
85 // Create a random number generator.
86 const auto local_seed =
87 global_grid.blockId() + ( seed % ( global_grid.blockId() + 1 ) );
88 using rnd_type = Kokkos::Random_XorShift64_Pool<ExecutionSpace>;
89 rnd_type pool;
90 pool.init( local_seed, owned_cells.size() );
91
92 // Get the aosoa.
93 auto& aosoa = particle_list.aosoa();
94
95 // Allocate enough space for the case the particles consume the entire
96 // local grid.
97 int num_particles = particles_per_cell * owned_cells.size();
98 aosoa.resize( num_particles );
99
100 // Creation count.
101 auto count = Kokkos::View<int*, memory_space>( "particle_count", 1 );
102
103 // Initialize particles.
104 Cajita::grid_parallel_for(
105 "Cajita::ParticleInit::Random", exec_space, owned_cells,
106 KOKKOS_LAMBDA( const int i, const int j, const int k ) {
107 // Compute the owned local cell id.
108 int i_own = i - owned_cells.min( Dim::I );
109 int j_own = j - owned_cells.min( Dim::J );
110 int k_own = k - owned_cells.min( Dim::K );
111 int cell_id =
112 i_own + owned_cells.extent( Dim::I ) *
113 ( j_own + k_own * owned_cells.extent( Dim::J ) );
114
115 // Get the coordinates of the low cell node.
116 int low_node[3] = { i, j, k };
117 double low_coords[3];
118 local_mesh.coordinates( Cajita::Node(), low_node, low_coords );
119
120 // Get the coordinates of the high cell node.
121 int high_node[3] = { i + 1, j + 1, k + 1 };
122 double high_coords[3];
123 local_mesh.coordinates( Cajita::Node(), high_node, high_coords );
124
125 // Random number generator.
126 auto rand = pool.get_state( cell_id );
127
128 // Particle coordinate.
129 double px[3];
130
131 // Particle volume.
132 double pv = local_mesh.measure( Cajita::Cell(), low_node ) /
133 particles_per_cell;
134
135 // Create particles.
136 for ( int p = 0; p < particles_per_cell; ++p )
137 {
138 // Local particle id.
139 int pid = cell_id * particles_per_cell + p;
140
141 // Select a random point in the cell for the particle
142 // location. These coordinates are logical.
143 for ( int d = 0; d < 3; ++d )
144 {
145 px[d] = Kokkos::rand<decltype( rand ), double>::draw(
146 rand, low_coords[d], high_coords[d] );
147 }
148
149 // Create a new particle with the given logical coordinates.
150 auto particle = particle_list.getParticle( pid );
151 bool created = create_functor( pid, px, pv, particle );
152
153 // If we created a new particle insert it into the list.
154 if ( created )
155 {
156 auto c = Kokkos::atomic_fetch_add( &count( 0 ), 1 );
157 particle_list.setParticle( particle, c );
158 }
159 }
160 } );
161 Kokkos::fence();
162
163 auto count_host =
164 Kokkos::create_mirror_view_and_copy( Kokkos::HostSpace(), count );
165 aosoa.resize( count_host( 0 ) );
166 if ( shrink_to_fit )
167 aosoa.shrinkToFit();
168 return count_host( 0 );
169}
