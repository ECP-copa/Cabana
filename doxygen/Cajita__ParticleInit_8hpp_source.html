64template <class ExecutionSpace, class InitFunctor, class ParticleListType,
65 class LocalGridType>
66int createParticles(
67 Cabana::InitRandom, const ExecutionSpace& exec_space,
68 const InitFunctor& create_functor, ParticleListType& particle_list,
69 const int particles_per_cell, LocalGridType& local_grid,
70 const std::size_t previous_num_particles = 0,
71 const bool shrink_to_fit = true, const uint64_t seed = 123456,
72 typename std::enable_if<Cajita::is_particle_list<ParticleListType>::value,
73 int>::type* = 0 )
74{
75 // Memory space.
76 using memory_space = typename ParticleListType::memory_space;
77
78 // Create a local mesh.
79 auto local_mesh = Cajita::createLocalMesh<ExecutionSpace>( local_grid );
80
81 // Get the global grid.
82 const auto& global_grid = local_grid.globalGrid();
83
84 // Get the local set of owned cell indices.
85 auto owned_cells =
86 local_grid.indexSpace( Cajita::Own(), Cajita::Cell(), Cajita::Local() );
87
88 // Create a random number generator.
89 const auto local_seed =
90 global_grid.blockId() + ( seed % ( global_grid.blockId() + 1 ) );
91 using rnd_type = Kokkos::Random_XorShift64_Pool<ExecutionSpace>;
92 rnd_type pool;
93 pool.init( local_seed, owned_cells.size() );
94
95 // Get the aosoa.
96 auto& aosoa = particle_list.aosoa();
97
98 // Allocate enough space for the case the particles consume the entire
99 // local grid.
100 int num_particles = particles_per_cell * owned_cells.size();
101 aosoa.resize( previous_num_particles + num_particles );
102
103 // Creation count (start from previous).
104 auto count = Kokkos::View<int*, memory_space>( "particle_count", 1 );
105 Kokkos::deep_copy( count, previous_num_particles );
106
107 // Initialize particles.
108 Cajita::grid_parallel_for(
109 "Cajita::ParticleInit::Random", exec_space, owned_cells,
110 KOKKOS_LAMBDA( const int i, const int j, const int k ) {
111 // Compute the owned local cell id.
112 int i_own = i - owned_cells.min( Dim::I );
113 int j_own = j - owned_cells.min( Dim::J );
114 int k_own = k - owned_cells.min( Dim::K );
115 int cell_id =
116 i_own + owned_cells.extent( Dim::I ) *
117 ( j_own + k_own * owned_cells.extent( Dim::J ) );
118
119 // Get the coordinates of the low cell node.
120 int low_node[3] = { i, j, k };
121 double low_coords[3];
122 local_mesh.coordinates( Cajita::Node(), low_node, low_coords );
123
124 // Get the coordinates of the high cell node.
125 int high_node[3] = { i + 1, j + 1, k + 1 };
126 double high_coords[3];
127 local_mesh.coordinates( Cajita::Node(), high_node, high_coords );
128
129 // Random number generator.
130 auto rand = pool.get_state( cell_id );
131
132 // Particle coordinate.
133 double px[3];
134
135 // Particle volume.
136 double pv = local_mesh.measure( Cajita::Cell(), low_node ) /
137 particles_per_cell;
138
139 // Create particles.
140 for ( int p = 0; p < particles_per_cell; ++p )
141 {
142 // Local particle id.
143 int pid =
144 previous_num_particles + cell_id * particles_per_cell + p;
145
146 // Select a random point in the cell for the particle
147 // location. These coordinates are logical.
148 for ( int d = 0; d < 3; ++d )
149 {
150 px[d] = Kokkos::rand<decltype( rand ), double>::draw(
151 rand, low_coords[d], high_coords[d] );
152 }
153
154 // Create a new particle with the given logical coordinates.
155 auto particle = particle_list.getParticle( pid );
156 bool created = create_functor( pid, px, pv, particle );
157
158 // If we created a new particle insert it into the list.
159 if ( created )
160 {
161 auto c = Kokkos::atomic_fetch_add( &count( 0 ), 1 );
162 particle_list.setParticle( particle, c );
163 }
164 }
165 } );
166 Kokkos::fence();
167
168 auto count_host =
169 Kokkos::create_mirror_view_and_copy( Kokkos::HostSpace(), count );
170 aosoa.resize( count_host( 0 ) );
171 if ( shrink_to_fit )
172 aosoa.shrinkToFit();
173 return count_host( 0 );
174}
