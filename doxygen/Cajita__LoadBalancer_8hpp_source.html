115  std::shared_ptr<GlobalGrid<mesh_type>> createBalancedGlobalGrid(
116  const std::shared_ptr<GlobalMesh<mesh_type>>& global_mesh,
117  const BlockPartitioner<NumSpaceDim>& partitioner,
118  const double local_work )
119  {
120  // Create new decomposition
121  _liball->setWork( local_work );
122  _liball->balance();
123  // Calculate new local cell offset and local extent
124  std::vector<ALL::Point<double>> updated_vertices =
125  _liball->getVertices();
126  std::array<double, NumSpaceDim> cell_size;
127  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
128  cell_size[d] = _global_grid->globalMesh().cellSize( d );
129  std::array<int, NumSpaceDim> cell_index_lo, cell_index_hi;
130  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
131  cell_index_lo[d] =
132  std::rint( updated_vertices.at( 0 )[d] / cell_size[d] );
133  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
134  cell_index_hi[d] =
135  std::rint( updated_vertices.at( 1 )[d] / cell_size[d] );
136  std::array<int, NumSpaceDim> num_cell;
137  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
138  num_cell[d] = cell_index_hi[d] - cell_index_lo[d];
139  // Create new global grid
140  // todo(sschulz): Can GlobalGrid be constructed with an already
141  // cartesian communicator? MPI_Cart_Create is called with the given
142  // comm.
143  std::array<bool, NumSpaceDim> periodic;
144  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
145  periodic[d] = _global_grid->isPeriodic( d );
146  std::shared_ptr<GlobalGrid<mesh_type>> global_grid =
147  createGlobalGrid( _comm, global_mesh, periodic, partitioner );
148  global_grid->setNumCellAndOffset( num_cell, cell_index_lo );
149  _global_grid = global_grid;
150  return _global_grid;
151  }

155  const std::array<double, NumSpaceDim * 2> getInternalVertices() const
156  {
157  std::array<double, NumSpaceDim * 2> internal_vertices;
158  std::vector<ALL::Point<double>> lb_vertices = _liball->getVertices();
159  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
160  internal_vertices[d] = lb_vertices.at( 0 )[d];
161  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
162  internal_vertices[d + NumSpaceDim] = lb_vertices.at( 1 )[d];
163  return internal_vertices;
164  // todo(sschulz): Is this ok to pass arrays?
165  }
166 
169  const std::array<double, NumSpaceDim * 2> getVertices() const
170  {
171  std::array<double, NumSpaceDim> cell_size;
172  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
173  cell_size[d] = _global_grid->globalMesh().cellSize( d );
174  std::array<double, NumSpaceDim * 2> vertices;
175  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
176  vertices[d] = _global_grid->globalOffset( d ) * cell_size[d];
177  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
178  vertices[d + NumSpaceDim] =
179  vertices[d] + _global_grid->ownedNumCell( d ) * cell_size[d];
180  return vertices;
181  // todo(sschulz): Is this ok to pass arrays?
182  }

186  double getImbalance() const
187  {
188  const double local_work = _liball->getWork();
189  double max_work, min_work;
190  MPI_Allreduce( &local_work, &max_work, 1, MPI_DOUBLE, MPI_MAX, _comm );
191  MPI_Allreduce( &local_work, &min_work, 1, MPI_DOUBLE, MPI_MIN, _comm );
192  return ( max_work - min_work ) / ( max_work + min_work );
193  }

204  private:
206  void setupLibrary()
207  {
208  // todo(sschulz): Investigate, why usage of num_space_dim instead of
209  // NumSpaceDim causes a linker error.
210  _liball = std::make_shared<ALL::ALL<double, double>>( ALL::TENSOR,
211  NumSpaceDim, 0 );
212  std::vector<int> loc( NumSpaceDim, 0 );
213  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
214  loc[d] = _global_grid->dimBlockId( d );
215  std::vector<int> size( NumSpaceDim, 0 );
216  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
217  size[d] = _global_grid->dimNumBlock( d );
218  _liball->setProcGridParams( loc, size );
219  _liball->setCommunicator( _comm );
220  int rank;
221  MPI_Comm_rank( _global_grid->comm(), &rank );
222  _liball->setProcTag( rank );
223  _liball->setup();
224  // Set initial vertices
225  std::array<double, NumSpaceDim> cell_size;
226  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
227  cell_size[d] = _global_grid->globalMesh().cellSize( d );
228  std::vector<ALL::Point<double>> lb_vertices(
229  2, ALL::Point<double>( NumSpaceDim ) );
230  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
231  lb_vertices.at( 0 )[d] =
232  _global_grid->globalOffset( d ) * cell_size[d];
233  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
234  lb_vertices.at( 1 )[d] =
235  lb_vertices.at( 0 )[d] +
236  _global_grid->ownedNumCell( d ) * cell_size[d];
237  _liball->setVertices( lb_vertices );
238  }

242  MPI_Comm _comm;
