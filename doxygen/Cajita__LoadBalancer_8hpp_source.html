117 std::shared_ptr<GlobalGrid<mesh_type>> createBalancedGlobalGrid(
118 const std::shared_ptr<GlobalMesh<mesh_type>>& global_mesh,
119 const BlockPartitioner<NumSpaceDim>& partitioner,
120 const double local_work )
121 {
122 Kokkos::Profiling::pushRegion( "Cajita::LoadBalancer::balance" );
123
124 // Create new decomposition
125 _liball->setWork( local_work );
126 _liball->balance();
127 // Calculate new local cell offset and local extent
128 std::vector<ALL::Point<double>> updated_vertices =
129 _liball->getVertices();
130 std::array<double, NumSpaceDim> cell_size;
131 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
132 cell_size[d] = _global_grid->globalMesh().cellSize( d );
133 std::array<int, NumSpaceDim> cell_index_lo, cell_index_hi;
134 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
135 cell_index_lo[d] =
136 std::rint( updated_vertices.at( 0 )[d] / cell_size[d] );
137 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
138 cell_index_hi[d] =
139 std::rint( updated_vertices.at( 1 )[d] / cell_size[d] );
140 std::array<int, NumSpaceDim> num_cell;
141 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
142 num_cell[d] = cell_index_hi[d] - cell_index_lo[d];
143 // Create new global grid
144 // todo(sschulz): Can GlobalGrid be constructed with an already
145 // cartesian communicator? MPI_Cart_Create is called with the given
146 // comm.
147 std::array<bool, NumSpaceDim> periodic;
148 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
149 periodic[d] = _global_grid->isPeriodic( d );
150 std::shared_ptr<GlobalGrid<mesh_type>> global_grid =
151 createGlobalGrid( _comm, global_mesh, periodic, partitioner );
152 global_grid->setNumCellAndOffset( num_cell, cell_index_lo );
153 _global_grid = global_grid;
154
155 Kokkos::Profiling::popRegion();
156 return _global_grid;
157 }
158
161 const std::array<double, NumSpaceDim * 2> getInternalVertices() const
162 {
163 std::array<double, NumSpaceDim * 2> internal_vertices;
164 std::vector<ALL::Point<double>> lb_vertices = _liball->getVertices();
165 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
166 internal_vertices[d] = lb_vertices.at( 0 )[d];
167 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
168 internal_vertices[d + NumSpaceDim] = lb_vertices.at( 1 )[d];
169 return internal_vertices;
170 // todo(sschulz): Is this ok to pass arrays?
171 }
172
175 const std::array<double, NumSpaceDim * 2> getVertices() const
176 {
177 std::array<double, NumSpaceDim> cell_size;
178 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
179 cell_size[d] = _global_grid->globalMesh().cellSize( d );
180 std::array<double, NumSpaceDim * 2> vertices;
181 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
182 vertices[d] = _global_grid->globalOffset( d ) * cell_size[d];
183 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
184 vertices[d + NumSpaceDim] =
185 vertices[d] + _global_grid->ownedNumCell( d ) * cell_size[d];
186 return vertices;
187 // todo(sschulz): Is this ok to pass arrays?
188 }

192 double getImbalance() const
193 {
194 const double local_work = _liball->getWork();
195 double max_work, min_work;
196 MPI_Allreduce( &local_work, &max_work, 1, MPI_DOUBLE, MPI_MAX, _comm );
197 MPI_Allreduce( &local_work, &min_work, 1, MPI_DOUBLE, MPI_MIN, _comm );
198 return ( max_work - min_work ) / ( max_work + min_work );
199 }

210 private:
212 void setupLibrary()
213 {
214 // todo(sschulz): Investigate, why usage of num_space_dim instead of
215 // NumSpaceDim causes a linker error.
216 _liball = std::make_shared<ALL::ALL<double, double>>( ALL::TENSOR,
217 NumSpaceDim, 0 );
218 std::vector<int> loc( NumSpaceDim, 0 );
219 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
220 loc[d] = _global_grid->dimBlockId( d );
221 std::vector<int> size( NumSpaceDim, 0 );
222 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
223 size[d] = _global_grid->dimNumBlock( d );
224 _liball->setProcGridParams( loc, size );
225 _liball->setCommunicator( _comm );
226 int rank;
227 MPI_Comm_rank( _global_grid->comm(), &rank );
228 _liball->setProcTag( rank );
229 _liball->setup();
230 // Set initial vertices
231 std::array<double, NumSpaceDim> cell_size;
232 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
233 cell_size[d] = _global_grid->globalMesh().cellSize( d );
234 std::vector<ALL::Point<double>> lb_vertices(
235 2, ALL::Point<double>( NumSpaceDim ) );
236 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
237 lb_vertices.at( 0 )[d] =
238 _global_grid->globalOffset( d ) * cell_size[d];
239 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
240 lb_vertices.at( 1 )[d] =
241 lb_vertices.at( 0 )[d] +
242 _global_grid->ownedNumCell( d ) * cell_size[d];
243 _liball->setVertices( lb_vertices );
244 }
245
246 std::shared_ptr<ALL::ALL<double, double>> _liball;
247 std::shared_ptr<GlobalGrid<mesh_type>> _global_grid;
248 MPI_Comm _comm;
249};
