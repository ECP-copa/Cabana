79template <std::size_t NumSpaceDim>
80class ManualBlockPartitioner : public BlockPartitioner<NumSpaceDim>
81{
82 public:
84 static constexpr std::size_t num_space_dim = NumSpaceDim;
85
90 ManualBlockPartitioner( const std::array<int, NumSpaceDim>& ranks_per_dim )
91 : _ranks_per_dim( ranks_per_dim )
92 {
93 }
94
99 std::array<int, NumSpaceDim>
100 ranksPerDimension( MPI_Comm comm,
101 const std::array<int, NumSpaceDim>& ) const override
102 {
103 int comm_size;
104 MPI_Comm_size( comm, &comm_size );
105 int nrank = 1;
106 for ( std::size_t d = 0; d < num_space_dim; ++d )
107 nrank *= _ranks_per_dim[d];
108 if ( comm_size != nrank )
109 throw std::runtime_error(
110 "ManualPartitioner ranks do not match comm size" );
111 return _ranks_per_dim;
112 }
113
124 void ownedCellInfo(
125 MPI_Comm cart_comm,
126 const std::array<int, num_space_dim>& global_cells_per_dim,
127 std::array<int, num_space_dim>& owned_num_cell,
128 std::array<int, num_space_dim>& global_cell_offset ) const override
129 {
130 // Get the cells per dimension and the remainder.
131 std::array<int, num_space_dim> cells_per_dim;
132 std::array<int, num_space_dim> dim_remainder;
133 std::array<int, num_space_dim> cart_rank;
134 averageCellInfo( cart_comm, global_cells_per_dim, cart_rank,
135 cells_per_dim, dim_remainder );
136
137 // Compute the global cell offset and the local low corner on this rank
138 // by computing the starting global cell index via exclusive scan.
139 global_cell_offset =
140 globalCellOffsetHelper( cart_rank, cells_per_dim, dim_remainder );
141
142 // Compute the number of local cells in this rank in each dimension.
143 owned_num_cell =
144 ownedCellsHelper( cart_rank, cells_per_dim, dim_remainder );
145 }
146
154 std::array<int, num_space_dim> ownedCellsPerDimension(
155 MPI_Comm cart_comm,
156 const std::array<int, num_space_dim>& global_cells_per_dim ) const
157 {
158 // Get the cells per dimension and the remainder.
159 std::array<int, num_space_dim> cells_per_dim;
160 std::array<int, num_space_dim> dim_remainder;
161 std::array<int, num_space_dim> cart_rank;
162 averageCellInfo( cart_comm, global_cells_per_dim, cart_rank,
163 cells_per_dim, dim_remainder );
164
165 // Compute the number of local cells in this rank in each dimension.
166 return ownedCellsHelper( cart_rank, cells_per_dim, dim_remainder );
167 }
168
169 private:
180 inline void
181 averageCellInfo( MPI_Comm cart_comm,
182 const std::array<int, num_space_dim>& global_cells_per_dim,
183 std::array<int, num_space_dim>& cart_rank,
184 std::array<int, num_space_dim>& cells_per_dim,
185 std::array<int, num_space_dim>& dim_remainder ) const
186 {
187 // Get the Cartesian size and topology index of this rank.
188 int linear_rank;
189 MPI_Comm_rank( cart_comm, &linear_rank );
190 MPI_Cart_coords( cart_comm, linear_rank, num_space_dim,
191 cart_rank.data() );
192 // Get the cells per dimension and the remainder.
193 for ( std::size_t d = 0; d < num_space_dim; ++d )
194 {
195 cells_per_dim[d] = global_cells_per_dim[d] / _ranks_per_dim[d];
196 dim_remainder[d] = global_cells_per_dim[d] % _ranks_per_dim[d];
197 }
198 }
199
209 inline std::array<int, num_space_dim> ownedCellsHelper(
210 const std::array<int, num_space_dim>& cart_rank,
211 const std::array<int, num_space_dim>& cells_per_dim,
212 const std::array<int, num_space_dim>& dim_remainder ) const
213 {
214 std::array<int, num_space_dim> owned_num_cell;
215 // Compute the number of local cells in this rank in each dimension.
216 for ( std::size_t d = 0; d < num_space_dim; ++d )
217 {
218 owned_num_cell[d] = cells_per_dim[d];
219 if ( dim_remainder[d] > cart_rank[d] )
220 ++owned_num_cell[d];
221 }
222 return owned_num_cell;
223 }
224
234 inline std::array<int, num_space_dim> globalCellOffsetHelper(
235 const std::array<int, num_space_dim>& cart_rank,
236 const std::array<int, num_space_dim>& cells_per_dim,
237 const std::array<int, num_space_dim>& dim_remainder ) const
238 {
239 std::array<int, num_space_dim> global_cell_offset;
240 // Compute the global cell offset and the local low corner on this rank
241 // by computing the starting global cell index via exclusive scan.
242 for ( std::size_t d = 0; d < num_space_dim; ++d )
243 {
244 global_cell_offset[d] = 0;
245 for ( int r = 0; r < cart_rank[d]; ++r )
246 {
247 global_cell_offset[d] += cells_per_dim[d];
248 if ( dim_remainder[d] > r )
249 ++global_cell_offset[d];
250 }
251 }
252 return global_cell_offset;
253 }
254
255 private:
256 std::array<int, NumSpaceDim> _ranks_per_dim;
257};

271template <std::size_t NumSpaceDim>
272class DimBlockPartitioner : public BlockPartitioner<NumSpaceDim>
273{
274 public:
276 static constexpr std::size_t num_space_dim = NumSpaceDim;
277
279 DimBlockPartitioner()
280 {
281 // Initialize so that all ranks will be modified when requested.
282 initUnpartitionedDims();
283 };
284
289 DimBlockPartitioner( const int dim )
290 {
291 initUnpartitionedDims();
292
293 // Only partition in the other dimensions.
294 _unpartitioned_dim[dim] = 1;
295 };
296
302 DimBlockPartitioner( const int dim_1, const int dim_2 )
303 {
304 static_assert(
305 NumSpaceDim > 2,
306 "Cannot partition 2d system with 2 unpartitioned dimensions." );
307
308 initUnpartitionedDims();
309
310 // Only partition in the third dimension.
311 _unpartitioned_dim[dim_1] = 1;
312 _unpartitioned_dim[dim_2] = 1;
313 };
314
319 std::array<int, NumSpaceDim>
320 ranksPerDimension( MPI_Comm comm,
321 const std::array<int, NumSpaceDim>& ) const override
322 {
323 int comm_size;
324 MPI_Comm_size( comm, &comm_size );
325 auto ranks_per_dim = _unpartitioned_dim;
326 MPI_Dims_create( comm_size, NumSpaceDim, ranks_per_dim.data() );
327
328 return ranks_per_dim;
329 }

407 inline void
408 averageCellInfo( MPI_Comm cart_comm,
409 const std::array<int, num_space_dim>& global_cells_per_dim,
410 std::array<int, num_space_dim>& cart_rank,
411 std::array<int, num_space_dim>& cells_per_dim,
412 std::array<int, num_space_dim>& dim_remainder ) const
413 {
414 // Get the Cartesian size and topology index of this rank.
415 std::array<int, num_space_dim> ranks_per_dim;
416 std::array<int, num_space_dim> cart_period;
417 MPI_Cart_get( cart_comm, num_space_dim, ranks_per_dim.data(),
418 cart_period.data(), cart_rank.data() );
419
420 // Get the cells per dimension and the remainder.
421 for ( std::size_t d = 0; d < num_space_dim; ++d )
422 {
423 cells_per_dim[d] = global_cells_per_dim[d] / ranks_per_dim[d];
424 dim_remainder[d] = global_cells_per_dim[d] % ranks_per_dim[d];
425 }
426 }
