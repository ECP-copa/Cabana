172 // Synchronously move data between a source and destination AoSoA by executing
173 // the forward communication plan.
174 template<class Distributor_t, class AoSoA_t>
175 void distributeData(
176  const Distributor_t& distributor,
177  const AoSoA_t& src,
178  AoSoA_t& dst,
179  int mpi_tag,
180  typename std::enable_if<(is_distributor<Distributor_t>::value &&
181  is_aosoa<AoSoA_t>::value),
182  int>::type * = 0 )
183 {
184  // Get the MPI rank we are currently on.
185  int my_rank = -1;
186  MPI_Comm_rank( distributor.comm(), &my_rank );
187 
188  // Get the number of neighbors.
189  int num_n = distributor.numNeighbor();
190 
191  // Calculate the number of elements that are staying on this rank and
192  // therefore can be directly copied. If any of the neighbor ranks are this
193  // rank it will be stored in first position (i.e. the first neighbor in
194  // the local list is always yourself if you are sending to yourself).
195  std::size_t num_stay = ( distributor.neighborRank(0) == my_rank )
196  ? distributor.numExport(0) : 0;
197 
198  // Allocate a send buffer.
199  std::size_t num_send = distributor.totalNumExport() - num_stay;
200  Kokkos::View<typename AoSoA_t::tuple_type*,
201  typename Distributor_t::memory_space>
202  send_buffer(
203  Kokkos::ViewAllocateWithoutInitializing("distributor_send_buffer"),
204  num_send );
205 
206  // Allocate a receive buffer.
207  Kokkos::View<typename AoSoA_t::tuple_type*,
208  typename Distributor_t::memory_space>
209  recv_buffer(
210  Kokkos::ViewAllocateWithoutInitializing("distributor_recv_buffer"),
211  distributor.totalNumImport() );
212 
213  // Get the steering vector for the sends.
214  auto steering = distributor.getExportSteering();
215 
216  // Gather the exports from the source AoSoA into the tuple-contiguous send
217  // buffer or the receive buffer if the data is staying. We know that the
218  // steering vector is ordered such that the data staying on this rank
219  // comes first.
220  auto build_send_buffer_func =
221  KOKKOS_LAMBDA( const std::size_t i )
222  {
223  auto tpl = src.getTuple( steering(i) );
224  if ( i < num_stay )
225  recv_buffer( i ) = tpl;
226  else
227  send_buffer( i - num_stay ) = tpl;
228  };
229  Kokkos::RangePolicy<typename Distributor_t::execution_space>
230  build_send_buffer_policy( 0, distributor.totalNumExport() );
231  Kokkos::parallel_for( "Cabana::Impl::distributeData::build_send_buffer",
232  build_send_buffer_policy,
233  build_send_buffer_func );
234  Kokkos::fence();
235 
236  // Post non-blocking receives.
237  std::vector<MPI_Request> requests;
238  requests.reserve( num_n );
239  std::pair<std::size_t,std::size_t> recv_range = { 0, 0 };
240  for ( int n = 0; n < num_n; ++n )
241  {
242  recv_range.second = recv_range.first + distributor.numImport(n);
243 
244  if ( (distributor.numImport(n) > 0) &&
245  (distributor.neighborRank(n) != my_rank) )
246  {
247  auto recv_subview = Kokkos::subview( recv_buffer, recv_range );
248 
249  requests.push_back( MPI_Request() );
250 
251  MPI_Irecv( recv_subview.data(),
252  recv_subview.size() * sizeof(typename AoSoA_t::tuple_type),
253  MPI_BYTE,
254  distributor.neighborRank(n),
255  mpi_tag,
256  distributor.comm(),
257  &(requests.back()) );
258  }
259 
260  recv_range.first = recv_range.second;
261  }
262 
263  // Do blocking sends.
264  std::pair<std::size_t,std::size_t> send_range = { 0, 0 };
265  for ( int n = 0; n < num_n; ++n )
266  {
267  if ( (distributor.numExport(n) > 0) &&
268  (distributor.neighborRank(n) != my_rank) )
269  {
270  send_range.second = send_range.first + distributor.numExport(n);
271 
272  auto send_subview = Kokkos::subview( send_buffer, send_range );
273 
274  MPI_Send( send_subview.data(),
275  send_subview.size() * sizeof(typename AoSoA_t::tuple_type),
276  MPI_BYTE,
277  distributor.neighborRank(n),
278  mpi_tag,
279  distributor.comm() );
280 
281  send_range.first = send_range.second;
282  }
283  }
284 
285  // Wait on non-blocking receives.
286  std::vector<MPI_Status> status( requests.size() );
287  const int ec =
288  MPI_Waitall( requests.size(), requests.data(), status.data() );
289  assert( MPI_SUCCESS == ec );
290 
291  // Extract the receive buffer into the destination AoSoA.
292  auto extract_recv_buffer_func =
293  KOKKOS_LAMBDA( const std::size_t i )
294  { dst.setTuple( i, recv_buffer(i) ); };
295  Kokkos::RangePolicy<typename Distributor_t::execution_space>
296  extract_recv_buffer_policy( 0, distributor.totalNumImport() );
297  Kokkos::parallel_for( "Cabana::Impl::distributeData::extract_recv_buffer",
298  extract_recv_buffer_policy,
299  extract_recv_buffer_func );
300  Kokkos::fence();
301 
302  // Barrier before completing to ensure synchronization.
303  MPI_Barrier( distributor.comm() );
304 }

336 template<class Distributor_t, class AoSoA_t>
337 void migrate( const Distributor_t& distributor,
338  const AoSoA_t& src,
339  AoSoA_t& dst,
340  int mpi_tag = 1001,
341  typename std::enable_if<(is_distributor<Distributor_t>::value &&
342  is_aosoa<AoSoA_t>::value),
343  int>::type * = 0 )
344 {
345  // Check that src and dst are the right size.
346  if ( src.size() != distributor.exportSize() )
347  throw std::runtime_error("Source is the wrong size for migration!");
348  if ( dst.size() != distributor.totalNumImport() )
349  throw std::runtime_error("Destination is the wrong size for migration!");
350 
351  // Move the data.
352  Impl::distributeData( distributor, src, dst, mpi_tag );
353 }

384 template<class Distributor_t, class AoSoA_t>
385 void migrate( const Distributor_t& distributor,
386  AoSoA_t& aosoa,
387  int mpi_tag = 1001,
388  typename std::enable_if<(is_distributor<Distributor_t>::value &&
389  is_aosoa<AoSoA_t>::value),
390  int>::type * = 0 )
391 {
392  // Check that the AoSoA is the right size.
393  if ( aosoa.size() != distributor.exportSize() )
394  throw std::runtime_error("AoSoA is the wrong size for migration!");
395 
396  // Determine if the source of destination decomposition has more data on
397  // this rank.
398  bool dst_is_bigger =
399  ( distributor.totalNumImport() > distributor.exportSize() );
400 
401  // If the destination decomposition is bigger than the source
402  // decomposition resize now so we have enough space to do the operation.
403  if ( dst_is_bigger )
404  aosoa.resize( distributor.totalNumImport() );
405 
406  // Move the data.
407  Impl::distributeData( distributor, aosoa, aosoa, mpi_tag );
408 
409  // If the destination decomposition is smaller than the source
410  // decomposition resize after we have moved the data.
411  if ( !dst_is_bigger )
412  aosoa.resize( distributor.totalNumImport() );
413 }
