169 // Synchronously move data between a source and destination AoSoA by executing
170 // the forward communication plan.
171 template <class Distributor_t, class AoSoA_t>
172 void distributeData(
173  const Distributor_t& distributor, const AoSoA_t& src, AoSoA_t& dst,
174  typename std::enable_if<( is_distributor<Distributor_t>::value &&
175  is_aosoa<AoSoA_t>::value ),
176  int>::type* = 0 )
177 {
178  // Get the MPI rank we are currently on.
179  int my_rank = -1;
180  MPI_Comm_rank( distributor.comm(), &my_rank );
181 
182  // Get the number of neighbors.
183  int num_n = distributor.numNeighbor();
184 
185  // Calculate the number of elements that are staying on this rank and
186  // therefore can be directly copied. If any of the neighbor ranks are this
187  // rank it will be stored in first position (i.e. the first neighbor in
188  // the local list is always yourself if you are sending to yourself).
189  std::size_t num_stay =
190  ( num_n > 0 && distributor.neighborRank( 0 ) == my_rank )
191  ? distributor.numExport( 0 )
192  : 0;
193 
194  // Allocate a send buffer.
195  std::size_t num_send = distributor.totalNumExport() - num_stay;
196  Kokkos::View<typename AoSoA_t::tuple_type*,
197  typename Distributor_t::memory_space>
198  send_buffer( Kokkos::ViewAllocateWithoutInitializing(
199  "distributor_send_buffer" ),
200  num_send );
201 
202  // Allocate a receive buffer.
203  Kokkos::View<typename AoSoA_t::tuple_type*,
204  typename Distributor_t::memory_space>
205  recv_buffer( Kokkos::ViewAllocateWithoutInitializing(
206  "distributor_recv_buffer" ),
207  distributor.totalNumImport() );
208 
209  // Get the steering vector for the sends.
210  auto steering = distributor.getExportSteering();
211 
212  // Gather the exports from the source AoSoA into the tuple-contiguous send
213  // buffer or the receive buffer if the data is staying. We know that the
214  // steering vector is ordered such that the data staying on this rank
215  // comes first.
216  auto build_send_buffer_func = KOKKOS_LAMBDA( const std::size_t i )
217  {
218  auto tpl = src.getTuple( steering( i ) );
219  if ( i < num_stay )
220  recv_buffer( i ) = tpl;
221  else
222  send_buffer( i - num_stay ) = tpl;
223  };
224  Kokkos::RangePolicy<typename Distributor_t::execution_space>
225  build_send_buffer_policy( 0, distributor.totalNumExport() );
226  Kokkos::parallel_for( "Cabana::Impl::distributeData::build_send_buffer",
227  build_send_buffer_policy, build_send_buffer_func );
228  Kokkos::fence();
229 
230  // The distributor has its own communication space so choose any tag.
231  const int mpi_tag = 1234;
232 
233  // Post non-blocking receives.
234  std::vector<MPI_Request> requests;
235  requests.reserve( num_n );
236  std::pair<std::size_t, std::size_t> recv_range = { 0, 0 };
237  for ( int n = 0; n < num_n; ++n )
238  {
239  recv_range.second = recv_range.first + distributor.numImport( n );
240 
241  if ( ( distributor.numImport( n ) > 0 ) &&
242  ( distributor.neighborRank( n ) != my_rank ) )
243  {
244  auto recv_subview = Kokkos::subview( recv_buffer, recv_range );
245 
246  requests.push_back( MPI_Request() );
247 
248  MPI_Irecv( recv_subview.data(),
249  recv_subview.size() *
250  sizeof( typename AoSoA_t::tuple_type ),
251  MPI_BYTE, distributor.neighborRank( n ), mpi_tag,
252  distributor.comm(), &( requests.back() ) );
253  }
254 
255  recv_range.first = recv_range.second;
256  }
257 
258  // Do blocking sends.
259  std::pair<std::size_t, std::size_t> send_range = { 0, 0 };
260  for ( int n = 0; n < num_n; ++n )
261  {
262  if ( ( distributor.numExport( n ) > 0 ) &&
263  ( distributor.neighborRank( n ) != my_rank ) )
264  {
265  send_range.second = send_range.first + distributor.numExport( n );
266 
267  auto send_subview = Kokkos::subview( send_buffer, send_range );
268 
269  MPI_Send( send_subview.data(),
270  send_subview.size() *
271  sizeof( typename AoSoA_t::tuple_type ),
272  MPI_BYTE, distributor.neighborRank( n ), mpi_tag,
273  distributor.comm() );
274 
275  send_range.first = send_range.second;
276  }
277  }
278 
279  // Wait on non-blocking receives.
280  std::vector<MPI_Status> status( requests.size() );
281  const int ec =
282  MPI_Waitall( requests.size(), requests.data(), status.data() );
283  if ( MPI_SUCCESS != ec )
284  throw std::logic_error( "Failed MPI Communication" );
285 
286  // Extract the receive buffer into the destination AoSoA.
287  auto extract_recv_buffer_func = KOKKOS_LAMBDA( const std::size_t i )
288  {
289  dst.setTuple( i, recv_buffer( i ) );
290  };
291  Kokkos::RangePolicy<typename Distributor_t::execution_space>
292  extract_recv_buffer_policy( 0, distributor.totalNumImport() );
293  Kokkos::parallel_for( "Cabana::Impl::distributeData::extract_recv_buffer",
294  extract_recv_buffer_policy,
295  extract_recv_buffer_func );
296  Kokkos::fence();
297 
298  // Barrier before completing to ensure synchronization.
299  MPI_Barrier( distributor.comm() );
300 }

327 template <class Distributor_t, class AoSoA_t>
328 void migrate( const Distributor_t& distributor, const AoSoA_t& src,
329  AoSoA_t& dst,
330  typename std::enable_if<( is_distributor<Distributor_t>::value &&
331  is_aosoa<AoSoA_t>::value ),
332  int>::type* = 0 )
333 {
334  // Check that src and dst are the right size.
335  if ( src.size() != distributor.exportSize() )
336  throw std::runtime_error( "Source is the wrong size for migration!" );
337  if ( dst.size() != distributor.totalNumImport() )
338  throw std::runtime_error(
339  "Destination is the wrong size for migration!" );
340 
341  // Move the data.
342  Impl::distributeData( distributor, src, dst );
343 }

369 template <class Distributor_t, class AoSoA_t>
370 void migrate( const Distributor_t& distributor, AoSoA_t& aosoa,
371  typename std::enable_if<( is_distributor<Distributor_t>::value &&
372  is_aosoa<AoSoA_t>::value ),
373  int>::type* = 0 )
374 {
375  // Check that the AoSoA is the right size.
376  if ( aosoa.size() != distributor.exportSize() )
377  throw std::runtime_error( "AoSoA is the wrong size for migration!" );
378 
379  // Determine if the source of destination decomposition has more data on
380  // this rank.
381  bool dst_is_bigger =
382  ( distributor.totalNumImport() > distributor.exportSize() );
383 
384  // If the destination decomposition is bigger than the source
385  // decomposition resize now so we have enough space to do the operation.
386  if ( dst_is_bigger )
387  aosoa.resize( distributor.totalNumImport() );
388 
389  // Move the data.
390  Impl::distributeData( distributor, aosoa, aosoa );
391 
392  // If the destination decomposition is smaller than the source
393  // decomposition resize after we have moved the data.
394  if ( !dst_is_bigger )
395  aosoa.resize( distributor.totalNumImport() );
396 }
