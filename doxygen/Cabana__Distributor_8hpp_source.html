169// Synchronously move data between a source and destination AoSoA by executing
170// the forward communication plan.
171template <class Distributor_t, class AoSoA_t>
172void distributeData(
173 const Distributor_t& distributor, const AoSoA_t& src, AoSoA_t& dst,
174 typename std::enable_if<( is_distributor<Distributor_t>::value &&
175 is_aosoa<AoSoA_t>::value ),
176 int>::type* = 0 )
177{
178 Kokkos::Profiling::pushRegion( "Cabana::migrate" );
179
180 // Get the MPI rank we are currently on.
181 int my_rank = -1;
182 MPI_Comm_rank( distributor.comm(), &my_rank );
183
184 // Get the number of neighbors.
185 int num_n = distributor.numNeighbor();
186
187 // Calculate the number of elements that are staying on this rank and
188 // therefore can be directly copied. If any of the neighbor ranks are this
189 // rank it will be stored in first position (i.e. the first neighbor in
190 // the local list is always yourself if you are sending to yourself).
191 std::size_t num_stay =
192 ( num_n > 0 && distributor.neighborRank( 0 ) == my_rank )
193 ? distributor.numExport( 0 )
194 : 0;
195
196 // Allocate a send buffer.
197 std::size_t num_send = distributor.totalNumExport() - num_stay;
198 Kokkos::View<typename AoSoA_t::tuple_type*,
199 typename Distributor_t::memory_space>
200 send_buffer( Kokkos::ViewAllocateWithoutInitializing(
201 "distributor_send_buffer" ),
202 num_send );
203
204 // Allocate a receive buffer.
205 Kokkos::View<typename AoSoA_t::tuple_type*,
206 typename Distributor_t::memory_space>
207 recv_buffer( Kokkos::ViewAllocateWithoutInitializing(
208 "distributor_recv_buffer" ),
209 distributor.totalNumImport() );
210
211 // Get the steering vector for the sends.
212 auto steering = distributor.getExportSteering();
213
214 // Gather the exports from the source AoSoA into the tuple-contiguous send
215 // buffer or the receive buffer if the data is staying. We know that the
216 // steering vector is ordered such that the data staying on this rank
217 // comes first.
218 auto build_send_buffer_func = KOKKOS_LAMBDA( const std::size_t i )
219 {
220 auto tpl = src.getTuple( steering( i ) );
221 if ( i < num_stay )
222 recv_buffer( i ) = tpl;
223 else
224 send_buffer( i - num_stay ) = tpl;
225 };
226 Kokkos::RangePolicy<typename Distributor_t::execution_space>
227 build_send_buffer_policy( 0, distributor.totalNumExport() );
228 Kokkos::parallel_for( "Cabana::Impl::distributeData::build_send_buffer",
229 build_send_buffer_policy, build_send_buffer_func );
230 Kokkos::fence();
231
232 // The distributor has its own communication space so choose any tag.
233 const int mpi_tag = 1234;
234
235 // Post non-blocking receives.
236 std::vector<MPI_Request> requests;
237 requests.reserve( num_n );
238 std::pair<std::size_t, std::size_t> recv_range = { 0, 0 };
239 for ( int n = 0; n < num_n; ++n )
240 {
241 recv_range.second = recv_range.first + distributor.numImport( n );
242
243 if ( ( distributor.numImport( n ) > 0 ) &&
244 ( distributor.neighborRank( n ) != my_rank ) )
245 {
246 auto recv_subview = Kokkos::subview( recv_buffer, recv_range );
247
248 requests.push_back( MPI_Request() );
249
250 MPI_Irecv( recv_subview.data(),
251 recv_subview.size() *
252 sizeof( typename AoSoA_t::tuple_type ),
253 MPI_BYTE, distributor.neighborRank( n ), mpi_tag,
254 distributor.comm(), &( requests.back() ) );
255 }
256
257 recv_range.first = recv_range.second;
258 }
259
260 // Do blocking sends.
261 std::pair<std::size_t, std::size_t> send_range = { 0, 0 };
262 for ( int n = 0; n < num_n; ++n )
263 {
264 if ( ( distributor.numExport( n ) > 0 ) &&
265 ( distributor.neighborRank( n ) != my_rank ) )
266 {
267 send_range.second = send_range.first + distributor.numExport( n );
268
269 auto send_subview = Kokkos::subview( send_buffer, send_range );
270
271 MPI_Send( send_subview.data(),
272 send_subview.size() *
273 sizeof( typename AoSoA_t::tuple_type ),
274 MPI_BYTE, distributor.neighborRank( n ), mpi_tag,
275 distributor.comm() );
276
277 send_range.first = send_range.second;
278 }
279 }
280
281 // Wait on non-blocking receives.
282 std::vector<MPI_Status> status( requests.size() );
283 const int ec =
284 MPI_Waitall( requests.size(), requests.data(), status.data() );
285 if ( MPI_SUCCESS != ec )
286 throw std::logic_error( "Failed MPI Communication" );
287
288 // Extract the receive buffer into the destination AoSoA.
289 auto extract_recv_buffer_func = KOKKOS_LAMBDA( const std::size_t i )
290 {
291 dst.setTuple( i, recv_buffer( i ) );
292 };
293 Kokkos::RangePolicy<typename Distributor_t::execution_space>
294 extract_recv_buffer_policy( 0, distributor.totalNumImport() );
295 Kokkos::parallel_for( "Cabana::Impl::distributeData::extract_recv_buffer",
296 extract_recv_buffer_policy,
297 extract_recv_buffer_func );
298 Kokkos::fence();
299
300 // Barrier before completing to ensure synchronization.
301 MPI_Barrier( distributor.comm() );
302 Kokkos::Profiling::popRegion();
303}

330template <class Distributor_t, class AoSoA_t>
331void migrate( const Distributor_t& distributor, const AoSoA_t& src,
332 AoSoA_t& dst,
333 typename std::enable_if<( is_distributor<Distributor_t>::value &&
334 is_aosoa<AoSoA_t>::value ),
335 int>::type* = 0 )
336{
337 // Check that src and dst are the right size.
338 if ( src.size() != distributor.exportSize() )
339 throw std::runtime_error( "Source is the wrong size for migration!" );
340 if ( dst.size() != distributor.totalNumImport() )
341 throw std::runtime_error(
342 "Destination is the wrong size for migration!" );
343
344 // Move the data.
345 Impl::distributeData( distributor, src, dst );
346}

372template <class Distributor_t, class AoSoA_t>
373void migrate( const Distributor_t& distributor, AoSoA_t& aosoa,
374 typename std::enable_if<( is_distributor<Distributor_t>::value &&
375 is_aosoa<AoSoA_t>::value ),
376 int>::type* = 0 )
377{
378 // Check that the AoSoA is the right size.
379 if ( aosoa.size() != distributor.exportSize() )
380 throw std::runtime_error( "AoSoA is the wrong size for migration!" );
381
382 // Determine if the source of destination decomposition has more data on
383 // this rank.
384 bool dst_is_bigger =
385 ( distributor.totalNumImport() > distributor.exportSize() );
386
387 // If the destination decomposition is bigger than the source
388 // decomposition resize now so we have enough space to do the operation.
389 if ( dst_is_bigger )
390 aosoa.resize( distributor.totalNumImport() );
391
392 // Move the data.
393 Impl::distributeData( distributor, aosoa, aosoa );
394
395 // If the destination decomposition is smaller than the source
396 // decomposition resize after we have moved the data.
397 if ( !dst_is_bigger )
398 aosoa.resize( distributor.totalNumImport() );
399}

424template <class Distributor_t, class Slice_t>
425void migrate( const Distributor_t& distributor, const Slice_t& src,
426 Slice_t& dst,
427 typename std::enable_if<( is_distributor<Distributor_t>::value &&
428 is_slice<Slice_t>::value ),
429 int>::type* = 0 )
430{
431 // Check that src and dst are the right size.
432 if ( src.size() != distributor.exportSize() )
433 throw std::runtime_error( "Source is the wrong size for migration!" );
434 if ( dst.size() != distributor.totalNumImport() )
435 throw std::runtime_error(
436 "Destination is the wrong size for migration!" );
437
438 // Get the number of components in the slices.
439 size_t num_comp = 1;
440 for ( size_t d = 2; d < src.viewRank(); ++d )
441 num_comp *= src.extent( d );
442
443 // Get the raw slice data.
444 auto src_data = src.data();
445 auto dst_data = dst.data();
446
447 // Get the MPI rank we are currently on.
448 int my_rank = -1;
449 MPI_Comm_rank( distributor.comm(), &my_rank );
450
451 // Get the number of neighbors.
452 int num_n = distributor.numNeighbor();
453
454 // Calculate the number of elements that are staying on this rank and
455 // therefore can be directly copied. If any of the neighbor ranks are this
456 // rank it will be stored in first position (i.e. the first neighbor in
457 // the local list is always yourself if you are sending to yourself).
458 std::size_t num_stay =
459 ( num_n > 0 && distributor.neighborRank( 0 ) == my_rank )
460 ? distributor.numExport( 0 )
461 : 0;
462
463 // Allocate a send buffer. Note this one is layout right so the components
464 // of each element are consecutive in memory.
465 std::size_t num_send = distributor.totalNumExport() - num_stay;
466 Kokkos::View<typename Slice_t::value_type**, Kokkos::LayoutRight,
467 typename Distributor_t::memory_space>
468 send_buffer( Kokkos::ViewAllocateWithoutInitializing(
469 "distributor_send_buffer" ),
470 num_send, num_comp );
471
472 // Allocate a receive buffer. Note this one is layout right so the
473 // components of each element are consecutive in memory.
474 Kokkos::View<typename Slice_t::value_type**, Kokkos::LayoutRight,
475 typename Distributor_t::memory_space>
476 recv_buffer( Kokkos::ViewAllocateWithoutInitializing(
477 "distributor_recv_buffer" ),
478 distributor.totalNumImport(), num_comp );
479
480 // Get the steering vector for the sends.
481 auto steering = distributor.getExportSteering();
482
483 // Gather from the source Slice into the contiguous send buffer or,
484 // if it is part of the local copy, put it directly in the destination
485 // Slice.
486 auto build_send_buffer_func = KOKKOS_LAMBDA( const std::size_t i )
487 {
488 auto s_src = Slice_t::index_type::s( steering( i ) );
489 auto a_src = Slice_t::index_type::a( steering( i ) );
490 std::size_t src_offset = s_src * src.stride( 0 ) + a_src;
491 if ( i < num_stay )
492 for ( std::size_t n = 0; n < num_comp; ++n )
493 recv_buffer( i, n ) =
494 src_data[src_offset + n * Slice_t::vector_length];
495 else
496 for ( std::size_t n = 0; n < num_comp; ++n )
497 send_buffer( i - num_stay, n ) =
498 src_data[src_offset + n * Slice_t::vector_length];
499 };
500 Kokkos::RangePolicy<typename Distributor_t::execution_space>
501 build_send_buffer_policy( 0, distributor.totalNumExport() );
502 Kokkos::parallel_for( "Cabana::migrate::build_send_buffer",
503 build_send_buffer_policy, build_send_buffer_func );
504 Kokkos::fence();
505
506 // The distributor has its own communication space so choose any tag.
507 const int mpi_tag = 1234;
508
509 // Post non-blocking receives.
510 std::vector<MPI_Request> requests;
511 requests.reserve( num_n );
512 std::pair<std::size_t, std::size_t> recv_range = { 0, 0 };
513 for ( int n = 0; n < num_n; ++n )
514 {
515 recv_range.second = recv_range.first + distributor.numImport( n );
516
517 if ( ( distributor.numImport( n ) > 0 ) &&
518 ( distributor.neighborRank( n ) != my_rank ) )
519 {
520 auto recv_subview =
521 Kokkos::subview( recv_buffer, recv_range, Kokkos::ALL );
522
523 requests.push_back( MPI_Request() );
524
525 MPI_Irecv( recv_subview.data(),
526 recv_subview.size() *
527 sizeof( typename Slice_t::value_type ),
528 MPI_BYTE, distributor.neighborRank( n ), mpi_tag,
529 distributor.comm(), &( requests.back() ) );
530 }
531
532 recv_range.first = recv_range.second;
533 }
534
535 // Do blocking sends.
536 std::pair<std::size_t, std::size_t> send_range = { 0, 0 };
537 for ( int n = 0; n < num_n; ++n )
538 {
539 if ( ( distributor.numExport( n ) > 0 ) &&
540 ( distributor.neighborRank( n ) != my_rank ) )
541 {
542 send_range.second = send_range.first + distributor.numExport( n );
543
544 auto send_subview =
545 Kokkos::subview( send_buffer, send_range, Kokkos::ALL );
546
547 MPI_Send( send_subview.data(),
548 send_subview.size() *
549 sizeof( typename Slice_t::value_type ),
550 MPI_BYTE, distributor.neighborRank( n ), mpi_tag,
551 distributor.comm() );
552
553 send_range.first = send_range.second;
554 }
555 }
556
557 // Wait on non-blocking receives.
558 std::vector<MPI_Status> status( requests.size() );
559 const int ec =
560 MPI_Waitall( requests.size(), requests.data(), status.data() );
561 if ( MPI_SUCCESS != ec )
562 throw std::logic_error( "Failed MPI Communication" );
563
564 // Extract the data from the receive buffer into the destination Slice.
565 auto extract_recv_buffer_func = KOKKOS_LAMBDA( const std::size_t i )
566 {
567 auto s = Slice_t::index_type::s( i );
568 auto a = Slice_t::index_type::a( i );
569 std::size_t dst_offset = s * dst.stride( 0 ) + a;
570 for ( std::size_t n = 0; n < num_comp; ++n )
571 dst_data[dst_offset + n * Slice_t::vector_length] =
572 recv_buffer( i, n );
573 };
574 Kokkos::RangePolicy<typename Distributor_t::execution_space>
575 extract_recv_buffer_policy( 0, distributor.totalNumImport() );
576 Kokkos::parallel_for( "Cabana::migrate::extract_recv_buffer",
577 extract_recv_buffer_policy,
578 extract_recv_buffer_func );
579 Kokkos::fence();
580
581 // Barrier before completing to ensure synchronization.
582 MPI_Barrier( distributor.comm() );
583}
