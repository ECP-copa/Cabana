165 // Synchronously move data between a source and destination AoSoA by executing
166 // the forward communication plan.
167 template <class Distributor_t, class AoSoA_t>
168 void distributeData(
169  const Distributor_t& distributor, const AoSoA_t& src, AoSoA_t& dst,
170  typename std::enable_if<( is_distributor<Distributor_t>::value &&
171  is_aosoa<AoSoA_t>::value ),
172  int>::type* = 0 )
173 {
174  // Get the MPI rank we are currently on.
175  int my_rank = -1;
176  MPI_Comm_rank( distributor.comm(), &my_rank );
177 
178  // Get the number of neighbors.
179  int num_n = distributor.numNeighbor();
180 
181  // Calculate the number of elements that are staying on this rank and
182  // therefore can be directly copied. If any of the neighbor ranks are this
183  // rank it will be stored in first position (i.e. the first neighbor in
184  // the local list is always yourself if you are sending to yourself).
185  std::size_t num_stay =
186  ( num_n > 0 && distributor.neighborRank( 0 ) == my_rank )
187  ? distributor.numExport( 0 )
188  : 0;
189 
190  // Allocate a send buffer.
191  std::size_t num_send = distributor.totalNumExport() - num_stay;
192  Kokkos::View<typename AoSoA_t::tuple_type*,
193  typename Distributor_t::memory_space>
194  send_buffer( Kokkos::ViewAllocateWithoutInitializing(
195  "distributor_send_buffer" ),
196  num_send );
197 
198  // Allocate a receive buffer.
199  Kokkos::View<typename AoSoA_t::tuple_type*,
200  typename Distributor_t::memory_space>
201  recv_buffer( Kokkos::ViewAllocateWithoutInitializing(
202  "distributor_recv_buffer" ),
203  distributor.totalNumImport() );
204 
205  // Get the steering vector for the sends.
206  auto steering = distributor.getExportSteering();
207 
208  // Gather the exports from the source AoSoA into the tuple-contiguous send
209  // buffer or the receive buffer if the data is staying. We know that the
210  // steering vector is ordered such that the data staying on this rank
211  // comes first.
212  auto build_send_buffer_func = KOKKOS_LAMBDA( const std::size_t i )
213  {
214  auto tpl = src.getTuple( steering( i ) );
215  if ( i < num_stay )
216  recv_buffer( i ) = tpl;
217  else
218  send_buffer( i - num_stay ) = tpl;
219  };
220  Kokkos::RangePolicy<typename Distributor_t::execution_space>
221  build_send_buffer_policy( 0, distributor.totalNumExport() );
222  Kokkos::parallel_for( "Cabana::Impl::distributeData::build_send_buffer",
223  build_send_buffer_policy, build_send_buffer_func );
224  Kokkos::fence();
225 
226  // The distributor has its own communication space so choose any tag.
227  const int mpi_tag = 1234;
228 
229  // Post non-blocking receives.
230  std::vector<MPI_Request> requests;
231  requests.reserve( num_n );
232  std::pair<std::size_t, std::size_t> recv_range = { 0, 0 };
233  for ( int n = 0; n < num_n; ++n )
234  {
235  recv_range.second = recv_range.first + distributor.numImport( n );
236 
237  if ( ( distributor.numImport( n ) > 0 ) &&
238  ( distributor.neighborRank( n ) != my_rank ) )
239  {
240  auto recv_subview = Kokkos::subview( recv_buffer, recv_range );
241 
242  requests.push_back( MPI_Request() );
243 
244  MPI_Irecv( recv_subview.data(),
245  recv_subview.size() *
246  sizeof( typename AoSoA_t::tuple_type ),
247  MPI_BYTE, distributor.neighborRank( n ), mpi_tag,
248  distributor.comm(), &( requests.back() ) );
249  }
250 
251  recv_range.first = recv_range.second;
252  }
253 
254  // Do blocking sends.
255  std::pair<std::size_t, std::size_t> send_range = { 0, 0 };
256  for ( int n = 0; n < num_n; ++n )
257  {
258  if ( ( distributor.numExport( n ) > 0 ) &&
259  ( distributor.neighborRank( n ) != my_rank ) )
260  {
261  send_range.second = send_range.first + distributor.numExport( n );
262 
263  auto send_subview = Kokkos::subview( send_buffer, send_range );
264 
265  MPI_Send( send_subview.data(),
266  send_subview.size() *
267  sizeof( typename AoSoA_t::tuple_type ),
268  MPI_BYTE, distributor.neighborRank( n ), mpi_tag,
269  distributor.comm() );
270 
271  send_range.first = send_range.second;
272  }
273  }
274 
275  // Wait on non-blocking receives.
276  std::vector<MPI_Status> status( requests.size() );
277  const int ec =
278  MPI_Waitall( requests.size(), requests.data(), status.data() );
279  if ( MPI_SUCCESS != ec )
280  throw std::logic_error( "Failed MPI Communication" );
281 
282  // Extract the receive buffer into the destination AoSoA.
283  auto extract_recv_buffer_func = KOKKOS_LAMBDA( const std::size_t i )
284  {
285  dst.setTuple( i, recv_buffer( i ) );
286  };
287  Kokkos::RangePolicy<typename Distributor_t::execution_space>
288  extract_recv_buffer_policy( 0, distributor.totalNumImport() );
289  Kokkos::parallel_for( "Cabana::Impl::distributeData::extract_recv_buffer",
290  extract_recv_buffer_policy,
291  extract_recv_buffer_func );
292  Kokkos::fence();
293 
294  // Barrier before completing to ensure synchronization.
295  MPI_Barrier( distributor.comm() );
296 }

323 template <class Distributor_t, class AoSoA_t>
324 void migrate( const Distributor_t& distributor, const AoSoA_t& src,
325  AoSoA_t& dst,
326  typename std::enable_if<( is_distributor<Distributor_t>::value &&
327  is_aosoa<AoSoA_t>::value ),
328  int>::type* = 0 )
329 {
330  // Check that src and dst are the right size.
331  if ( src.size() != distributor.exportSize() )
332  throw std::runtime_error( "Source is the wrong size for migration!" );
333  if ( dst.size() != distributor.totalNumImport() )
334  throw std::runtime_error(
335  "Destination is the wrong size for migration!" );
336 
337  // Move the data.
338  Impl::distributeData( distributor, src, dst );
339 }

365 template <class Distributor_t, class AoSoA_t>
366 void migrate( const Distributor_t& distributor, AoSoA_t& aosoa,
367  typename std::enable_if<( is_distributor<Distributor_t>::value &&
368  is_aosoa<AoSoA_t>::value ),
369  int>::type* = 0 )
370 {
371  // Check that the AoSoA is the right size.
372  if ( aosoa.size() != distributor.exportSize() )
373  throw std::runtime_error( "AoSoA is the wrong size for migration!" );
374 
375  // Determine if the source of destination decomposition has more data on
376  // this rank.
377  bool dst_is_bigger =
378  ( distributor.totalNumImport() > distributor.exportSize() );
379 
380  // If the destination decomposition is bigger than the source
381  // decomposition resize now so we have enough space to do the operation.
382  if ( dst_is_bigger )
383  aosoa.resize( distributor.totalNumImport() );
384 
385  // Move the data.
386  Impl::distributeData( distributor, aosoa, aosoa );
387 
388  // If the destination decomposition is smaller than the source
389  // decomposition resize after we have moved the data.
390  if ( !dst_is_bigger )
391  aosoa.resize( distributor.totalNumImport() );
392 }

417 template <class Distributor_t, class Slice_t>
418 void migrate( const Distributor_t& distributor, const Slice_t& src,
419  Slice_t& dst,
420  typename std::enable_if<( is_distributor<Distributor_t>::value &&
421  is_slice<Slice_t>::value ),
422  int>::type* = 0 )
423 {
424  // Check that src and dst are the right size.
425  if ( src.size() != distributor.exportSize() )
426  throw std::runtime_error( "Source is the wrong size for migration!" );
427  if ( dst.size() != distributor.totalNumImport() )
428  throw std::runtime_error(
429  "Destination is the wrong size for migration!" );
430 
431  // Get the number of components in the slices.
432  size_t num_comp = 1;
433  for ( size_t d = 2; d < src.rank(); ++d )
434  num_comp *= src.extent( d );
435 
436  // Get the raw slice data.
437  auto src_data = src.data();
438  auto dst_data = dst.data();
439

441  int my_rank = -1;
442  MPI_Comm_rank( distributor.comm(), &my_rank );
443 
444  // Get the number of neighbors.
445  int num_n = distributor.numNeighbor();
446

448  // therefore can be directly copied. If any of the neighbor ranks are this
449  // rank it will be stored in first position (i.e. the first neighbor in
450  // the local list is always yourself if you are sending to yourself).

452  ( num_n > 0 && distributor.neighborRank( 0 ) == my_rank )
453  ? distributor.numExport( 0 )
454  : 0;
455 
456  // Allocate a send buffer. Note this one is layout right so the components
457  // of each element are consecutive in memory.
458  std::size_t num_send = distributor.totalNumExport() - num_stay;
459  Kokkos::View<typename Slice_t::value_type**, Kokkos::LayoutRight,
460  typename Distributor_t::memory_space>
461  send_buffer( Kokkos::ViewAllocateWithoutInitializing(
462  "distributor_send_buffer" ),
463  num_send, num_comp );
464 
465  // Allocate a receive buffer. Note this one is layout right so the
466  // components of each element are consecutive in memory.
467  Kokkos::View<typename Slice_t::value_type**, Kokkos::LayoutRight,
468  typename Distributor_t::memory_space>
469  recv_buffer( Kokkos::ViewAllocateWithoutInitializing(
470  "distributor_recv_buffer" ),
471  distributor.totalNumImport(), num_comp );
472 
473  // Get the steering vector for the sends.
474  auto steering = distributor.getExportSteering();
475 
476  // Gather from the source Slice into the contiguous send buffer or,
477  // if it is part of the local copy, put it directly in the destination
478  // Slice.
479  auto build_send_buffer_func = KOKKOS_LAMBDA( const std::size_t i )
480  {
481  auto s_src = Slice_t::index_type::s( steering( i ) );
482  auto a_src = Slice_t::index_type::a( steering( i ) );
483  std::size_t src_offset = s_src * src.stride( 0 ) + a_src;
484  if ( i < num_stay )
485  for ( std::size_t n = 0; n < num_comp; ++n )
486  recv_buffer( i, n ) =
487  src_data[src_offset + n * Slice_t::vector_length];
488  else
489  for ( std::size_t n = 0; n < num_comp; ++n )
490  send_buffer( i - num_stay, n ) =
491  src_data[src_offset + n * Slice_t::vector_length];
492  };
493  Kokkos::RangePolicy<typename Distributor_t::execution_space>
494  build_send_buffer_policy( 0, distributor.totalNumExport() );
495  Kokkos::parallel_for( "Cabana::migrate::build_send_buffer",
496  build_send_buffer_policy, build_send_buffer_func );
497  Kokkos::fence();

506  for ( int n = 0; n < num_n; ++n )
507  {
508  recv_range.second = recv_range.first + distributor.numImport( n );
509 
510  if ( ( distributor.numImport( n ) > 0 ) &&
511  ( distributor.neighborRank( n ) != my_rank ) )
512  {
513  auto recv_subview =
514  Kokkos::subview( recv_buffer, recv_range, Kokkos::ALL );
515 
516  requests.push_back( MPI_Request() );

521  MPI_BYTE, distributor.neighborRank( n ), mpi_tag,
522  distributor.comm(), &( requests.back() ) );
523  }

530  for ( int n = 0; n < num_n; ++n )

534  {

554  if ( MPI_SUCCESS != ec )
555  throw std::logic_error( "Failed MPI Communication" );

558  auto extract_recv_buffer_func = KOKKOS_LAMBDA( const std::size_t i )
559  {

562  std::size_t dst_offset = s * dst.stride( 0 ) + a;
