50template <class GlobalGridType, class CoordSliceType, class... FieldSliceTypes>
51void writePartialRangeTimeStep( const std::string& prefix,
52 const GlobalGridType& global_grid,
53 const int time_step_index, const double time,
54 const std::size_t begin, const std::size_t end,
55 const CoordSliceType& coords,
56 FieldSliceTypes&&... fields )
57{
58 // Pick a number of groups. We want to write approximately the N^3 blocks
59 // to N^2 groups. Pick the block dimension with the largest number of
60 // ranks as the number of groups. We may want to tweak this as an optional
61 // input later with this behavior as the default.
62 int num_group = 0;
63 for ( int d = 0; d < 3; ++d )
64 if ( global_grid.dimNumBlock( d ) > num_group )
65 num_group = global_grid.dimNumBlock( d );
66
67 Cabana::Experimental::SiloParticleOutput::writePartialRangeTimeStep(
68 prefix, global_grid.comm(), num_group, time_step_index, time, begin,
69 end, coords, fields... );
70}
