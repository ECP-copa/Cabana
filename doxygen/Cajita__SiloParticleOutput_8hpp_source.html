51 {
52  // Pick a number of groups. We want to write approximately the N^3 blocks
53  // to N^2 groups. Pick the block dimension with the largest number of
54  // ranks as the number of groups. We may want to tweak this as an optional
55  // input later with this behavior as the default.
56  int num_group = 0;
57  for ( int d = 0; d < 3; ++d )
58  if ( global_grid.dimNumBlock( d ) > num_group )
59  num_group = global_grid.dimNumBlock( d );
