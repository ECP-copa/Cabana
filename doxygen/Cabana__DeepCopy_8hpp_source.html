155 template <class DstAoSoA, class SrcAoSoA>
156 inline void
157 deep_copy( DstAoSoA& dst, const SrcAoSoA& src,
158  typename std::enable_if<( is_aosoa<DstAoSoA>::value &&
159  is_aosoa<SrcAoSoA>::value )>::type* = 0 )
160 {
161  using dst_type = DstAoSoA;
162  using src_type = SrcAoSoA;
163  using dst_memory_space = typename dst_type::memory_space;
164  using src_memory_space = typename src_type::memory_space;
165  using dst_soa_type = typename dst_type::soa_type;
166  using src_soa_type = typename src_type::soa_type;
167 
168  // Check that the data types are the same.
169  static_assert(
170  std::is_same<typename dst_type::member_types,
171  typename src_type::member_types>::value,
172  "Attempted to deep copy AoSoA objects of different member types" );
173 
174  // Check for the same number of values.
175  if ( dst.size() != src.size() )
176  {
177  throw std::runtime_error(
178  "Attempted to deep copy AoSoA objects of different sizes" );
179  }
180 
181  // Get the pointers to the beginning of the data blocks.
182  void* dst_data = dst.data();
183  const void* src_data = src.data();
184 
185  // Return if both pointers are null.
186  if ( dst_data == nullptr && src_data == nullptr )
187  {
188  return;
189  }
190 
191  // Get the number of SoA's in each object.
192  auto dst_num_soa = dst.numSoA();
193  auto src_num_soa = src.numSoA();
194 
195  // Return if the AoSoA memory occupies the same space.
196  if ( ( dst_data == src_data ) && ( dst_num_soa * sizeof( dst_soa_type ) ==
197  src_num_soa * sizeof( src_soa_type ) ) )
198  {
199  return;
200  }
201 
202  // If the inner array size is the same and both AoSoAs have the same number
203  // of values then we can do a byte-wise copy directly.
204  if ( std::is_same<dst_soa_type, src_soa_type>::value )
205  {
206  Kokkos::deep_copy( Kokkos::View<char*, dst_memory_space>(
207  reinterpret_cast<char*>( dst.data() ),
208  dst.numSoA() * sizeof( dst_soa_type ) ),
209  Kokkos::View<char*, src_memory_space>(
210  reinterpret_cast<char*>( src.data() ),
211  src.numSoA() * sizeof( src_soa_type ) ) );
212  }
213 
214  // Otherwise copy the data element-by-element because the data layout is
215  // different.
216  else
217  {
218  // Create an AoSoA in the destination space with the same data layout
219  // as the source.
220  auto src_copy_on_dst = create_mirror_view_and_copy(
221  typename dst_type::memory_space(), src );
222 
223  // Copy via tuples.
224  auto copy_func = KOKKOS_LAMBDA( const std::size_t i )
225  {
226  dst.setTuple( i, src_copy_on_dst.getTuple( i ) );
227  };
228  Kokkos::RangePolicy<typename dst_type::execution_space> exec_policy(
229  0, dst.size() );
230  Kokkos::parallel_for( "Cabana::deep_copy", exec_policy, copy_func );
231  Kokkos::fence();
232  }
233 }

271 template <class DstSlice, class SrcSlice>
272 inline void
273 deep_copy( DstSlice& dst, const SrcSlice& src,
274  typename std::enable_if<( is_slice<DstSlice>::value &&
275  is_slice<SrcSlice>::value )>::type* = 0 )
276 {
277  using dst_type = DstSlice;
278  using src_type = SrcSlice;
279 
280  // Check that the data types are the same.
281  static_assert(
282  std::is_same<typename dst_type::value_type,
283  typename src_type::value_type>::value,
284  "Attempted to deep copy Slice objects of different value types" );
285 
286  // Check that the element dimensions are the same.
287  static_assert( SrcSlice::view_layout::D0 == SrcSlice::view_layout::D0,
288  "Slice dimension 0 is different" );
289  static_assert( SrcSlice::view_layout::D1 == SrcSlice::view_layout::D1,
290  "Slice dimension 1 is different" );
291  static_assert( SrcSlice::view_layout::D2 == SrcSlice::view_layout::D2,
292  "Slice dimension 2 is different" );
293  static_assert( SrcSlice::view_layout::D3 == SrcSlice::view_layout::D3,
294  "Slice dimension 3 is different" );
295  static_assert( SrcSlice::view_layout::D4 == SrcSlice::view_layout::D4,
296  "Slice dimension 4 is different" );
297  static_assert( SrcSlice::view_layout::D5 == SrcSlice::view_layout::D5,
298  "Slice dimension 5 is different" );
299 
300  // Check for the same number of elements.
301  if ( dst.size() != src.size() )
302  {
303  throw std::runtime_error(
304  "Attempted to deep copy Slice objects of different sizes" );
305  }
306 
307  // Get the pointers to the beginning of the data blocks.
308  auto dst_data = dst.data();
309  const auto src_data = src.data();
310 
311  // Return if both pointers are null.
312  if ( dst_data == nullptr && src_data == nullptr )
313  {
314  return;
315  }
316 
317  // Get the number of SoA's in each object.
318  auto dst_num_soa = dst.numSoA();
319  auto src_num_soa = src.numSoA();
320 
321  // Return if the slice memory occupies the same space.
322  if ( ( dst_data == src_data ) &&
323  ( dst_num_soa * dst.stride( 0 ) == src_num_soa * src.stride( 0 ) ) )
324  {
325  return;
326  }
327 
328  // Get the number of components in each slice element.
329  std::size_t num_comp = 1;
330  for ( std::size_t d = 2; d < dst.rank(); ++d )
331  num_comp *= dst.extent( d );
332 
333  // Gather the slice data in a flat view in the source space and copy it to
334  // the destination space.
335  Kokkos::View<typename dst_type::value_type*,
336  typename dst_type::memory_space>
337  gather_dst( "gather_dst", num_comp * dst.size() );
338  {
339  Kokkos::View<typename src_type::value_type*,
340  typename src_type::memory_space>
341  gather_src( "gather_src", num_comp * src.size() );
342  auto gather_func = KOKKOS_LAMBDA( const std::size_t i )
343  {
344  auto src_offset = SrcSlice::index_type::s( i ) * src.stride( 0 ) +
345  SrcSlice::index_type::a( i );
346  for ( std::size_t n = 0; n < num_comp; ++n )
347  gather_src( i * num_comp + n ) =
348  src_data[src_offset + SrcSlice::vector_length * n];
349  };
350  Kokkos::RangePolicy<typename src_type::execution_space> gather_policy(
351  0, src.size() );
352  Kokkos::parallel_for( "Cabana::deep_copy::gather", gather_policy,
353  gather_func );
354  Kokkos::fence();
355  Kokkos::deep_copy( gather_dst, gather_src );
356  }
357 
358  // Scatter back into the destination slice from the gathered slice.
359  auto scatter_func = KOKKOS_LAMBDA( const std::size_t i )
360  {
361  auto dst_offset = DstSlice::index_type::s( i ) * dst.stride( 0 ) +
362  DstSlice::index_type::a( i );
363  for ( std::size_t n = 0; n < num_comp; ++n )
364  dst_data[dst_offset + DstSlice::vector_length * n] =
365  gather_dst( i * num_comp + n );
366  };
367  Kokkos::RangePolicy<typename dst_type::execution_space> scatter_policy(
368  0, dst.size() );
369  Kokkos::parallel_for( "Cabana::deep_copy::scatter", scatter_policy,
370  scatter_func );
371  Kokkos::fence();
372 }
