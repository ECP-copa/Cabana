153 template <class DstAoSoA, class SrcAoSoA>
154 inline void
155 deep_copy( DstAoSoA& dst, const SrcAoSoA& src,
156  typename std::enable_if<( is_aosoa<DstAoSoA>::value &&
157  is_aosoa<SrcAoSoA>::value )>::type* = 0 )
158 {
159  using dst_type = DstAoSoA;
160  using src_type = SrcAoSoA;
161  using dst_memory_space = typename dst_type::memory_space;
162  using src_memory_space = typename src_type::memory_space;
163  using dst_soa_type = typename dst_type::soa_type;
164  using src_soa_type = typename src_type::soa_type;
165 
166  // Check that the data types are the same.
167  static_assert(
168  std::is_same<typename dst_type::member_types,
169  typename src_type::member_types>::value,
170  "Attempted to deep copy AoSoA objects of different member types" );
171 
172  // Check for the same number of values.
173  if ( dst.size() != src.size() )
174  {
175  throw std::runtime_error(
176  "Attempted to deep copy AoSoA objects of different sizes" );
177  }
178 
179  // Get the pointers to the beginning of the data blocks.
180  void* dst_data = dst.data();
181  const void* src_data = src.data();
182 
183  // Return if both pointers are null.
184  if ( dst_data == nullptr && src_data == nullptr )
185  {
186  return;
187  }
188 
189  // Get the number of SoA's in each object.
190  auto dst_num_soa = dst.numSoA();
191  auto src_num_soa = src.numSoA();
192 
193  // Return if the AoSoA memory occupies the same space.
194  if ( ( dst_data == src_data ) && ( dst_num_soa * sizeof( dst_soa_type ) ==
195  src_num_soa * sizeof( src_soa_type ) ) )
196  {
197  return;
198  }
199 
200  // If the inner array size is the same and both AoSoAs have the same number
201  // of values then we can do a byte-wise copy directly.
202  if ( std::is_same<dst_soa_type, src_soa_type>::value )
203  {
204  Kokkos::Impl::DeepCopy<dst_memory_space, src_memory_space>(
205  dst_data, src_data, dst_num_soa * sizeof( dst_soa_type ) );
206  }
207 
208  // Otherwise copy the data element-by-element because the data layout is
209  // different.
210  else
211  {
212  // Create an AoSoA in the destination space with the same data layout
213  // as the source.
214  auto src_copy_on_dst = create_mirror_view_and_copy(
215  typename dst_type::memory_space(), src );
216 
217  // Copy via tuples.
218  auto copy_func = KOKKOS_LAMBDA( const std::size_t i )
219  {
220  dst.setTuple( i, src_copy_on_dst.getTuple( i ) );
221  };
222  Kokkos::RangePolicy<typename dst_type::execution_space> exec_policy(
223  0, dst.size() );
224  Kokkos::parallel_for( "Cabana::deep_copy", exec_policy, copy_func );
225  Kokkos::fence();
226  }
227 }

265 template <class DstSlice, class SrcSlice>
266 inline void
267 deep_copy( DstSlice& dst, const SrcSlice& src,
268  typename std::enable_if<( is_slice<DstSlice>::value &&
269  is_slice<SrcSlice>::value )>::type* = 0 )
270 {
271  using dst_type = DstSlice;
272  using src_type = SrcSlice;
273 
274  // Check that the data types are the same.
275  static_assert(
276  std::is_same<typename dst_type::value_type,
277  typename src_type::value_type>::value,
278  "Attempted to deep copy Slice objects of different value types" );
279 
280  // Check that the element dimensions are the same.
281  static_assert( SrcSlice::view_layout::D0 == SrcSlice::view_layout::D0,
282  "Slice dimension 0 is different" );
283  static_assert( SrcSlice::view_layout::D1 == SrcSlice::view_layout::D1,
284  "Slice dimension 1 is different" );
285  static_assert( SrcSlice::view_layout::D2 == SrcSlice::view_layout::D2,
286  "Slice dimension 2 is different" );
287  static_assert( SrcSlice::view_layout::D3 == SrcSlice::view_layout::D3,
288  "Slice dimension 3 is different" );
289  static_assert( SrcSlice::view_layout::D4 == SrcSlice::view_layout::D4,
290  "Slice dimension 4 is different" );
291  static_assert( SrcSlice::view_layout::D5 == SrcSlice::view_layout::D5,
292  "Slice dimension 5 is different" );
293 
294  // Check for the same number of elements.
295  if ( dst.size() != src.size() )
296  {
297  throw std::runtime_error(
298  "Attempted to deep copy Slice objects of different sizes" );
299  }
300 
301  // Get the pointers to the beginning of the data blocks.
302  auto dst_data = dst.data();
303  const auto src_data = src.data();
304 
305  // Return if both pointers are null.
306  if ( dst_data == nullptr && src_data == nullptr )
307  {
308  return;
309  }
310 
311  // Get the number of SoA's in each object.
312  auto dst_num_soa = dst.numSoA();
313  auto src_num_soa = src.numSoA();
314 
315  // Return if the slice memory occupies the same space.
316  if ( ( dst_data == src_data ) &&
317  ( dst_num_soa * dst.stride( 0 ) == src_num_soa * src.stride( 0 ) ) )
318  {
319  return;
320  }
321 
322  // Get the number of components in each slice element.
323  std::size_t num_comp = 1;
324  for ( std::size_t d = 2; d < dst.rank(); ++d )
325  num_comp *= dst.extent( d );
326 
327  // Gather the slice data in a flat view in the source space and copy it to
328  // the destination space.
329  Kokkos::View<typename dst_type::value_type*,
330  typename dst_type::memory_space>
331  gather_dst( "gather_dst", num_comp * dst.size() );
332  {
333  Kokkos::View<typename src_type::value_type*,
334  typename src_type::memory_space>
335  gather_src( "gather_src", num_comp * src.size() );
336  auto gather_func = KOKKOS_LAMBDA( const std::size_t i )
337  {
338  auto src_offset = SrcSlice::index_type::s( i ) * src.stride( 0 ) +
339  SrcSlice::index_type::a( i );
340  for ( std::size_t n = 0; n < num_comp; ++n )
341  gather_src( i * num_comp + n ) =
342  src_data[src_offset + SrcSlice::vector_length * n];
343  };
344  Kokkos::RangePolicy<typename src_type::execution_space> gather_policy(
345  0, src.size() );
346  Kokkos::parallel_for( "Cabana::deep_copy::gather", gather_policy,
347  gather_func );
348  Kokkos::fence();
349  Kokkos::deep_copy( gather_dst, gather_src );
350  }
351 
352  // Scatter back into the destination slice from the gathered slice.
353  auto scatter_func = KOKKOS_LAMBDA( const std::size_t i )
354  {
355  auto dst_offset = DstSlice::index_type::s( i ) * dst.stride( 0 ) +
356  DstSlice::index_type::a( i );
357  for ( std::size_t n = 0; n < num_comp; ++n )
358  dst_data[dst_offset + DstSlice::vector_length * n] =
359  gather_dst( i * num_comp + n );
360  };
361  Kokkos::RangePolicy<typename dst_type::execution_space> scatter_policy(
362  0, dst.size() );
363  Kokkos::parallel_for( "Cabana::deep_copy::scatter", scatter_policy,
364  scatter_func );
365  Kokkos::fence();
366 }
