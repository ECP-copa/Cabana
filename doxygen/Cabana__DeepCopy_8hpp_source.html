37 template<class DstAoSoA, class SrcAoSoA>
38 inline void deep_copy(
39  DstAoSoA& dst,
40  const SrcAoSoA& src,
41  typename std::enable_if<(is_aosoa<DstAoSoA>::value &&
42  is_aosoa<SrcAoSoA>::value)>::type * = 0 )
43 {
44  using dst_type = DstAoSoA;
45  using src_type = SrcAoSoA;
46  using dst_memory_space =
47  typename dst_type::memory_space::kokkos_memory_space;
48  using src_memory_space =
49  typename src_type::memory_space::kokkos_memory_space;
50  using dst_soa_type = typename dst_type::soa_type;
51  using src_soa_type = typename src_type::soa_type;
52 
53  // Check that the data types are the same.
54  static_assert(
55  std::is_same<typename dst_type::member_types,
56  typename src_type::member_types>::value,
57  "Attempted to deep copy AoSoA objects of different member types" );
58 
59  // Check for the same number of values.
60  if ( dst.size() != src.size() )
61  {
62  throw std::runtime_error(
63  "Attempted to deep copy AoSoA objects of different sizes" );
64  }
65 
66  // Get the pointers to the beginning of the data blocks.
67  void* dst_data = dst.ptr();
68  const void* src_data = src.ptr();
69 
70  // Return if both pointers are null.
71  if ( dst_data == nullptr && src_data == nullptr )
72  {
73  Kokkos::fence();
74  return;
75  }
76 
77  // Get the number of SoA's in each object.
78  auto dst_num_soa = dst.numSoA();
79  auto src_num_soa = src.numSoA();
80 
81  // Return if the AoSoA memory occupies the same space.
82  if ( (dst_data == src_data) &&
83  (dst_num_soa * sizeof(dst_soa_type) ==
84  src_num_soa * sizeof(src_soa_type)) )
85  {
86  Kokkos::fence();
87  return;
88  }
89 
90  // If the inner array size is the same and both AoSoAs have the same number
91  // of values then we can do a byte-wise copy directly.
92  if ( std::is_same<dst_soa_type,src_soa_type>::value &&
93  ( dst_type::vector_length == src_type::vector_length ) )
94  {
95  Kokkos::fence();
96  Kokkos::Impl::DeepCopy<dst_memory_space,src_memory_space>(
97  dst_data, src_data, dst_num_soa * sizeof(dst_soa_type) );
98  Kokkos::fence();
99  }
100 
101  // Otherwise copy the data element-by-element because the data layout is
102  // different.
103  else
104  {
105  // Define a AoSoA type in the destination space with the same data
106  // layout as the source.
107  using src_mirror_type = AoSoA<typename src_type::member_types,
108  typename dst_type::memory_space,
109  src_type::vector_length>;
110  static_assert(
111  std::is_same<src_soa_type,typename src_mirror_type::soa_type>::value,
112  "Incompatible source mirror type in destination space" );
113 
114  // Create an AoSoA in the destination space with the same data layout
115  // as the source.
116  src_mirror_type src_copy_on_dst( src.size() );
117 
118  // Copy the source to the destination space.
119  Kokkos::fence();
120  Kokkos::Impl::DeepCopy<dst_memory_space,src_memory_space>(
121  src_copy_on_dst.ptr(),
122  src_data,
123  src_num_soa * sizeof(src_soa_type) );
124  Kokkos::fence();
125 
126  // Copy via tuples.
127  auto copy_func =
128  KOKKOS_LAMBDA( const std::size_t i )
129  { dst.setTuple( i, src_copy_on_dst.getTuple(i) ); };
130  Kokkos::RangePolicy<typename dst_memory_space::execution_space>
131  exec_policy( 0, dst.size() );
132  Kokkos::parallel_for( "Cabana::deep_copy", exec_policy, copy_func );
133  Kokkos::fence();
134  }
135 }
