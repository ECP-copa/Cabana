261  void solve( const Array_t& b, Array_t& x ) override
262  {
263  // Get the local grid.
264  auto local_grid = _vectors->layout()->localGrid();
265 
266  // Print banner
267  if ( 1 <= _print_level && 0 == local_grid->globalGrid().blockId() )
268  std::cout << std::endl
269  << "Preconditioned conjugate gradient" << std::endl;
270 
271  // Index space.
272  auto entity_space =
273  local_grid->indexSpace( Own(), EntityType(), Local() );
274 
275  // Subarrays.
276  auto p_old = createSubarray( *_vectors, 0, 1 );
277  auto z = createSubarray( *_vectors, 1, 2 );
278  auto r_old = createSubarray( *_vectors, 2, 3 );
279  auto q = createSubarray( *_vectors, 3, 4 );
280  auto p_new = createSubarray( *_vectors, 4, 5 );
281  auto r_new = createSubarray( *_vectors, 5, 6 );
282  auto A_halo_vectors = createSubarray( *_vectors, 0, 2 );
283  auto M_halo_vectors = createSubarray( *_vectors, 2, 4 );
284 
285  // Views.
286  auto x_view = x.view();
287  auto b_view = b.view();
288  auto M_view = _M->view();
289  auto A_view = _A->view();
290  auto p_old_view = p_old->view();
291  auto z_view = z->view();
292  auto r_old_view = r_old->view();
293  auto q_view = q->view();
294  auto p_new_view = p_new->view();
295  auto r_new_view = r_new->view();
296 
297  // Reset iteration count.
298  _num_iter = 0;
299 
300  // Compute the norm of the RHS.
301  std::vector<Scalar> b_norm( 1 );
302  ArrayOp::norm2( b, b_norm );
303 
304  // Copy the LHS into p so we can gather it.
305  Kokkos::deep_copy( p_old_view, x_view );
306 
307  // Gather the LHS through gatheing p and z.
308  _A_halo->gather( execution_space(), *A_halo_vectors );
309 
310  // Compute the initial residual and norm.
311  _residual_norm = 0.0;
312  auto compute_r0 = createComputeR0( _A_stencil, A_view, p_old_view,
313  b_view, r_old_view );
314  grid_parallel_reduce(
315  "compute_r0", execution_space(), entity_space,
316  std::integral_constant<std::size_t, num_space_dim>{}, compute_r0,
317  _residual_norm );
318 
319  // Finish the global norm reduction.
320  MPI_Allreduce( MPI_IN_PLACE, &_residual_norm, 1,
321  MpiTraits<Scalar>::type(), MPI_SUM,
322  local_grid->globalGrid().comm() );
323 
324  // If we already have met our criteria then return.
325  _residual_norm = std::sqrt( _residual_norm ) / b_norm[0];
326  if ( 2 == _print_level && 0 == local_grid->globalGrid().blockId() )
327  std::cout << "Iteration " << _num_iter
328  << ": |r|_2 / |b|_2 = " << _residual_norm << std::endl;
329  if ( _residual_norm <= _tol )
330  return;
331 
332  // r and q.
333  _M_halo->gather( execution_space(), *M_halo_vectors );
334 
335  // Compute the initial preconditioned residual.
336  Scalar zTr_old = 0.0;
337  auto compute_z0 = createComputeZ0( _M_stencil, M_view, p_old_view,
338  p_new_view, r_old_view, z_view );
339  grid_parallel_reduce(
340  "compute_z0", execution_space(), entity_space,
341  std::integral_constant<std::size_t, num_space_dim>{}, compute_z0,
342  zTr_old );
343 
344  // Finish computation of zTr
345  MPI_Allreduce( MPI_IN_PLACE, &zTr_old, 1, MpiTraits<Scalar>::type(),
346  MPI_SUM, local_grid->globalGrid().comm() );
347 
348  // Gather the LHS through gatheing p and z.
349  _A_halo->gather( execution_space(), *A_halo_vectors );
350 
351  // Compute A*p and pT*A*p.
352  Scalar pTAp = 0.0;
353  auto compute_q0 =
354  createComputeQ0( _A_stencil, A_view, p_old_view, q_view );
355  grid_parallel_reduce(
356  "compute_q0", execution_space(), entity_space,
357  std::integral_constant<std::size_t, num_space_dim>{}, compute_q0,
358  pTAp );
359 
360  // Finish the global reduction on pTAp.
361  MPI_Allreduce( MPI_IN_PLACE, &pTAp, 1, MpiTraits<Scalar>::type(),
362  MPI_SUM, local_grid->globalGrid().comm() );
363 
364  // Iterate.
365  bool converged = false;
366  Scalar zTr_new = 0.0;
367  Scalar alpha;
368  Scalar beta;
369  while ( _residual_norm > _tol && _num_iter < _max_iter )
370  {
371  // Gather r and q.
372  _M_halo->gather( execution_space(), *M_halo_vectors );
373 
374  // Kernel 1: Compute x, r, residual norm, and zTr
375  alpha = zTr_old / pTAp;
376  zTr_new = 0.0;
377  auto cg_kernel_1 = createKernel1(
378  _M_stencil, M_view, x_view, r_new_view, r_old_view, p_new_view,
379  p_old_view, z_view, q_view, alpha );
380  grid_parallel_reduce(
381  "cg_kernel_1", execution_space(), entity_space,
382  std::integral_constant<std::size_t, num_space_dim>{},
383  cg_kernel_1, zTr_new );
384 
385  // Finish the global reduction on zTr and r_norm.
386  MPI_Allreduce( MPI_IN_PLACE, &zTr_new, 1, MpiTraits<Scalar>::type(),
387  MPI_SUM, local_grid->globalGrid().comm() );
388 
389  // Update residual norm
390  _residual_norm = std::sqrt( fabs( zTr_new ) ) / b_norm[0];
391 
392  // Increment iteration count.
393  _num_iter++;
394 
395  // Output result
396  if ( 2 == _print_level && 0 == local_grid->globalGrid().blockId() )
397  std::cout << "Iteration " << _num_iter
398  << ": |r|_2 / |b|_2 = " << _residual_norm
399  << std::endl;
400 
401  // Check for convergence.
402  if ( _residual_norm <= _tol )
403  {
404  converged = true;
405  break;
406  }
407 
408  // Gather p and z.
409  _A_halo->gather( execution_space(), *A_halo_vectors );
410 
411  // Kernel 2: Compute p, A*p, and p^T*A*p
412  beta = zTr_new / zTr_old;
413  pTAp = 0.0;
414  auto cg_kernel_2 = createKernel2(
415  _A_stencil, A_view, x_view, r_new_view, r_old_view, p_new_view,
416  p_old_view, z_view, q_view, beta );
417  grid_parallel_reduce(
418  "cg_kernel_2", execution_space(), entity_space,
419  std::integral_constant<std::size_t, num_space_dim>{},
420  cg_kernel_2, pTAp );
421 
422  // Finish the global reduction on pTAp.
423  MPI_Allreduce( MPI_IN_PLACE, &pTAp, 1, MpiTraits<Scalar>::type(),
424  MPI_SUM, local_grid->globalGrid().comm() );
425 
426  // Update zTr
427  zTr_old = zTr_new;
428  }
429 
430  // Output end state.
431  if ( 1 <= _print_level && 0 == local_grid->globalGrid().blockId() )
432  std::cout << "Finished in " << _num_iter
433  << " iterations, converged to " << _residual_norm
434  << std::endl
435  << std::endl;
436 
437  // If we didn't converge throw.
438  if ( !converged )
439  throw std::runtime_error( "CG solver did not converge" );
440  }

462  KOKKOS_INLINE_FUNCTION void
463  operator()( const std::integral_constant<std::size_t, 3>&, const int i,
464  const int j, const int k, value_type& result ) const
465  {
466  // Compute the local contribution from matrix-vector
467  // multiplication. Note that we copied x into p for this
468  // operation to easily perform the gather. Only apply the
469  // stencil entry if it is greater than 0.
470  value_type Ax = 0.0;
471  for ( unsigned c = 0; c < A_stencil.extent( 0 ); ++c )
472  if ( fabs( A_view( i, j, k, c ) ) > 0.0 )
473  Ax += A_view( i, j, k, c ) *
474  p_old_view( i + A_stencil( c, Dim::I ),
475  j + A_stencil( c, Dim::J ),
476  k + A_stencil( c, Dim::K ), 0 );
477 
478  // Compute the residual.
479  auto r_new = b_view( i, j, k, 0 ) - Ax;
480 
481  // Assign the residual.
482  r_old_view( i, j, k, 0 ) = r_new;
483 
484  // Contribute to the reduction.
485  result += r_new * r_new;
486  }

662  {
663  StencilM M_stencil;
664  ViewM M_view;

674  KOKKOS_INLINE_FUNCTION void
675  operator()( const std::integral_constant<std::size_t, 3>&, const int i,
676  const int j, const int k, ValueType& result ) const
677  {
678  // Compute the local contribution from matrix-vector
679  // multiplication. This computes the updated q vector
680  // in-line to avoid another kernel launch. Only apply the
681  // stencil entry if it is greater than 0.
682  ValueType Mr = 0.0;
683  for ( unsigned c = 0; c < M_stencil.extent( 0 ); ++c )
684  if ( fabs( M_view( i, j, k, c ) ) > 0.0 )
685  Mr += M_view( i, j, k, c ) *
686  ( r_old_view( i + M_stencil( c, Dim::I ),
687  j + M_stencil( c, Dim::J ),
688  k + M_stencil( c, Dim::K ), 0 ) -
689  alpha * q_view( i + M_stencil( c, Dim::I ),
690  j + M_stencil( c, Dim::J ),
691  k + M_stencil( c, Dim::K ), 0 ) );
692 
693  // Compute the updated x.
694  ValueType x_new =
695  x_view( i, j, k, 0 ) + alpha * p_new_view( i, j, k, 0 );
696 
697  // Compute the updated residual.
698  ValueType r_new =
699  r_old_view( i, j, k, 0 ) - alpha * q_view( i, j, k, 0 );
700 
701  // Write to old p vector.
702  p_old_view( i, j, k, 0 ) = p_new_view( i, j, k, 0 );
703 
704  // Write values.
705  x_view( i, j, k, 0 ) = x_new;
706  r_new_view( i, j, k, 0 ) = r_new;
707  z_view( i, j, k, 0 ) = Mr;
708 
709  // Compute contribution to the zTr.
710  result += Mr * r_new;
711  }

713  KOKKOS_INLINE_FUNCTION void
714  operator()( const std::integral_constant<std::size_t, 2>&, const int i,
715  const int j, ValueType& result ) const
716  {
717  // Compute the local contribution from matrix-vector
718  // multiplication. This computes the updated q vector
719  // in-line to avoid another kernel launch. Only apply the
720  // stencil entry if it is greater than 0.
721  ValueType Mr = 0.0;
722  for ( unsigned c = 0; c < M_stencil.extent( 0 ); ++c )
723  if ( fabs( M_view( i, j, c ) ) > 0.0 )
724  Mr += M_view( i, j, c ) *
725  ( r_old_view( i + M_stencil( c, Dim::I ),
726  j + M_stencil( c, Dim::J ), 0 ) -
727  alpha * q_view( i + M_stencil( c, Dim::I ),
728  j + M_stencil( c, Dim::J ), 0 ) );
729 
730  // Compute the updated x.
731  ValueType x_new = x_view( i, j, 0 ) + alpha * p_new_view( i, j, 0 );
732 
733  // Compute the updated residual.
734  ValueType r_new = r_old_view( i, j, 0 ) - alpha * q_view( i, j, 0 );
735 
736  // Write to old p vector.
737  p_old_view( i, j, 0 ) = p_new_view( i, j, 0 );
738 
739  // Write values.
740  x_view( i, j, 0 ) = x_new;
741  r_new_view( i, j, 0 ) = r_new;
742  z_view( i, j, 0 ) = Mr;
743 
744  // Compute contribution to the zTr.
745  result += Mr * r_new;
746  }
747  };

780  KOKKOS_INLINE_FUNCTION void
781  operator()( const std::integral_constant<std::size_t, 3>&, const int i,
782  const int j, const int k, ValueType& result ) const
783  {
784  // Compute the local contribution from matrix-vector
785  // multiplication. This computes the updated p vector
786  // in-line to avoid another kernel launch. Only apply the
787  // stencil entry if it is greater than 0.
788  ValueType Ap = 0.0;
789  for ( unsigned c = 0; c < A_stencil.extent( 0 ); ++c )
790  if ( fabs( A_view( i, j, k, c ) ) > 0.0 )
791  Ap +=
792  A_view( i, j, k, c ) *
793  ( z_view( i + A_stencil( c, Dim::I ),
794  j + A_stencil( c, Dim::J ),
795  k + A_stencil( c, Dim::K ), 0 ) +
796  beta * p_old_view( i + A_stencil( c, Dim::I ),
797  j + A_stencil( c, Dim::J ),
798  k + A_stencil( c, Dim::K ), 0 ) );
799 
800  // Compute the updated p.
801  ValueType p_new =
802  z_view( i, j, k, 0 ) + beta * p_old_view( i, j, k, 0 );
803 
804  // Write to old residual.
805  r_old_view( i, j, k, 0 ) = r_new_view( i, j, k, 0 );
806 
807  // Write values.
808  q_view( i, j, k, 0 ) = Ap;
809  p_new_view( i, j, k, 0 ) = p_new;
810 
811  // Compute contribution to the dot product.
812  result += p_new * Ap;
813  }

865  // Set the stencil of a matrix.
866  void
867  setStencil( const std::vector<std::array<int, num_space_dim>>& stencil,
868  const bool is_symmetric,
869  Kokkos::View<int* [num_space_dim], DeviceType>& device_stencil,
870  std::shared_ptr<Halo<memory_space>>& halo,
871  std::shared_ptr<Array_t>& matrix )
872  {
873  // For now we don't support symmetry.
874  if ( is_symmetric )
875  throw std::logic_error(
876  "Reference CG currently does not support symmetry" );
877 
878  // Get the local grid.
879  auto local_grid = _vectors->layout()->localGrid();
880 
881  // Copy stencil to the device.
882  device_stencil = Kokkos::View<int* [num_space_dim], DeviceType>(
883  Kokkos::ViewAllocateWithoutInitializing( "stencil" ),
884  stencil.size() );
885  auto stencil_mirror =
886  Kokkos::create_mirror_view( Kokkos::HostSpace(), device_stencil );
887  for ( unsigned s = 0; s < stencil.size(); ++s )
888  for ( std::size_t d = 0; d < num_space_dim; ++d )
889  stencil_mirror( s, d ) = stencil[s][d];
890  Kokkos::deep_copy( device_stencil, stencil_mirror );
891 
892  // Compose the halo pattern and compute how wide the halo needs to be
893  // to gather all elements accessed by the stencil.
894  std::set<std::array<int, num_space_dim>> neighbor_set;
895  std::array<int, num_space_dim> neighbor;
896  int width = 0;
897  for ( auto s : stencil )
898  {
899  // Compse a set of the neighbor ranks based on the stencil.
900  for ( std::size_t d = 0; d < num_space_dim; ++d )
901  neighbor[d] = ( s[d] == 0 ) ? 0 : s[d] / std::abs( s[d] );
902  neighbor_set.emplace( neighbor );
903 
904  // Compute the width of the halo needed to apply the stencil.
905  for ( std::size_t d = 0; d < num_space_dim; ++d )
906  width = std::max( width, std::abs( s[d] ) );
907  }
908  std::vector<std::array<int, num_space_dim>> halo_neighbors(
909  neighbor_set.size() );
910  std::copy( neighbor_set.begin(), neighbor_set.end(),
911  halo_neighbors.begin() );
912 
913  // Build the halo. We put 2 entries as each operator application will
914  // gather 2 vectors with fused kernels.
915  auto halo_layout = createArrayLayout( local_grid, 2, EntityType() );
916  HaloPattern<num_space_dim> pattern;
917  pattern.setNeighbors( halo_neighbors );
918  halo = createHalo<Scalar, DeviceType>( *halo_layout, pattern, width );
919 
920  // Create a new layout.
921  auto matrix_layout =
922  createArrayLayout( local_grid, stencil.size(), EntityType() );
923 
924  // Allocate the matrix.
925  matrix = createArray<Scalar, DeviceType>( "matrix", matrix_layout );
926  }
