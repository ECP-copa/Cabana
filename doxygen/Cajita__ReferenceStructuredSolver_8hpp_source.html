128  : public ReferenceStructuredSolver<Scalar, EntityType, MeshType, DeviceType>
129 {
130  public:
131  // Types.
132  using entity_type = EntityType;
133  using device_type = DeviceType;

236  void solve( const Array_t& b, Array_t& x ) override
237  {
238  // Get the local grid.
239  auto local_grid = _vectors->layout()->localGrid();
240 
241  // Print banner
242  if ( 1 <= _print_level && 0 == local_grid->globalGrid().blockId() )
243  std::cout << std::endl
244  << "Preconditioned conjugate gradient" << std::endl;
245 
246  // Index space.
247  auto entity_space =
248  local_grid->indexSpace( Own(), EntityType(), Local() );
249 
250  // Subarrays.
251  auto p_old = createSubarray( *_vectors, 0, 1 );
252  auto z = createSubarray( *_vectors, 1, 2 );
253  auto r_old = createSubarray( *_vectors, 2, 3 );
254  auto q = createSubarray( *_vectors, 3, 4 );
255  auto p_new = createSubarray( *_vectors, 4, 5 );
256  auto r_new = createSubarray( *_vectors, 5, 6 );
257  auto A_halo_vectors = createSubarray( *_vectors, 0, 2 );
258  auto M_halo_vectors = createSubarray( *_vectors, 2, 4 );
259 
260  // Views.
261  auto x_view = x.view();
262  auto b_view = b.view();
263  auto M_view = _M->view();
264  auto A_view = _A->view();
265  auto p_old_view = p_old->view();
266  auto z_view = z->view();
267  auto r_old_view = r_old->view();
268  auto q_view = q->view();
269  auto p_new_view = p_new->view();
270  auto r_new_view = r_new->view();
271 
272  // Reset iteration count.
273  _num_iter = 0;
274 
275  // Compute the norm of the RHS.
276  std::vector<Scalar> b_norm( 1 );
277  ArrayOp::norm2( b, b_norm );
278 
279  // Copy the LHS into p so we can gather it.
280  Kokkos::deep_copy( p_old_view, x_view );
281 
282  // Gather the LHS through gatheing p and z.
283  _A_halo->gather( execution_space(), *A_halo_vectors );
284 
285  // Compute the initial residual and norm.
286  _residual_norm = 0.0;
287  Kokkos::parallel_reduce(
288  "compute_r0",
289  createExecutionPolicy( entity_space, execution_space() ),
290  KOKKOS_LAMBDA( const int i, const int j, const int k,
291  Scalar& result ) {
292  // Compute the local contribution from matrix-vector
293  // multiplication. Note that we copied x into p for this
294  // operation to easily perform the gather. Only apply the
295  // stencil entry if it is greater than 0.
296  Scalar Ax = 0.0;
297  for ( unsigned c = 0; c < _A_stencil.extent( 0 ); ++c )
298  if ( fabs( A_view( i, j, k, c ) ) > 0.0 )
299  Ax += A_view( i, j, k, c ) *
300  p_old_view( i + _A_stencil( c, Dim::I ),
301  j + _A_stencil( c, Dim::J ),
302  k + _A_stencil( c, Dim::K ), 0 );
303 
304  // Compute the residual.
305  auto r_new = b_view( i, j, k, 0 ) - Ax;
306 
307  // Assign the residual.
308  r_old_view( i, j, k, 0 ) = r_new;
309 
310  // Contribute to the reduction.
311  result += r_new * r_new;
312  },
313  _residual_norm );
314 
315  // Finish the global norm reduction.
316  MPI_Allreduce( MPI_IN_PLACE, &_residual_norm, 1,
317  MpiTraits<Scalar>::type(), MPI_SUM,
318  local_grid->globalGrid().comm() );
319 
320  // If we already have met our criteria then return.
321  _residual_norm = std::sqrt( _residual_norm ) / b_norm[0];
322  if ( 2 == _print_level && 0 == local_grid->globalGrid().blockId() )
323  std::cout << "Iteration " << _num_iter
324  << ": |r|_2 / |b|_2 = " << _residual_norm << std::endl;
325  if ( _residual_norm <= _tol )
326  return;
327 
328  // r and q.
329  _M_halo->gather( execution_space(), *M_halo_vectors );
330 
331  // Compute the initial preconditioned residual.
332  Scalar zTr_old = 0.0;
333  Kokkos::parallel_reduce(
334  "compute_z0",
335  createExecutionPolicy( entity_space, execution_space() ),
336  KOKKOS_LAMBDA( const int i, const int j, const int k,
337  Scalar& result ) {
338  // Compute the local contribution from matrix-vector
339  // multiplication. Only apply the stencil entry if it is
340  // greater than 0.
341  Scalar Mr = 0.0;
342  for ( unsigned c = 0; c < _M_stencil.extent( 0 ); ++c )
343  if ( fabs( M_view( i, j, k, c ) ) > 0.0 )
344  Mr += M_view( i, j, k, c ) *
345  r_old_view( i + _M_stencil( c, Dim::I ),
346  j + _M_stencil( c, Dim::J ),
347  k + _M_stencil( c, Dim::K ), 0 );
348  // Write values.
349  z_view( i, j, k, 0 ) = Mr;
350  p_old_view( i, j, k, 0 ) = Mr;
351  p_new_view( i, j, k, 0 ) = Mr;
352 
353  // Compute zTr
354  result += Mr * r_old_view( i, j, k, 0 );
355  },
356  zTr_old );
357 
358  // Finish computation of zTr
359  MPI_Allreduce( MPI_IN_PLACE, &zTr_old, 1, MpiTraits<Scalar>::type(),
360  MPI_SUM, local_grid->globalGrid().comm() );
361 
362  // Gather the LHS through gatheing p and z.
363  _A_halo->gather( execution_space(), *A_halo_vectors );
364 
365  // Compute A*p and pT*A*p.
366  Scalar pTAp = 0.0;
367  Kokkos::parallel_reduce(
368  "compute_q0",
369  createExecutionPolicy( entity_space, execution_space() ),
370  KOKKOS_LAMBDA( const int i, const int j, const int k,
371  Scalar& result ) {
372  // Compute the local contribution from matrix-vector
373  // multiplication. This computes the updated p vector
374  // in-line to avoid another kernel launch. Only apply the
375  // stencil entry if it is greater than 0.
376  Scalar Ap = 0.0;
377  for ( unsigned c = 0; c < _A_stencil.extent( 0 ); ++c )
378  if ( fabs( A_view( i, j, k, c ) ) > 0.0 )
379  Ap += A_view( i, j, k, c ) *
380  ( p_old_view( i + _A_stencil( c, Dim::I ),
381  j + _A_stencil( c, Dim::J ),
382  k + _A_stencil( c, Dim::K ), 0 ) );
383 
384  // Write values.
385  q_view( i, j, k, 0 ) = Ap;
386 
387  // Compute contribution to the dot product.
388  result += p_old_view( i, j, k, 0 ) * Ap;
389  },
390  pTAp );
391 
392  // Finish the global reduction on pTAp.
393  MPI_Allreduce( MPI_IN_PLACE, &pTAp, 1, MpiTraits<Scalar>::type(),
394  MPI_SUM, local_grid->globalGrid().comm() );
395 
396  // Iterate.
397  bool converged = false;
398  Scalar zTr_new = 0.0;
399  Scalar alpha;
400  Scalar beta;
401  while ( _residual_norm > _tol && _num_iter < _max_iter )
402  {
403  // Gather r and q.
404  _M_halo->gather( execution_space(), *M_halo_vectors );
405 
406  // Kernel 1: Compute x, r, residual norm, and zTr
407  alpha = zTr_old / pTAp;
408  zTr_new = 0.0;
409  Kokkos::parallel_reduce(
410  "cg_kernel_1",
411  createExecutionPolicy( entity_space, execution_space() ),
412  KOKKOS_LAMBDA( const int i, const int j, const int k,
413  Scalar& result ) {
414  // Compute the local contribution from matrix-vector
415  // multiplication. This computes the updated q vector
416  // in-line to avoid another kernel launch. Only apply the
417  // stencil entry if it is greater than 0.
418  Scalar Mr = 0.0;
419  for ( unsigned c = 0; c < _M_stencil.extent( 0 ); ++c )
420  if ( fabs( M_view( i, j, k, c ) ) > 0.0 )
421  Mr +=
422  M_view( i, j, k, c ) *
423  ( r_old_view( i + _M_stencil( c, Dim::I ),
424  j + _M_stencil( c, Dim::J ),
425  k + _M_stencil( c, Dim::K ), 0 ) -
426  alpha * q_view( i + _M_stencil( c, Dim::I ),
427  j + _M_stencil( c, Dim::J ),
428  k + _M_stencil( c, Dim::K ),
429  0 ) );
430 
431  // Compute the updated x.
432  Scalar x_new =
433  x_view( i, j, k, 0 ) + alpha * p_new_view( i, j, k, 0 );
434 
435  // Compute the updated residual.
436  Scalar r_new =
437  r_old_view( i, j, k, 0 ) - alpha * q_view( i, j, k, 0 );
438 
439  // Write to old p vector.
440  p_old_view( i, j, k, 0 ) = p_new_view( i, j, k, 0 );
441 
442  // Write values.
443  x_view( i, j, k, 0 ) = x_new;
444  r_new_view( i, j, k, 0 ) = r_new;
445  z_view( i, j, k, 0 ) = Mr;
446 
447  // Compute contribution to the zTr.
448  result += Mr * r_new;
449  },
450  zTr_new );
451 
452  // Finish the global reduction on zTr and r_norm.
453  MPI_Allreduce( MPI_IN_PLACE, &zTr_new, 1, MpiTraits<Scalar>::type(),
454  MPI_SUM, local_grid->globalGrid().comm() );
455 
456  // Update residual norm
457  _residual_norm = std::sqrt( fabs( zTr_new ) ) / b_norm[0];
458 
459  // Increment iteration count.
460  _num_iter++;
461 
462  // Output result
463  if ( 2 == _print_level && 0 == local_grid->globalGrid().blockId() )
464  std::cout << "Iteration " << _num_iter
465  << ": |r|_2 / |b|_2 = " << _residual_norm
466  << std::endl;
467 
468  // Check for convergence.
469  if ( _residual_norm <= _tol )
470  {
471  converged = true;
472  break;
473  }
474 
475  // Gather p and z.
476  _A_halo->gather( execution_space(), *A_halo_vectors );
477 
478  // Kernel 2: Compute p, A*p, and p^T*A*p
479  beta = zTr_new / zTr_old;
480  pTAp = 0.0;
481  Kokkos::parallel_reduce(
482  "cg_kernel_2",
483  createExecutionPolicy( entity_space, execution_space() ),
484  KOKKOS_LAMBDA( const int i, const int j, const int k,
485  Scalar& result ) {
486  // Compute the local contribution from matrix-vector
487  // multiplication. This computes the updated p vector
488  // in-line to avoid another kernel launch. Only apply the
489  // stencil entry if it is greater than 0.
490  Scalar Ap = 0.0;
491  for ( unsigned c = 0; c < _A_stencil.extent( 0 ); ++c )
492  if ( fabs( A_view( i, j, k, c ) ) > 0.0 )
493  Ap += A_view( i, j, k, c ) *
494  ( z_view( i + _A_stencil( c, Dim::I ),
495  j + _A_stencil( c, Dim::J ),
496  k + _A_stencil( c, Dim::K ), 0 ) +
497  beta *
498  p_old_view( i + _A_stencil( c, Dim::I ),
499  j + _A_stencil( c, Dim::J ),
500  k + _A_stencil( c, Dim::K ),
501  0 ) );
502 
503  // Compute the updated p.
504  Scalar p_new =
505  z_view( i, j, k, 0 ) + beta * p_old_view( i, j, k, 0 );
506 
507  // Write to old residual.
508  r_old_view( i, j, k, 0 ) = r_new_view( i, j, k, 0 );
509 
510  // Write values.
511  q_view( i, j, k, 0 ) = Ap;
512  p_new_view( i, j, k, 0 ) = p_new;
513 
514  // Compute contribution to the dot product.
515  result += p_new * Ap;
516  },
517  pTAp );
518 
519  // Finish the global reduction on pTAp.
520  MPI_Allreduce( MPI_IN_PLACE, &pTAp, 1, MpiTraits<Scalar>::type(),
521  MPI_SUM, local_grid->globalGrid().comm() );
522 
523  // Update zTr
524  zTr_old = zTr_new;
525  }
526 
527  // Output end state.
528  if ( 1 <= _print_level && 0 == local_grid->globalGrid().blockId() )
529  std::cout << "Finished in " << _num_iter
530  << " iterations, converged to " << _residual_norm
531  << std::endl
532  << std::endl;
533 
534  // If we didn't converge throw.
535  if ( !converged )
536  throw std::runtime_error( "CG solver did not converge" );
537  }

540  int getNumIter() override { return _num_iter; }
541 
542  // Get the relative residual norm achieved on the last solve.
543  double getFinalRelativeResidualNorm() override { return _residual_norm; }

545  private:
546  // Set the stencil of a matrix.
547  void setStencil( const std::vector<std::array<int, 3>>& stencil,
548  const bool is_symmetric,
549  Kokkos::View<int* [3], DeviceType>& device_stencil,
550  std::shared_ptr<Halo<memory_space>>& halo,
551  std::shared_ptr<Array_t>& matrix )
552  {
553  // For now we don't support symmetry.
554  if ( is_symmetric )
555  throw std::logic_error(
556  "Reference CG currently does not support symmetry" );
557 
558  // Get the local grid.
559  auto local_grid = _vectors->layout()->localGrid();
560 
561  // Copy stencil to the device.
562  device_stencil = Kokkos::View<int* [3], DeviceType>(
563  Kokkos::ViewAllocateWithoutInitializing( "stencil" ),
564  stencil.size() );
565  auto stencil_mirror =
566  Kokkos::create_mirror_view( Kokkos::HostSpace(), device_stencil );
567  for ( unsigned s = 0; s < stencil.size(); ++s )
568  for ( int d = 0; d < 3; ++d )
569  stencil_mirror( s, d ) = stencil[s][d];
570  Kokkos::deep_copy( device_stencil, stencil_mirror );
571 
572  // Compose the halo pattern and compute how wide the halo needs to be
573  // to gather all elements accessed by the stencil.
574  std::set<std::array<int, 3>> neighbor_set;
575  std::array<int, 3> neighbor;
576  int width = 0;
577  for ( auto s : stencil )
578  {
579  // Compse a set of the neighbor ranks based on the stencil.
580  for ( int d = 0; d < 3; ++d )
581  neighbor[d] = ( s[d] == 0 ) ? 0 : s[d] / std::abs( s[d] );
582  neighbor_set.emplace( neighbor );
583 
584  // Compute the width of the halo needed to apply the stencil.
585  for ( int d = 0; d < 3; ++d )
586  width = std::max( width, std::abs( s[d] ) );
587  }
588  std::vector<std::array<int, 3>> halo_neighbors( neighbor_set.size() );
589  std::copy( neighbor_set.begin(), neighbor_set.end(),
590  halo_neighbors.begin() );
591 
592  // Build the halo. We put 2 entries as each operator application will
593  // gather 2 vectors with fused kernels.
594  auto halo_layout = createArrayLayout( local_grid, 2, EntityType() );
595  HaloPattern pattern;
596  pattern.setNeighbors( halo_neighbors );
597  halo = createHalo<Scalar, DeviceType>( *halo_layout, pattern, width );
598 
599  // Create a new layout.
600  auto matrix_layout =
601  createArrayLayout( local_grid, stencil.size(), EntityType() );
602 
603  // Allocate the matrix.
604  matrix = createArray<Scalar, DeviceType>( "matrix", matrix_layout );
605  }
606 
607  private:
608  Scalar _tol;
609  int _max_iter;
610  int _print_level;
611  int _num_iter;
612  Scalar _residual_norm;
613  int _diag_entry;
614  Kokkos::View<int* [3], DeviceType> _A_stencil;
615  Kokkos::View<int* [3], DeviceType> _M_stencil;
616  std::shared_ptr<Halo<memory_space>> _A_halo;
617  std::shared_ptr<Halo<memory_space>> _M_halo;
618  std::shared_ptr<Array_t> _A;
619  std::shared_ptr<Array_t> _M;
620  std::shared_ptr<Array_t> _vectors;
621 };
