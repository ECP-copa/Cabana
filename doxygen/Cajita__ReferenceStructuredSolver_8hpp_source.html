241  void solve( const Array_t& b, Array_t& x ) override
242  {
243  // Get the local grid.
244  auto local_grid = _vectors->layout()->localGrid();
245 
246  // Print banner
247  if ( 1 <= _print_level && 0 == local_grid->globalGrid().blockId() )
248  std::cout << std::endl
249  << "Preconditioned conjugate gradient" << std::endl;
250 
251  // Index space.
252  auto entity_space =
253  local_grid->indexSpace( Own(), EntityType(), Local() );
254 
255  // Subarrays.
256  auto p_old = createSubarray( *_vectors, 0, 1 );
257  auto z = createSubarray( *_vectors, 1, 2 );
258  auto r_old = createSubarray( *_vectors, 2, 3 );
259  auto q = createSubarray( *_vectors, 3, 4 );
260  auto p_new = createSubarray( *_vectors, 4, 5 );
261  auto r_new = createSubarray( *_vectors, 5, 6 );
262  auto A_halo_vectors = createSubarray( *_vectors, 0, 2 );
263  auto M_halo_vectors = createSubarray( *_vectors, 2, 4 );
264 
265  // Views.
266  auto x_view = x.view();
267  auto b_view = b.view();
268  auto M_view = _M->view();
269  auto A_view = _A->view();
270  auto p_old_view = p_old->view();
271  auto z_view = z->view();
272  auto r_old_view = r_old->view();
273  auto q_view = q->view();
274  auto p_new_view = p_new->view();
275  auto r_new_view = r_new->view();
276 
277  // Reset iteration count.
278  _num_iter = 0;
279 
280  // Compute the norm of the RHS.
281  std::vector<Scalar> b_norm( 1 );
282  ArrayOp::norm2( b, b_norm );
283 
284  // Copy the LHS into p so we can gather it.
285  Kokkos::deep_copy( p_old_view, x_view );
286 
287  // Gather the LHS through gatheing p and z.
288  _A_halo->gather( execution_space(), *A_halo_vectors );
289 
290  // Compute the initial residual and norm.
291  _residual_norm = 0.0;
292  auto compute_r0 = createComputeR0( _A_stencil, A_view, p_old_view,
293  b_view, r_old_view );
294  grid_parallel_reduce(
295  "compute_r0", execution_space(), entity_space,
296  std::integral_constant<std::size_t, num_space_dim>{}, compute_r0,
297  _residual_norm );
298 
299  // Finish the global norm reduction.
300  MPI_Allreduce( MPI_IN_PLACE, &_residual_norm, 1,
301  MpiTraits<Scalar>::type(), MPI_SUM,
302  local_grid->globalGrid().comm() );
303 
304  // If we already have met our criteria then return.
305  _residual_norm = std::sqrt( _residual_norm ) / b_norm[0];
306  if ( 2 == _print_level && 0 == local_grid->globalGrid().blockId() )
307  std::cout << "Iteration " << _num_iter
308  << ": |r|_2 / |b|_2 = " << _residual_norm << std::endl;
309  if ( _residual_norm <= _tol )
310  return;
311 
312  // r and q.
313  _M_halo->gather( execution_space(), *M_halo_vectors );
314 
315  // Compute the initial preconditioned residual.
316  Scalar zTr_old = 0.0;
317  auto compute_z0 = createComputeZ0( _M_stencil, M_view, p_old_view,
318  p_new_view, r_old_view, z_view );
319  grid_parallel_reduce(
320  "compute_z0", execution_space(), entity_space,
321  std::integral_constant<std::size_t, num_space_dim>{}, compute_z0,
322  zTr_old );
323 
324  // Finish computation of zTr
325  MPI_Allreduce( MPI_IN_PLACE, &zTr_old, 1, MpiTraits<Scalar>::type(),
326  MPI_SUM, local_grid->globalGrid().comm() );
327 
328  // Gather the LHS through gatheing p and z.
329  _A_halo->gather( execution_space(), *A_halo_vectors );
330 
331  // Compute A*p and pT*A*p.
332  Scalar pTAp = 0.0;
333  auto compute_q0 =
334  createComputeQ0( _A_stencil, A_view, p_old_view, q_view );
335  grid_parallel_reduce(
336  "compute_q0", execution_space(), entity_space,
337  std::integral_constant<std::size_t, num_space_dim>{}, compute_q0,
338  pTAp );
339 
340  // Finish the global reduction on pTAp.
341  MPI_Allreduce( MPI_IN_PLACE, &pTAp, 1, MpiTraits<Scalar>::type(),
342  MPI_SUM, local_grid->globalGrid().comm() );
343 
344  // Iterate.
345  bool converged = false;
346  Scalar zTr_new = 0.0;
347  Scalar alpha;
348  Scalar beta;
349  while ( _residual_norm > _tol && _num_iter < _max_iter )
350  {
351  // Gather r and q.
352  _M_halo->gather( execution_space(), *M_halo_vectors );
353 
354  // Kernel 1: Compute x, r, residual norm, and zTr
355  alpha = zTr_old / pTAp;
356  zTr_new = 0.0;
357  auto cg_kernel_1 = createKernel1(
358  _M_stencil, M_view, x_view, r_new_view, r_old_view, p_new_view,
359  p_old_view, z_view, q_view, alpha );
360  grid_parallel_reduce(
361  "cg_kernel_1", execution_space(), entity_space,
362  std::integral_constant<std::size_t, num_space_dim>{},
363  cg_kernel_1, zTr_new );
364 
365  // Finish the global reduction on zTr and r_norm.
366  MPI_Allreduce( MPI_IN_PLACE, &zTr_new, 1, MpiTraits<Scalar>::type(),
367  MPI_SUM, local_grid->globalGrid().comm() );
368 
369  // Update residual norm
370  _residual_norm = std::sqrt( fabs( zTr_new ) ) / b_norm[0];
371 
372  // Increment iteration count.
373  _num_iter++;
374 
375  // Output result
376  if ( 2 == _print_level && 0 == local_grid->globalGrid().blockId() )
377  std::cout << "Iteration " << _num_iter
378  << ": |r|_2 / |b|_2 = " << _residual_norm
379  << std::endl;
380 
381  // Check for convergence.
382  if ( _residual_norm <= _tol )
383  {
384  converged = true;
385  break;
386  }
387 
388  // Gather p and z.
389  _A_halo->gather( execution_space(), *A_halo_vectors );
390 
391  // Kernel 2: Compute p, A*p, and p^T*A*p
392  beta = zTr_new / zTr_old;
393  pTAp = 0.0;
394  auto cg_kernel_2 = createKernel2(
395  _A_stencil, A_view, x_view, r_new_view, r_old_view, p_new_view,
396  p_old_view, z_view, q_view, beta );
397  grid_parallel_reduce(
398  "cg_kernel_2", execution_space(), entity_space,
399  std::integral_constant<std::size_t, num_space_dim>{},
400  cg_kernel_2, pTAp );
401 
402  // Finish the global reduction on pTAp.
403  MPI_Allreduce( MPI_IN_PLACE, &pTAp, 1, MpiTraits<Scalar>::type(),
404  MPI_SUM, local_grid->globalGrid().comm() );
405 
406  // Update zTr
407  zTr_old = zTr_new;
408  }
409 
410  // Output end state.
411  if ( 1 <= _print_level && 0 == local_grid->globalGrid().blockId() )
412  std::cout << "Finished in " << _num_iter
413  << " iterations, converged to " << _residual_norm
414  << std::endl
415  << std::endl;
416 
417  // If we didn't converge throw.
418  if ( !converged )
419  throw std::runtime_error( "CG solver did not converge" );
420  }

429  template <class StencilA, class ViewA, class ViewOldP, class ViewB,
430  class ViewOldR>
431  struct ComputeR0
432  {
433  StencilA A_stencil;
434  ViewA A_view;
435  ViewOldP p_old_view;
436  ViewB b_view;
437  ViewOldR r_old_view;

439  using value_type = typename ViewB::value_type;

441  KOKKOS_INLINE_FUNCTION void
442  operator()( const std::integral_constant<std::size_t, 3>&, const int i,
443  const int j, const int k, value_type& result ) const
444  {
445  // Compute the local contribution from matrix-vector
446  // multiplication. Note that we copied x into p for this
447  // operation to easily perform the gather. Only apply the
448  // stencil entry if it is greater than 0.
449  value_type Ax = 0.0;
450  for ( unsigned c = 0; c < A_stencil.extent( 0 ); ++c )
451  if ( fabs( A_view( i, j, k, c ) ) > 0.0 )
452  Ax += A_view( i, j, k, c ) *
453  p_old_view( i + A_stencil( c, Dim::I ),
454  j + A_stencil( c, Dim::J ),
455  k + A_stencil( c, Dim::K ), 0 );
456 
457  // Compute the residual.
458  auto r_new = b_view( i, j, k, 0 ) - Ax;
459 
460  // Assign the residual.
461  r_old_view( i, j, k, 0 ) = r_new;
462 
463  // Contribute to the reduction.
464  result += r_new * r_new;
465  }

585  const int j, const int k, value_type& result ) const
586  {
587  // Compute the local contribution from matrix-vector

597  k + A_stencil( c, Dim::K ), 0 ) );
598 
599  // Write values.
600  q_view( i, j, k, 0 ) = Ap;
601 
602  // Compute contribution to the dot product.
603  result += p_old_view( i, j, k, 0 ) * Ap;
604  }

637  template <class StencilM, class ViewM, class ViewX, class ViewNewR,
638  class ViewOldR, class ViewOldP, class ViewNewP, class ViewZ,
639  class ViewQ, class ValueType>
640  struct Kernel1
641  {
642  StencilM M_stencil;
643  ViewM M_view;
644  ViewX x_view;
645  ViewNewR r_new_view;
646  ViewOldR r_old_view;
647  ViewNewP p_new_view;
648  ViewOldP p_old_view;
649  ViewZ z_view;
650  ViewQ q_view;
651  ValueType alpha;
652 
653  KOKKOS_INLINE_FUNCTION void
654  operator()( const std::integral_constant<std::size_t, 3>&, const int i,
655  const int j, const int k, ValueType& result ) const
656  {
657  // Compute the local contribution from matrix-vector
658  // multiplication. This computes the updated q vector
659  // in-line to avoid another kernel launch. Only apply the
660  // stencil entry if it is greater than 0.
661  ValueType Mr = 0.0;
662  for ( unsigned c = 0; c < M_stencil.extent( 0 ); ++c )
663  if ( fabs( M_view( i, j, k, c ) ) > 0.0 )
664  Mr += M_view( i, j, k, c ) *
665  ( r_old_view( i + M_stencil( c, Dim::I ),
666  j + M_stencil( c, Dim::J ),
667  k + M_stencil( c, Dim::K ), 0 ) -
668  alpha * q_view( i + M_stencil( c, Dim::I ),
669  j + M_stencil( c, Dim::J ),
670  k + M_stencil( c, Dim::K ), 0 ) );
671 
672  // Compute the updated x.
673  ValueType x_new =
674  x_view( i, j, k, 0 ) + alpha * p_new_view( i, j, k, 0 );
675 
676  // Compute the updated residual.
677  ValueType r_new =
678  r_old_view( i, j, k, 0 ) - alpha * q_view( i, j, k, 0 );
679 
680  // Write to old p vector.
681  p_old_view( i, j, k, 0 ) = p_new_view( i, j, k, 0 );
682 
683  // Write values.
684  x_view( i, j, k, 0 ) = x_new;
685  r_new_view( i, j, k, 0 ) = r_new;
686  z_view( i, j, k, 0 ) = Mr;
687 
688  // Compute contribution to the zTr.
689  result += Mr * r_new;
690  }

693  operator()( const std::integral_constant<std::size_t, 2>&, const int i,
694  const int j, ValueType& result ) const
695  {
696  // Compute the local contribution from matrix-vector
697  // multiplication. This computes the updated q vector
698  // in-line to avoid another kernel launch. Only apply the
699  // stencil entry if it is greater than 0.
700  ValueType Mr = 0.0;
701  for ( unsigned c = 0; c < M_stencil.extent( 0 ); ++c )
702  if ( fabs( M_view( i, j, c ) ) > 0.0 )
703  Mr += M_view( i, j, c ) *

705  j + M_stencil( c, Dim::J ), 0 ) -
706  alpha * q_view( i + M_stencil( c, Dim::I ),
707  j + M_stencil( c, Dim::J ), 0 ) );

843  // Set the stencil of a matrix.
844  void
845  setStencil( const std::vector<std::array<int, num_space_dim>>& stencil,
846  const bool is_symmetric,
847  Kokkos::View<int* [num_space_dim], DeviceType>& device_stencil,
848  std::shared_ptr<Halo<memory_space>>& halo,
849  std::shared_ptr<Array_t>& matrix )
850  {
851  // For now we don't support symmetry.
852  if ( is_symmetric )
853  throw std::logic_error(
854  "Reference CG currently does not support symmetry" );
855 
856  // Get the local grid.
857  auto local_grid = _vectors->layout()->localGrid();
858 
859  // Copy stencil to the device.
860  device_stencil = Kokkos::View<int* [num_space_dim], DeviceType>(
861  Kokkos::ViewAllocateWithoutInitializing( "stencil" ),
862  stencil.size() );
863  auto stencil_mirror =
864  Kokkos::create_mirror_view( Kokkos::HostSpace(), device_stencil );
865  for ( unsigned s = 0; s < stencil.size(); ++s )
866  for ( std::size_t d = 0; d < num_space_dim; ++d )
867  stencil_mirror( s, d ) = stencil[s][d];
868  Kokkos::deep_copy( device_stencil, stencil_mirror );
869 
870  // Compose the halo pattern and compute how wide the halo needs to be
871  // to gather all elements accessed by the stencil.
872  std::set<std::array<int, num_space_dim>> neighbor_set;
873  std::array<int, num_space_dim> neighbor;
874  int width = 0;
875  for ( auto s : stencil )
876  {
877  // Compse a set of the neighbor ranks based on the stencil.
878  for ( std::size_t d = 0; d < num_space_dim; ++d )
879  neighbor[d] = ( s[d] == 0 ) ? 0 : s[d] / std::abs( s[d] );
880  neighbor_set.emplace( neighbor );
881 
882  // Compute the width of the halo needed to apply the stencil.
883  for ( std::size_t d = 0; d < num_space_dim; ++d )
884  width = std::max( width, std::abs( s[d] ) );
885  }
886  std::vector<std::array<int, num_space_dim>> halo_neighbors(
887  neighbor_set.size() );
888  std::copy( neighbor_set.begin(), neighbor_set.end(),
889  halo_neighbors.begin() );
890 
891  // Build the halo. We put 2 entries as each operator application will
892  // gather 2 vectors with fused kernels.
893  auto halo_layout = createArrayLayout( local_grid, 2, EntityType() );
894  HaloPattern<num_space_dim> pattern;
895  pattern.setNeighbors( halo_neighbors );
896  halo = createHalo<Scalar, DeviceType>( *halo_layout, pattern, width );
897 
898  // Create a new layout.
899  auto matrix_layout =
900  createArrayLayout( local_grid, stencil.size(), EntityType() );
901 
902  // Allocate the matrix.
903  matrix = createArray<Scalar, DeviceType>( "matrix", matrix_layout );
904  }
