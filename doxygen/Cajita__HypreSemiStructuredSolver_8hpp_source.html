70 template <class ArrayLayout_t>
71 HypreSemiStructuredSolver( const ArrayLayout_t& layout, int n_vars,
72 const bool is_preconditioner = false )
73 : _comm( layout.localGrid()->globalGrid().comm() )
74 , _is_preconditioner( is_preconditioner )
75 {
76 HYPRE_Init();
77
78 static_assert( is_array_layout<ArrayLayout_t>::value,
79 "Must use an array layout" );
80 static_assert(
81 std::is_same<typename ArrayLayout_t::entity_type,
82 entity_type>::value,
83 "Array layout entity type mush match solver entity type" );
84
85 // Spatial dimension.
86 const std::size_t num_space_dim = ArrayLayout_t::num_space_dim;
87
88 // Only a single part grid is supported initially
89 int n_parts = 1;
90 int part = 0;
91
92 // Only create data structures if this is not a preconditioner.
93 if ( !_is_preconditioner )
94 {
95 // Create the grid.
96 auto error = HYPRE_SStructGridCreate( _comm, num_space_dim, n_parts,
97 &_grid );
98 checkHypreError( error );
99
100 // Get the global index space spanned by the local grid on this
101 // rank. Note that the upper bound is not a bound but rather the
102 // last index as this is what Hypre wants. Note that we reordered
103 // this to KJI from IJK to be consistent with HYPRE ordering. By
104 // setting up the grid like this, HYPRE will then want layout-right
105 // data indexed as (i,j,k) or (i,j,k,l) which will allow us to
106 // directly use Kokkos::deep_copy to move data between Cajita arrays
107 // and HYPRE data structures.
108 auto global_space = layout.indexSpace( Own(), Global() );
109 _lower.resize( num_space_dim );
110 _upper.resize( num_space_dim );
111 for ( std::size_t d = 0; d < num_space_dim; ++d )
112 {
113 _lower[d] = static_cast<HYPRE_Int>(
114 global_space.min( num_space_dim - d - 1 ) );
115 _upper[d] = static_cast<HYPRE_Int>(
116 global_space.max( num_space_dim - d - 1 ) - 1 );
117 }
118 error = HYPRE_SStructGridSetExtents( _grid, part, _lower.data(),
119 _upper.data() );
120 checkHypreError( error );
121
122 // Get periodicity. Note we invert the order of this to KJI as well.
123 const auto& global_grid = layout.localGrid()->globalGrid();
124 HYPRE_Int periodic[num_space_dim];
125 for ( std::size_t d = 0; d < num_space_dim; ++d )
126 periodic[num_space_dim - 1 - d] =
127 global_grid.isPeriodic( d )
128 ? layout.localGrid()->globalGrid().globalNumEntity(
129 EntityType(), d )
130 : 0;
131 error = HYPRE_SStructGridSetPeriodic( _grid, part, periodic );
132 checkHypreError( error );
133
134 // Set the variables on the HYPRE grid
135 std::vector<HYPRE_SStructVariable> vartypes;
136 vartypes.resize( n_vars );
137 for ( int i = 0; i < n_vars; ++i )
138 {
139 vartypes[i] = HYPRE_SSTRUCT_VARIABLE_CELL;
140 }
141 error = HYPRE_SStructGridSetVariables( _grid, part, n_vars,
142 vartypes.data() );
143
144 // Assemble the grid.
145 error = HYPRE_SStructGridAssemble( _grid );
146 checkHypreError( error );
147
148 // Allocate LHS and RHS vectors and initialize to zero. Note that we
149 // are fixing the views under these vectors to layout-right.
150
151 std::array<long, num_space_dim + 1> reorder_size;
152 for ( std::size_t d = 0; d < num_space_dim; ++d )
153 {
154 reorder_size[d] = global_space.extent( d );
155 }
156 reorder_size.back() = n_vars;
157 IndexSpace<num_space_dim + 1> reorder_space( reorder_size );
158 auto vector_values =
159 createView<HYPRE_Complex, Kokkos::LayoutRight, memory_space>(
160 "vector_values0", reorder_space );
161 Kokkos::deep_copy( vector_values, 0.0 );
162
163 _stencils.resize( n_vars );
164 _stencil_size.resize( n_vars );
165 _stencil_index.resize( n_vars,
166 std::vector<unsigned>( n_vars + 1 ) );
167
168 error = HYPRE_SStructVectorCreate( _comm, _grid, &_b );
169 checkHypreError( error );
170 error = HYPRE_SStructVectorSetObjectType( _b, object_type );
171 checkHypreError( error );
172 error = HYPRE_SStructVectorInitialize( _b );
173 checkHypreError( error );
174 for ( int i = 0; i < n_vars; ++i )
175 {
176 error = HYPRE_SStructVectorSetBoxValues(
177 _b, part, _lower.data(), _upper.data(), i,
178 vector_values.data() );
179 checkHypreError( error );
180 }
181 error = HYPRE_SStructVectorAssemble( _b );
182 checkHypreError( error );
183
184 error = HYPRE_SStructVectorCreate( _comm, _grid, &_x );
185 checkHypreError( error );
186 error = HYPRE_SStructVectorSetObjectType( _x, object_type );
187 checkHypreError( error );
188 error = HYPRE_SStructVectorInitialize( _x );
189 checkHypreError( error );
190 for ( int i = 0; i < n_vars; ++i )
191 {
192 error = HYPRE_SStructVectorSetBoxValues(
193 _x, part, _lower.data(), _upper.data(), i,
194 vector_values.data() );
195 checkHypreError( error );
196 }
197 checkHypreError( error );
198 error = HYPRE_SStructVectorAssemble( _x );
199 checkHypreError( error );
200 }
201 }

236 void createMatrixStencil( int NumSpaceDim, int var = 0, int n_vars = 3,
237 std::vector<unsigned> stencil_length = { 7, 7,
238 7 } )
239 {
240 // This function is only valid for non-preconditioners.
241 if ( _is_preconditioner )
242 throw std::logic_error(
243 "Cannot call createMatrixStencil() on preconditioners" );
244
245 // Generate the stencil indexing
246 unsigned index = 0;
247 for ( int i = 0; i < n_vars; ++i )
248 {
249 _stencil_index[var][i] = index;
250 index += stencil_length[i];
251 }
252 _stencil_index[var][n_vars] = index;
253
254 // Create the stencil.
255 _stencil_size[var] = index;
256 auto error = HYPRE_SStructStencilCreate(
257 NumSpaceDim, _stencil_size[var], &_stencils[var] );
258 checkHypreError( error );
259 }

270 template <std::size_t NumSpaceDim>
271 void
272 setMatrixStencil( const std::vector<std::array<int, NumSpaceDim>>& stencil,
273 int var = 0, int dep = 0 )
274 {
275 // This function is only valid for non-preconditioners.
276 if ( _is_preconditioner )
277 throw std::logic_error(
278 "Cannot call setMatrixStencil() on preconditioners" );
279
280 std::array<HYPRE_Int, NumSpaceDim> offset;
281
282 auto index = _stencil_index[var][dep];
283 for ( unsigned n = index; n < index + stencil.size(); ++n )
284 {
285 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
286 offset[d] = stencil[n - index][d];
287 auto error = HYPRE_SStructStencilSetEntry( _stencils[var], n,
288 offset.data(), dep );
289 checkHypreError( error );
290 }

359 template <class Array_t>
360 void setMatrixValues( const Array_t& values, int v_x, int v_h )
361 {
362 static_assert( is_array<Array_t>::value, "Must use an array" );
363 static_assert(
364 std::is_same<typename Array_t::entity_type, entity_type>::value,
365 "Array entity type mush match solver entity type" );
366 static_assert(
367 std::is_same<typename Array_t::memory_space, MemorySpace>::value,
368 "Array device type and solver device type are different." );
369
370 static_assert(
371 std::is_same<typename Array_t::value_type, value_type>::value,
372 "Array value type and solver value type are different." );
373
374 // This function is only valid for non-preconditioners.
375 if ( _is_preconditioner )
376 throw std::logic_error(
377 "Cannot call setMatrixValues() on preconditioners" );
378
379 int index_size =
380 _stencil_index[v_h][v_x + 1] - _stencil_index[v_h][v_x];
381
382 // Ensure the values array matches up in dimension with the stencil size
383 if ( values.layout()->dofsPerEntity() !=
384 static_cast<int>( index_size ) )
385 throw std::runtime_error(
386 "Number of matrix values does not match stencil size" );
387
388 // Spatial dimension.
389 const std::size_t num_space_dim = Array_t::num_space_dim;
390
391 int part = 0;
392
393 // Copy the matrix entries into HYPRE. The HYPRE layout is fixed as
394 // layout-right.
395 auto owned_space = values.layout()->indexSpace( Own(), Local() );
396 std::array<long, num_space_dim + 1> reorder_size;
397 for ( std::size_t d = 0; d < num_space_dim; ++d )
398 {
399 reorder_size[d] = owned_space.extent( d );
400 }
401
402 reorder_size.back() = index_size;
403 IndexSpace<num_space_dim + 1> reorder_space( reorder_size );
404 auto a_values =
405 createView<HYPRE_Complex, Kokkos::LayoutRight, memory_space>(
406 "a_values", reorder_space );
407
408 auto values_subv = createSubview( values.view(), owned_space );
409 Kokkos::deep_copy( a_values, values_subv );
410
411 // Insert values into the HYPRE matrix.
412 std::vector<HYPRE_Int> indices( index_size );
413 int start = _stencil_index[v_h][v_x];
414 std::iota( indices.begin(), indices.end(), start );
415 auto error = HYPRE_SStructMatrixSetBoxValues(
416 _A, part, _lower.data(), _upper.data(), v_h, indices.size(),
417 indices.data(), a_values.data() );
418 checkHypreError( error );
419 }

497 template <class Array_t>
498 void solve( const Array_t& b, Array_t& x, int n_vars = 3 )
499 {
500 Kokkos::Profiling::pushRegion(
501 "Cajita::HypreSemiStructuredSolver::solve" );
502
503 static_assert( is_array<Array_t>::value, "Must use an array" );
504 static_assert(
505 std::is_same<typename Array_t::entity_type, entity_type>::value,
506 "Array entity type mush match solver entity type" );
507 static_assert(
508 std::is_same<typename Array_t::memory_space, MemorySpace>::value,
509 "Array device type and solver device type are different." );
510
511 static_assert(
512 std::is_same<typename Array_t::value_type, value_type>::value,
513 "Array value type and solver value type are different." );
514
515 // This function is only valid for non-preconditioners.
516 if ( _is_preconditioner )
517 throw std::logic_error( "Cannot call solve() on preconditioners" );
518
519 // Spatial dimension.
520 const std::size_t num_space_dim = Array_t::num_space_dim;
521
522 int part = 0;
523
524 // Initialize the RHS.
525 auto error = HYPRE_SStructVectorInitialize( _b );
526 checkHypreError( error );
527
528 // Copy the RHS into HYPRE. The HYPRE layout is fixed as layout-right.
529 auto owned_space = b.layout()->indexSpace( Own(), Local() );
530 std::array<long, num_space_dim + 1> reorder_min;
531 std::array<long, num_space_dim + 1> reorder_max;
532 for ( std::size_t d = 0; d < num_space_dim; ++d )
533 {
534 reorder_min[d] = owned_space.min( d );
535 reorder_max[d] = owned_space.max( d );
536 }
537
538 // Insert b values into the HYPRE vector.
539 // The process of creating the view and then deep copying each
540 // variable is functional, but we should avoid this process
541 // for performance if possible
542 for ( int var = 0; var < n_vars; ++var )
543 {
544 reorder_min.back() = var;
545 reorder_max.back() = var + 1;
546
547 IndexSpace<num_space_dim + 1> reorder_space( reorder_min,
548 reorder_max );
549 auto b_values =
550 createView<HYPRE_Complex, Kokkos::LayoutRight, memory_space>(
551 "vector_values", reorder_space );
552 // Extract one variable at at time.
553 auto b_subv = createSubview( b.view(), reorder_space );
554
555 Kokkos::deep_copy( b_values, b_subv );
556 error = HYPRE_SStructVectorSetBoxValues(
557 _b, part, _lower.data(), _upper.data(), var, b_values.data() );
558 checkHypreError( error );
559 }
560
561 error = HYPRE_SStructVectorAssemble( _b );
562 checkHypreError( error );
563
564 // Solve the problem
565 this->solveImpl( _A, _b, _x );
566
567 // Extract the solution from the LHS
568 for ( int var = 0; var < n_vars; ++var )
569 {
570 reorder_min.back() = var;
571 reorder_max.back() = var + 1;
572
573 IndexSpace<num_space_dim + 1> reorder_space( reorder_min,
574 reorder_max );
575 auto x_values =
576 createView<HYPRE_Complex, Kokkos::LayoutRight, memory_space>(
577 "vector_values", reorder_space );
578
579 // Extract one variable at at time.
580 // Use a pair here to retain the view rank.
581
582 error = HYPRE_SStructVectorGetBoxValues(
583 _x, part, _lower.data(), _upper.data(), var, x_values.data() );
584 checkHypreError( error );
585
586 // Copy the HYPRE solution to the LHS.
587 auto x_subv = createSubview( x.view(), reorder_space );
588 Kokkos::deep_copy( x_subv, x_values );
589 }
590
591 Kokkos::Profiling::popRegion();
592 }
