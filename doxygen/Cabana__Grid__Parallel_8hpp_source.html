197template <class FunctorType, class ExecutionSpace, std::size_t NumSpace>
198inline void
199grid_parallel_for( const std::string& label, const ExecutionSpace& exec_space,
200 const Kokkos::Array<IndexSpace<4>, NumSpace>& index_spaces,
201 const FunctorType& functor )
202{
203 Kokkos::Profiling::pushRegion( "Cabana::Grid::grid_parallel_for" );
204
205 // Compute the total number of threads needed and the index space offsets
206 // via inclusive scan.
207 int size = index_spaces[0].size();
208 Kokkos::Array<int, NumSpace> exclusive_offsets;
209 Kokkos::Array<int, NumSpace> inclusive_offsets;
210 exclusive_offsets[0] = 0;
211 inclusive_offsets[0] = size;
212 for ( std::size_t sp = 1; sp < NumSpace; ++sp )
213 {
214 size += index_spaces[sp].size();
215 exclusive_offsets[sp] =
216 exclusive_offsets[sp - 1] + index_spaces[sp - 1].size();
217 inclusive_offsets[sp] =
218 inclusive_offsets[sp - 1] + index_spaces[sp].size();
219 }
220
221 // Unroll the index spaces into a linear parallel for.
222 Kokkos::parallel_for(
223 label, Kokkos::RangePolicy<ExecutionSpace>( exec_space, 0, size ),
224 KOKKOS_LAMBDA( const int n ) {
225 // Get the index space id.
226 int s = -1;
227 for ( std::size_t sp = 0; sp < NumSpace; ++sp )
228 {
229 if ( n < inclusive_offsets[sp] )
230 {
231 s = sp;
232 break;
233 }
234 }
235
236 // Linear id in index space.
237 int linear_id = n - exclusive_offsets[s];
238
239 // Compute entity id.
240 int extent_l = index_spaces[s].extent( 3 );
241 int extent_kl = extent_l * index_spaces[s].extent( Dim::K );
242 int extent_jkl = extent_kl * index_spaces[s].extent( Dim::J );
243 int i_base = linear_id / extent_jkl;
244 int stride_j = extent_jkl * i_base;
245 int j_base = ( linear_id - stride_j ) / extent_kl;
246 int stride_k = ( stride_j + extent_kl * j_base );
247 int k_base = ( linear_id - stride_k ) / extent_l;
248 int l_base = linear_id - stride_k - k_base * extent_l;
249
250 // Execute the user functor. Provide the index space id so they
251 // can discern which space they are in within the functor.
252 functor( s, i_base + index_spaces[s].min( Dim::I ),
253 j_base + index_spaces[s].min( Dim::J ),
254 k_base + index_spaces[s].min( Dim::K ),
255 l_base + index_spaces[s].min( 3 ) );
256 } );
257
258 Kokkos::Profiling::popRegion();
259}
