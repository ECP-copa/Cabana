309  void resize( const size_type n )
310  {
311  static_assert( !memory_traits::is_unmanaged,
312  "Cannot resize unmanaged memory" );
313 
314  // Reserve memory if needed.
315  reserve( n );
316 
317  // Update the sizes of the data. This is potentially different than
318  // the amount of allocated data.
319  _size = n;
320  _num_soa = std::floor( n / vector_length );
321  if ( 0 < n % vector_length )
322  ++_num_soa;
323  }
324 
339  void reserve( const size_type n )
340  {
341  static_assert( !memory_traits::is_unmanaged,
342  "Cannot reserve unmanaged memory" );
343 
344  // If we aren't asking for more memory then we have nothing to do.
345  if ( n <= _capacity )
346  return;
347 
348  // Figure out the new capacity.
349  size_type num_soa_alloc = std::floor( n / vector_length );
350  if ( 0 < n % vector_length )
351  ++num_soa_alloc;
352 
353  // If we aren't asking for any more SoA objects then we still have
354  // nothing to do.
355  if ( num_soa_alloc <= _num_soa )
356  return;
357 
358  // Assign the new capacity.
359  _capacity = num_soa_alloc * vector_length;
360 
361  // If we need more SoA objects then resize.
362  Kokkos::resize( _data, num_soa_alloc );
363  }
