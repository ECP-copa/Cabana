73template <std::size_t M, class AoSoA_t>
74typename AoSoA_t::template member_slice_type<M>
75slice( const AoSoA_t& aosoa, const std::string& slice_label = "" )
76{
77 static_assert(
78 0 == sizeof( typename AoSoA_t::soa_type ) %
79 sizeof( typename AoSoA_t::template member_value_type<M> ),
80 "Slice stride cannot be calculated for misaligned memory!" );
81
82 return typename AoSoA_t::template member_slice_type<M>(
83 Impl::soaMemberPtr<M>( aosoa.data() ), aosoa.size(), aosoa.numSoA(),
84 slice_label );
85}

322 void resize( const size_type n )
323 {
324 static_assert( !memory_traits::is_unmanaged,
325 "Cannot resize unmanaged memory" );
326
327 // Reserve memory if needed.
328 reserve( n );
329
330 // Update the sizes of the data. This is potentially different than
331 // the amount of allocated data.
332 _size = n;
333 _num_soa = std::ceil( static_cast<double>( n ) / vector_length );
334 }
335
354 void reserve( const size_type n )
355 {
356 static_assert( !memory_traits::is_unmanaged,
357 "Cannot reserve unmanaged memory" );
358
359 // If we aren't asking for more memory then we have nothing to do.
360 if ( n <= _capacity )
361 return;
362
363 // Figure out the new capacity.
364 size_type num_soa_alloc = std::floor( n / vector_length );
365 if ( 0 < n % vector_length )
366 ++num_soa_alloc;
367
368 // If we aren't asking for any more SoA objects then we still have
369 // nothing to do.
370 if ( num_soa_alloc <= _num_soa )
371 return;
372
373 // Assign the new capacity.
374 _capacity = num_soa_alloc * vector_length;
375
376 // We need more SoA objects so allocate a new view and copy the
377 // existing data.
378 soa_view resized_data(
379 Kokkos::ViewAllocateWithoutInitializing( _data.label() ),
380 num_soa_alloc );
381 if ( _num_soa > 0 )
382 Kokkos::deep_copy(
383 Kokkos::subview(
384 resized_data,
385 Kokkos::pair<size_type, size_type>( 0, _num_soa ) ),
386 Kokkos::subview( _data, Kokkos::pair<size_type, size_type>(
387 0, _num_soa ) ) );
388 _data = resized_data;
389 }
390
399 void shrinkToFit()
400 {
401 static_assert( !memory_traits::is_unmanaged,
402 "Cannot shrink unmanaged memory" );
403
404 // If we aren't asking for any fewer SoA objects then we have nothing
405 // to do. The amount of allocated data has to be at least as big as
406 // _num_soa so we just need to check here that they are equivalent. If
407 // they are equivalent, the container is already as small as it can be.
408 if ( _data.size() == _num_soa )
409 return;
410
411 // Assign the new capacity.
412 _capacity = _num_soa * vector_length;
413
414 // We need fewer SoA objects so allocate a new view and copy the
415 // existing data.
416 soa_view resized_data(
417 Kokkos::ViewAllocateWithoutInitializing( _data.label() ),
418 _num_soa );
419 if ( _num_soa > 0 )
420 Kokkos::deep_copy(
421 resized_data,
422 Kokkos::subview( _data, Kokkos::pair<size_type, size_type>(
423 0, _num_soa ) ) );
424 _data = resized_data;
425 }

442 KOKKOS_INLINE_FUNCTION
443 size_type arraySize( const size_type s ) const
444 {
445 // the SoA struct size should be full size, i.e. vector_length if:
446 // 1) s is not the last SoA struct
447 // or 2) if _size = _num_soa * vector_length
448 return ( ( (size_type)s < _num_soa - 1 ) ||
449 ( _size % vector_length == 0 ) )
450 ? vector_length // if s is a full SoA struct
451 : ( _size % vector_length ); // if s is the last SoA struct
452 }
