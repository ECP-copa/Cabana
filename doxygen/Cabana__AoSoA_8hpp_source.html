310  void resize( const size_type n )
311  {
312  static_assert( !memory_traits::is_unmanaged,
313  "Cannot resize unmanaged memory" );
314 
315  // Reserve memory if needed.
316  reserve( n );
317 
318  // Update the sizes of the data. This is potentially different than
319  // the amount of allocated data.
320  _size = n;
321  _num_soa = std::floor( n / vector_length );
322  if ( 0 < n % vector_length )
323  ++_num_soa;
324  }
325 
344  void reserve( const size_type n )
345  {
346  static_assert( !memory_traits::is_unmanaged,
347  "Cannot reserve unmanaged memory" );
348 
349  // If we aren't asking for more memory then we have nothing to do.
350  if ( n <= _capacity )
351  return;
352 
353  // Figure out the new capacity.
354  size_type num_soa_alloc = std::floor( n / vector_length );
355  if ( 0 < n % vector_length )
356  ++num_soa_alloc;
357 
358  // If we aren't asking for any more SoA objects then we still have
359  // nothing to do.
360  if ( num_soa_alloc <= _num_soa )
361  return;
362 
363  // Assign the new capacity.
364  _capacity = num_soa_alloc * vector_length;
365 
366  // We need more SoA objects so allocate a new view and copy the
367  // existing data.
368  soa_view resized_data(
369  Kokkos::ViewAllocateWithoutInitializing( _data.label() ),
370  num_soa_alloc );
371  if ( _num_soa > 0 )
372  Kokkos::deep_copy(
373  Kokkos::subview(
374  resized_data,
375  Kokkos::pair<size_type, size_type>( 0, _num_soa ) ),
376  Kokkos::subview( _data, Kokkos::pair<size_type, size_type>(
377  0, _num_soa ) ) );
378  _data = resized_data;
379  }

389  void shrinkToFit()
390  {
391  static_assert( !memory_traits::is_unmanaged,
392  "Cannot shrink unmanaged memory" );
393 
394  // If we aren't asking for any fewer SoA objects then we have nothing
395  // to do. The amount of allocated data has to be at least as big as
396  // _num_soa so we just need to check here that they are equivalent. If
397  // they are equivalent, the container is already as small as it can be.
398  if ( _data.size() == _num_soa )
399  return;
400 
401  // Assign the new capacity.
402  _capacity = _num_soa * vector_length;
403 
404  // We need fewer SoA objects so allocate a new view and copy the
405  // existing data.
406  soa_view resized_data(
407  Kokkos::ViewAllocateWithoutInitializing( _data.label() ),
408  _num_soa );
409  if ( _num_soa > 0 )
410  Kokkos::deep_copy(
411  resized_data,
412  Kokkos::subview( _data, Kokkos::pair<size_type, size_type>(
413  0, _num_soa ) ) );
414  _data = resized_data;
415  }
