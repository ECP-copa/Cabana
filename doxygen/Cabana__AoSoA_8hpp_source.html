329  void resize( const size_type n )
330  {
331  static_assert( !memory_traits::is_unmanaged,
332  "Cannot resize unmanaged memory" );
333 
334  // Reserve memory if needed.
335  reserve( n );
336 
337  // Update the sizes of the data. This is potentially different than
338  // the amount of allocated data.
339  _size = n;
340  _num_soa = std::floor( n / vector_length );
341  if ( 0 < n % vector_length )
342  ++_num_soa;
343  }
344 
363  void reserve( const size_type n )
364  {
365  static_assert( !memory_traits::is_unmanaged,
366  "Cannot reserve unmanaged memory" );
367 
368  // If we aren't asking for more memory then we have nothing to do.
369  if ( n <= _capacity )
370  return;
371 
372  // Figure out the new capacity.
373  size_type num_soa_alloc = std::floor( n / vector_length );
374  if ( 0 < n % vector_length )
375  ++num_soa_alloc;
376 
377  // If we aren't asking for any more SoA objects then we still have
378  // nothing to do.
379  if ( num_soa_alloc <= _num_soa )
380  return;
381 
382  // Assign the new capacity.
383  _capacity = num_soa_alloc * vector_length;
384 
385  // We need more SoA objects so allocate a new view and copy the
386  // existing data.
387  soa_view resized_data(
388  Kokkos::ViewAllocateWithoutInitializing( _data.label() ),
389  num_soa_alloc );
390  if ( _num_soa > 0 )
391  Kokkos::deep_copy(
392  Kokkos::subview(
393  resized_data,
394  Kokkos::pair<size_type, size_type>( 0, _num_soa ) ),
395  _data );
396  _data = resized_data;
397  }

407  void shrinkToFit()
408  {
409  static_assert( !memory_traits::is_unmanaged,
410  "Cannot shrink unmanaged memory" );
411 
412  // If we aren't asking for any fewer SoA objects then we have nothing
413  // to do. The amount of allocated data has to be at least as big as
414  // _num_soa so we just need to check here that they are equivalent. If
415  // they are equivalent, the container is already as small as it can be.
416  if ( _data.size() == _num_soa )
417  return;
418 
419  // Assign the new capacity.
420  _capacity = _num_soa * vector_length;
421 
422  // We need fewer SoA objects so allocate a new view and copy the
423  // existing data.
424  soa_view resized_data(
425  Kokkos::ViewAllocateWithoutInitializing( _data.label() ),
426  _num_soa );
427  if ( _num_soa > 0 )
428  Kokkos::deep_copy(
429  resized_data,
430  Kokkos::subview( _data, Kokkos::pair<size_type, size_type>(
431  0, _num_soa ) ) );
432  _data = resized_data;
433  }
