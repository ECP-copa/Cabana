310  void resize( const size_type n )
311  {
312  static_assert( !memory_traits::Unmanaged,
313  "Cannot resize unmanaged memory" );
314 
315  // Reserve memory if needed.
316  reserve( n );
317 
318  // Update the sizes of the data. This is potentially different than
319  // the amount of allocated data.
320  _size = n;
321  _num_soa = std::floor( n / vector_length );
322  if ( 0 < n % vector_length ) ++_num_soa;
323  }

339  void reserve( const size_type n )
340  {
341  static_assert( !memory_traits::Unmanaged,
342  "Cannot reserve unmanaged memory" );
343 
344  // If we aren't asking for more memory then we have nothing to do.
345  if ( n <= _capacity ) return;
346 
347  // Figure out the new capacity.
348  size_type num_soa_alloc = std::floor( n / vector_length );
349  if ( 0 < n % vector_length ) ++num_soa_alloc;
350 
351  // If we aren't asking for any more SoA objects then we still have
352  // nothing to do.
353  if ( num_soa_alloc <= _num_soa ) return;
354 
355  // Assign the new capacity.
356  _capacity = num_soa_alloc * vector_length;
357 
358  // If we need more SoA objects then resize.
359  Kokkos::resize( _data, num_soa_alloc );
360  }
