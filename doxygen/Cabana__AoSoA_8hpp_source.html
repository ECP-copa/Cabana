307  void resize( const std::size_t n )
308  {
309  static_assert( !memory_traits::Unmanaged,
310  "Cannot resize unmanaged memory" );
311 
312  // Reserve memory if needed.
313  reserve( n );
314 
315  // Update the sizes of the data. This is potentially different than
316  // the amount of allocated data.
317  _size = n;
318  _num_soa = std::floor( n / vector_length );
319  if ( 0 < n % vector_length ) ++_num_soa;
320  }
321 
336  void reserve( const std::size_t n )
337  {
338  static_assert( !memory_traits::Unmanaged,
339  "Cannot reserve unmanaged memory" );
340 
341  // If we aren't asking for more memory then we have nothing to do.
342  if ( n <= _capacity ) return;
343 
344  // Figure out the new capacity.
345  std::size_t num_soa_alloc = std::floor( n / vector_length );
346  if ( 0 < n % vector_length ) ++num_soa_alloc;
347 
348  // If we aren't asking for any more SoA objects then we still have
349  // nothing to do.
350  if ( num_soa_alloc <= _num_soa ) return;
351 
352  // Assign the new capacity.
353  _capacity = num_soa_alloc * vector_length;
354 
355  // If we need more SoA objects then resize.
356  Kokkos::resize( _data, num_soa_alloc );
357  }
