226  void reserve( const std::size_t n )
227  {
228  // If we aren't asking for more memory then we have nothing to do.
229  if ( n <= _capacity ) return;
230 
231  // Figure out the new capacity.
232  std::size_t num_soa_alloc = std::floor( n / vector_length );
233  if ( 0 < n % vector_length ) ++num_soa_alloc;
234 
235  // If we aren't asking for any more SoA objects then we still have
236  // nothing to do.
237  if ( num_soa_alloc <= _num_soa ) return;
238 
239  // Assign the new capacity.
240  _capacity = num_soa_alloc * vector_length;
241 
242  // If we need more SoA objects then resize.
243  Kokkos::resize( _data, num_soa_alloc );
244 
245  // Get new pointers and strides for the members.
246  storePointersAndStrides(
247  std::integral_constant<std::size_t,number_of_members-1>() );
