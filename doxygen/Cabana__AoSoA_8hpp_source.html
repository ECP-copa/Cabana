308  void resize( const size_type n )
309  {
310  static_assert( !memory_traits::is_unmanaged,
311  "Cannot resize unmanaged memory" );
312 
313  // Reserve memory if needed.
314  reserve( n );
315 
316  // Update the sizes of the data. This is potentially different than
317  // the amount of allocated data.
318  _size = n;
319  _num_soa = std::floor( n / vector_length );
320  if ( 0 < n % vector_length )
321  ++_num_soa;
322  }

342  void reserve( const size_type n )
343  {
344  static_assert( !memory_traits::is_unmanaged,
345  "Cannot reserve unmanaged memory" );
346 
347  // If we aren't asking for more memory then we have nothing to do.
348  if ( n <= _capacity )
349  return;
350 
351  // Figure out the new capacity.
352  size_type num_soa_alloc = std::floor( n / vector_length );
353  if ( 0 < n % vector_length )
354  ++num_soa_alloc;
355 
356  // If we aren't asking for any more SoA objects then we still have
357  // nothing to do.
358  if ( num_soa_alloc <= _num_soa )
359  return;
360 
361  // Assign the new capacity.
362  _capacity = num_soa_alloc * vector_length;
363 
364  // We need more SoA objects so allocate a new view and copy the
365  // existing data.
366  soa_view resized_data(
367  Kokkos::ViewAllocateWithoutInitializing( _data.label() ),
368  num_soa_alloc );
369  if ( _num_soa > 0 )
370  Kokkos::deep_copy(
371  Kokkos::subview(
372  resized_data,
373  Kokkos::pair<size_type, size_type>( 0, _num_soa ) ),
374  _data );
375  _data = resized_data;
376  }

386  void shrinkToFit()
387  {
388  static_assert( !memory_traits::is_unmanaged,
389  "Cannot shrink unmanaged memory" );
390 
391  // If we aren't asking for any fewer SoA objects then we have nothing
392  // to do. The amount of allocated data has to be at least as big as
393  // _num_soa so we just need to check here that they are equivalent. If
394  // they are equivalent, the container is already as small as it can be.
395  if ( _data.size() == _num_soa )
396  return;
397 
398  // Assign the new capacity.
399  _capacity = _num_soa * vector_length;
400 
401  // We need fewer SoA objects so allocate a new view and copy the
402  // existing data.
403  soa_view resized_data(
404  Kokkos::ViewAllocateWithoutInitializing( _data.label() ),
405  _num_soa );
406  if ( _num_soa > 0 )
407  Kokkos::deep_copy(
408  resized_data,
409  Kokkos::subview( _data, Kokkos::pair<size_type, size_type>(
410  0, _num_soa ) ) );
411  _data = resized_data;
412  }
