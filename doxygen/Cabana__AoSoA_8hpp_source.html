187  void resize( const std::size_t n )
188  {
189  // Reserve memory if needed.
190  reserve( n );

192  // Update the sizes of the data. This is potentially different than
193  // the amount of allocated data.

213  void reserve( const std::size_t n )
214  {
215  // If we aren't asking for more memory then we have nothing to do.
216  if ( n <= _capacity ) return;
217 
218  // Figure out the new capacity.
219  std::size_t num_soa_alloc = std::floor( n / vector_length );
220  if ( 0 < n % vector_length ) ++num_soa_alloc;
221 
222  // If we aren't asking for any more SoA objects then we still have
223  // nothing to do.
224  if ( num_soa_alloc <= _num_soa ) return;
225 
226  // Assign the new capacity.
227  _capacity = num_soa_alloc * vector_length;
228 
229  // If we need more SoA objects then resize.
230  Kokkos::resize( _data, num_soa_alloc );
231  }
