64 if ( input_int )

82//---------------------------------------------------------------------------//
89template <typename Integer>
90KOKKOS_INLINE_FUNCTION constexpr Integer
91binaryReverse( Integer input_int, char loc = sizeof( Integer ) * 8 - 1 )
92{
93 if ( input_int == 0 )
94 return 0;
95 return ( ( input_int & 1 ) << loc ) |
96 binaryReverse( input_int >> 1, loc - 1 );
97}

99//---------------------------------------------------------------------------//
105template <typename Integer>
106KOKKOS_INLINE_FUNCTION constexpr unsigned countLeadingZeros( Integer input_int )
107{
108 unsigned res{ 0 };
109 input_int = binaryReverse( input_int );
110 if ( input_int == 0 )
111 return sizeof( Integer ) * 8;
112 while ( ( input_int & 1 ) == 0 )
113 res++, input_int >>= 1;
114 return res;
115}

117//---------------------------------------------------------------------------//
124KOKKOS_INLINE_FUNCTION
125constexpr int bitPack( const uint64_t mask, const uint64_t data )
126{
127 uint64_t slresult = 0;
128 uint64_t& ulresult{ slresult };
129 uint64_t uldata = data;
130 int count = 0;
131 ulresult = 0;
132
133 uint64_t rmask = binaryReverse( mask );
134 unsigned char lz{ 0 };
135
136 while ( rmask )
137 {
138 lz = countLeadingZeros( rmask );
139 uldata >>= lz;
140 ulresult <<= 1;
141 count++;
142 ulresult |= ( uldata & 1 );
143 uldata >>= 1;
144 rmask <<= lz + 1;
145 }
146 ulresult <<= 64 - count; // 64 bits (maybe not use a constant 64 ...?)
147 ulresult = binaryReverse( ulresult );
148 return (int)slresult;
149}
150
151//---------------------------------------------------------------------------//
158KOKKOS_INLINE_FUNCTION
159constexpr uint64_t bitSpread( const uint64_t mask, const int data )
160{
161 uint64_t rmask = binaryReverse( mask );
162 int dat = data;
163 uint64_t result = 0;
164 unsigned char lz{ 0 };
165 while ( rmask )
166 {
167 lz = countLeadingZeros( rmask ) + 1;
168 result = result << lz | ( dat & 1 );
169 dat >>= 1, rmask <<= lz;
170 }
171 result = binaryReverse( result ) >> countLeadingZeros( mask );
172 return result;
173}

1029 KOKKOS_INLINE_FUNCTION static constexpr auto fromCoord( Coords&&... coords )
1030 -> uint64_t
1031 {
1032 static_assert( sizeof...( Coords ) == rank,
1033 "Dimension of coordinate mismatch" );
