85 template <typename Integer>
86 KOKKOS_INLINE_FUNCTION constexpr Integer
87 binaryReverse( Integer input_int, char loc = sizeof( Integer ) * 8 - 1 )
88 {
89  if ( input_int == 0 )
90  return 0;
91  return ( ( input_int & 1 ) << loc ) |
92  binaryReverse( input_int >> 1, loc - 1 );
93 }

101 template <typename Integer>
102 KOKKOS_INLINE_FUNCTION constexpr unsigned countLeadingZeros( Integer input_int )
103 {
104  unsigned res{ 0 };
105  input_int = binaryReverse( input_int );
106  if ( input_int == 0 )
107  return sizeof( Integer ) * 8;
108  while ( ( input_int & 1 ) == 0 )
109  res++, input_int >>= 1;
110  return res;
111 }

113 //---------------------------------------------------------------------------//
120 KOKKOS_INLINE_FUNCTION
121 constexpr int bitPack( const uint64_t mask, const uint64_t data )
122 {
123  uint64_t slresult = 0;
124  uint64_t& ulresult{ slresult };
125  uint64_t uldata = data;
126  int count = 0;
127  ulresult = 0;
128 
129  uint64_t rmask = binaryReverse( mask );
130  unsigned char lz{ 0 };
131 
132  while ( rmask )
133  {
134  lz = countLeadingZeros( rmask );
135  uldata >>= lz;
136  ulresult <<= 1;
137  count++;
138  ulresult |= ( uldata & 1 );
139  uldata >>= 1;
140  rmask <<= lz + 1;
141  }
142  ulresult <<= 64 - count; // 64 bits (maybe not use a constant 64 ...?)
143  ulresult = binaryReverse( ulresult );
144  return (int)slresult;
145 }

147 //---------------------------------------------------------------------------//
154 KOKKOS_INLINE_FUNCTION
155 constexpr uint64_t bitSpread( const uint64_t mask, const int data )
156 {
157  uint64_t rmask = binaryReverse( mask );
158  int dat = data;
159  uint64_t result = 0;
160  unsigned char lz{ 0 };
161  while ( rmask )
162  {
163  lz = countLeadingZeros( rmask ) + 1;
164  result = result << lz | ( dat & 1 );
165  dat >>= 1, rmask <<= lz;
166  }
167  result = binaryReverse( result ) >> countLeadingZeros( mask );
168  return result;
169 }
