84 template <typename Integer>
85 KOKKOS_INLINE_FUNCTION constexpr Integer
86 binaryReverse( Integer input_int, char loc = sizeof( Integer ) * 8 - 1 )
87 {
88  if ( input_int == 0 )
89  return 0;
90  return ( ( input_int & 1 ) << loc ) |
91  binaryReverse( input_int >> 1, loc - 1 );
92 }

100 template <typename Integer>
101 KOKKOS_INLINE_FUNCTION constexpr unsigned countLeadingZeros( Integer input_int )
102 {
103  unsigned res{ 0 };
104  input_int = binaryReverse( input_int );
105  if ( input_int == 0 )
106  return sizeof( Integer ) * 8;
107  while ( ( input_int & 1 ) == 0 )
108  res++, input_int >>= 1;
109  return res;
110 }

112 //---------------------------------------------------------------------------//
119 KOKKOS_INLINE_FUNCTION
120 constexpr int bitPack( const uint64_t mask, const uint64_t data )
121 {
122  uint64_t slresult = 0;
123  uint64_t& ulresult{ slresult };
124  uint64_t uldata = data;
125  int count = 0;
126  ulresult = 0;
127 
128  uint64_t rmask = binaryReverse( mask );
129  unsigned char lz{ 0 };
130 
131  while ( rmask )
132  {
133  lz = countLeadingZeros( rmask );
134  uldata >>= lz;
135  ulresult <<= 1;
136  count++;
137  ulresult |= ( uldata & 1 );
138  uldata >>= 1;
139  rmask <<= lz + 1;
140  }
141  ulresult <<= 64 - count; // 64 bits (maybe not use a constant 64 ...?)
142  ulresult = binaryReverse( ulresult );
143  return (int)slresult;
144 }

146 //---------------------------------------------------------------------------//
153 KOKKOS_INLINE_FUNCTION
154 constexpr uint64_t bitSpread( const uint64_t mask, const int data )
155 {
156  uint64_t rmask = binaryReverse( mask );
157  int dat = data;
158  uint64_t result = 0;
159  unsigned char lz{ 0 };
160  while ( rmask )
161  {
162  lz = countLeadingZeros( rmask ) + 1;
163  result = result << lz | ( dat & 1 );
164  dat >>= 1, rmask <<= lz;
165  }
166  result = binaryReverse( result ) >> countLeadingZeros( mask );
167  return result;
168 }
