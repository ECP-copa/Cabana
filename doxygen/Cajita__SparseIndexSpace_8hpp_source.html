77 template <typename Integer>
78 KOKKOS_INLINE_FUNCTION constexpr Integer
79 binaryReverse( Integer input_int, char loc = sizeof( Integer ) * 8 - 1 )
80 {
81  if ( input_int == 0 )
82  return 0;
83  return ( ( input_int & 1 ) << loc ) |
84  binaryReverse( input_int >> 1, loc - 1 );
85 }

87 //---------------------------------------------------------------------------//
92 template <typename Integer>
93 KOKKOS_INLINE_FUNCTION constexpr unsigned countLeadingZeros( Integer input_int )
94 {
95  unsigned res{ 0 };
96  input_int = binaryReverse( input_int );
97  if ( input_int == 0 )
98  return sizeof( Integer ) * 8;
99  while ( ( input_int & 1 ) == 0 )
100  res++, input_int >>= 1;
101  return res;
102 }

104 //---------------------------------------------------------------------------//
109 KOKKOS_INLINE_FUNCTION
110 constexpr int bitPack( const uint64_t mask, const uint64_t data )
111 {
112  uint64_t slresult = 0;
113  uint64_t& ulresult{ slresult };
114  uint64_t uldata = data;
115  int count = 0;
116  ulresult = 0;
117 
118  uint64_t rmask = binaryReverse( mask );
119  unsigned char lz{ 0 };
120 
121  while ( rmask )
122  {
123  lz = countLeadingZeros( rmask );
124  uldata >>= lz;
125  ulresult <<= 1;
126  count++;
127  ulresult |= ( uldata & 1 );
128  uldata >>= 1;
129  rmask <<= lz + 1;
130  }
131  ulresult <<= 64 - count; // 64 bits (maybe not use a constant 64 ...?)
132  ulresult = binaryReverse( ulresult );
133  return (int)slresult;
134 }

136 //---------------------------------------------------------------------------//
141 KOKKOS_INLINE_FUNCTION
142 constexpr uint64_t bitSpread( const uint64_t mask, const int data )
143 {
144  uint64_t rmask = binaryReverse( mask );
145  int dat = data;
146  uint64_t result = 0;
147  unsigned char lz{ 0 };
148  while ( rmask )
149  {
150  lz = countLeadingZeros( rmask ) + 1;
151  result = result << lz | ( dat & 1 );
152  dat >>= 1, rmask <<= lz;
153  }
154  result = binaryReverse( result ) >> countLeadingZeros( mask );
155  return result;
156 }

393  1 << cell_bits_per_tile_dim;

696  KOKKOS_INLINE_FUNCTION
697  void insert( int tile_i, int tile_j, int tile_k ) const
698  {
699  // Note: reallocation will be performed after insert
700  // tile ijk => tile key
701  key_type tile_key = ijk2key( tile_i, tile_j, tile_k );
702  // Try to insert the tile key into the map. Give it a dummy value for
703  // now.

705  // If the tile key was actually inserted, atomically increment the

707  // it

709  _tile_table.value_at( insert_result.index() ) =
710  Kokkos::atomic_fetch_add( &( _tile_table_info( 0 ) ), 1 );
711  }
