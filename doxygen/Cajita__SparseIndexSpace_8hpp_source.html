76 template <typename Integer>
77 KOKKOS_INLINE_FUNCTION constexpr Integer
78 binaryReverse( Integer input_int, char loc = sizeof( Integer ) * 8 - 1 )
79 {
80  if ( input_int == 0 )
81  return 0;
82  return ( ( input_int & 1 ) << loc ) |
83  binaryReverse( input_int >> 1, loc - 1 );
84 }

86 //---------------------------------------------------------------------------//
91 template <typename Integer>
92 KOKKOS_INLINE_FUNCTION constexpr unsigned countLeadingZeros( Integer input_int )
93 {
94  unsigned res{ 0 };
95  input_int = binaryReverse( input_int );
96  if ( input_int == 0 )
97  return sizeof( Integer ) * 8;
98  while ( ( input_int & 1 ) == 0 )
99  res++, input_int >>= 1;
100  return res;
101 }

103 //---------------------------------------------------------------------------//
108 KOKKOS_INLINE_FUNCTION
109 constexpr int bitPack( const uint64_t mask, const uint64_t data )
110 {
111  uint64_t slresult = 0;
112  uint64_t& ulresult{ slresult };
113  uint64_t uldata = data;
114  int count = 0;
115  ulresult = 0;
116 
117  uint64_t rmask = binaryReverse( mask );
118  unsigned char lz{ 0 };
119 
120  while ( rmask )
121  {
122  lz = countLeadingZeros( rmask );
123  uldata >>= lz;
124  ulresult <<= 1;
125  count++;
126  ulresult |= ( uldata & 1 );
127  uldata >>= 1;
128  rmask <<= lz + 1;
129  }
130  ulresult <<= 64 - count; // 64 bits (maybe not use a constant 64 ...?)
131  ulresult = binaryReverse( ulresult );
132  return (int)slresult;
133 }

135 //---------------------------------------------------------------------------//
140 KOKKOS_INLINE_FUNCTION
141 constexpr uint64_t bitSpread( const uint64_t mask, const int data )
142 {
143  uint64_t rmask = binaryReverse( mask );
144  int dat = data;
145  uint64_t result = 0;
146  unsigned char lz{ 0 };
147  while ( rmask )
148  {
149  lz = countLeadingZeros( rmask ) + 1;
150  result = result << lz | ( dat & 1 );
151  dat >>= 1, rmask <<= lz;
152  }
153  result = binaryReverse( result ) >> countLeadingZeros( mask );
154  return result;
155 }

719  KOKKOS_INLINE_FUNCTION
720  void insert( int tile_i, int tile_j, int tile_k ) const
721  {
722  // Note: reallocation will be performed after insert
723  // tile ijk => tile key
724  key_type tile_key = ijk2key( tile_i, tile_j, tile_k );
725  // Try to insert the tile key into the map. Give it a dummy value for
726  // now.
727  auto insert_result = _tile_table.insert( tile_key, 0 );
728  // If the tile key was actually inserted, atomically increment the
729  // counter Only threads that actually did a successful insert will call
730  // it
731  if ( !insert_result.existing() )
732  _tile_table.value_at( insert_result.index() ) =
733  Kokkos::atomic_fetch_add( &( _tile_table_info( 0 ) ), 1 );
734  }
