190 template <class FunctorType, class ExecutionSpace, std::size_t NumSpace>
191 inline void
192 grid_parallel_for( const std::string& label, const ExecutionSpace& exec_space,
193  const Kokkos::Array<IndexSpace<4>, NumSpace>& index_spaces,
194  const FunctorType& functor )
195 {
196  // Compute the total number of threads needed and the index space offsets
197  // via inclusive scan.
198  int size = index_spaces[0].size();
199  Kokkos::Array<int, NumSpace> exclusive_offsets;
200  Kokkos::Array<int, NumSpace> inclusive_offsets;
201  exclusive_offsets[0] = 0;
202  inclusive_offsets[0] = size;
203  for ( std::size_t sp = 1; sp < NumSpace; ++sp )
204  {
205  size += index_spaces[sp].size();
206  exclusive_offsets[sp] =
207  exclusive_offsets[sp - 1] + index_spaces[sp - 1].size();
208  inclusive_offsets[sp] =
209  inclusive_offsets[sp - 1] + index_spaces[sp].size();
210  }
211 
212  // Unroll the index spaces into a linear parallel for.
213  Kokkos::parallel_for(
214  label, Kokkos::RangePolicy<ExecutionSpace>( exec_space, 0, size ),
215  KOKKOS_LAMBDA( const int n ) {
216  // Get the index space id.
217  int s = -1;
218  for ( std::size_t sp = 0; sp < NumSpace; ++sp )
219  {
220  if ( n < inclusive_offsets[sp] )
221  {
222  s = sp;
223  break;
224  }
225  }
226 
227  // Linear id in index space.
228  int linear_id = n - exclusive_offsets[s];
229 
230  // Compute entity id.
231  int extent_l = index_spaces[s].extent( 3 );
232  int extent_kl = extent_l * index_spaces[s].extent( Dim::K );
233  int extent_jkl = extent_kl * index_spaces[s].extent( Dim::J );
234  int i_base = linear_id / extent_jkl;
235  int stride_j = extent_jkl * i_base;
236  int j_base = ( linear_id - stride_j ) / extent_kl;
237  int stride_k = ( stride_j + extent_kl * j_base );
238  int k_base = ( linear_id - stride_k ) / extent_l;
239  int l_base = linear_id - stride_k - k_base * extent_l;
240 
241  // Execute the user functor. Provide the index space id so they
242  // can discern which space they are in within the functor.
243  functor( s, i_base + index_spaces[s].min( Dim::I ),
244  j_base + index_spaces[s].min( Dim::J ),
245  k_base + index_spaces[s].min( Dim::K ),
246  l_base + index_spaces[s].min( 3 ) );
247  } );
248 }
