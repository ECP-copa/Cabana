186 template <class FunctorType, class ExecutionSpace, std::size_t NumSpace>
187 inline void
188 grid_parallel_for( const std::string& label, const ExecutionSpace& exec_space,
189  const Kokkos::Array<IndexSpace<4>, NumSpace>& index_spaces,
190  const FunctorType& functor )
191 {
192  // Compute the total number of threads needed and the index space offsets
193  // via inclusive scan.
194  int size = index_spaces[0].size();
195  Kokkos::Array<int, NumSpace> exclusive_offsets;
196  Kokkos::Array<int, NumSpace> inclusive_offsets;
197  exclusive_offsets[0] = 0;
198  inclusive_offsets[0] = size;
199  for ( std::size_t sp = 1; sp < NumSpace; ++sp )
200  {
201  size += index_spaces[sp].size();
202  exclusive_offsets[sp] =
203  exclusive_offsets[sp - 1] + index_spaces[sp - 1].size();
204  inclusive_offsets[sp] =
205  inclusive_offsets[sp - 1] + index_spaces[sp].size();
206  }
207 
208  // Unroll the index spaces into a linear parallel for.
209  Kokkos::parallel_for(
210  label, Kokkos::RangePolicy<ExecutionSpace>( exec_space, 0, size ),
211  KOKKOS_LAMBDA( const int n ) {
212  // Get the index space id.
213  int s = -1;
214  for ( std::size_t sp = 0; sp < NumSpace; ++sp )
215  {
216  if ( n < inclusive_offsets[sp] )
217  {
218  s = sp;
219  break;
220  }
221  }
222 
223  // Linear id in index space.
224  int linear_id = n - exclusive_offsets[s];
225 
226  // Compute entity id.
227  int extent_l = index_spaces[s].extent( 3 );
228  int extent_kl = extent_l * index_spaces[s].extent( Dim::K );
229  int extent_jkl = extent_kl * index_spaces[s].extent( Dim::J );
230  int i_base = linear_id / extent_jkl;
231  int stride_j = extent_jkl * i_base;
232  int j_base = ( linear_id - stride_j ) / extent_kl;
233  int stride_k = ( stride_j + extent_kl * j_base );
234  int k_base = ( linear_id - stride_k ) / extent_l;
235  int l_base = linear_id - stride_k - k_base * extent_l;
236 
237  // Execute the user functor. Provide the index space id so they
238  // can discern which space they are in within the functor.
239  functor( s, i_base + index_spaces[s].min( Dim::I ),
240  j_base + index_spaces[s].min( Dim::J ),
241  k_base + index_spaces[s].min( Dim::K ),
242  l_base + index_spaces[s].min( 3 ) );
243  } );
244 }
