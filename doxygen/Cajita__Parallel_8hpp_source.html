194template <class FunctorType, class ExecutionSpace, std::size_t NumSpace>
195inline void
196grid_parallel_for( const std::string& label, const ExecutionSpace& exec_space,
197 const Kokkos::Array<IndexSpace<4>, NumSpace>& index_spaces,
198 const FunctorType& functor )
199{
200 Kokkos::Profiling::pushRegion( "Cajita::grid_parallel_for" );
201
202 // Compute the total number of threads needed and the index space offsets
203 // via inclusive scan.
204 int size = index_spaces[0].size();
205 Kokkos::Array<int, NumSpace> exclusive_offsets;
206 Kokkos::Array<int, NumSpace> inclusive_offsets;
207 exclusive_offsets[0] = 0;
208 inclusive_offsets[0] = size;
209 for ( std::size_t sp = 1; sp < NumSpace; ++sp )
210 {
211 size += index_spaces[sp].size();
212 exclusive_offsets[sp] =
213 exclusive_offsets[sp - 1] + index_spaces[sp - 1].size();
214 inclusive_offsets[sp] =
215 inclusive_offsets[sp - 1] + index_spaces[sp].size();
216 }
217
218 // Unroll the index spaces into a linear parallel for.
219 Kokkos::parallel_for(
220 label, Kokkos::RangePolicy<ExecutionSpace>( exec_space, 0, size ),
221 KOKKOS_LAMBDA( const int n ) {
222 // Get the index space id.
223 int s = -1;
224 for ( std::size_t sp = 0; sp < NumSpace; ++sp )
225 {
226 if ( n < inclusive_offsets[sp] )
227 {
228 s = sp;
229 break;
230 }
231 }
232
233 // Linear id in index space.
234 int linear_id = n - exclusive_offsets[s];
235
236 // Compute entity id.
237 int extent_l = index_spaces[s].extent( 3 );
238 int extent_kl = extent_l * index_spaces[s].extent( Dim::K );
239 int extent_jkl = extent_kl * index_spaces[s].extent( Dim::J );
240 int i_base = linear_id / extent_jkl;
241 int stride_j = extent_jkl * i_base;
242 int j_base = ( linear_id - stride_j ) / extent_kl;
243 int stride_k = ( stride_j + extent_kl * j_base );
244 int k_base = ( linear_id - stride_k ) / extent_l;
245 int l_base = linear_id - stride_k - k_base * extent_l;
246
247 // Execute the user functor. Provide the index space id so they
248 // can discern which space they are in within the functor.
249 functor( s, i_base + index_spaces[s].min( Dim::I ),
250 j_base + index_spaces[s].min( Dim::J ),
251 k_base + index_spaces[s].min( Dim::K ),
252 l_base + index_spaces[s].min( 3 ) );
253 } );
254
255 Kokkos::Profiling::popRegion();
256}
