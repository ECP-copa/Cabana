144 template <class ExecSpace, class PositionSliceType>
145 void registerSparseMap( PositionSliceType& positions,
146 const int particle_num, const int p2g_radius = 1 )
147 {
148 // get references
149 Kokkos::Array<scalar_type, 3> dx_inv = {
150 (scalar_type)1.0 / _cell_size[0], (scalar_type)1.0 / _cell_size[1],
151 (scalar_type)1.0 / _cell_size[2] };
152 auto& map = _map;
153 auto& low_corner = _global_low_corner;
154 // register sparse map in sparse array layout
155 Kokkos::parallel_for(
156 "register sparse map in sparse array layout",
157 Kokkos::RangePolicy<ExecSpace>( 0, particle_num ),
158 KOKKOS_LAMBDA( const int pid ) {
159 scalar_type pos[3] = { positions( pid, 0 ) - low_corner[0],
160 positions( pid, 1 ) - low_corner[1],
161 positions( pid, 2 ) - low_corner[2] };
162 int grid_base[3] = {
163 static_cast<int>( std::lround( pos[0] * dx_inv[0] ) -
164 p2g_radius ),
165 static_cast<int>( std::lround( pos[1] * dx_inv[1] ) -
166 p2g_radius ),
167 static_cast<int>( std::lround( pos[2] * dx_inv[2] ) -
168 p2g_radius ) };
169 // register grids that will have data transfer with the particle
170 const int p2g_size = p2g_radius * 2;
171 for ( int i = 0; i <= p2g_size; ++i )
172 for ( int j = 0; j <= p2g_size; ++j )
173 for ( int k = 0; k <= p2g_size; ++k )
174 {
175 int cell_id[3] = { grid_base[0] + i,
176 grid_base[1] + j,
177 grid_base[2] + k };
178 map.insertCell( cell_id[0], cell_id[1],
179 cell_id[2] );
180 }
181 } );
182 }
