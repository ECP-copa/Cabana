145 template <class ExecSpace, class PositionSliceType>
146 void registerSparseMap( PositionSliceType& positions,
147 const int particle_num, const int p2g_radius = 1 )
148 {
149 // get references
150 Kokkos::Array<scalar_type, 3> dx_inv = {
151 (scalar_type)1.0 / _cell_size[0], (scalar_type)1.0 / _cell_size[1],
152 (scalar_type)1.0 / _cell_size[2] };
153 auto& map = _map;
154 auto& low_corner = _global_low_corner;
155 // register sparse map in sparse array layout
156 Kokkos::parallel_for(
157 "register sparse map in sparse array layout",
158 Kokkos::RangePolicy<ExecSpace>( 0, particle_num ),
159 KOKKOS_LAMBDA( const int pid ) {
160 scalar_type pos[3] = { positions( pid, 0 ) - low_corner[0],
161 positions( pid, 1 ) - low_corner[1],
162 positions( pid, 2 ) - low_corner[2] };
163 int grid_base[3] = {
164 static_cast<int>( std::lround( pos[0] * dx_inv[0] ) -
165 p2g_radius ),
166 static_cast<int>( std::lround( pos[1] * dx_inv[1] ) -
167 p2g_radius ),
168 static_cast<int>( std::lround( pos[2] * dx_inv[2] ) -
169 p2g_radius ) };
170 // register grids that will have data transfer with the particle
171 const int p2g_size = p2g_radius * 2;
172 for ( int i = 0; i <= p2g_size; ++i )
173 for ( int j = 0; j <= p2g_size; ++j )
174 for ( int k = 0; k <= p2g_size; ++k )
175 {
176 int cell_id[3] = { grid_base[0] + i,
177 grid_base[1] + j,
178 grid_base[2] + k };
179 map.insertCell( cell_id[0], cell_id[1],
180 cell_id[2] );
181 }
182 } );
183 }
