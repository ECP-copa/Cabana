137 template <class ExecSpace, class PositionSliceType>
138 void registerSparseMap( PositionSliceType& positions,
139 const int particle_num, const int p2g_radius = 1 )
140 {
141 // get references
142 Kokkos::Array<scalar_type, 3> dx_inv = {
143 (scalar_type)1.0 / _cell_size[0], (scalar_type)1.0 / _cell_size[1],
144 (scalar_type)1.0 / _cell_size[2] };
145 auto& map = _map;
146 auto& low_corner = _global_low_corner;
147 // register sparse map in sparse array layout
148 Kokkos::parallel_for(
149 "register sparse map in sparse array layout",
150 Kokkos::RangePolicy<ExecSpace>( 0, particle_num ),
151 KOKKOS_LAMBDA( const int pid ) {
152 scalar_type pos[3] = { positions( pid, 0 ) - low_corner[0],
153 positions( pid, 1 ) - low_corner[1],
154 positions( pid, 2 ) - low_corner[2] };
155 int grid_base[3] = {
156 static_cast<int>( std::lround( pos[0] * dx_inv[0] ) -
157 p2g_radius ),
158 static_cast<int>( std::lround( pos[1] * dx_inv[1] ) -
159 p2g_radius ),
160 static_cast<int>( std::lround( pos[2] * dx_inv[2] ) -
161 p2g_radius ) };
162 // register grids that will have data transfer with the particle
163 const int p2g_size = p2g_radius * 2;
164 for ( int i = 0; i <= p2g_size; ++i )
165 for ( int j = 0; j <= p2g_size; ++j )
166 for ( int k = 0; k <= p2g_size; ++k )
167 {
168 int cell_id[3] = { grid_base[0] + i,
169 grid_base[1] + j,
170 grid_base[2] + k };
171 map.insertCell( cell_id[0], cell_id[1],
172 cell_id[2] );
173 }
174 } );
175 }
