54  LocalMesh( const LocalGrid<UniformMesh<Scalar, num_space_dim>>& local_grid )
55  {
56  const auto& global_grid = local_grid.globalGrid();
57  const auto& global_mesh = global_grid.globalMesh();
58 
59  // Get the cell size.
60  for ( std::size_t d = 0; d < num_space_dim; ++d )
61  _cell_size[d] = global_mesh.cellSize( d );
62 
63  // Compute face area.
64  if ( 3 == num_space_dim )
65  {
66  _face_area[Dim::I] = _cell_size[Dim::J] * _cell_size[Dim::K];
67  _face_area[Dim::J] = _cell_size[Dim::I] * _cell_size[Dim::K];
68  _face_area[Dim::K] = _cell_size[Dim::I] * _cell_size[Dim::J];
69  }
70  else if ( 2 == num_space_dim )
71  {
72  _face_area[Dim::I] = _cell_size[Dim::J];
73  _face_area[Dim::J] = _cell_size[Dim::I];
74  }
75 
76  // Compute cell volume.
77  _cell_volume = 1.0;
78  for ( std::size_t d = 0; d < num_space_dim; ++d )
79  _cell_volume *= _cell_size[d];
80 
81  // Compute the owned low corner
82  for ( std::size_t d = 0; d < num_space_dim; ++d )
83  _own_low_corner[d] = global_mesh.lowCorner( d ) +
84  _cell_size[d] * global_grid.globalOffset( d );
85 
86  // Compute the owned high corner
87  for ( std::size_t d = 0; d < num_space_dim; ++d )
88  _own_high_corner[d] =
89  global_mesh.lowCorner( d ) +
90  _cell_size[d] * ( global_grid.globalOffset( d ) +
91  global_grid.ownedNumCell( d ) );
92 
93  // Compute the ghosted low corner
94  for ( std::size_t d = 0; d < num_space_dim; ++d )
95  _ghost_low_corner[d] = lowCorner( Own(), d ) -
96  local_grid.haloCellWidth() * _cell_size[d];
97 
98  // Compute the ghosted high corner
99  for ( std::size_t d = 0; d < num_space_dim; ++d )
100  _ghost_high_corner[d] = highCorner( Own(), d ) +
101  local_grid.haloCellWidth() * _cell_size[d];
102 
103  // Periodicity
104  for ( std::size_t d = 0; d < num_space_dim; ++d )
105  _periodic[d] = global_grid.isPeriodic( d );
106 
107  // Determine if a block is on the low or high boundaries.
108  for ( std::size_t d = 0; d < num_space_dim; ++d )
109  {
110  _boundary_lo[d] = global_grid.onLowBoundary( d );
111  _boundary_hi[d] = global_grid.onHighBoundary( d );
112  }
113  }

284 template <class Scalar, class Device, std::size_t NumSpaceDim>
285 class LocalMesh<Device, NonUniformMesh<Scalar, NumSpaceDim>>
286 {
287  public:
289  using mesh_type = NonUniformMesh<Scalar, NumSpaceDim>;
290 
292  using scalar_type = Scalar;
293 
295  static constexpr std::size_t num_space_dim = NumSpaceDim;
296 
298  using device_type = Device;
300  using memory_space = typename Device::memory_space;
302  using execution_space = typename Device::execution_space;
303 
305  LocalMesh(
306  const LocalGrid<NonUniformMesh<Scalar, num_space_dim>>& local_grid )
307  {
308  const auto& global_grid = local_grid.globalGrid();
309  const auto& global_mesh = global_grid.globalMesh();
310 
311  // Compute the owned low corner.
312  for ( std::size_t d = 0; d < num_space_dim; ++d )
313  _own_low_corner[d] =
314  global_mesh.nonUniformEdge( d )[global_grid.globalOffset( d )];
315 
316  // Compute the owned high corner
317  for ( std::size_t d = 0; d < num_space_dim; ++d )
318  _own_high_corner[d] =
319  global_mesh.nonUniformEdge( d )[global_grid.globalOffset( d ) +
320  global_grid.ownedNumCell( d )];
321 
322  // Compute the ghosted low corner
323  for ( std::size_t d = 0; d < num_space_dim; ++d )
324  {
325  // Interior domain case.
326  if ( !global_grid.onLowBoundary( d ) )
327  {
328  _ghost_low_corner[d] = global_mesh.nonUniformEdge(
329  d )[global_grid.globalOffset( d ) -
330  local_grid.haloCellWidth()];
331  }
332 
333  // Periodic boundary. Use cells on other side of boundary to
334  // generate geometry.
335  else if ( global_grid.isPeriodic( d ) )
336  {
337  int nedge = global_mesh.nonUniformEdge( d ).size();
338  _ghost_low_corner[d] =
339  global_mesh.nonUniformEdge( d ).front() -
340  ( global_mesh.nonUniformEdge( d ).back() -
341  global_mesh.nonUniformEdge(
342  d )[nedge - local_grid.haloCellWidth() - 1] );
343  }
344 
345  // In the non-periodic boundary case we extrapolate halo cells to
346  // have the same width as the boundary cell.
347  else
348  {
349  Scalar dx = global_mesh.nonUniformEdge( d )[1] -
350  global_mesh.nonUniformEdge( d )[0];
351  _ghost_low_corner[d] = global_mesh.nonUniformEdge( d ).front() -
352  dx * local_grid.haloCellWidth();
353  }
354  }
355 
356  // Compute the ghosted high corner
357  for ( std::size_t d = 0; d < num_space_dim; ++d )
358  {
359  // Interior domain case.
360  if ( !global_grid.onHighBoundary( d ) )
361  {
362  _ghost_high_corner[d] = global_mesh.nonUniformEdge(
363  d )[global_grid.globalOffset( d ) +
364  global_grid.ownedNumCell( d ) +
365  local_grid.haloCellWidth()];
366  }
367 
368  // Periodic boundary. Use cells on other side of boundary to
369  // generate geometry.
370  else if ( global_grid.isPeriodic( d ) )
371  {
372  _ghost_high_corner[d] =
373  global_mesh.nonUniformEdge( d ).back() +
374  ( global_mesh.nonUniformEdge(
375  d )[local_grid.haloCellWidth()] -
376  global_mesh.nonUniformEdge( d ).front() );
377  }
378 
379  // In the non-periodic boundary case we extrapolate halo cells to
380  // have the same width as the boundary cell.
381  else
382  {
383  int nedge = global_mesh.nonUniformEdge( d ).size();
384  Scalar dx = global_mesh.nonUniformEdge( d )[nedge - 1] -
385  global_mesh.nonUniformEdge( d )[nedge - 2];
386  _ghost_high_corner[d] = global_mesh.nonUniformEdge( d ).back() +
387  dx * local_grid.haloCellWidth();
388  }
389  }
390 
391  // Get the node index spaces.
392  auto owned_nodes_local =
393  local_grid.indexSpace( Own(), Node(), Local() );
394  auto ghosted_nodes_local =
395  local_grid.indexSpace( Ghost(), Node(), Local() );
396  auto owned_nodes_global =
397  local_grid.indexSpace( Own(), Node(), Global() );
398  for ( std::size_t d = 0; d < num_space_dim; ++d )
399  {
400  // Allocate edges on the device for this dimension.
401  const auto& global_edge = global_mesh.nonUniformEdge( d );
402  int nedge = ghosted_nodes_local.extent( d );
403  int nedge_global = global_edge.size();
404  _local_edges[d] = Kokkos::View<Scalar*, Device>(
405  Kokkos::ViewAllocateWithoutInitializing( "local_edges" ),
406  nedge );
407 
408  // Compute edges on the host.
409  auto edge_mirror = Kokkos::create_mirror_view( Kokkos::HostSpace(),
410  _local_edges[d] );
411 
412  // Compute the owned edges.
413  for ( int n = owned_nodes_local.min( d );
414  n < owned_nodes_local.max( d ); ++n )
415  {
416  edge_mirror( n ) = global_edge[owned_nodes_global.min( d ) + n -
417  owned_nodes_local.min( d )];
418  }
419 
420  // Compute the lower boundary edges.
421  if ( !global_grid.onLowBoundary( d ) )
422  {
423  // Interior block gets edges from neighbors.
424  for ( int n = 0; n < owned_nodes_local.min( d ); ++n )
425  {
426  edge_mirror( n ) =
427  global_edge[owned_nodes_global.min( d ) + n -
428  owned_nodes_local.min( d )];
429  }
430  }
431  else if ( global_grid.isPeriodic( d ) )
432  {
433  // Periodic boundary block gets edges from neighbor on
434  // opposite side of boundary.
435  for ( int n = 0; n < owned_nodes_local.min( d ); ++n )
436  {
437  edge_mirror( n ) =
438  global_edge.front() - global_edge.back() +
439  global_edge[global_edge.size() - 1 -
440  local_grid.haloCellWidth() + n];
441  }
442  }
443  else
444  {
445  // Non-periodic boundary block extrapolates edges using
446  // boundary cell width.
447  for ( int n = 0; n < owned_nodes_local.min( d ); ++n )
448  {
449  Scalar dx = global_edge[1] - global_edge[0];
450  edge_mirror( n ) = global_edge.front() -
451  ( owned_nodes_local.min( d ) - n ) * dx;
452  }
453  }
454 
455  // Compute the upper boundary edges.
456  if ( !global_grid.onHighBoundary( d ) )
457  {
458  // Interior block gets edges from neighbors.
459  for ( int n = owned_nodes_local.max( d );
460  n < ghosted_nodes_local.max( d ); ++n )
461  {
462  edge_mirror( n ) =
463  global_edge[owned_nodes_global.min( d ) + n -
464  owned_nodes_local.min( d )];
465  }
466  }
467  else if ( global_grid.isPeriodic( d ) )
468  {
469  // Periodic boundary block gets edges from neighbor on
470  // opposite side of boundary.
471  for ( int n = 0; n < ghosted_nodes_local.max( d ) -
472  owned_nodes_local.max( d );
473  ++n )
474  {
475  edge_mirror( owned_nodes_local.max( d ) + n ) =
476  global_edge.back() + global_edge[n] -
477  global_edge.front();
478  }
479  }
480  else
481  {
482  // Non-periodic boundary block extrapolates edges using
483  // boundary cell width.
484  for ( int n = owned_nodes_local.max( d );
485  n < ghosted_nodes_local.max( d ); ++n )
486  {
487  Scalar dx = global_edge[nedge_global - 1] -
488  global_edge[nedge_global - 2];
489  edge_mirror( n ) =
490  global_edge.back() +
491  ( n - owned_nodes_local.max( d ) + 1 ) * dx;
492  }
493  }
494 
495  // Copy edges to the device.
496  Kokkos::deep_copy( _local_edges[d], edge_mirror );
497  }
498 
499  // Periodicity
500  for ( std::size_t d = 0; d < num_space_dim; ++d )
501  _periodic[d] = global_grid.isPeriodic( d );
502 
503  // Determine if a block is on the low or high boundaries.
504  for ( std::size_t d = 0; d < num_space_dim; ++d )
505  {
506  _boundary_lo[d] = global_grid.onLowBoundary( d );
507  _boundary_hi[d] = global_grid.onHighBoundary( d );
508  }
509  }
