57  LocalMesh( const LocalGrid<UniformMesh<Scalar, num_space_dim>>& local_grid )
58  {
59  const auto& global_grid = local_grid.globalGrid();
60  const auto& global_mesh = global_grid.globalMesh();
61 
62  // Get the cell size.
63  for ( std::size_t d = 0; d < num_space_dim; ++d )
64  _cell_size[d] = global_mesh.cellSize( d );
65 
66  // Compute face area.
67  if ( 3 == num_space_dim )
68  {
69  _face_area[Dim::I] = _cell_size[Dim::J] * _cell_size[Dim::K];
70  _face_area[Dim::J] = _cell_size[Dim::I] * _cell_size[Dim::K];
71  _face_area[Dim::K] = _cell_size[Dim::I] * _cell_size[Dim::J];
72  }
73  else if ( 2 == num_space_dim )
74  {
75  _face_area[Dim::I] = _cell_size[Dim::J];
76  _face_area[Dim::J] = _cell_size[Dim::I];
77  }
78 
79  // Compute cell volume.
80  _cell_volume = 1.0;
81  for ( std::size_t d = 0; d < num_space_dim; ++d )
82  _cell_volume *= _cell_size[d];
83 
84  // Compute the owned low corner
85  for ( std::size_t d = 0; d < num_space_dim; ++d )
86  _own_low_corner[d] = global_mesh.lowCorner( d ) +
87  _cell_size[d] * global_grid.globalOffset( d );
88 
89  // Compute the owned high corner
90  for ( std::size_t d = 0; d < num_space_dim; ++d )
91  _own_high_corner[d] =
92  global_mesh.lowCorner( d ) +
93  _cell_size[d] * ( global_grid.globalOffset( d ) +
94  global_grid.ownedNumCell( d ) );
95 
96  // Compute the ghosted low corner
97  for ( std::size_t d = 0; d < num_space_dim; ++d )
98  _ghost_low_corner[d] = lowCorner( Own(), d ) -
99  local_grid.haloCellWidth() * _cell_size[d];
100 
101  // Compute the ghosted high corner
102  for ( std::size_t d = 0; d < num_space_dim; ++d )
103  _ghost_high_corner[d] = highCorner( Own(), d ) +
104  local_grid.haloCellWidth() * _cell_size[d];
105 
106  // Periodicity
107  for ( std::size_t d = 0; d < num_space_dim; ++d )
108  _periodic[d] = global_grid.isPeriodic( d );
109 
110  // Determine if a block is on the low or high boundaries.
111  for ( std::size_t d = 0; d < num_space_dim; ++d )
112  {
113  _boundary_lo[d] = global_grid.onLowBoundary( d );
114  _boundary_hi[d] = global_grid.onHighBoundary( d );
115  }
116  }

287 template <class Scalar, class Device, std::size_t NumSpaceDim>
288 class LocalMesh<Device, NonUniformMesh<Scalar, NumSpaceDim>>
289 {
290  public:
292  using mesh_type = NonUniformMesh<Scalar, NumSpaceDim>;
293 
295  using scalar_type = Scalar;
296 
298  static constexpr std::size_t num_space_dim = NumSpaceDim;
299 
301  using device_type = Device;
303  using memory_space = typename Device::memory_space;
305  using execution_space = typename Device::execution_space;

308  LocalMesh(
309  const LocalGrid<NonUniformMesh<Scalar, num_space_dim>>& local_grid )
310  {
311  const auto& global_grid = local_grid.globalGrid();
312  const auto& global_mesh = global_grid.globalMesh();
313 
314  // Compute the owned low corner.
315  for ( std::size_t d = 0; d < num_space_dim; ++d )
316  _own_low_corner[d] =
317  global_mesh.nonUniformEdge( d )[global_grid.globalOffset( d )];
318 
319  // Compute the owned high corner
320  for ( std::size_t d = 0; d < num_space_dim; ++d )
321  _own_high_corner[d] =
322  global_mesh.nonUniformEdge( d )[global_grid.globalOffset( d ) +
323  global_grid.ownedNumCell( d )];
324 
325  // Compute the ghosted low corner
326  for ( std::size_t d = 0; d < num_space_dim; ++d )
327  {
328  // Interior domain case.
329  if ( !global_grid.onLowBoundary( d ) )
330  {
331  _ghost_low_corner[d] = global_mesh.nonUniformEdge(
332  d )[global_grid.globalOffset( d ) -
333  local_grid.haloCellWidth()];
334  }
335 
336  // Periodic boundary. Use cells on other side of boundary to
337  // generate geometry.
338  else if ( global_grid.isPeriodic( d ) )
339  {
340  int nedge = global_mesh.nonUniformEdge( d ).size();
341  _ghost_low_corner[d] =
342  global_mesh.nonUniformEdge( d ).front() -
343  ( global_mesh.nonUniformEdge( d ).back() -
344  global_mesh.nonUniformEdge(
345  d )[nedge - local_grid.haloCellWidth() - 1] );
346  }
347 
348  // In the non-periodic boundary case we extrapolate halo cells to
349  // have the same width as the boundary cell.
350  else
351  {
352  Scalar dx = global_mesh.nonUniformEdge( d )[1] -
353  global_mesh.nonUniformEdge( d )[0];
354  _ghost_low_corner[d] = global_mesh.nonUniformEdge( d ).front() -
355  dx * local_grid.haloCellWidth();
356  }
357  }
358 
359  // Compute the ghosted high corner
360  for ( std::size_t d = 0; d < num_space_dim; ++d )
361  {
362  // Interior domain case.
363  if ( !global_grid.onHighBoundary( d ) )
364  {
365  _ghost_high_corner[d] = global_mesh.nonUniformEdge(
366  d )[global_grid.globalOffset( d ) +
367  global_grid.ownedNumCell( d ) +
368  local_grid.haloCellWidth()];
369  }
370 
371  // Periodic boundary. Use cells on other side of boundary to
372  // generate geometry.
373  else if ( global_grid.isPeriodic( d ) )
374  {
375  _ghost_high_corner[d] =
376  global_mesh.nonUniformEdge( d ).back() +
377  ( global_mesh.nonUniformEdge(
378  d )[local_grid.haloCellWidth()] -
379  global_mesh.nonUniformEdge( d ).front() );
380  }
381 
382  // In the non-periodic boundary case we extrapolate halo cells to
383  // have the same width as the boundary cell.
384  else
385  {
386  int nedge = global_mesh.nonUniformEdge( d ).size();
387  Scalar dx = global_mesh.nonUniformEdge( d )[nedge - 1] -
388  global_mesh.nonUniformEdge( d )[nedge - 2];
389  _ghost_high_corner[d] = global_mesh.nonUniformEdge( d ).back() +
390  dx * local_grid.haloCellWidth();
391  }
392  }
393 
394  // Get the node index spaces.
395  auto owned_nodes_local =
396  local_grid.indexSpace( Own(), Node(), Local() );
397  auto ghosted_nodes_local =
398  local_grid.indexSpace( Ghost(), Node(), Local() );
399  auto owned_nodes_global =
400  local_grid.indexSpace( Own(), Node(), Global() );
401  for ( std::size_t d = 0; d < num_space_dim; ++d )
402  {
403  // Allocate edges on the device for this dimension.
404  const auto& global_edge = global_mesh.nonUniformEdge( d );
405  int nedge = ghosted_nodes_local.extent( d );
406  int nedge_global = global_edge.size();
407  _local_edges[d] = Kokkos::View<Scalar*, Device>(
408  Kokkos::ViewAllocateWithoutInitializing( "local_edges" ),
409  nedge );
410 
411  // Compute edges on the host.
412  auto edge_mirror = Kokkos::create_mirror_view( Kokkos::HostSpace(),
413  _local_edges[d] );
414 
415  // Compute the owned edges.
416  for ( int n = owned_nodes_local.min( d );
417  n < owned_nodes_local.max( d ); ++n )
418  {
419  edge_mirror( n ) = global_edge[owned_nodes_global.min( d ) + n -
420  owned_nodes_local.min( d )];
421  }
422 
423  // Compute the lower boundary edges.
424  if ( !global_grid.onLowBoundary( d ) )
425  {
426  // Interior block gets edges from neighbors.
427  for ( int n = 0; n < owned_nodes_local.min( d ); ++n )
428  {
429  edge_mirror( n ) =
430  global_edge[owned_nodes_global.min( d ) + n -
431  owned_nodes_local.min( d )];
432  }
433  }
434  else if ( global_grid.isPeriodic( d ) )
435  {
436  // Periodic boundary block gets edges from neighbor on
437  // opposite side of boundary.
438  for ( int n = 0; n < owned_nodes_local.min( d ); ++n )
439  {
440  edge_mirror( n ) =
441  global_edge.front() - global_edge.back() +
442  global_edge[global_edge.size() - 1 -
443  local_grid.haloCellWidth() + n];
444  }
445  }
446  else
447  {
448  // Non-periodic boundary block extrapolates edges using
449  // boundary cell width.
450  for ( int n = 0; n < owned_nodes_local.min( d ); ++n )
451  {
452  Scalar dx = global_edge[1] - global_edge[0];
453  edge_mirror( n ) = global_edge.front() -
454  ( owned_nodes_local.min( d ) - n ) * dx;
455  }
456  }
457 
458  // Compute the upper boundary edges.
459  if ( !global_grid.onHighBoundary( d ) )
460  {
461  // Interior block gets edges from neighbors.
462  for ( int n = owned_nodes_local.max( d );
463  n < ghosted_nodes_local.max( d ); ++n )
464  {
465  edge_mirror( n ) =
466  global_edge[owned_nodes_global.min( d ) + n -
467  owned_nodes_local.min( d )];
468  }
469  }
470  else if ( global_grid.isPeriodic( d ) )
471  {
472  // Periodic boundary block gets edges from neighbor on
473  // opposite side of boundary.
474  for ( int n = 0; n < ghosted_nodes_local.max( d ) -
475  owned_nodes_local.max( d );
476  ++n )
477  {
478  edge_mirror( owned_nodes_local.max( d ) + n ) =
479  global_edge.back() + global_edge[n] -
480  global_edge.front();
481  }
482  }
483  else
484  {
485  // Non-periodic boundary block extrapolates edges using
486  // boundary cell width.
487  for ( int n = owned_nodes_local.max( d );
488  n < ghosted_nodes_local.max( d ); ++n )
489  {
490  Scalar dx = global_edge[nedge_global - 1] -
491  global_edge[nedge_global - 2];
492  edge_mirror( n ) =
493  global_edge.back() +
494  ( n - owned_nodes_local.max( d ) + 1 ) * dx;
495  }
496  }
497 
498  // Copy edges to the device.
499  Kokkos::deep_copy( _local_edges[d], edge_mirror );
500  }
501 
502  // Periodicity
503  for ( std::size_t d = 0; d < num_space_dim; ++d )
504  _periodic[d] = global_grid.isPeriodic( d );
505 
506  // Determine if a block is on the low or high boundaries.
507  for ( std::size_t d = 0; d < num_space_dim; ++d )
508  {
509  _boundary_lo[d] = global_grid.onLowBoundary( d );
510  _boundary_hi[d] = global_grid.onHighBoundary( d );
511  }
512  }
