35template <class Scalar, class MemorySpace, std::size_t NumSpaceDim>
36class LocalMesh<MemorySpace, UniformMesh<Scalar, NumSpaceDim>>
37{
38 public:
40 using mesh_type = UniformMesh<Scalar, NumSpaceDim>;
41
43 using scalar_type = Scalar;
44
46 static constexpr std::size_t num_space_dim = NumSpaceDim;
47
48 // FIXME: extracting the self type for backwards compatibility with previous
49 // template on DeviceType. Should simply be MemorySpace after next release.
51 using memory_space = typename MemorySpace::memory_space;
52 // FIXME: replace warning with memory space assert after next release.
53 static_assert( Cabana::Impl::warn( Kokkos::is_device<MemorySpace>() ) );
54
56 using device_type [[deprecated]] = typename memory_space::device_type;
58 using execution_space = typename memory_space::execution_space;

64 LocalMesh( const LocalGrid<UniformMesh<Scalar, num_space_dim>>& local_grid )
65 {
66 const auto& global_grid = local_grid.globalGrid();
67 const auto& global_mesh = global_grid.globalMesh();
68
69 // Get the cell size.
70 for ( std::size_t d = 0; d < num_space_dim; ++d )
71 _cell_size[d] = global_mesh.cellSize( d );
72
73 // Compute face area.
74 if ( 3 == num_space_dim )
75 {
76 _face_area[Dim::I] = _cell_size[Dim::J] * _cell_size[Dim::K];
77 _face_area[Dim::J] = _cell_size[Dim::I] * _cell_size[Dim::K];
78 _face_area[Dim::K] = _cell_size[Dim::I] * _cell_size[Dim::J];
79 }
80 else if ( 2 == num_space_dim )
81 {
82 _face_area[Dim::I] = _cell_size[Dim::J];
83 _face_area[Dim::J] = _cell_size[Dim::I];
84 }
85
86 // Compute cell volume.
87 _cell_volume = 1.0;
88 for ( std::size_t d = 0; d < num_space_dim; ++d )
89 _cell_volume *= _cell_size[d];
90
91 // Compute the owned low corner
92 for ( std::size_t d = 0; d < num_space_dim; ++d )
93 _own_low_corner[d] = global_mesh.lowCorner( d ) +
94 _cell_size[d] * global_grid.globalOffset( d );
95
96 // Compute the owned high corner
97 for ( std::size_t d = 0; d < num_space_dim; ++d )
98 _own_high_corner[d] =
99 global_mesh.lowCorner( d ) +
100 _cell_size[d] * ( global_grid.globalOffset( d ) +
101 global_grid.ownedNumCell( d ) );
102
103 // Compute the ghosted low corner
104 for ( std::size_t d = 0; d < num_space_dim; ++d )
105 _ghost_low_corner[d] = lowCorner( Own(), d ) -
106 local_grid.haloCellWidth() * _cell_size[d];
107
108 // Compute the ghosted high corner
109 for ( std::size_t d = 0; d < num_space_dim; ++d )
110 _ghost_high_corner[d] = highCorner( Own(), d ) +
111 local_grid.haloCellWidth() * _cell_size[d];
112
113 // Periodicity
114 for ( std::size_t d = 0; d < num_space_dim; ++d )
115 _periodic[d] = global_grid.isPeriodic( d );
116
117 // Determine if a block is on the low or high boundaries.
118 for ( std::size_t d = 0; d < num_space_dim; ++d )
119 {
120 _boundary_lo[d] = global_grid.onLowBoundary( d );
121 _boundary_hi[d] = global_grid.onHighBoundary( d );
122 }
123 }

301template <class Scalar, class MemorySpace, std::size_t NumSpaceDim>
302class LocalMesh<MemorySpace, NonUniformMesh<Scalar, NumSpaceDim>>
303{
304 public:
306 using mesh_type = NonUniformMesh<Scalar, NumSpaceDim>;
307
309 using scalar_type = Scalar;
310
312 static constexpr std::size_t num_space_dim = NumSpaceDim;
313
314 // FIXME: extracting the self type for backwards compatibility with previous
315 // template on DeviceType. Should simply be MemorySpace after next release.
317 using memory_space = typename MemorySpace::memory_space;
318 // FIXME: replace warning with memory space assert after next release.
319 static_assert( Cabana::Impl::warn( Kokkos::is_device<MemorySpace>() ) );
320
322 using device_type [[deprecated]] = typename memory_space::device_type;
324 using execution_space = typename memory_space::execution_space;
325
327 LocalMesh(
328 const LocalGrid<NonUniformMesh<Scalar, num_space_dim>>& local_grid )
329 {
330 const auto& global_grid = local_grid.globalGrid();
331 const auto& global_mesh = global_grid.globalMesh();
332
333 // Compute the owned low corner.
334 for ( std::size_t d = 0; d < num_space_dim; ++d )
335 _own_low_corner[d] =
336 global_mesh.nonUniformEdge( d )[global_grid.globalOffset( d )];
337
338 // Compute the owned high corner
339 for ( std::size_t d = 0; d < num_space_dim; ++d )
340 _own_high_corner[d] =
341 global_mesh.nonUniformEdge( d )[global_grid.globalOffset( d ) +
342 global_grid.ownedNumCell( d )];
343
344 // Compute the ghosted low corner
345 for ( std::size_t d = 0; d < num_space_dim; ++d )
346 {
347 // Interior domain case.
348 if ( !global_grid.onLowBoundary( d ) )
349 {
350 _ghost_low_corner[d] = global_mesh.nonUniformEdge(
351 d )[global_grid.globalOffset( d ) -
352 local_grid.haloCellWidth()];
353 }
354
355 // Periodic boundary. Use cells on other side of boundary to
356 // generate geometry.
357 else if ( global_grid.isPeriodic( d ) )
358 {
359 int nedge = global_mesh.nonUniformEdge( d ).size();
360 _ghost_low_corner[d] =
361 global_mesh.nonUniformEdge( d ).front() -
362 ( global_mesh.nonUniformEdge( d ).back() -
363 global_mesh.nonUniformEdge(
364 d )[nedge - local_grid.haloCellWidth() - 1] );
365 }
366
367 // In the non-periodic boundary case we extrapolate halo cells to
368 // have the same width as the boundary cell.
369 else
370 {
371 Scalar dx = global_mesh.nonUniformEdge( d )[1] -
372 global_mesh.nonUniformEdge( d )[0];
373 _ghost_low_corner[d] = global_mesh.nonUniformEdge( d ).front() -
374 dx * local_grid.haloCellWidth();
375 }
376 }
377
378 // Compute the ghosted high corner
379 for ( std::size_t d = 0; d < num_space_dim; ++d )
380 {
381 // Interior domain case.
382 if ( !global_grid.onHighBoundary( d ) )
383 {
384 _ghost_high_corner[d] = global_mesh.nonUniformEdge(
385 d )[global_grid.globalOffset( d ) +
386 global_grid.ownedNumCell( d ) +
387 local_grid.haloCellWidth()];
388 }
389
390 // Periodic boundary. Use cells on other side of boundary to
391 // generate geometry.
392 else if ( global_grid.isPeriodic( d ) )
393 {
394 _ghost_high_corner[d] =
395 global_mesh.nonUniformEdge( d ).back() +
396 ( global_mesh.nonUniformEdge(
397 d )[local_grid.haloCellWidth()] -
398 global_mesh.nonUniformEdge( d ).front() );
399 }
400
401 // In the non-periodic boundary case we extrapolate halo cells to
402 // have the same width as the boundary cell.
403 else
404 {
405 int nedge = global_mesh.nonUniformEdge( d ).size();
406 Scalar dx = global_mesh.nonUniformEdge( d )[nedge - 1] -
407 global_mesh.nonUniformEdge( d )[nedge - 2];
408 _ghost_high_corner[d] = global_mesh.nonUniformEdge( d ).back() +
409 dx * local_grid.haloCellWidth();
410 }
411 }
412
413 // Get the node index spaces.
414 auto owned_nodes_local =
415 local_grid.indexSpace( Own(), Node(), Local() );
416 auto ghosted_nodes_local =
417 local_grid.indexSpace( Ghost(), Node(), Local() );
418 auto owned_nodes_global =
419 local_grid.indexSpace( Own(), Node(), Global() );
420 for ( std::size_t d = 0; d < num_space_dim; ++d )
421 {
422 // Allocate edges on the device for this dimension.
423 const auto& global_edge = global_mesh.nonUniformEdge( d );
424 int nedge = ghosted_nodes_local.extent( d );
425 int nedge_global = global_edge.size();
426 _local_edges[d] = Kokkos::View<Scalar*, MemorySpace>(
427 Kokkos::ViewAllocateWithoutInitializing( "local_edges" ),
428 nedge );
429
430 // Compute edges on the host.
431 auto edge_mirror = Kokkos::create_mirror_view( Kokkos::HostSpace(),
432 _local_edges[d] );
433
434 // Compute the owned edges.
435 for ( int n = owned_nodes_local.min( d );
436 n < owned_nodes_local.max( d ); ++n )
437 {
438 edge_mirror( n ) = global_edge[owned_nodes_global.min( d ) + n -
439 owned_nodes_local.min( d )];
440 }
441
442 // Compute the lower boundary edges.
443 if ( !global_grid.onLowBoundary( d ) )
444 {
445 // Interior block gets edges from neighbors.
446 for ( int n = 0; n < owned_nodes_local.min( d ); ++n )
447 {
448 edge_mirror( n ) =
449 global_edge[owned_nodes_global.min( d ) + n -
450 owned_nodes_local.min( d )];
451 }
452 }
453 else if ( global_grid.isPeriodic( d ) )
454 {
455 // Periodic boundary block gets edges from neighbor on
456 // opposite side of boundary.
457 for ( int n = 0; n < owned_nodes_local.min( d ); ++n )
458 {
459 edge_mirror( n ) =
460 global_edge.front() - global_edge.back() +
461 global_edge[global_edge.size() - 1 -
462 local_grid.haloCellWidth() + n];
463 }
464 }
465 else
466 {
467 // Non-periodic boundary block extrapolates edges using
468 // boundary cell width.
469 for ( int n = 0; n < owned_nodes_local.min( d ); ++n )
470 {
471 Scalar dx = global_edge[1] - global_edge[0];
472 edge_mirror( n ) = global_edge.front() -
473 ( owned_nodes_local.min( d ) - n ) * dx;
474 }
475 }
476
477 // Compute the upper boundary edges.
478 if ( !global_grid.onHighBoundary( d ) )
479 {
480 // Interior block gets edges from neighbors.
481 for ( int n = owned_nodes_local.max( d );
482 n < ghosted_nodes_local.max( d ); ++n )
483 {
484 edge_mirror( n ) =
485 global_edge[owned_nodes_global.min( d ) + n -
486 owned_nodes_local.min( d )];
487 }
488 }
489 else if ( global_grid.isPeriodic( d ) )
490 {
491 // Periodic boundary block gets edges from neighbor on
492 // opposite side of boundary.
493 for ( int n = 0; n < ghosted_nodes_local.max( d ) -
494 owned_nodes_local.max( d );
495 ++n )
496 {
497 edge_mirror( owned_nodes_local.max( d ) + n ) =
498 global_edge.back() + global_edge[n] -
499 global_edge.front();
500 }
501 }
502 else
503 {
504 // Non-periodic boundary block extrapolates edges using
505 // boundary cell width.
506 for ( int n = owned_nodes_local.max( d );
507 n < ghosted_nodes_local.max( d ); ++n )
508 {
509 Scalar dx = global_edge[nedge_global - 1] -
510 global_edge[nedge_global - 2];
511 edge_mirror( n ) =
512 global_edge.back() +
513 ( n - owned_nodes_local.max( d ) + 1 ) * dx;
514 }
515 }
516
517 // Copy edges to the device.
518 Kokkos::deep_copy( _local_edges[d], edge_mirror );
519 }
520
521 // Periodicity
522 for ( std::size_t d = 0; d < num_space_dim; ++d )
523 _periodic[d] = global_grid.isPeriodic( d );
524
525 // Determine if a block is on the low or high boundaries.
526 for ( std::size_t d = 0; d < num_space_dim; ++d )
527 {
528 _boundary_lo[d] = global_grid.onLowBoundary( d );
529 _boundary_hi[d] = global_grid.onHighBoundary( d );
530 }
531 }
