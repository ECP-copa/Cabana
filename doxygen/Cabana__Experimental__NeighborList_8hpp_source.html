78template <typename ExecutionSpace, typename D, typename... P>
79typename Kokkos::View<D, P...>::non_const_value_type
80max_reduce( ExecutionSpace const& space, Kokkos::View<D, P...> const& v )
81{
82 using V = Kokkos::View<D, P...>;
83 static_assert( V::rank == 1 );
84 static_assert( Kokkos::is_execution_space<ExecutionSpace>::value );
85 static_assert(
86 is_accessible_from<typename V::memory_space, ExecutionSpace>::value );
87 using Ret = typename Kokkos::View<D, P...>::non_const_value_type;
88 Ret max_val;
89 Kokkos::parallel_reduce(
90 Kokkos::RangePolicy<ExecutionSpace>( space, 0, v.extent( 0 ) ),
91 KOKKOS_LAMBDA( int i, Ret& partial_max ) {
92 if ( v( i ) > partial_max )
93 {
94 partial_max = v( i );
95 }
96 },
97 Kokkos::Max<Ret>( max_val ) );
98 return max_val;
99}

185// Custom callback for ArborX::BVH::query()
186template <typename Tag>
187struct NeighborDiscriminatorCallback
188{
189 template <typename Predicate, typename OutputFunctor>
190 KOKKOS_FUNCTION void operator()( Predicate const& predicate,
191 int primitive_index,
192 OutputFunctor const& out ) const
193 {
194 int const predicate_index = getData( predicate );
195 if ( CollisionFilter<Tag>::keep( predicate_index, primitive_index ) )
196 {
197 out( primitive_index );
198 }
199 }
200};
201
202// Count in the first pass
203template <typename Counts, typename Tag>
204struct NeighborDiscriminatorCallback2D_FirstPass
205{
206 Counts counts;
207 template <typename Predicate>
208 KOKKOS_FUNCTION void operator()( Predicate const& predicate,
209 int primitive_index ) const
210 {
211 int const predicate_index = getData( predicate );
212 if ( CollisionFilter<Tag>::keep( predicate_index, primitive_index ) )
213 {
214 ++counts( predicate_index ); // WARNING see below**
215 }
216 }
217};
218
219// Preallocate and attempt fill in the first pass
220template <typename Counts, typename Neighbors, typename Tag>
221struct NeighborDiscriminatorCallback2D_FirstPass_BufferOptimization
222{
223 Counts counts;
224 Neighbors neighbors;
225 template <typename Predicate>
226 KOKKOS_FUNCTION void operator()( Predicate const& predicate,
227 int primitive_index ) const
228 {
229 int const predicate_index = getData( predicate );
230 auto& count = counts( predicate_index );
231 if ( CollisionFilter<Tag>::keep( predicate_index, primitive_index ) )
232 {
233 if ( count < (int)neighbors.extent( 1 ) )
234 {
235 neighbors( predicate_index, count++ ) =
236 primitive_index; // WARNING see below**
237 }
238 else
239 {
240 count++;
241 }
242 }
243 }
244};
245
246// Fill in the second pass
247template <typename Counts, typename Neighbors, typename Tag>
248struct NeighborDiscriminatorCallback2D_SecondPass
249{
250 Counts counts;
251 Neighbors neighbors;
252 template <typename Predicate>
253 KOKKOS_FUNCTION void operator()( Predicate const& predicate,
254 int primitive_index ) const
255 {
256 int const predicate_index = getData( predicate );
257 auto& count = counts( predicate_index );
258 if ( CollisionFilter<Tag>::keep( predicate_index, primitive_index ) )
259 {
260 assert( count < (int)neighbors.extent( 1 ) );
261 neighbors( predicate_index, count++ ) =
262 primitive_index; // WARNING see below**
263 }
264 }
265};
266
267// NOTE** Taking advantage of the knowledge that one predicate is processed by a
268// single thread. Count increment should be atomic otherwise.

309template <typename ExecutionSpace, typename Positions, typename Tag>
310auto makeNeighborList( ExecutionSpace space, Tag, Positions const& positions,
311 typename Positions::size_type first,
312 typename Positions::size_type last,
313 typename Positions::value_type radius,
314 int buffer_size = 0 )
315{
316 assert( buffer_size >= 0 );
317 assert( last >= first );
318 assert( last <= positions.size() );
319
320 using memory_space = typename Positions::memory_space;
321
322 ArborX::BVH<memory_space> bvh( space, positions );
323
324 Kokkos::View<int*, memory_space> indices(
325 Kokkos::view_alloc( "indices", Kokkos::WithoutInitializing ), 0 );
326 Kokkos::View<int*, memory_space> offset(
327 Kokkos::view_alloc( "offset", Kokkos::WithoutInitializing ), 0 );
328 bvh.query(
329 space, Impl::makePredicates( positions, first, last, radius ),
330 Impl::NeighborDiscriminatorCallback<Tag>{}, indices, offset,
331 ArborX::Experimental::TraversalPolicy().setBufferSize( buffer_size ) );
332
333 return CrsGraph<memory_space, Tag>{
334 std::move( indices ), std::move( offset ), first, bvh.size() };
335}

434template <typename ExecutionSpace, typename Positions, typename Tag>
435auto make2DNeighborList( ExecutionSpace space, Tag, Positions const& positions,
436 typename Positions::size_type first,
437 typename Positions::size_type last,
438 typename Positions::value_type radius,
439 int buffer_size = 0 )
440{
441 assert( buffer_size >= 0 );
442 assert( last >= first );
443 assert( last <= positions.size() );
444
445 using memory_space = typename Positions::memory_space;
446
447 ArborX::BVH<memory_space> bvh( space, positions );
448
449 auto const predicates =
450 Impl::makePredicates( positions, first, last, radius );
451
452 auto const n_queries =
453 ArborX::AccessTraits<std::remove_const_t<decltype( predicates )>,
454 ArborX::PredicatesTag>::size( predicates );
455
456 Kokkos::View<int**, memory_space> neighbors;
457 Kokkos::View<int*, memory_space> counts( "counts", n_queries );
458 if ( buffer_size > 0 )
459 {
460 neighbors = Kokkos::View<int**, memory_space>(
461 Kokkos::view_alloc( "neighbors", Kokkos::WithoutInitializing ),
462 n_queries, buffer_size );
463 bvh.query(
464 space, predicates,
465 Impl::NeighborDiscriminatorCallback2D_FirstPass_BufferOptimization<
466 decltype( counts ), decltype( neighbors ), Tag>{ counts,
467 neighbors } );
468 }
469 else
470 {
471 bvh.query(
472 space, predicates,
473 Impl::NeighborDiscriminatorCallback2D_FirstPass<decltype( counts ),
474 Tag>{ counts } );
475 }
476
477 auto const max_neighbors = Impl::max_reduce( space, counts );
478 if ( max_neighbors <= buffer_size )
479 {
480 // NOTE We do not bother shrinking to eliminate the excess allocation.
481 // NOTE If buffer_size is 0, neighbors is default constructed. This is
482 // fine with the current design/implementation of NeighborList access
483 // traits.
484 return Dense<memory_space, Tag>{ counts, neighbors, first, bvh.size() };
485 }
486
487 neighbors = Kokkos::View<int**, memory_space>(
488 Kokkos::view_alloc( "neighbors", Kokkos::WithoutInitializing ),
489 n_queries, max_neighbors ); // realloc storage for neighbors
490 Kokkos::deep_copy( counts, 0 ); // reset counts to zero
491 bvh.query( space, predicates,
492 Impl::NeighborDiscriminatorCallback2D_SecondPass<
493 decltype( counts ), decltype( neighbors ), Tag>{
494 counts, neighbors } );
495
496 return Dense<memory_space, Tag>{ counts, neighbors, first, bvh.size() };
497}

565template <typename MemorySpace, typename Tag>
566class NeighborList<Experimental::CrsGraph<MemorySpace, Tag>>
567{
569 using size_type = std::size_t;
571 using crs_graph_type = Experimental::CrsGraph<MemorySpace, Tag>;
572
573 public:
575 using memory_space = MemorySpace;
576
578 KOKKOS_INLINE_FUNCTION
579 static size_type totalNeighbor( crs_graph_type const& crs_graph )
580 {
581 return Impl::totalNeighbor( crs_graph, crs_graph.total );
582 }
583
585 KOKKOS_INLINE_FUNCTION
586 static size_type maxNeighbor( crs_graph_type const& crs_graph )
587 {
588 return Impl::maxNeighbor( crs_graph, crs_graph.total );
589 }
590
592 static KOKKOS_FUNCTION size_type
593 numNeighbor( crs_graph_type const& crs_graph, size_type p )
594 {
595 assert( (int)p >= 0 && p < crs_graph.total );
596 p -= crs_graph.shift;
597 if ( (int)p < 0 || p >= crs_graph.row_ptr.size() - 1 )
598 return 0;
599 return crs_graph.row_ptr( p + 1 ) - crs_graph.row_ptr( p );
600 }
602 static KOKKOS_FUNCTION size_type
603 getNeighbor( crs_graph_type const& crs_graph, size_type p, size_type n )
604 {
605 assert( n < numNeighbor( crs_graph, p ) );
606 p -= crs_graph.shift;
607 return crs_graph.col_ind( crs_graph.row_ptr( p ) + n );
608 }
609};

654 return d.val( p, n );
