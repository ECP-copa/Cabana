74template <typename ExecutionSpace, typename D, typename... P>
75typename Kokkos::View<D, P...>::non_const_value_type
76max_reduce( ExecutionSpace const& space, Kokkos::View<D, P...> const& v )
77{
78 using V = Kokkos::View<D, P...>;
79 static_assert( V::rank == 1 );
80 static_assert( Kokkos::is_execution_space<ExecutionSpace>::value );
81 static_assert(
82 is_accessible_from<typename V::memory_space, ExecutionSpace>::value );
83 using Ret = typename Kokkos::View<D, P...>::non_const_value_type;
84 Ret max_val;
85 Kokkos::parallel_reduce(
86 Kokkos::RangePolicy<ExecutionSpace>( space, 0, v.extent( 0 ) ),
87 KOKKOS_LAMBDA( int i, Ret& partial_max ) {
88 if ( v( i ) > partial_max )
89 {
90 partial_max = v( i );
91 }
92 },
93 Kokkos::Max<Ret>( max_val ) );
94 return max_val;
95}

173// Custom callback for ArborX::BVH::query()
174template <typename Tag>
175struct NeighborDiscriminatorCallback
176{
177 template <typename Predicate, typename OutputFunctor>
178 KOKKOS_FUNCTION void operator()( Predicate const& predicate,
179 int primitive_index,
180 OutputFunctor const& out ) const
181 {
182 int const predicate_index = getData( predicate );
183 if ( CollisionFilter<Tag>::keep( predicate_index, primitive_index ) )
184 {
185 out( primitive_index );
186 }
187 }
188};
189
190// Count in the first pass
191template <typename Counts, typename Tag>
192struct NeighborDiscriminatorCallback2D_FirstPass
193{
194 Counts counts;
195 template <typename Predicate>
196 KOKKOS_FUNCTION void operator()( Predicate const& predicate,
197 int primitive_index ) const
198 {
199 int const predicate_index = getData( predicate );
200 if ( CollisionFilter<Tag>::keep( predicate_index, primitive_index ) )
201 {
202 ++counts( predicate_index ); // WARNING see below**
203 }
204 }
205};
206
207// Preallocate and attempt fill in the first pass
208template <typename Counts, typename Neighbors, typename Tag>
209struct NeighborDiscriminatorCallback2D_FirstPass_BufferOptimization
210{
211 Counts counts;
212 Neighbors neighbors;
213 template <typename Predicate>
214 KOKKOS_FUNCTION void operator()( Predicate const& predicate,
215 int primitive_index ) const
216 {
217 int const predicate_index = getData( predicate );
218 auto& count = counts( predicate_index );
219 if ( CollisionFilter<Tag>::keep( predicate_index, primitive_index ) )
220 {
221 if ( count < (int)neighbors.extent( 1 ) )
222 {
223 neighbors( predicate_index, count++ ) =
224 primitive_index; // WARNING see below**
225 }
226 else
227 {
228 count++;
229 }
230 }
231 }
232};
233
234// Fill in the second pass
235template <typename Counts, typename Neighbors, typename Tag>
236struct NeighborDiscriminatorCallback2D_SecondPass
237{
238 Counts counts;
239 Neighbors neighbors;
240 template <typename Predicate>
241 KOKKOS_FUNCTION void operator()( Predicate const& predicate,
242 int primitive_index ) const
243 {
244 int const predicate_index = getData( predicate );
245 auto& count = counts( predicate_index );
246 if ( CollisionFilter<Tag>::keep( predicate_index, primitive_index ) )
247 {
248 assert( count < (int)neighbors.extent( 1 ) );
249 neighbors( predicate_index, count++ ) =
250 primitive_index; // WARNING see below**
251 }
252 }
253};
254
255// NOTE** Taking advantage of the knowledge that one predicate is processed by a
256// single thread. Count increment should be atomic otherwise.

297template <typename ExecutionSpace, typename Slice, typename Tag>
298auto makeNeighborList( ExecutionSpace space, Tag, Slice const& coordinate_slice,
299 typename Slice::size_type first,
300 typename Slice::size_type last,
301 typename Slice::value_type radius, int buffer_size = 0 )
302{
303 assert( buffer_size >= 0 );
304 assert( last >= first );
305 assert( last <= coordinate_slice.size() );
306
307 using memory_space = typename Slice::memory_space;
308
309 ArborX::BVH<memory_space> bvh( space, coordinate_slice );
310
311 Kokkos::View<int*, memory_space> indices(
312 Kokkos::view_alloc( "indices", Kokkos::WithoutInitializing ), 0 );
313 Kokkos::View<int*, memory_space> offset(
314 Kokkos::view_alloc( "offset", Kokkos::WithoutInitializing ), 0 );
315 bvh.query(
316 space, Impl::makePredicates( coordinate_slice, first, last, radius ),
317 Impl::NeighborDiscriminatorCallback<Tag>{}, indices, offset,
318 ArborX::Experimental::TraversalPolicy().setBufferSize( buffer_size ) );
319
320 return CrsGraph<memory_space, Tag>{
321 std::move( indices ), std::move( offset ), first, bvh.size() };
322}

420template <typename ExecutionSpace, typename Slice, typename Tag>
421auto make2DNeighborList( ExecutionSpace space, Tag,
422 Slice const& coordinate_slice,
423 typename Slice::size_type first,
424 typename Slice::size_type last,
425 typename Slice::value_type radius,
426 int buffer_size = 0 )
427{
428 assert( buffer_size >= 0 );
429 assert( last >= first );
430 assert( last <= coordinate_slice.size() );
431
432 using memory_space = typename Slice::memory_space;
433
434 ArborX::BVH<memory_space> bvh( space, coordinate_slice );
435
436 auto const predicates =
437 Impl::makePredicates( coordinate_slice, first, last, radius );
438
439 auto const n_queries =
440 ArborX::AccessTraits<decltype( predicates ),
441 ArborX::PredicatesTag>::size( predicates );
442
443 Kokkos::View<int**, memory_space> neighbors;
444 Kokkos::View<int*, memory_space> counts( "counts", n_queries );
445 if ( buffer_size > 0 )
446 {
447 neighbors = Kokkos::View<int**, memory_space>(
448 Kokkos::view_alloc( "neighbors", Kokkos::WithoutInitializing ),
449 n_queries, buffer_size );
450 bvh.query(
451 space, predicates,
452 Impl::NeighborDiscriminatorCallback2D_FirstPass_BufferOptimization<
453 decltype( counts ), decltype( neighbors ), Tag>{ counts,
454 neighbors } );
455 }
456 else
457 {
458 bvh.query(
459 space, predicates,
460 Impl::NeighborDiscriminatorCallback2D_FirstPass<decltype( counts ),
461 Tag>{ counts } );
462 }
463
464 auto const max_neighbors = Impl::max_reduce( space, counts );
465 if ( max_neighbors <= buffer_size )
466 {
467 // NOTE We do not bother shrinking to eliminate the excess allocation.
468 // NOTE If buffer_size is 0, neighbors is default constructed. This is
469 // fine with the current design/implementation of NeighborList access
470 // traits.
471 return Dense<memory_space, Tag>{ counts, neighbors, first, bvh.size() };
472 }
473
474 neighbors = Kokkos::View<int**, memory_space>(
475 Kokkos::view_alloc( "neighbors", Kokkos::WithoutInitializing ),
476 n_queries, max_neighbors ); // realloc storage for neighbors
477 Kokkos::deep_copy( counts, 0 ); // reset counts to zero
478 bvh.query( space, predicates,
479 Impl::NeighborDiscriminatorCallback2D_SecondPass<
480 decltype( counts ), decltype( neighbors ), Tag>{
481 counts, neighbors } );
482
483 return Dense<memory_space, Tag>{ counts, neighbors, first, bvh.size() };
484}

552template <typename MemorySpace, typename Tag>
553class NeighborList<Experimental::CrsGraph<MemorySpace, Tag>>
554{
556 using size_type = std::size_t;
558 using crs_graph_type = Experimental::CrsGraph<MemorySpace, Tag>;
559
560 public:
562 using memory_space = MemorySpace;
563
565 KOKKOS_INLINE_FUNCTION
566 static size_type totalNeighbor( crs_graph_type const& crs_graph )
567 {
568 return Impl::totalNeighbor( crs_graph, crs_graph.total );
569 }
570
572 KOKKOS_INLINE_FUNCTION
573 static size_type maxNeighbor( crs_graph_type const& crs_graph )
574 {
575 return Impl::maxNeighbor( crs_graph, crs_graph.total );
576 }
577
579 static KOKKOS_FUNCTION size_type
580 numNeighbor( crs_graph_type const& crs_graph, size_type p )
581 {
582 assert( (int)p >= 0 && p < crs_graph.total );
583 p -= crs_graph.shift;
584 if ( (int)p < 0 || p >= crs_graph.row_ptr.size() - 1 )
585 return 0;
586 return crs_graph.row_ptr( p + 1 ) - crs_graph.row_ptr( p );
587 }
589 static KOKKOS_FUNCTION size_type
590 getNeighbor( crs_graph_type const& crs_graph, size_type p, size_type n )
591 {
592 assert( n < numNeighbor( crs_graph, p ) );
593 p -= crs_graph.shift;
594 return crs_graph.col_ind( crs_graph.row_ptr( p ) + n );
595 }
596};

641 return d.val( p, n );
