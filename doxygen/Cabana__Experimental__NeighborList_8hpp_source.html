74template <typename ExecutionSpace, typename D, typename... P>
75typename Kokkos::View<D, P...>::non_const_value_type
76max_reduce( ExecutionSpace const& space, Kokkos::View<D, P...> const& v )
77{
78 using V = Kokkos::View<D, P...>;
79 static_assert( V::rank == 1 );
80 static_assert( Kokkos::is_execution_space<ExecutionSpace>::value );
81 static_assert(
82 is_accessible_from<typename V::memory_space, ExecutionSpace>::value );
83 using Ret = typename Kokkos::View<D, P...>::non_const_value_type;
84 Ret max_val;
85 Kokkos::parallel_reduce(
86 Kokkos::RangePolicy<ExecutionSpace>( space, 0, v.extent( 0 ) ),
87 KOKKOS_LAMBDA( int i, Ret& partial_max ) {
88 if ( v( i ) > partial_max )
89 {
90 partial_max = v( i );
91 }
92 },
93 Kokkos::Max<Ret>( max_val ) );
94 return max_val;
95}

176// Custom callback for ArborX::BVH::query()
177template <typename Tag>
178struct NeighborDiscriminatorCallback
179{
180 template <typename Predicate, typename OutputFunctor>
181 KOKKOS_FUNCTION void operator()( Predicate const& predicate,
182 int primitive_index,
183 OutputFunctor const& out ) const
184 {
185 int const predicate_index = getData( predicate );
186 if ( CollisionFilter<Tag>::keep( predicate_index, primitive_index ) )
187 {
188 out( primitive_index );
189 }
190 }
191};
192
193// Count in the first pass
194template <typename Counts, typename Tag>
195struct NeighborDiscriminatorCallback2D_FirstPass
196{
197 Counts counts;
198 template <typename Predicate>
199 KOKKOS_FUNCTION void operator()( Predicate const& predicate,
200 int primitive_index ) const
201 {
202 int const predicate_index = getData( predicate );
203 if ( CollisionFilter<Tag>::keep( predicate_index, primitive_index ) )
204 {
205 ++counts( predicate_index ); // WARNING see below**
206 }
207 }
208};
209
210// Preallocate and attempt fill in the first pass
211template <typename Counts, typename Neighbors, typename Tag>
212struct NeighborDiscriminatorCallback2D_FirstPass_BufferOptimization
213{
214 Counts counts;
215 Neighbors neighbors;
216 template <typename Predicate>
217 KOKKOS_FUNCTION void operator()( Predicate const& predicate,
218 int primitive_index ) const
219 {
220 int const predicate_index = getData( predicate );
221 auto& count = counts( predicate_index );
222 if ( CollisionFilter<Tag>::keep( predicate_index, primitive_index ) )
223 {
224 if ( count < (int)neighbors.extent( 1 ) )
225 {
226 neighbors( predicate_index, count++ ) =
227 primitive_index; // WARNING see below**
228 }
229 else
230 {
231 count++;
232 }
233 }
234 }
235};
236
237// Fill in the second pass
238template <typename Counts, typename Neighbors, typename Tag>
239struct NeighborDiscriminatorCallback2D_SecondPass
240{
241 Counts counts;
242 Neighbors neighbors;
243 template <typename Predicate>
244 KOKKOS_FUNCTION void operator()( Predicate const& predicate,
245 int primitive_index ) const
246 {
247 int const predicate_index = getData( predicate );
248 auto& count = counts( predicate_index );
249 if ( CollisionFilter<Tag>::keep( predicate_index, primitive_index ) )
250 {
251 assert( count < (int)neighbors.extent( 1 ) );
252 neighbors( predicate_index, count++ ) =
253 primitive_index; // WARNING see below**
254 }
255 }
256};
257
258// NOTE** Taking advantage of the knowledge that one predicate is processed by a
259// single thread. Count increment should be atomic otherwise.

300template <typename ExecutionSpace, typename Slice, typename Tag>
301auto makeNeighborList( ExecutionSpace space, Tag, Slice const& coordinate_slice,
302 typename Slice::size_type first,
303 typename Slice::size_type last,
304 typename Slice::value_type radius, int buffer_size = 0 )
305{
306 assert( buffer_size >= 0 );
307 assert( last >= first );
308 assert( last <= coordinate_slice.size() );
309
310 using memory_space = typename Slice::memory_space;
311
312 ArborX::BVH<memory_space> bvh( space, coordinate_slice );
313
314 Kokkos::View<int*, memory_space> indices(
315 Kokkos::view_alloc( "indices", Kokkos::WithoutInitializing ), 0 );
316 Kokkos::View<int*, memory_space> offset(
317 Kokkos::view_alloc( "offset", Kokkos::WithoutInitializing ), 0 );
318 bvh.query(
319 space, Impl::makePredicates( coordinate_slice, first, last, radius ),
320 Impl::NeighborDiscriminatorCallback<Tag>{}, indices, offset,
321 ArborX::Experimental::TraversalPolicy().setBufferSize( buffer_size ) );
322
323 return CrsGraph<memory_space, Tag>{
324 std::move( indices ), std::move( offset ), first, bvh.size() };
325}

423template <typename ExecutionSpace, typename Slice, typename Tag>
424auto make2DNeighborList( ExecutionSpace space, Tag,
425 Slice const& coordinate_slice,
426 typename Slice::size_type first,
427 typename Slice::size_type last,
428 typename Slice::value_type radius,
429 int buffer_size = 0 )
430{
431 assert( buffer_size >= 0 );
432 assert( last >= first );
433 assert( last <= coordinate_slice.size() );
434
435 using memory_space = typename Slice::memory_space;
436
437 ArborX::BVH<memory_space> bvh( space, coordinate_slice );
438
439 auto const predicates =
440 Impl::makePredicates( coordinate_slice, first, last, radius );
441
442 auto const n_queries =
443 ArborX::AccessTraits<std::remove_const_t<decltype( predicates )>,
444 ArborX::PredicatesTag>::size( predicates );
445
446 Kokkos::View<int**, memory_space> neighbors;
447 Kokkos::View<int*, memory_space> counts( "counts", n_queries );
448 if ( buffer_size > 0 )
449 {
450 neighbors = Kokkos::View<int**, memory_space>(
451 Kokkos::view_alloc( "neighbors", Kokkos::WithoutInitializing ),
452 n_queries, buffer_size );
453 bvh.query(
454 space, predicates,
455 Impl::NeighborDiscriminatorCallback2D_FirstPass_BufferOptimization<
456 decltype( counts ), decltype( neighbors ), Tag>{ counts,
457 neighbors } );
458 }
459 else
460 {
461 bvh.query(
462 space, predicates,
463 Impl::NeighborDiscriminatorCallback2D_FirstPass<decltype( counts ),
464 Tag>{ counts } );
465 }
466
467 auto const max_neighbors = Impl::max_reduce( space, counts );
468 if ( max_neighbors <= buffer_size )
469 {
470 // NOTE We do not bother shrinking to eliminate the excess allocation.
471 // NOTE If buffer_size is 0, neighbors is default constructed. This is
472 // fine with the current design/implementation of NeighborList access
473 // traits.
474 return Dense<memory_space, Tag>{ counts, neighbors, first, bvh.size() };
475 }
476
477 neighbors = Kokkos::View<int**, memory_space>(
478 Kokkos::view_alloc( "neighbors", Kokkos::WithoutInitializing ),
479 n_queries, max_neighbors ); // realloc storage for neighbors
480 Kokkos::deep_copy( counts, 0 ); // reset counts to zero
481 bvh.query( space, predicates,
482 Impl::NeighborDiscriminatorCallback2D_SecondPass<
483 decltype( counts ), decltype( neighbors ), Tag>{
484 counts, neighbors } );
485
486 return Dense<memory_space, Tag>{ counts, neighbors, first, bvh.size() };
487}

555template <typename MemorySpace, typename Tag>
556class NeighborList<Experimental::CrsGraph<MemorySpace, Tag>>
557{
559 using size_type = std::size_t;
561 using crs_graph_type = Experimental::CrsGraph<MemorySpace, Tag>;
562
563 public:
565 using memory_space = MemorySpace;
566
568 KOKKOS_INLINE_FUNCTION
569 static size_type totalNeighbor( crs_graph_type const& crs_graph )
570 {
571 return Impl::totalNeighbor( crs_graph, crs_graph.total );
572 }
573
575 KOKKOS_INLINE_FUNCTION
576 static size_type maxNeighbor( crs_graph_type const& crs_graph )
577 {
578 return Impl::maxNeighbor( crs_graph, crs_graph.total );
579 }
580
582 static KOKKOS_FUNCTION size_type
583 numNeighbor( crs_graph_type const& crs_graph, size_type p )
584 {
585 assert( (int)p >= 0 && p < crs_graph.total );
586 p -= crs_graph.shift;
587 if ( (int)p < 0 || p >= crs_graph.row_ptr.size() - 1 )
588 return 0;
589 return crs_graph.row_ptr( p + 1 ) - crs_graph.row_ptr( p );
590 }
592 static KOKKOS_FUNCTION size_type
593 getNeighbor( crs_graph_type const& crs_graph, size_type p, size_type n )
594 {
595 assert( n < numNeighbor( crs_graph, p ) );
596 p -= crs_graph.shift;
597 return crs_graph.col_ind( crs_graph.row_ptr( p ) + n );
598 }
599};

631 {
