78template <typename ExecutionSpace, typename D, typename... P>
79typename Kokkos::View<D, P...>::non_const_value_type
80max_reduce( ExecutionSpace const& space, Kokkos::View<D, P...> const& v )
81{
82 using V = Kokkos::View<D, P...>;
83 static_assert( V::rank == 1 );
84 static_assert( Kokkos::is_execution_space<ExecutionSpace>::value );
85 static_assert(
86 is_accessible_from<typename V::memory_space, ExecutionSpace>::value );
87 using Ret = typename Kokkos::View<D, P...>::non_const_value_type;
88 Ret max_val;
89 Kokkos::parallel_reduce(
90 Kokkos::RangePolicy<ExecutionSpace>( space, 0, v.extent( 0 ) ),
91 KOKKOS_LAMBDA( int i, Ret& partial_max ) {
92 if ( v( i ) > partial_max )
93 {
94 partial_max = v( i );
95 }
96 },
97 Kokkos::Max<Ret>( max_val ) );
98 return max_val;
99}

394template <typename ExecutionSpace, typename Positions, typename Tag>
395auto makeNeighborList( ExecutionSpace space, Tag, Positions const& positions,
396 typename Positions::size_type first,
397 typename Positions::size_type last,
398 typename Positions::value_type radius,
399 int buffer_size = 0 )
400{
401 assert( buffer_size >= 0 );
402 assert( last >= first );
403 assert( last <= positions.size() );
404
405 using memory_space = typename Positions::memory_space;
406
407#if ARBORX_VERSION >= 10799
408 ArborX::BoundingVolumeHierarchy bvh(
409 space, ArborX::Experimental::attach_indices<int>( positions ) );
410#else
411 ArborX::BVH<memory_space> bvh( space, positions );
412#endif
413
414 Kokkos::View<int*, memory_space> indices(
415 Kokkos::view_alloc( "indices", Kokkos::WithoutInitializing ), 0 );
416 Kokkos::View<int*, memory_space> offset(
417 Kokkos::view_alloc( "offset", Kokkos::WithoutInitializing ), 0 );
418 bvh.query(
419 space, Impl::makePredicates( positions, first, last, radius ),
420 Impl::NeighborDiscriminatorCallback<Tag>{}, indices, offset,
421 ArborX::Experimental::TraversalPolicy().setBufferSize( buffer_size ) );
422
423 return CrsGraph<memory_space, Tag>{
424 std::move( indices ), std::move( offset ), first, bvh.size() };
425}

524template <typename ExecutionSpace, typename Positions, typename Tag>
525auto make2DNeighborList( ExecutionSpace space, Tag, Positions const& positions,
526 typename Positions::size_type first,
527 typename Positions::size_type last,
528 typename Positions::value_type radius,
529 int buffer_size = 0 )
530{
531 assert( buffer_size >= 0 );
532 assert( last >= first );
533 assert( last <= positions.size() );
534
535 using memory_space = typename Positions::memory_space;
536
537#if ARBORX_VERSION >= 10799
538 ArborX::BoundingVolumeHierarchy bvh(
539 space, ArborX::Experimental::attach_indices<int>( positions ) );
540#else
541 ArborX::BVH<memory_space> bvh( space, positions );
542#endif
543
544 auto const predicates =
545 Impl::makePredicates( positions, first, last, radius );
546
547 auto const n_queries =
548 ArborX::AccessTraits<std::remove_const_t<decltype( predicates )>
549#if ARBORX_VERSION < 10799
550 ,
551 ArborX::PredicatesTag
552#endif
553 >::size( predicates );
554
555 Kokkos::View<int**, memory_space> neighbors;
556 Kokkos::View<int*, memory_space> counts( "counts", n_queries );
557 if ( buffer_size > 0 )
558 {
559 neighbors = Kokkos::View<int**, memory_space>(
560 Kokkos::view_alloc( "neighbors", Kokkos::WithoutInitializing ),
561 n_queries, buffer_size );
562 bvh.query(
563 space, predicates,
564 Impl::NeighborDiscriminatorCallback2D_FirstPass_BufferOptimization<
565 decltype( counts ), decltype( neighbors ), Tag>{ counts,
566 neighbors } );
567 }
568 else
569 {
570 bvh.query(
571 space, predicates,
572 Impl::NeighborDiscriminatorCallback2D_FirstPass<decltype( counts ),
573 Tag>{ counts } );
574 }
575
576 auto const max_neighbors = Impl::max_reduce( space, counts );
577 if ( max_neighbors <= buffer_size )
578 {
579 // NOTE We do not bother shrinking to eliminate the excess allocation.
580 // NOTE If buffer_size is 0, neighbors is default constructed. This is
581 // fine with the current design/implementation of NeighborList access
582 // traits.
583 return Dense<memory_space, Tag>{ counts, neighbors, first, bvh.size() };
584 }
585
586 neighbors = Kokkos::View<int**, memory_space>(
587 Kokkos::view_alloc( "neighbors", Kokkos::WithoutInitializing ),
588 n_queries, max_neighbors ); // realloc storage for neighbors
589 Kokkos::deep_copy( counts, 0 ); // reset counts to zero
590 bvh.query( space, predicates,
591 Impl::NeighborDiscriminatorCallback2D_SecondPass<
592 decltype( counts ), decltype( neighbors ), Tag>{
593 counts, neighbors } );
594
595 return Dense<memory_space, Tag>{ counts, neighbors, first, bvh.size() };
596}

664template <typename MemorySpace, typename Tag>
665class NeighborList<Experimental::CrsGraph<MemorySpace, Tag>>
666{
668 using size_type = std::size_t;
670 using crs_graph_type = Experimental::CrsGraph<MemorySpace, Tag>;
671
672 public:
674 using memory_space = MemorySpace;
675
677 KOKKOS_INLINE_FUNCTION
678 static size_type totalNeighbor( crs_graph_type const& crs_graph )
679 {
680 return Impl::totalNeighbor( crs_graph, crs_graph.total );
681 }
682
684 KOKKOS_INLINE_FUNCTION
685 static size_type maxNeighbor( crs_graph_type const& crs_graph )
686 {
687 return Impl::maxNeighbor( crs_graph, crs_graph.total );
688 }
689
691 static KOKKOS_FUNCTION size_type
692 numNeighbor( crs_graph_type const& crs_graph, size_type p )
693 {
694 assert( (int)p >= 0 && p < crs_graph.total );
695 p -= crs_graph.shift;
696 if ( (int)p < 0 || p >= crs_graph.row_ptr.size() - 1 )
697 return 0;
698 return crs_graph.row_ptr( p + 1 ) - crs_graph.row_ptr( p );
699 }
700
701 static KOKKOS_FUNCTION size_type
702 getNeighbor( crs_graph_type const& crs_graph, size_type p, size_type n )
703 {
704 assert( n < numNeighbor( crs_graph, p ) );
705 p -= crs_graph.shift;
706 return crs_graph.col_ind( crs_graph.row_ptr( p ) + n );
707 }
708};
