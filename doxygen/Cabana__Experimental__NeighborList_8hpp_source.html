99template <typename SliceLike>
100struct AccessTraits<SliceLike, PredicatesTag>
101{
103 using memory_space = typename SliceLike::memory_space;
105 using size_type = typename SliceLike::size_type;
107 static KOKKOS_FUNCTION size_type size( SliceLike const& x )
108 {
109 return x.last - x.first;
110 }
112 static KOKKOS_FUNCTION auto get( SliceLike const& x, size_type i )
113 {
114 assert( i < size( x ) );
115 auto const point =
116 AccessTraits<typename SliceLike::slice_type, PrimitivesTag>::get(
117 x.slice, x.first + i );
118 return attach( intersects( Sphere{ point, x.radius } ), (int)i );
119 }
120};

149// Custom callback for ArborX::BVH::query()
150template <typename Tag>
151struct NeighborDiscriminatorCallback
152{
153 template <typename Predicate, typename OutputFunctor>
154 KOKKOS_FUNCTION void operator()( Predicate const& predicate,
155 int primitive_index,
156 OutputFunctor const& out ) const
157 {
158 int const predicate_index = getData( predicate );
159 if ( CollisionFilter<Tag>::keep( predicate_index, primitive_index ) )
160 {
161 out( primitive_index );
162 }
163 }
164};
165
166// Count in the first pass
167template <typename Counts, typename Tag>
168struct NeighborDiscriminatorCallback2D_FirstPass
169{
170 Counts counts;
171 template <typename Predicate>
172 KOKKOS_FUNCTION void operator()( Predicate const& predicate,
173 int primitive_index ) const
174 {
175 int const predicate_index = getData( predicate );
176 if ( CollisionFilter<Tag>::keep( predicate_index, primitive_index ) )
177 {
178 ++counts( predicate_index ); // WARNING see below**
179 }
180 }
181};
182
183// Preallocate and attempt fill in the first pass
184template <typename Counts, typename Neighbors, typename Tag>
185struct NeighborDiscriminatorCallback2D_FirstPass_BufferOptimization
186{
187 Counts counts;
188 Neighbors neighbors;
189 template <typename Predicate>
190 KOKKOS_FUNCTION void operator()( Predicate const& predicate,
191 int primitive_index ) const
192 {
193 int const predicate_index = getData( predicate );
194 auto& count = counts( predicate_index );
195 if ( CollisionFilter<Tag>::keep( predicate_index, primitive_index ) )
196 {
197 if ( count < (int)neighbors.extent( 1 ) )
198 {
199 neighbors( predicate_index, count++ ) =
200 primitive_index; // WARNING see below**
201 }
202 else
203 {
204 count++;
205 }
206 }
207 }
208};
209
210// Fill in the second pass
211template <typename Counts, typename Neighbors, typename Tag>
212struct NeighborDiscriminatorCallback2D_SecondPass
213{
214 Counts counts;
215 Neighbors neighbors;
216 template <typename Predicate>
217 KOKKOS_FUNCTION void operator()( Predicate const& predicate,
218 int primitive_index ) const
219 {
220 int const predicate_index = getData( predicate );
221 auto& count = counts( predicate_index );
222 if ( CollisionFilter<Tag>::keep( predicate_index, primitive_index ) )
223 {
224 assert( count < (int)neighbors.extent( 1 ) );
225 neighbors( predicate_index, count++ ) =
226 primitive_index; // WARNING see below**
227 }
228 }
229};
230
231// NOTE** Taking advantage of the knowledge that one predicate is processed by a
232// single thread. Count increment should be atomic otherwise.

272template <typename DeviceType, typename Slice, typename Tag>
273auto makeNeighborList( Tag, Slice const& coordinate_slice,
274 typename Slice::size_type first,
275 typename Slice::size_type last,
276 typename Slice::value_type radius, int buffer_size = 0 )
277{
278 assert( buffer_size >= 0 );
279 assert( last >= first );
280 assert( last <= coordinate_slice.size() );
281
282 using MemorySpace = typename DeviceType::memory_space;
283 using ExecutionSpace = typename DeviceType::execution_space;
284 ExecutionSpace space{};
285
286 ArborX::BVH<MemorySpace> bvh( space, coordinate_slice );
287
288 Kokkos::View<int*, DeviceType> indices(
289 Kokkos::view_alloc( "indices", Kokkos::WithoutInitializing ), 0 );
290 Kokkos::View<int*, DeviceType> offset(
291 Kokkos::view_alloc( "offset", Kokkos::WithoutInitializing ), 0 );
292 bvh.query(
293 space, Impl::makePredicates( coordinate_slice, first, last, radius ),
294 Impl::NeighborDiscriminatorCallback<Tag>{}, indices, offset,
295 ArborX::Experimental::TraversalPolicy().setBufferSize( buffer_size ) );
296
297 return CrsGraph<MemorySpace, Tag>{ std::move( indices ),
298 std::move( offset ), first, bvh.size() };
299}

337template <typename DeviceType, typename Slice, typename Tag>
338auto make2DNeighborList( Tag, Slice const& coordinate_slice,
339 typename Slice::size_type first,
340 typename Slice::size_type last,
341 typename Slice::value_type radius,
342 int buffer_size = 0 )
343{
344 assert( buffer_size >= 0 );
345 assert( last >= first );
346 assert( last <= coordinate_slice.size() );
347
348 using MemorySpace = typename DeviceType::memory_space;
349 using ExecutionSpace = typename DeviceType::execution_space;
350 ExecutionSpace space{};
351
352 ArborX::BVH<MemorySpace> bvh( space, coordinate_slice );
353
354 auto const predicates =
355 Impl::makePredicates( coordinate_slice, first, last, radius );
356
357 auto const n_queries =
358 ArborX::AccessTraits<decltype( predicates ),
359 ArborX::PredicatesTag>::size( predicates );
360
361 Kokkos::View<int**, DeviceType> neighbors;
362 Kokkos::View<int*, DeviceType> counts( "counts", n_queries );
363 if ( buffer_size > 0 )
364 {
365 neighbors = Kokkos::View<int**, DeviceType>(
366 Kokkos::view_alloc( "neighbors", Kokkos::WithoutInitializing ),
367 n_queries, buffer_size );
368 bvh.query(
369 space, predicates,
370 Impl::NeighborDiscriminatorCallback2D_FirstPass_BufferOptimization<
371 decltype( counts ), decltype( neighbors ), Tag>{ counts,
372 neighbors } );
373 }
374 else
375 {
376 bvh.query(
377 space, predicates,
378 Impl::NeighborDiscriminatorCallback2D_FirstPass<decltype( counts ),
379 Tag>{ counts } );
380 }
381
382 auto const max_neighbors = ArborX::max( space, counts );
383 if ( max_neighbors <= buffer_size )
384 {
385 // NOTE We do not bother shrinking to eliminate the excess allocation.
386 // NOTE If buffer_size is 0, neighbors is default constructed. This is
387 // fine with the current design/implementation of NeighborList access
388 // traits.
389 return Dense<MemorySpace, Tag>{ counts, neighbors, first, bvh.size() };
390 }
391
392 neighbors = Kokkos::View<int**, DeviceType>(
393 Kokkos::view_alloc( "neighbors", Kokkos::WithoutInitializing ),
394 n_queries, max_neighbors ); // realloc storage for neighbors
395 Kokkos::deep_copy( counts, 0 ); // reset counts to zero
396 bvh.query( space, predicates,
397 Impl::NeighborDiscriminatorCallback2D_SecondPass<
398 decltype( counts ), decltype( neighbors ), Tag>{
399 counts, neighbors } );
400
401 return Dense<MemorySpace, Tag>{ counts, neighbors, first, bvh.size() };
402}

407template <typename MemorySpace, typename Tag>
408class NeighborList<Experimental::CrsGraph<MemorySpace, Tag>>
409{
411 using size_type = std::size_t;
413 using crs_graph_type = Experimental::CrsGraph<MemorySpace, Tag>;
414
415 public:
417 using memory_space = MemorySpace;
419 static KOKKOS_FUNCTION size_type
420 numNeighbor( crs_graph_type const& crs_graph, size_type p )
421 {
422 assert( (int)p >= 0 && p < crs_graph.total );
423 p -= crs_graph.shift;
424 if ( (int)p < 0 || p >= crs_graph.row_ptr.size() - 1 )
425 return 0;
426 return crs_graph.row_ptr( p + 1 ) - crs_graph.row_ptr( p );
427 }
429 static KOKKOS_FUNCTION size_type
430 getNeighbor( crs_graph_type const& crs_graph, size_type p, size_type n )
431 {
432 assert( n < numNeighbor( crs_graph, p ) );
433 p -= crs_graph.shift;
434 return crs_graph.col_ind( crs_graph.row_ptr( p ) + n );
435 }
436};

439template <typename MemorySpace, typename Tag>
440class NeighborList<Experimental::Dense<MemorySpace, Tag>>
441{
443 using size_type = std::size_t;
445 using specialization_type = Experimental::Dense<MemorySpace, Tag>;
446
447 public:
449 using memory_space = MemorySpace;
451 static KOKKOS_FUNCTION size_type numNeighbor( specialization_type const& d,
452 size_type p )
453 {
454 assert( (int)p >= 0 && p < d.total );
455 p -= d.shift;
456 if ( (int)p < 0 || p >= d.cnt.size() )
457 return 0;
458 return d.cnt( p );
459 }
461 static KOKKOS_FUNCTION size_type getNeighbor( specialization_type const& d,
462 size_type p, size_type n )
463 {
464 assert( n < numNeighbor( d, p ) );
465 p -= d.shift;
466 return d.val( p, n );
467 }
468};
