149// Custom callback for ArborX::BVH::query()
150template <typename Tag>
151struct NeighborDiscriminatorCallback
152{
153 template <typename Predicate, typename OutputFunctor>
154 KOKKOS_FUNCTION void operator()( Predicate const& predicate,
155 int primitive_index,
156 OutputFunctor const& out ) const
157 {
158 int const predicate_index = getData( predicate );
159 if ( CollisionFilter<Tag>::keep( predicate_index, primitive_index ) )
160 {
161 out( primitive_index );
162 }
163 }
164};
165
166// Count in the first pass
167template <typename Counts, typename Tag>
168struct NeighborDiscriminatorCallback2D_FirstPass
169{
170 Counts counts;
171 template <typename Predicate>
172 KOKKOS_FUNCTION void operator()( Predicate const& predicate,
173 int primitive_index ) const
174 {
175 int const predicate_index = getData( predicate );
176 if ( CollisionFilter<Tag>::keep( predicate_index, primitive_index ) )
177 {
178 ++counts( predicate_index ); // WARNING see below**
179 }
180 }
181};
182
183// Preallocate and attempt fill in the first pass
184template <typename Counts, typename Neighbors, typename Tag>
185struct NeighborDiscriminatorCallback2D_FirstPass_BufferOptimization
186{
187 Counts counts;
188 Neighbors neighbors;
189 template <typename Predicate>
190 KOKKOS_FUNCTION void operator()( Predicate const& predicate,
191 int primitive_index ) const
192 {
193 int const predicate_index = getData( predicate );
194 auto& count = counts( predicate_index );
195 if ( CollisionFilter<Tag>::keep( predicate_index, primitive_index ) )
196 {
197 if ( count < (int)neighbors.extent( 1 ) )
198 {
199 neighbors( predicate_index, count++ ) =
200 primitive_index; // WARNING see below**
201 }
202 else
203 {
204 count++;
205 }
206 }
207 }
208};
209
210// Fill in the second pass
211template <typename Counts, typename Neighbors, typename Tag>
212struct NeighborDiscriminatorCallback2D_SecondPass
213{
214 Counts counts;
215 Neighbors neighbors;
216 template <typename Predicate>
217 KOKKOS_FUNCTION void operator()( Predicate const& predicate,
218 int primitive_index ) const
219 {
220 int const predicate_index = getData( predicate );
221 auto& count = counts( predicate_index );
222 if ( CollisionFilter<Tag>::keep( predicate_index, primitive_index ) )
223 {
224 assert( count < (int)neighbors.extent( 1 ) );
225 neighbors( predicate_index, count++ ) =
226 primitive_index; // WARNING see below**
227 }
228 }
229};
230
231// NOTE** Taking advantage of the knowledge that one predicate is processed by a
232// single thread. Count increment should be atomic otherwise.
233

273template <typename ExecutionSpace, typename Slice, typename Tag>
274auto makeNeighborList( ExecutionSpace space, Tag, Slice const& coordinate_slice,
275 typename Slice::size_type first,
276 typename Slice::size_type last,
277 typename Slice::value_type radius, int buffer_size = 0 )
278{
279 assert( buffer_size >= 0 );
280 assert( last >= first );
281 assert( last <= coordinate_slice.size() );
282
283 using memory_space = typename Slice::memory_space;
284
285 ArborX::BVH<memory_space> bvh( space, coordinate_slice );
286
287 Kokkos::View<int*, memory_space> indices(
288 Kokkos::view_alloc( "indices", Kokkos::WithoutInitializing ), 0 );
289 Kokkos::View<int*, memory_space> offset(
290 Kokkos::view_alloc( "offset", Kokkos::WithoutInitializing ), 0 );
291 bvh.query(
292 space, Impl::makePredicates( coordinate_slice, first, last, radius ),
293 Impl::NeighborDiscriminatorCallback<Tag>{}, indices, offset,
294 ArborX::Experimental::TraversalPolicy().setBufferSize( buffer_size ) );
295
296 return CrsGraph<memory_space, Tag>{
297 std::move( indices ), std::move( offset ), first, bvh.size() };
298}

396template <typename ExecutionSpace, typename Slice, typename Tag>
397auto make2DNeighborList( ExecutionSpace space, Tag,
398 Slice const& coordinate_slice,
399 typename Slice::size_type first,
400 typename Slice::size_type last,
401 typename Slice::value_type radius,
402 int buffer_size = 0 )
403{
404 assert( buffer_size >= 0 );
405 assert( last >= first );
406 assert( last <= coordinate_slice.size() );
407
408 using memory_space = typename Slice::memory_space;
409
410 ArborX::BVH<memory_space> bvh( space, coordinate_slice );
411
412 auto const predicates =
413 Impl::makePredicates( coordinate_slice, first, last, radius );
414
415 auto const n_queries =
416 ArborX::AccessTraits<decltype( predicates ),
417 ArborX::PredicatesTag>::size( predicates );
418
419 Kokkos::View<int**, memory_space> neighbors;
420 Kokkos::View<int*, memory_space> counts( "counts", n_queries );
421 if ( buffer_size > 0 )
422 {
423 neighbors = Kokkos::View<int**, memory_space>(
424 Kokkos::view_alloc( "neighbors", Kokkos::WithoutInitializing ),
425 n_queries, buffer_size );
426 bvh.query(
427 space, predicates,
428 Impl::NeighborDiscriminatorCallback2D_FirstPass_BufferOptimization<
429 decltype( counts ), decltype( neighbors ), Tag>{ counts,
430 neighbors } );
431 }
432 else
433 {
434 bvh.query(
435 space, predicates,
436 Impl::NeighborDiscriminatorCallback2D_FirstPass<decltype( counts ),
437 Tag>{ counts } );
438 }
439
440 auto const max_neighbors = ArborX::max( space, counts );
441 if ( max_neighbors <= buffer_size )
442 {
443 // NOTE We do not bother shrinking to eliminate the excess allocation.
444 // NOTE If buffer_size is 0, neighbors is default constructed. This is
445 // fine with the current design/implementation of NeighborList access
446 // traits.
447 return Dense<memory_space, Tag>{ counts, neighbors, first, bvh.size() };
448 }
449
450 neighbors = Kokkos::View<int**, memory_space>(
451 Kokkos::view_alloc( "neighbors", Kokkos::WithoutInitializing ),
452 n_queries, max_neighbors ); // realloc storage for neighbors
453 Kokkos::deep_copy( counts, 0 ); // reset counts to zero
454 bvh.query( space, predicates,
455 Impl::NeighborDiscriminatorCallback2D_SecondPass<
456 decltype( counts ), decltype( neighbors ), Tag>{
457 counts, neighbors } );
458
459 return Dense<memory_space, Tag>{ counts, neighbors, first, bvh.size() };
460}

528template <typename MemorySpace, typename Tag>
529class NeighborList<Experimental::CrsGraph<MemorySpace, Tag>>
530{
532 using size_type = std::size_t;
534 using crs_graph_type = Experimental::CrsGraph<MemorySpace, Tag>;
535
536 public:
538 using memory_space = MemorySpace;
539
541 KOKKOS_INLINE_FUNCTION
542 static size_type totalNeighbor( crs_graph_type const& crs_graph )
543 {
544 return Impl::totalNeighbor( crs_graph, crs_graph.total );
545 }
546
548 KOKKOS_INLINE_FUNCTION
549 static size_type maxNeighbor( crs_graph_type const& crs_graph )
550 {
551 return Impl::maxNeighbor( crs_graph, crs_graph.total );
552 }
553
555 static KOKKOS_FUNCTION size_type
556 numNeighbor( crs_graph_type const& crs_graph, size_type p )
557 {
558 assert( (int)p >= 0 && p < crs_graph.total );
559 p -= crs_graph.shift;
560 if ( (int)p < 0 || p >= crs_graph.row_ptr.size() - 1 )
561 return 0;
562 return crs_graph.row_ptr( p + 1 ) - crs_graph.row_ptr( p );
563 }
565 static KOKKOS_FUNCTION size_type
566 getNeighbor( crs_graph_type const& crs_graph, size_type p, size_type n )
567 {
568 assert( n < numNeighbor( crs_graph, p ) );
569 p -= crs_graph.shift;
570 return crs_graph.col_ind( crs_graph.row_ptr( p ) + n );
571 }
572};

603 size_type p )
604 {
605 assert( (int)p >= 0 && p < d.total );
606 p -= d.shift;

615 assert( n < numNeighbor( d, p ) );
616 p -= d.shift;
617 return d.val( p, n );
618 }
