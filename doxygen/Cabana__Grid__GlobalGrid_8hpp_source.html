217template <class Scalar, std::size_t NumSpaceDim>
218std::shared_ptr<GlobalGrid<SparseMesh<Scalar, NumSpaceDim>>> createGlobalGrid(
219 MPI_Comm comm,
220 const std::shared_ptr<GlobalMesh<SparseMesh<Scalar, NumSpaceDim>>>&
221 global_mesh,
222 const std::array<bool, SparseMesh<Scalar, NumSpaceDim>::num_space_dim>&
223 periodic,
224 const BlockPartitioner<SparseMesh<Scalar, NumSpaceDim>::num_space_dim>&
225 partitioner )
226{
227 for ( long unsigned int d = 0; d < NumSpaceDim; ++d )
228 if ( periodic[d] )
229 std::runtime_error(
230 "Sparse grid doesn't support periodic BC so far!" );
231 return std::make_shared<GlobalGrid<SparseMesh<Scalar, NumSpaceDim>>>(
232 comm, global_mesh, periodic, partitioner );
233}
