217 template <class Pattern, class... ArrayTypes>
218 Halo( const Pattern& pattern, const int width, const ArrayTypes&... arrays )
219 {
220 // Spatial dimension.
221 const std::size_t num_space_dim = Pattern::num_space_dim;
222
223 // Get the local grid.
224 auto local_grid = getLocalGrid( arrays... );
225
226 // Function to get the local id of the neighbor.
227 auto neighbor_id = []( const std::array<int, num_space_dim>& ijk )
228 {
229 int id = ijk[0];
230 for ( std::size_t d = 1; d < num_space_dim; ++d )
231 id += num_space_dim * id + ijk[d];
232 return id;
233 };
234
235 // Neighbor id flip function. This lets us compute what neighbor we
236 // are relative to a given neighbor.
237 auto flip_id = [=]( const std::array<int, num_space_dim>& ijk )
238 {
239 std::array<int, num_space_dim> flip_ijk;
240 for ( std::size_t d = 0; d < num_space_dim; ++d )
241 flip_ijk[d] = -ijk[d];
242 return flip_ijk;
243 };
244
245 // Get the neighbor ranks we will exchange with in the halo and
246 // allocate buffers. If any of the exchanges are self sends mark these
247 // so we know which send buffers correspond to which receive buffers.
248 auto neighbors = pattern.getNeighbors();
249 for ( const auto& n : neighbors )
250 {
251 // Get the rank of the neighbor.
252 int rank = local_grid->neighborRank( n );
253
254 // If this is a valid rank add it as a neighbor.
255 if ( rank >= 0 )
256 {
257 // Add the rank.
258 _neighbor_ranks.push_back( rank );
259
260 // Set the tag we will use to send data to this neighbor. The
261 // receiving rank should have a matching tag.
262 _send_tags.push_back( neighbor_id( n ) );
263
264 // Set the tag we will use to receive data from this
265 // neighbor. The sending rank should have a matching tag.
266 _receive_tags.push_back( neighbor_id( flip_id( n ) ) );
267
268 // Create communication data for owned entities.
269 buildCommData( Own(), width, n, _owned_buffers, _owned_steering,
270 arrays... );
271
272 // Create communication data for ghosted entities.
273 buildCommData( Ghost(), width, n, _ghosted_buffers,
274 _ghosted_steering, arrays... );
275 }
276 }
277 }

289 template <class ExecutionSpace, class... ArrayTypes>
290 void gather( const ExecutionSpace& exec_space,
291 const ArrayTypes&... arrays ) const
292 {
293 Kokkos::Profiling::pushRegion( "Cajita::gather" );
294
295 // Get the number of neighbors. Return if we have none.
296 int num_n = _neighbor_ranks.size();
297 if ( 0 == num_n )
298 return;
299
300 // Get the MPI communicator.
301 auto comm = getComm( arrays... );
302
303 // Allocate requests.
304 std::vector<MPI_Request> requests( 2 * num_n, MPI_REQUEST_NULL );
305
306 // Pick a tag to use for communication. This object has its own
307 // communication space so any tag will do.
308 const int mpi_tag = 1234;
309
310 // Post receives.
311 for ( int n = 0; n < num_n; ++n )
312 {
313 // Only process this neighbor if there is work to do.
314 if ( 0 < _ghosted_buffers[n].size() )
315 {
316 MPI_Irecv( _ghosted_buffers[n].data(),
317 _ghosted_buffers[n].size(), MPI_BYTE,
318 _neighbor_ranks[n], mpi_tag + _receive_tags[n], comm,
319 &requests[n] );
320 }
321 }
322
323 // Pack send buffers and post sends.
324 for ( int n = 0; n < num_n; ++n )
325 {
326 // Only process this neighbor if there is work to do.
327 if ( 0 < _owned_buffers[n].size() )
328 {
329 // Pack the send buffer.
330 packBuffer( exec_space, _owned_buffers[n], _owned_steering[n],
331 arrays.view()... );
332
333 // Post a send.
334 MPI_Isend( _owned_buffers[n].data(), _owned_buffers[n].size(),
335 MPI_BYTE, _neighbor_ranks[n],
336 mpi_tag + _send_tags[n], comm,
337 &requests[num_n + n] );
338 }
339 }
340
341 // Unpack receive buffers.
342 bool unpack_complete = false;
343 while ( !unpack_complete )
344 {
345 // Get the next buffer to unpack.
346 int unpack_index = MPI_UNDEFINED;
347 MPI_Waitany( num_n, requests.data(), &unpack_index,
348 MPI_STATUS_IGNORE );
349
350 // If there are no more buffers to unpack we are done.
351 if ( MPI_UNDEFINED == unpack_index )
352 {
353 unpack_complete = true;
354 }
355
356 // Otherwise unpack the next buffer.
357 else
358 {
359 unpackBuffer( ScatterReduce::Replace(), exec_space,
360 _ghosted_buffers[unpack_index],
361 _ghosted_steering[unpack_index],
362 arrays.view()... );
363 }
364 }
365
366 // Wait on send requests.
367 MPI_Waitall( num_n, requests.data() + num_n, MPI_STATUSES_IGNORE );
368 Kokkos::Profiling::popRegion();
369 }

378 template <class ExecutionSpace, class ReduceOp, class... ArrayTypes>
379 void scatter( const ExecutionSpace& exec_space, const ReduceOp& reduce_op,
380 const ArrayTypes&... arrays ) const
381 {
382 Kokkos::Profiling::pushRegion( "Cajita::scatter" );
383
384 // Get the number of neighbors. Return if we have none.
385 int num_n = _neighbor_ranks.size();
386 if ( 0 == num_n )
387 return;
388
389 // Get the MPI communicator.
390 auto comm = getComm( arrays... );
391
392 // Requests.
393 std::vector<MPI_Request> requests( 2 * num_n, MPI_REQUEST_NULL );
394
395 // Pick a tag to use for communication. This object has its own
396 // communication space so any tag will do.
397 const int mpi_tag = 2345;
398
399 // Post receives for all neighbors that are not self sends.
400 for ( int n = 0; n < num_n; ++n )
401 {
402 // Only process this neighbor if there is work to do.
403 if ( 0 < _owned_buffers[n].size() )
404 {
405 MPI_Irecv( _owned_buffers[n].data(), _owned_buffers[n].size(),
406 MPI_BYTE, _neighbor_ranks[n],
407 mpi_tag + _receive_tags[n], comm, &requests[n] );
408 }
409 }
410
411 // Pack send buffers and post sends.
412 for ( int n = 0; n < num_n; ++n )
413 {
414 // Only process this neighbor if there is work to do.
415 if ( 0 < _ghosted_buffers[n].size() )
416 {
417 // Pack the send buffer.
418 packBuffer( exec_space, _ghosted_buffers[n],
419 _ghosted_steering[n], arrays.view()... );
420
421 // Post a send.
422 MPI_Isend( _ghosted_buffers[n].data(),
423 _ghosted_buffers[n].size(), MPI_BYTE,
424 _neighbor_ranks[n], mpi_tag + _send_tags[n], comm,
425 &requests[num_n + n] );
426 }
427 }
428
429 // Unpack receive buffers.
430 bool unpack_complete = false;
431 while ( !unpack_complete )
432 {
433 // Get the next buffer to unpack.
434 int unpack_index = MPI_UNDEFINED;
435 MPI_Waitany( num_n, requests.data(), &unpack_index,
436 MPI_STATUS_IGNORE );
437
438 // If there are no more buffers to unpack we are done.
439 if ( MPI_UNDEFINED == unpack_index )
440 {
441 unpack_complete = true;
442 }
443
444 // Otherwise unpack the next buffer and apply the reduce operation.
445 else
446 {
447 unpackBuffer( reduce_op, exec_space,
448 _owned_buffers[unpack_index],
449 _owned_steering[unpack_index], arrays.view()... );
450 }
451
452 // Wait on send requests.
453 MPI_Waitall( num_n, requests.data() + num_n, MPI_STATUSES_IGNORE );
454 }
455 Kokkos::Profiling::popRegion();
456 }

467 template <class Array_t, class... ArrayTypes>
468 MPI_Comm getComm( const Array_t& array, const ArrayTypes&... arrays ) const
469 {
470 auto comm = getComm( array );
471
472 // Check that the communicator of this array is the same as the other
473 // arrays.
474 int result;
475 MPI_Comm_compare( comm, getComm( arrays... ), &result );
476 if ( result != MPI_IDENT && result != MPI_CONGRUENT )
477 throw std::runtime_error( "Arrays have different communicators" );
478
479 return comm;

509 template <class DecompositionTag, std::size_t NumSpaceDim,
510 class... ArrayTypes>
511 void
512 buildCommData( DecompositionTag decomposition_tag, const int width,
513 const std::array<int, NumSpaceDim>& nid,
514 std::vector<Kokkos::View<char*, memory_space>>& buffers,
515 std::vector<Kokkos::View<int**, memory_space>>& steering,
516 const ArrayTypes&... arrays )
517 {
518 // Number of arrays.
519 const std::size_t num_array = sizeof...( ArrayTypes );
520
521 // Get the byte sizes of array value types.
522 std::array<std::size_t, num_array> value_byte_sizes = {
523 sizeof( typename ArrayTypes::value_type )... };
524
525 // Get the index spaces we share with this neighbor. We
526 // get a shared index space for each array.
527 std::array<IndexSpace<NumSpaceDim + 1>, num_array> spaces = {
528 ( arrays.layout()->sharedIndexSpace( decomposition_tag, nid,
529 width ) )... };
530
531 // Compute the buffer size of this neighbor and the
532 // number of elements in the buffer.
533 int buffer_bytes = 0;
534 int buffer_num_element = 0;
535 for ( std::size_t a = 0; a < num_array; ++a )
536 {
537 buffer_bytes += value_byte_sizes[a] * spaces[a].size();
538 buffer_num_element += spaces[a].size();
539 }
540
541 // Allocate the buffer of data that we share with this neighbor. All
542 // arrays will be packed into a single buffer.
543 buffers.push_back(
544 Kokkos::View<char*, memory_space>( "halo_buffer", buffer_bytes ) );
545
546 // Allocate the steering vector for building the buffer.
547 steering.push_back( Kokkos::View<int**, memory_space>(
548 "steering", buffer_num_element, 3 + NumSpaceDim ) );
549
550 // Build steering vector.
551 buildSteeringVector( spaces, value_byte_sizes, buffer_bytes,
552 buffer_num_element, steering );
553 }

556 template <std::size_t NumArray>
557 void buildSteeringVector(
558 const std::array<IndexSpace<4>, NumArray>& spaces,
559 const std::array<std::size_t, NumArray>& value_byte_sizes,
560 const int buffer_bytes, const int buffer_num_element,
561 std::vector<Kokkos::View<int**, memory_space>>& steering )
562 {
563 // Create the steering vector. For each element in the buffer it gives
564 // the starting byte location of the element, the array the element is
565 // in, and the ijkl structured index in the array of the element.
566 auto host_steering =
567 Kokkos::create_mirror_view( Kokkos::HostSpace(), steering.back() );
568 int elem_counter = 0;
569 int byte_counter = 0;
570 for ( std::size_t a = 0; a < NumArray; ++a )
571 {
572 for ( int i = spaces[a].min( 0 ); i < spaces[a].max( 0 ); ++i )
573 {
574 for ( int j = spaces[a].min( 1 ); j < spaces[a].max( 1 ); ++j )
575 {
576 for ( int k = spaces[a].min( 2 ); k < spaces[a].max( 2 );
577 ++k )
578 {
579 for ( int l = spaces[a].min( 3 );
580 l < spaces[a].max( 3 ); ++l )
581 {
582 // Byte starting location in buffer.
583 host_steering( elem_counter, 0 ) = byte_counter;
584
585 // Array location of element.
586 host_steering( elem_counter, 1 ) = a;
587
588 // Structured index in array of element.
589 host_steering( elem_counter, 2 ) = i;
590 host_steering( elem_counter, 3 ) = j;
591 host_steering( elem_counter, 4 ) = k;
592 host_steering( elem_counter, 5 ) = l;
593
594 // Update element id.
595 ++elem_counter;
596
597 // Update buffer position.
598 byte_counter += value_byte_sizes[a];
599 }
600 }
601 }
602 }
603 }
604
605 // Check that all elements and bytes are accounted for.
606 if ( byte_counter != buffer_bytes )
607 throw std::logic_error( "Steering vector contains different number "
608 "of bytes than buffer" );
609 if ( elem_counter != buffer_num_element )
610 throw std::logic_error( "Steering vector contains different number "
611 "of elements than buffer" );
612
613 // Copy steering vector to device.
614 Kokkos::deep_copy( steering.back(), host_steering );
615 }

829 KOKKOS_INLINE_FUNCTION static std::enable_if_t<3 == ArrayView::rank, void>
830 unpackElement( const ReduceOp& reduce_op,
831 const Kokkos::View<char*, memory_space>& buffer,
832 const Kokkos::View<int**, memory_space>& steering,
833 const int element_idx, const ArrayView& array_view )
834 {
835 typename ArrayView::value_type elem;
836 char* elem_ptr = reinterpret_cast<char*>( &elem );
837 for ( std::size_t b = 0; b < sizeof( typename ArrayView::value_type );
838 ++b )
839 {
840 *( elem_ptr + b ) = buffer( steering( element_idx, 0 ) + b );
841 }
842 unpackOp( reduce_op, elem,
843 array_view( steering( element_idx, 2 ),
844 steering( element_idx, 3 ),
845 steering( element_idx, 4 ) ) );
846 }

849 template <class ReduceOp, class... ArrayViews>
850 KOKKOS_INLINE_FUNCTION static void
851 unpackArray( const ReduceOp& reduce_op,
852 const Kokkos::View<char*, memory_space>& buffer,
853 const Kokkos::View<int**, memory_space>& steering,
854 const int element_idx,
855 const std::integral_constant<std::size_t, 0>,
856 const Cabana::ParameterPack<ArrayViews...>& array_views )
857 {
858 // If the unpack element_idx is in the current array, unpack it.
859 if ( 0 == steering( element_idx, 1 ) )
860 unpackElement( reduce_op, buffer, steering, element_idx,
861 Cabana::get<0>( array_views ) );
862 }

865 template <class ReduceOp, std::size_t N, class... ArrayViews>
866 KOKKOS_INLINE_FUNCTION static void
867 unpackArray( const ReduceOp reduce_op,
868 const Kokkos::View<char*, memory_space>& buffer,
869 const Kokkos::View<int**, memory_space>& steering,
870 const int element_idx,
871 const std::integral_constant<std::size_t, N>,
872 const Cabana::ParameterPack<ArrayViews...>& array_views )
873 {
874 // If the unpack element_idx is in the current array, unpack it.
875 if ( N == steering( element_idx, 1 ) )
876 {
877 unpackElement( reduce_op, buffer, steering, element_idx,
878 Cabana::get<N>( array_views ) );
879 return;
880 }
881
882 // Recurse.
883 unpackArray( reduce_op, buffer, steering, element_idx,
884 std::integral_constant<std::size_t, N - 1>(),
885 array_views );
886 }

889 template <class ExecutionSpace, class ReduceOp, class... ArrayViews>
890 void unpackBuffer( const ReduceOp& reduce_op,
891 const ExecutionSpace& exec_space,
892 const Kokkos::View<char*, memory_space>& buffer,
893 const Kokkos::View<int**, memory_space>& steering,
894 ArrayViews... array_views ) const
895 {
896 auto pp = Cabana::makeParameterPack( array_views... );
897 Kokkos::parallel_for(
898 "Cajita::Halo::unpack_buffer",
899 Kokkos::RangePolicy<ExecutionSpace>( exec_space, 0,
900 steering.extent( 0 ) ),
901 KOKKOS_LAMBDA( const int i ) {
902 unpackArray(
903 reduce_op, buffer, steering, i,
904 std::integral_constant<std::size_t,
905 sizeof...( ArrayViews ) - 1>(),
906 pp );
907 } );
908 }
