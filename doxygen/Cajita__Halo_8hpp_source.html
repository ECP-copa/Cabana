217  template <class Pattern, class... ArrayTypes>
218  Halo( const Pattern& pattern, const int width, const ArrayTypes&... arrays )
219  {
220  // Spatial dimension.
221  const std::size_t num_space_dim = Pattern::num_space_dim;
222 
223  // Get the MPI communicator. All arrays must have the same
224  // communicator.
225  getComm( arrays... );
226 
227  // Get the local grid.
228  auto local_grid = getLocalGrid( arrays... );
229 
230  // Function to get the local id of the neighbor.
231  auto neighbor_id = []( const std::array<int, num_space_dim>& ijk )
232  {
233  int id = ijk[0];
234  for ( std::size_t d = 1; d < num_space_dim; ++d )
235  id += num_space_dim * id + ijk[d];
236  return id;
237  };
238 
239  // Neighbor id flip function. This lets us compute what neighbor we
240  // are relative to a given neighbor.
241  auto flip_id = [=]( const std::array<int, num_space_dim>& ijk )
242  {
243  std::array<int, num_space_dim> flip_ijk;
244  for ( std::size_t d = 0; d < num_space_dim; ++d )
245  flip_ijk[d] = -ijk[d];
246  return flip_ijk;
247  };
248 
249  // Get the neighbor ranks we will exchange with in the halo and
250  // allocate buffers. If any of the exchanges are self sends mark these
251  // so we know which send buffers correspond to which receive buffers.
252  auto neighbors = pattern.getNeighbors();
253  for ( const auto& n : neighbors )
254  {
255  // Get the rank of the neighbor.
256  int rank = local_grid->neighborRank( n );
257 
258  // If this is a valid rank add it as a neighbor.
259  if ( rank >= 0 )
260  {
261  // Add the rank.
262  _neighbor_ranks.push_back( rank );
263 
264  // Set the tag we will use to send data to this neighbor. The
265  // receiving rank should have a matching tag.
266  _send_tags.push_back( neighbor_id( n ) );
267 
268  // Set the tag we will use to receive data from this
269  // neighbor. The sending rank should have a matching tag.
270  _receive_tags.push_back( neighbor_id( flip_id( n ) ) );
271 
272  // Create communication data for owned entities.
273  buildCommData( Own(), width, n, _owned_buffers, _owned_steering,
274  arrays... );
275 
276  // Create communication data for ghosted entities.
277  buildCommData( Ghost(), width, n, _ghosted_buffers,
278  _ghosted_steering, arrays... );
279  }
280  }
281  }

284  ~Halo() { MPI_Comm_free( &_comm ); }

296  template <class ExecutionSpace, class... ArrayTypes>
297  void gather( const ExecutionSpace& exec_space,
298  const ArrayTypes&... arrays ) const
299  {
300  // Get the number of neighbors. Return if we have none.
301  int num_n = _neighbor_ranks.size();
302  if ( 0 == num_n )
303  return;
304 
305  // Allocate requests.
306  std::vector<MPI_Request> requests( 2 * num_n, MPI_REQUEST_NULL );
307 
308  // Pick a tag to use for communication. This object has its own
309  // communication space so any tag will do.
310  const int mpi_tag = 1234;
311 
312  // Post receives.
313  for ( int n = 0; n < num_n; ++n )
314  {
315  // Only process this neighbor if there is work to do.
316  if ( 0 < _ghosted_buffers[n].size() )
317  {
318  MPI_Irecv( _ghosted_buffers[n].data(),
319  _ghosted_buffers[n].size(), MPI_BYTE,
320  _neighbor_ranks[n], mpi_tag + _receive_tags[n],
321  _comm, &requests[n] );
322  }
323  }
324 
325  // Pack send buffers and post sends.
326  for ( int n = 0; n < num_n; ++n )
327  {
328  // Only process this neighbor if there is work to do.
329  if ( 0 < _owned_buffers[n].size() )
330  {
331  // Pack the send buffer.
332  packBuffer( exec_space, _owned_buffers[n], _owned_steering[n],
333  arrays.view()... );
334 
335  // Post a send.
336  MPI_Isend( _owned_buffers[n].data(), _owned_buffers[n].size(),
337  MPI_BYTE, _neighbor_ranks[n],
338  mpi_tag + _send_tags[n], _comm,
339  &requests[num_n + n] );
340  }
341  }
342 
343  // Unpack receive buffers.
344  bool unpack_complete = false;
345  while ( !unpack_complete )
346  {
347  // Get the next buffer to unpack.
348  int unpack_index = MPI_UNDEFINED;
349  MPI_Waitany( num_n, requests.data(), &unpack_index,
350  MPI_STATUS_IGNORE );
351 
352  // If there are no more buffers to unpack we are done.
353  if ( MPI_UNDEFINED == unpack_index )
354  {
355  unpack_complete = true;
356  }
357 
358  // Otherwise unpack the next buffer.
359  else
360  {
361  unpackBuffer( ScatterReduce::Replace(), exec_space,
362  _ghosted_buffers[unpack_index],
363  _ghosted_steering[unpack_index],
364  arrays.view()... );
365  }
366  }
367 
368  // Wait on send requests.
369  MPI_Waitall( num_n, requests.data() + num_n, MPI_STATUSES_IGNORE );
370  }

506  template <class DecompositionTag, std::size_t NumSpaceDim,
507  class... ArrayTypes>
508  void
509  buildCommData( DecompositionTag decomposition_tag, const int width,
510  const std::array<int, NumSpaceDim>& nid,
511  std::vector<Kokkos::View<char*, memory_space>>& buffers,
512  std::vector<Kokkos::View<int**, memory_space>>& steering,
513  const ArrayTypes&... arrays )
514  {
515  // Number of arrays.
516  const std::size_t num_array = sizeof...( ArrayTypes );
517 
518  // Get the byte sizes of array value types.
519  std::array<std::size_t, num_array> value_byte_sizes = {
520  sizeof( typename ArrayTypes::value_type )... };
521 
522  // Get the index spaces we share with this neighbor. We
523  // get a shared index space for each array.
524  std::array<IndexSpace<NumSpaceDim + 1>, num_array> spaces = {
525  ( arrays.layout()->sharedIndexSpace( decomposition_tag, nid,
526  width ) )... };
527 
528  // Compute the buffer size of this neighbor and the
529  // number of elements in the buffer.
530  int buffer_bytes = 0;
531  int buffer_num_element = 0;
532  for ( std::size_t a = 0; a < num_array; ++a )
533  {
534  buffer_bytes += value_byte_sizes[a] * spaces[a].size();
535  buffer_num_element += spaces[a].size();
536  }
537 
538  // Allocate the buffer of data that we share with this neighbor. All
539  // arrays will be packed into a single buffer.
540  buffers.push_back(
541  Kokkos::View<char*, memory_space>( "halo_buffer", buffer_bytes ) );
542 
543  // Allocate the steering vector for building the buffer.
544  steering.push_back( Kokkos::View<int**, memory_space>(
545  "steering", buffer_num_element, 3 + NumSpaceDim ) );
546 
547  // Build steering vector.
548  buildSteeringVector( spaces, value_byte_sizes, buffer_bytes,
549  buffer_num_element, steering );
550  }

553  template <std::size_t NumArray>
554  void buildSteeringVector(
555  const std::array<IndexSpace<4>, NumArray>& spaces,
556  const std::array<std::size_t, NumArray>& value_byte_sizes,
557  const int buffer_bytes, const int buffer_num_element,
558  std::vector<Kokkos::View<int**, memory_space>>& steering )
559  {
560  // Create the steering vector. For each element in the buffer it gives
561  // the starting byte location of the element, the array the element is
562  // in, and the ijkl structured index in the array of the element.
563  auto host_steering =
564  Kokkos::create_mirror_view( Kokkos::HostSpace(), steering.back() );
565  int elem_counter = 0;
566  int byte_counter = 0;
567  for ( std::size_t a = 0; a < NumArray; ++a )
568  {
569  for ( int i = spaces[a].min( 0 ); i < spaces[a].max( 0 ); ++i )
570  {
571  for ( int j = spaces[a].min( 1 ); j < spaces[a].max( 1 ); ++j )
572  {
573  for ( int k = spaces[a].min( 2 ); k < spaces[a].max( 2 );
574  ++k )
575  {
576  for ( int l = spaces[a].min( 3 );
577  l < spaces[a].max( 3 ); ++l )
578  {
579  // Byte starting location in buffer.
580  host_steering( elem_counter, 0 ) = byte_counter;
581 
582  // Array location of element.
583  host_steering( elem_counter, 1 ) = a;
584 
585  // Structured index in array of element.
586  host_steering( elem_counter, 2 ) = i;
587  host_steering( elem_counter, 3 ) = j;
588  host_steering( elem_counter, 4 ) = k;
589  host_steering( elem_counter, 5 ) = l;
590 
591  // Update element id.
592  ++elem_counter;
593 
594  // Update buffer position.
595  byte_counter += value_byte_sizes[a];
596  }
597  }
598  }
599  }
600  }
601 
602  // Check that all elements and bytes are accounted for.
603  if ( byte_counter != buffer_bytes )
604  throw std::logic_error( "Steering vector contains different number "
605  "of bytes than buffer" );
606  if ( elem_counter != buffer_num_element )
607  throw std::logic_error( "Steering vector contains different number "
608  "of elements than buffer" );
609 
610  // Copy steering vector to device.
611  Kokkos::deep_copy( steering.back(), host_steering );
612  }

863  KOKKOS_INLINE_FUNCTION static void
864  unpackArray( const ReduceOp reduce_op,
865  const Kokkos::View<char*, memory_space>& buffer,
866  const Kokkos::View<int**, memory_space>& steering,
867  const int element_idx,
868  const std::integral_constant<std::size_t, N>,
869  const Cabana::ParameterPack<ArrayViews...>& array_views )
870  {
871  // If the unpack element_idx is in the current array, unpack it.
872  if ( N == steering( element_idx, 1 ) )
873  {
874  unpackElement( reduce_op, buffer, steering, element_idx,
875  Cabana::get<N>( array_views ) );
876  return;
877  }
878 
879  // Recurse.
880  unpackArray( reduce_op, buffer, steering, element_idx,
881  std::integral_constant<std::size_t, N - 1>(),
882  array_views );
883  }

890  const Kokkos::View<int**, memory_space>& steering,

902  sizeof...( ArrayViews ) - 1>(),

904  } );
