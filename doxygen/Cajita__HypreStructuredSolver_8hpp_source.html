52  template <class ArrayLayout_t>
53  HypreStructuredSolver( const ArrayLayout_t& layout,
54  const bool is_preconditioner = false )
55  : _comm( layout.localGrid()->globalGrid().comm() )
56  , _is_preconditioner( is_preconditioner )
57  {
58  static_assert( is_array_layout<ArrayLayout_t>::value,
59  "Must use an array layout" );
60  static_assert(
61  std::is_same<typename ArrayLayout_t::entity_type,
62  entity_type>::value,
63  "Array layout entity type mush match solver entity type" );
64 
65  // Spatial dimension.
66  const std::size_t num_space_dim = ArrayLayout_t::num_space_dim;
67 
68  // Only create data structures if this is not a preconditioner.
69  if ( !_is_preconditioner )
70  {
71  // Create the grid.
72  auto error = HYPRE_StructGridCreate( _comm, num_space_dim, &_grid );
73  checkHypreError( error );
74 
75  // Get the global index space spanned by the local grid on this
76  // rank. Note that the upper bound is not a bound but rather the
77  // last index as this is what Hypre wants. Note that we reordered
78  // this to KJI from IJK to be consistent with HYPRE ordering. By
79  // setting up the grid like this, HYPRE will then want layout-right
80  // data indexed as (i,j,k) or (i,j,k,l) which will allow us to
81  // directly use Kokkos::deep_copy to move data between Cajita arrays
82  // and HYPRE data structures.
83  auto global_space = layout.indexSpace( Own(), Global() );
84  _lower.resize( num_space_dim );
85  _upper.resize( num_space_dim );
86  for ( std::size_t d = 0; d < num_space_dim; ++d )
87  {
88  _lower[d] = static_cast<HYPRE_Int>(
89  global_space.min( num_space_dim - d - 1 ) );
90  _upper[d] = static_cast<HYPRE_Int>(
91  global_space.max( num_space_dim - d - 1 ) - 1 );
92  }
93  error = HYPRE_StructGridSetExtents( _grid, _lower.data(),
94  _upper.data() );
95  checkHypreError( error );
96 
97  // Get periodicity. Note we invert the order of this to KJI as well.
98  const auto& global_grid = layout.localGrid()->globalGrid();
99  HYPRE_Int periodic[num_space_dim];
100  for ( std::size_t d = 0; d < num_space_dim; ++d )
101  periodic[num_space_dim - 1 - d] =
102  global_grid.isPeriodic( d )
103  ? layout.localGrid()->globalGrid().globalNumEntity(
104  EntityType(), d )
105  : 0;
106  error = HYPRE_StructGridSetPeriodic( _grid, periodic );
107  checkHypreError( error );
108 
109  // Assemble the grid.
110  error = HYPRE_StructGridAssemble( _grid );
111  checkHypreError( error );
112 
113  // Allocate LHS and RHS vectors and initialize to zero. Note that we
114  // are fixing the views under these vectors to layout-right.
115  std::array<long, num_space_dim> reorder_size;
116  for ( std::size_t d = 0; d < num_space_dim; ++d )
117  {
118  reorder_size[d] = global_space.extent( d );
119  }
120  IndexSpace<num_space_dim> reorder_space( reorder_size );
121  auto vector_values =
122  createView<HYPRE_Complex, Kokkos::LayoutRight,
123  Kokkos::HostSpace>( "vector_values", reorder_space );
124  Kokkos::deep_copy( vector_values, 0.0 );
125 
126  error = HYPRE_StructVectorCreate( _comm, _grid, &_b );
127  checkHypreError( error );
128  error = HYPRE_StructVectorInitialize( _b );
129  checkHypreError( error );
130  error = HYPRE_StructVectorSetBoxValues(
131  _b, _lower.data(), _upper.data(), vector_values.data() );
132  checkHypreError( error );
133  error = HYPRE_StructVectorAssemble( _b );
134  checkHypreError( error );
135 
136  error = HYPRE_StructVectorCreate( _comm, _grid, &_x );
137  checkHypreError( error );
138  error = HYPRE_StructVectorInitialize( _x );
139  checkHypreError( error );
140  error = HYPRE_StructVectorSetBoxValues(
141  _x, _lower.data(), _upper.data(), vector_values.data() );
142  checkHypreError( error );
143  error = HYPRE_StructVectorAssemble( _x );
144  checkHypreError( error );
145  }
146  }

173  template <std::size_t NumSpaceDim>
174  void
175  setMatrixStencil( const std::vector<std::array<int, NumSpaceDim>>& stencil,
176  const bool is_symmetric = false )
177  {
178  // This function is only valid for non-preconditioners.
179  if ( _is_preconditioner )
180  throw std::logic_error(
181  "Cannot call setMatrixStencil() on preconditioners" );
182 
183  // Create the stencil.
184  _stencil_size = stencil.size();
185  auto error =
186  HYPRE_StructStencilCreate( NumSpaceDim, _stencil_size, &_stencil );
187  checkHypreError( error );
188  std::array<HYPRE_Int, NumSpaceDim> offset;
189  for ( unsigned n = 0; n < stencil.size(); ++n )
190  {
191  for ( std::size_t d = 0; d < NumSpaceDim; ++d )
192  offset[d] = stencil[n][d];
193  error = HYPRE_StructStencilSetElement( _stencil, n, offset.data() );
194  checkHypreError( error );
195  }
196 
197  // Create the matrix.
198  error = HYPRE_StructMatrixCreate( _comm, _grid, _stencil, &_A );
199  checkHypreError( error );
200  error = HYPRE_StructMatrixSetSymmetric( _A, is_symmetric );
201  checkHypreError( error );
202  }

212  template <class Array_t>
213  void setMatrixValues( const Array_t& values )
214  {
215  static_assert( is_array<Array_t>::value, "Must use an array" );
216  static_assert(
217  std::is_same<typename Array_t::entity_type, entity_type>::value,
218  "Array entity type mush match solver entity type" );
219  static_assert(
220  std::is_same<typename Array_t::device_type, DeviceType>::value,
221  "Array device type and solver device type are different." );
222 
223  static_assert(
224  std::is_same<typename Array_t::value_type, value_type>::value,
225  "Array value type and solver value type are different." );
226 
227  // This function is only valid for non-preconditioners.
228  if ( _is_preconditioner )
229  throw std::logic_error(
230  "Cannot call setMatrixValues() on preconditioners" );
231 
232  if ( values.layout()->dofsPerEntity() !=
233  static_cast<int>( _stencil_size ) )
234  throw std::runtime_error(
235  "Number of matrix values does not match stencil size" );
236 
237  // Spatial dimension.
238  const std::size_t num_space_dim = Array_t::num_space_dim;
239 
240  // Intialize the matrix for setting values.
241  auto error = HYPRE_StructMatrixInitialize( _A );
242  checkHypreError( error );
243 
244  // Get a view of the matrix values on the host.
245  auto values_mirror = Kokkos::create_mirror_view_and_copy(
246  Kokkos::HostSpace(), values.view() );
247 
248  // Copy the matrix entries into HYPRE. The HYPRE layout is fixed as
249  // layout-right.
250  auto owned_space = values.layout()->indexSpace( Own(), Local() );
251  std::array<long, num_space_dim + 1> reorder_size;
252  for ( std::size_t d = 0; d < num_space_dim; ++d )
253  {
254  reorder_size[d] = owned_space.extent( d );
255  }
256  reorder_size.back() = _stencil_size;
257  IndexSpace<num_space_dim + 1> reorder_space( reorder_size );
258  auto a_values =
259  createView<HYPRE_Complex, Kokkos::LayoutRight, Kokkos::HostSpace>(
260  "a_values", reorder_space );
261  auto values_mirror_subv = createSubview( values_mirror, owned_space );
262  Kokkos::deep_copy( a_values, values_mirror_subv );
263 
264  // Insert values into the HYPRE matrix.
265  std::vector<HYPRE_Int> indices( _stencil_size );
266  std::iota( indices.begin(), indices.end(), 0 );
267  error = HYPRE_StructMatrixSetBoxValues(
268  _A, _lower.data(), _upper.data(), indices.size(), indices.data(),
269  a_values.data() );
270  checkHypreError( error );
271  error = HYPRE_StructMatrixAssemble( _A );
272  checkHypreError( error );
273  }

320  template <class Array_t>
321  void solve( const Array_t& b, Array_t& x )
322  {
323  static_assert( is_array<Array_t>::value, "Must use an array" );
324  static_assert(
325  std::is_same<typename Array_t::entity_type, entity_type>::value,
326  "Array entity type mush match solver entity type" );
327  static_assert(
328  std::is_same<typename Array_t::device_type, DeviceType>::value,
329  "Array device type and solver device type are different." );
330 
331  static_assert(
332  std::is_same<typename Array_t::value_type, value_type>::value,
333  "Array value type and solver value type are different." );
334 
335  // This function is only valid for non-preconditioners.
336  if ( _is_preconditioner )
337  throw std::logic_error( "Cannot call solve() on preconditioners" );
338 
339  if ( b.layout()->dofsPerEntity() != 1 ||
340  x.layout()->dofsPerEntity() != 1 )
341  throw std::runtime_error(
342  "Structured solver only for scalar fields" );
343 
344  // Spatial dimension.
345  const std::size_t num_space_dim = Array_t::num_space_dim;
346 
347  // Initialize the RHS.
348  auto error = HYPRE_StructVectorInitialize( _b );
349  checkHypreError( error );
350 
351  // Get a local view of RHS on the host.
352  auto b_mirror = Kokkos::create_mirror_view_and_copy(
353  Kokkos::HostSpace(), b.view() );
354 
355  // Copy the RHS into HYPRE. The HYPRE layout is fixed as layout-right.
356  auto owned_space = b.layout()->indexSpace( Own(), Local() );
357  std::array<long, num_space_dim + 1> reorder_size;
358  for ( std::size_t d = 0; d < num_space_dim; ++d )
359  {
360  reorder_size[d] = owned_space.extent( d );
361  }
362  reorder_size.back() = 1;
363  IndexSpace<num_space_dim + 1> reorder_space( reorder_size );
364  auto vector_values =
365  createView<HYPRE_Complex, Kokkos::LayoutRight, Kokkos::HostSpace>(
366  "vector_values", reorder_space );
367  auto b_mirror_subv = createSubview( b_mirror, owned_space );
368  Kokkos::deep_copy( vector_values, b_mirror_subv );
369 
370  // Insert b values into the HYPRE vector.
371  error = HYPRE_StructVectorSetBoxValues(
372  _b, _lower.data(), _upper.data(), vector_values.data() );
373  checkHypreError( error );
374  error = HYPRE_StructVectorAssemble( _b );
375  checkHypreError( error );
376 
377  // Solve the problem
378  this->solveImpl( _A, _b, _x );
379 
380  // Extract the solution from the LHS
381  error = HYPRE_StructVectorGetBoxValues(
382  _x, _lower.data(), _upper.data(), vector_values.data() );
383  checkHypreError( error );
384 
385  // Get a local view of x on the host.
386  auto x_mirror =
387  Kokkos::create_mirror_view( Kokkos::HostSpace(), x.view() );
388 
389  // Copy the HYPRE solution to the LHS.
390  auto x_mirror_subv = createSubview( x_mirror, owned_space );
391  Kokkos::deep_copy( x_mirror_subv, vector_values );
392 
393  // Copy back to the device.
394  Kokkos::deep_copy( x.view(), x_mirror );
395  }

440  // Check a hypre error.
441  void checkHypreError( const int error ) const
442  {
443  if ( error > 0 )
444  {
445  char error_msg[256];
446  HYPRE_DescribeError( error, error_msg );
447  std::stringstream out;
448  out << "HYPRE structured solver error: ";
449  out << error << " " << error_msg;
450  HYPRE_ClearError( error );
451  throw std::runtime_error( out.str() );
452  }
453  }
