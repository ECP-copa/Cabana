119  template <class ArrayLayout_t>
120  HypreStructuredSolver( const ArrayLayout_t& layout,
121  const bool is_preconditioner = false )
122  : _comm( layout.localGrid()->globalGrid().comm() )
123  , _is_preconditioner( is_preconditioner )
124  {
125  static_assert( is_array_layout<ArrayLayout_t>::value,
126  "Must use an array layout" );
127  static_assert(
128  std::is_same<typename ArrayLayout_t::entity_type,
129  entity_type>::value,
130  "Array layout entity type mush match solver entity type" );
131 
132  // Spatial dimension.
133  const std::size_t num_space_dim = ArrayLayout_t::num_space_dim;
134 
135  // Only create data structures if this is not a preconditioner.
136  if ( !_is_preconditioner )
137  {
138  // Create the grid.
139  auto error = HYPRE_StructGridCreate( _comm, num_space_dim, &_grid );
140  checkHypreError( error );
141 
142  // Get the global index space spanned by the local grid on this
143  // rank. Note that the upper bound is not a bound but rather the
144  // last index as this is what Hypre wants. Note that we reordered
145  // this to KJI from IJK to be consistent with HYPRE ordering. By
146  // setting up the grid like this, HYPRE will then want layout-right
147  // data indexed as (i,j,k) or (i,j,k,l) which will allow us to
148  // directly use Kokkos::deep_copy to move data between Cajita arrays
149  // and HYPRE data structures.
150  auto global_space = layout.indexSpace( Own(), Global() );
151  _lower.resize( num_space_dim );
152  _upper.resize( num_space_dim );
153  for ( std::size_t d = 0; d < num_space_dim; ++d )
154  {
155  _lower[d] = static_cast<HYPRE_Int>(
156  global_space.min( num_space_dim - d - 1 ) );
157  _upper[d] = static_cast<HYPRE_Int>(
158  global_space.max( num_space_dim - d - 1 ) - 1 );
159  }
160  error = HYPRE_StructGridSetExtents( _grid, _lower.data(),
161  _upper.data() );
162  checkHypreError( error );
163 
164  // Get periodicity. Note we invert the order of this to KJI as well.
165  const auto& global_grid = layout.localGrid()->globalGrid();
166  HYPRE_Int periodic[num_space_dim];
167  for ( std::size_t d = 0; d < num_space_dim; ++d )
168  periodic[num_space_dim - 1 - d] =
169  global_grid.isPeriodic( d )
170  ? layout.localGrid()->globalGrid().globalNumEntity(
171  EntityType(), d )
172  : 0;
173  error = HYPRE_StructGridSetPeriodic( _grid, periodic );
174  checkHypreError( error );
175 
176  // Assemble the grid.
177  error = HYPRE_StructGridAssemble( _grid );
178  checkHypreError( error );
179 
180  // Allocate LHS and RHS vectors and initialize to zero. Note that we
181  // are fixing the views under these vectors to layout-right.
182  std::array<long, num_space_dim> reorder_size;
183  for ( std::size_t d = 0; d < num_space_dim; ++d )
184  {
185  reorder_size[d] = global_space.extent( d );
186  }
187  IndexSpace<num_space_dim> reorder_space( reorder_size );
188  auto vector_values =
189  createView<HYPRE_Complex, Kokkos::LayoutRight, memory_space>(
190  "vector_values", reorder_space );
191  Kokkos::deep_copy( vector_values, 0.0 );
192 
193  error = HYPRE_StructVectorCreate( _comm, _grid, &_b );
194  checkHypreError( error );
195  error = HYPRE_StructVectorInitialize( _b );
196  checkHypreError( error );
197  error = HYPRE_StructVectorSetBoxValues(
198  _b, _lower.data(), _upper.data(), vector_values.data() );
199  checkHypreError( error );
200  error = HYPRE_StructVectorAssemble( _b );
201  checkHypreError( error );
202 
203  error = HYPRE_StructVectorCreate( _comm, _grid, &_x );
204  checkHypreError( error );
205  error = HYPRE_StructVectorInitialize( _x );
206  checkHypreError( error );
207  error = HYPRE_StructVectorSetBoxValues(
208  _x, _lower.data(), _upper.data(), vector_values.data() );
209  checkHypreError( error );
210  error = HYPRE_StructVectorAssemble( _x );
211  checkHypreError( error );
212  }
213  }

242  setMatrixStencil( const std::vector<std::array<int, NumSpaceDim>>& stencil,
243  const bool is_symmetric = false )
244  {

279  template <class Array_t>
280  void setMatrixValues( const Array_t& values )
281  {
282  static_assert( is_array<Array_t>::value, "Must use an array" );
283  static_assert(
284  std::is_same<typename Array_t::entity_type, entity_type>::value,
285  "Array entity type mush match solver entity type" );
286  static_assert(
287  std::is_same<typename Array_t::memory_space, MemorySpace>::value,
288  "Array device type and solver device type are different." );
289 
290  static_assert(
291  std::is_same<typename Array_t::value_type, value_type>::value,
292  "Array value type and solver value type are different." );
293 
294  // This function is only valid for non-preconditioners.
295  if ( _is_preconditioner )
296  throw std::logic_error(
297  "Cannot call setMatrixValues() on preconditioners" );
298 
299  if ( values.layout()->dofsPerEntity() !=
300  static_cast<int>( _stencil_size ) )
301  throw std::runtime_error(
302  "Number of matrix values does not match stencil size" );
303 
304  // Spatial dimension.
305  const std::size_t num_space_dim = Array_t::num_space_dim;
306 
307  // Intialize the matrix for setting values.
308  auto error = HYPRE_StructMatrixInitialize( _A );
309  checkHypreError( error );
310 
311  // Copy the matrix entries into HYPRE. The HYPRE layout is fixed as
312  // layout-right.
313  auto owned_space = values.layout()->indexSpace( Own(), Local() );
314  std::array<long, num_space_dim + 1> reorder_size;
315  for ( std::size_t d = 0; d < num_space_dim; ++d )
316  {
317  reorder_size[d] = owned_space.extent( d );
318  }
319  reorder_size.back() = _stencil_size;
320  IndexSpace<num_space_dim + 1> reorder_space( reorder_size );
321  auto a_values =
322  createView<HYPRE_Complex, Kokkos::LayoutRight, memory_space>(
323  "a_values", reorder_space );
324  auto values_subv = createSubview( values.view(), owned_space );
325  Kokkos::deep_copy( a_values, values_subv );
326 
327  // Insert values into the HYPRE matrix.
328  std::vector<HYPRE_Int> indices( _stencil_size );
329  std::iota( indices.begin(), indices.end(), 0 );
330  error = HYPRE_StructMatrixSetBoxValues(
331  _A, _lower.data(), _upper.data(), indices.size(), indices.data(),
332  a_values.data() );
333  checkHypreError( error );
334  error = HYPRE_StructMatrixAssemble( _A );
335  checkHypreError( error );
336  }

383  template <class Array_t>
384  void solve( const Array_t& b, Array_t& x )
385  {
386  static_assert( is_array<Array_t>::value, "Must use an array" );
387  static_assert(
388  std::is_same<typename Array_t::entity_type, entity_type>::value,
389  "Array entity type mush match solver entity type" );
390  static_assert(
391  std::is_same<typename Array_t::memory_space, MemorySpace>::value,
392  "Array device type and solver device type are different." );
393 
394  static_assert(
395  std::is_same<typename Array_t::value_type, value_type>::value,
396  "Array value type and solver value type are different." );
397 
398  // This function is only valid for non-preconditioners.
399  if ( _is_preconditioner )
400  throw std::logic_error( "Cannot call solve() on preconditioners" );
401 
402  if ( b.layout()->dofsPerEntity() != 1 ||
403  x.layout()->dofsPerEntity() != 1 )
404  throw std::runtime_error(
405  "Structured solver only for scalar fields" );
406 
407  // Spatial dimension.
408  const std::size_t num_space_dim = Array_t::num_space_dim;
409 
410  // Initialize the RHS.
411  auto error = HYPRE_StructVectorInitialize( _b );
412  checkHypreError( error );
413 
414  // Copy the RHS into HYPRE. The HYPRE layout is fixed as layout-right.
415  auto owned_space = b.layout()->indexSpace( Own(), Local() );
416  std::array<long, num_space_dim + 1> reorder_size;
417  for ( std::size_t d = 0; d < num_space_dim; ++d )
418  {
419  reorder_size[d] = owned_space.extent( d );
420  }
421  reorder_size.back() = 1;
422  IndexSpace<num_space_dim + 1> reorder_space( reorder_size );
423  auto vector_values =
424  createView<HYPRE_Complex, Kokkos::LayoutRight, memory_space>(
425  "vector_values", reorder_space );
426  auto b_subv = createSubview( b.view(), owned_space );
427  Kokkos::deep_copy( vector_values, b_subv );
428 
429  // Insert b values into the HYPRE vector.
430  error = HYPRE_StructVectorSetBoxValues(
431  _b, _lower.data(), _upper.data(), vector_values.data() );
432  checkHypreError( error );
433  error = HYPRE_StructVectorAssemble( _b );
434  checkHypreError( error );
435 
436  // Solve the problem
437  this->solveImpl( _A, _b, _x );
438 
439  // Extract the solution from the LHS
440  error = HYPRE_StructVectorGetBoxValues(
441  _x, _lower.data(), _upper.data(), vector_values.data() );
442  checkHypreError( error );
443 
444  // Copy the HYPRE solution to the LHS.
445  auto x_subv = createSubview( x.view(), owned_space );
446  Kokkos::deep_copy( x_subv, vector_values );
447  }
