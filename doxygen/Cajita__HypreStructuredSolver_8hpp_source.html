98class HypreStructuredSolver
99{
100 public:
102 using entity_type = EntityType;
104 using memory_space = MemorySpace;
106 using value_type = Scalar;
108 static_assert( HypreIsCompatibleWithMemorySpace<memory_space>::value,
109 "HYPRE not compatible with solver memory space" );

117 template <class ArrayLayout_t>
118 HypreStructuredSolver( const ArrayLayout_t& layout,
119 const bool is_preconditioner = false )
120 : _comm( layout.localGrid()->globalGrid().comm() )
121 , _is_preconditioner( is_preconditioner )
122 {
123 static_assert( is_array_layout<ArrayLayout_t>::value,
124 "Must use an array layout" );
125 static_assert(
126 std::is_same<typename ArrayLayout_t::entity_type,
127 entity_type>::value,
128 "Array layout entity type mush match solver entity type" );
129
130 // Spatial dimension.
131 const std::size_t num_space_dim = ArrayLayout_t::num_space_dim;
132
133 // Only create data structures if this is not a preconditioner.
134 if ( !_is_preconditioner )
135 {
136 // Create the grid.
137 auto error = HYPRE_StructGridCreate( _comm, num_space_dim, &_grid );
138 checkHypreError( error );
139
140 // Get the global index space spanned by the local grid on this
141 // rank. Note that the upper bound is not a bound but rather the
142 // last index as this is what Hypre wants. Note that we reordered
143 // this to KJI from IJK to be consistent with HYPRE ordering. By
144 // setting up the grid like this, HYPRE will then want layout-right
145 // data indexed as (i,j,k) or (i,j,k,l) which will allow us to
146 // directly use Kokkos::deep_copy to move data between Cajita arrays
147 // and HYPRE data structures.
148 auto global_space = layout.indexSpace( Own(), Global() );
149 _lower.resize( num_space_dim );
150 _upper.resize( num_space_dim );
151 for ( std::size_t d = 0; d < num_space_dim; ++d )
152 {
153 _lower[d] = static_cast<HYPRE_Int>(
154 global_space.min( num_space_dim - d - 1 ) );
155 _upper[d] = static_cast<HYPRE_Int>(
156 global_space.max( num_space_dim - d - 1 ) - 1 );
157 }
158 error = HYPRE_StructGridSetExtents( _grid, _lower.data(),
159 _upper.data() );
160 checkHypreError( error );
161
162 // Get periodicity. Note we invert the order of this to KJI as well.
163 const auto& global_grid = layout.localGrid()->globalGrid();
164 HYPRE_Int periodic[num_space_dim];
165 for ( std::size_t d = 0; d < num_space_dim; ++d )
166 periodic[num_space_dim - 1 - d] =
167 global_grid.isPeriodic( d )
168 ? layout.localGrid()->globalGrid().globalNumEntity(
169 EntityType(), d )
170 : 0;
171 error = HYPRE_StructGridSetPeriodic( _grid, periodic );
172 checkHypreError( error );
173
174 // Assemble the grid.
175 error = HYPRE_StructGridAssemble( _grid );
176 checkHypreError( error );
177
178 // Allocate LHS and RHS vectors and initialize to zero. Note that we
179 // are fixing the views under these vectors to layout-right.
180 std::array<long, num_space_dim> reorder_size;
181 for ( std::size_t d = 0; d < num_space_dim; ++d )
182 {
183 reorder_size[d] = global_space.extent( d );
184 }
185 IndexSpace<num_space_dim> reorder_space( reorder_size );
186 auto vector_values =
187 createView<HYPRE_Complex, Kokkos::LayoutRight, memory_space>(
188 "vector_values", reorder_space );
189 Kokkos::deep_copy( vector_values, 0.0 );
190
191 error = HYPRE_StructVectorCreate( _comm, _grid, &_b );
192 checkHypreError( error );
193 error = HYPRE_StructVectorInitialize( _b );
194 checkHypreError( error );
195 error = HYPRE_StructVectorSetBoxValues(
196 _b, _lower.data(), _upper.data(), vector_values.data() );
197 checkHypreError( error );
198 error = HYPRE_StructVectorAssemble( _b );
199 checkHypreError( error );
200
201 error = HYPRE_StructVectorCreate( _comm, _grid, &_x );
202 checkHypreError( error );
203 error = HYPRE_StructVectorInitialize( _x );
204 checkHypreError( error );
205 error = HYPRE_StructVectorSetBoxValues(
206 _x, _lower.data(), _upper.data(), vector_values.data() );
207 checkHypreError( error );
208 error = HYPRE_StructVectorAssemble( _x );
209 checkHypreError( error );
210 }
211 }

238 template <std::size_t NumSpaceDim>
239 void
240 setMatrixStencil( const std::vector<std::array<int, NumSpaceDim>>& stencil,
241 const bool is_symmetric = false )
242 {
243 // This function is only valid for non-preconditioners.
244 if ( _is_preconditioner )
245 throw std::logic_error(
246 "Cannot call setMatrixStencil() on preconditioners" );
247
248 // Create the stencil.
249 _stencil_size = stencil.size();
250 auto error =
251 HYPRE_StructStencilCreate( NumSpaceDim, _stencil_size, &_stencil );
252 checkHypreError( error );
253 std::array<HYPRE_Int, NumSpaceDim> offset;
254 for ( unsigned n = 0; n < stencil.size(); ++n )
255 {
256 for ( std::size_t d = 0; d < NumSpaceDim; ++d )
257 offset[d] = stencil[n][d];
258 error = HYPRE_StructStencilSetElement( _stencil, n, offset.data() );
259 checkHypreError( error );
260 }
261
262 // Create the matrix.
263 error = HYPRE_StructMatrixCreate( _comm, _grid, _stencil, &_A );
264 checkHypreError( error );
265 error = HYPRE_StructMatrixSetSymmetric( _A, is_symmetric );
266 checkHypreError( error );
267 }

277 template <class Array_t>
278 void setMatrixValues( const Array_t& values )
279 {
280 static_assert( is_array<Array_t>::value, "Must use an array" );
281 static_assert(
282 std::is_same<typename Array_t::entity_type, entity_type>::value,
283 "Array entity type mush match solver entity type" );
284 static_assert(
285 std::is_same<typename Array_t::memory_space, MemorySpace>::value,
286 "Array device type and solver device type are different." );
287
288 static_assert(
289 std::is_same<typename Array_t::value_type, value_type>::value,
290 "Array value type and solver value type are different." );
291
292 // This function is only valid for non-preconditioners.
293 if ( _is_preconditioner )
294 throw std::logic_error(
295 "Cannot call setMatrixValues() on preconditioners" );
296
297 if ( values.layout()->dofsPerEntity() !=
298 static_cast<int>( _stencil_size ) )
299 throw std::runtime_error(
300 "Number of matrix values does not match stencil size" );
301
302 // Spatial dimension.
303 const std::size_t num_space_dim = Array_t::num_space_dim;
304
305 // Intialize the matrix for setting values.
306 auto error = HYPRE_StructMatrixInitialize( _A );
307 checkHypreError( error );
308
309 // Copy the matrix entries into HYPRE. The HYPRE layout is fixed as
310 // layout-right.
311 auto owned_space = values.layout()->indexSpace( Own(), Local() );
312 std::array<long, num_space_dim + 1> reorder_size;
313 for ( std::size_t d = 0; d < num_space_dim; ++d )
314 {
315 reorder_size[d] = owned_space.extent( d );
316 }
317 reorder_size.back() = _stencil_size;
318 IndexSpace<num_space_dim + 1> reorder_space( reorder_size );
319 auto a_values =
320 createView<HYPRE_Complex, Kokkos::LayoutRight, memory_space>(
321 "a_values", reorder_space );
322 auto values_subv = createSubview( values.view(), owned_space );
323 Kokkos::deep_copy( a_values, values_subv );
324
325 // Insert values into the HYPRE matrix.
326 std::vector<HYPRE_Int> indices( _stencil_size );
327 std::iota( indices.begin(), indices.end(), 0 );
328 error = HYPRE_StructMatrixSetBoxValues(
329 _A, _lower.data(), _upper.data(), indices.size(), indices.data(),
330 a_values.data() );
331 checkHypreError( error );
332 error = HYPRE_StructMatrixAssemble( _A );
333 checkHypreError( error );
334 }

381 template <class Array_t>
382 void solve( const Array_t& b, Array_t& x )
383 {
384 Kokkos::Profiling::pushRegion( "Cajita::HypreStructuredSolver::solve" );
385
386 static_assert( is_array<Array_t>::value, "Must use an array" );
387 static_assert(
388 std::is_same<typename Array_t::entity_type, entity_type>::value,
389 "Array entity type mush match solver entity type" );
390 static_assert(
391 std::is_same<typename Array_t::memory_space, MemorySpace>::value,
392 "Array device type and solver device type are different." );
393
394 static_assert(
395 std::is_same<typename Array_t::value_type, value_type>::value,
396 "Array value type and solver value type are different." );
397
398 // This function is only valid for non-preconditioners.
399 if ( _is_preconditioner )
400 throw std::logic_error( "Cannot call solve() on preconditioners" );
401
402 if ( b.layout()->dofsPerEntity() != 1 ||
403 x.layout()->dofsPerEntity() != 1 )
404 throw std::runtime_error(
405 "Structured solver only for scalar fields" );
406
407 // Spatial dimension.
408 const std::size_t num_space_dim = Array_t::num_space_dim;
409
410 // Initialize the RHS.
411 auto error = HYPRE_StructVectorInitialize( _b );
412 checkHypreError( error );
413
414 // Copy the RHS into HYPRE. The HYPRE layout is fixed as layout-right.
415 auto owned_space = b.layout()->indexSpace( Own(), Local() );
416 std::array<long, num_space_dim + 1> reorder_size;
417 for ( std::size_t d = 0; d < num_space_dim; ++d )
418 {
419 reorder_size[d] = owned_space.extent( d );
420 }
421 reorder_size.back() = 1;
422 IndexSpace<num_space_dim + 1> reorder_space( reorder_size );
423 auto vector_values =
424 createView<HYPRE_Complex, Kokkos::LayoutRight, memory_space>(
425 "vector_values", reorder_space );
426 auto b_subv = createSubview( b.view(), owned_space );
427 Kokkos::deep_copy( vector_values, b_subv );
428
429 // Insert b values into the HYPRE vector.
430 error = HYPRE_StructVectorSetBoxValues(
431 _b, _lower.data(), _upper.data(), vector_values.data() );
432 checkHypreError( error );
433 error = HYPRE_StructVectorAssemble( _b );
434 checkHypreError( error );
435
436 // Solve the problem
437 this->solveImpl( _A, _b, _x );
438
439 // Extract the solution from the LHS
440 error = HYPRE_StructVectorGetBoxValues(
441 _x, _lower.data(), _upper.data(), vector_values.data() );
442 checkHypreError( error );
443
444 // Copy the HYPRE solution to the LHS.
445 auto x_subv = createSubview( x.view(), owned_space );
446 Kokkos::deep_copy( x_subv, vector_values );
447
448 Kokkos::Profiling::popRegion();
449 }

497 void checkHypreError( const int error ) const

501 char error_msg[256];

503 std::stringstream out;
504 out << "HYPRE structured solver error: ";
505 out << error << " " << error_msg;
506 HYPRE_ClearError( error );
507 throw std::runtime_error( out.str() );
508 }

1518 layout, is_preconditioner );
1519 else if ( "PFMG" == solver_type )

1521 is_preconditioner );

1524 is_preconditioner );

1526 return createHypreStructJacobi<Scalar, MemorySpace>(
1527 layout, is_preconditioner );

1529 return createHypreStructDiagonal<Scalar, MemorySpace>(
1530 layout, is_preconditioner );
