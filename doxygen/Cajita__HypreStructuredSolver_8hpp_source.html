58  template <class ArrayLayout_t>
59  HypreStructuredSolver( const ArrayLayout_t& layout,
60  const bool is_preconditioner = false )
61  : _comm( layout.localGrid()->globalGrid().comm() )
62  , _is_preconditioner( is_preconditioner )
63  {
64  static_assert( is_array_layout<ArrayLayout_t>::value,
65  "Must use an array layout" );
66  static_assert(
67  std::is_same<typename ArrayLayout_t::entity_type,
68  entity_type>::value,
69  "Array layout entity type mush match solver entity type" );
70 
71  // Spatial dimension.
72  const std::size_t num_space_dim = ArrayLayout_t::num_space_dim;
73 
74  // Only create data structures if this is not a preconditioner.
75  if ( !_is_preconditioner )
76  {
77  // Create the grid.
78  auto error = HYPRE_StructGridCreate( _comm, num_space_dim, &_grid );
79  checkHypreError( error );
80 
81  // Get the global index space spanned by the local grid on this
82  // rank. Note that the upper bound is not a bound but rather the
83  // last index as this is what Hypre wants. Note that we reordered
84  // this to KJI from IJK to be consistent with HYPRE ordering. By
85  // setting up the grid like this, HYPRE will then want layout-right
86  // data indexed as (i,j,k) or (i,j,k,l) which will allow us to
87  // directly use Kokkos::deep_copy to move data between Cajita arrays
88  // and HYPRE data structures.
89  auto global_space = layout.indexSpace( Own(), Global() );
90  _lower.resize( num_space_dim );
91  _upper.resize( num_space_dim );
92  for ( std::size_t d = 0; d < num_space_dim; ++d )
93  {
94  _lower[d] = static_cast<HYPRE_Int>(
95  global_space.min( num_space_dim - d - 1 ) );
96  _upper[d] = static_cast<HYPRE_Int>(
97  global_space.max( num_space_dim - d - 1 ) - 1 );
98  }
99  error = HYPRE_StructGridSetExtents( _grid, _lower.data(),
100  _upper.data() );
101  checkHypreError( error );
102 
103  // Get periodicity. Note we invert the order of this to KJI as well.
104  const auto& global_grid = layout.localGrid()->globalGrid();
105  HYPRE_Int periodic[num_space_dim];
106  for ( std::size_t d = 0; d < num_space_dim; ++d )
107  periodic[num_space_dim - 1 - d] =
108  global_grid.isPeriodic( d )
109  ? layout.localGrid()->globalGrid().globalNumEntity(
110  EntityType(), d )
111  : 0;
112  error = HYPRE_StructGridSetPeriodic( _grid, periodic );
113  checkHypreError( error );
114 
115  // Assemble the grid.
116  error = HYPRE_StructGridAssemble( _grid );
117  checkHypreError( error );
118 
119  // Allocate LHS and RHS vectors and initialize to zero. Note that we
120  // are fixing the views under these vectors to layout-right.
121  std::array<long, num_space_dim> reorder_size;
122  for ( std::size_t d = 0; d < num_space_dim; ++d )
123  {
124  reorder_size[d] = global_space.extent( d );
125  }
126  IndexSpace<num_space_dim> reorder_space( reorder_size );
127  auto vector_values =
128  createView<HYPRE_Complex, Kokkos::LayoutRight,
129  Kokkos::HostSpace>( "vector_values", reorder_space );
130  Kokkos::deep_copy( vector_values, 0.0 );
131 
132  error = HYPRE_StructVectorCreate( _comm, _grid, &_b );
133  checkHypreError( error );
134  error = HYPRE_StructVectorInitialize( _b );
135  checkHypreError( error );
136  error = HYPRE_StructVectorSetBoxValues(
137  _b, _lower.data(), _upper.data(), vector_values.data() );
138  checkHypreError( error );
139  error = HYPRE_StructVectorAssemble( _b );
140  checkHypreError( error );
141 
142  error = HYPRE_StructVectorCreate( _comm, _grid, &_x );
143  checkHypreError( error );
144  error = HYPRE_StructVectorInitialize( _x );
145  checkHypreError( error );
146  error = HYPRE_StructVectorSetBoxValues(
147  _x, _lower.data(), _upper.data(), vector_values.data() );
148  checkHypreError( error );
149  error = HYPRE_StructVectorAssemble( _x );
150  checkHypreError( error );
151  }
152  }

181  setMatrixStencil( const std::vector<std::array<int, NumSpaceDim>>& stencil,
182  const bool is_symmetric = false )

218  template <class Array_t>
219  void setMatrixValues( const Array_t& values )
220  {
221  static_assert( is_array<Array_t>::value, "Must use an array" );
222  static_assert(
223  std::is_same<typename Array_t::entity_type, entity_type>::value,
224  "Array entity type mush match solver entity type" );
225  static_assert(
226  std::is_same<typename Array_t::device_type, DeviceType>::value,
227  "Array device type and solver device type are different." );
228 
229  static_assert(
230  std::is_same<typename Array_t::value_type, value_type>::value,
231  "Array value type and solver value type are different." );
232 
233  // This function is only valid for non-preconditioners.
234  if ( _is_preconditioner )
235  throw std::logic_error(
236  "Cannot call setMatrixValues() on preconditioners" );
237 
238  if ( values.layout()->dofsPerEntity() !=
239  static_cast<int>( _stencil_size ) )
240  throw std::runtime_error(
241  "Number of matrix values does not match stencil size" );
242 
243  // Spatial dimension.
244  const std::size_t num_space_dim = Array_t::num_space_dim;
245 
246  // Intialize the matrix for setting values.
247  auto error = HYPRE_StructMatrixInitialize( _A );
248  checkHypreError( error );
249 
250  // Get a view of the matrix values on the host.
251  auto values_mirror = Kokkos::create_mirror_view_and_copy(
252  Kokkos::HostSpace(), values.view() );
253 
254  // Copy the matrix entries into HYPRE. The HYPRE layout is fixed as
255  // layout-right.
256  auto owned_space = values.layout()->indexSpace( Own(), Local() );
257  std::array<long, num_space_dim + 1> reorder_size;
258  for ( std::size_t d = 0; d < num_space_dim; ++d )
259  {
260  reorder_size[d] = owned_space.extent( d );
261  }
262  reorder_size.back() = _stencil_size;
263  IndexSpace<num_space_dim + 1> reorder_space( reorder_size );
264  auto a_values =
265  createView<HYPRE_Complex, Kokkos::LayoutRight, Kokkos::HostSpace>(
266  "a_values", reorder_space );
267  auto values_mirror_subv = createSubview( values_mirror, owned_space );
268  Kokkos::deep_copy( a_values, values_mirror_subv );
269 
270  // Insert values into the HYPRE matrix.
271  std::vector<HYPRE_Int> indices( _stencil_size );
272  std::iota( indices.begin(), indices.end(), 0 );
273  error = HYPRE_StructMatrixSetBoxValues(
274  _A, _lower.data(), _upper.data(), indices.size(), indices.data(),
275  a_values.data() );
276  checkHypreError( error );
277  error = HYPRE_StructMatrixAssemble( _A );
278  checkHypreError( error );
279  }

326  template <class Array_t>
327  void solve( const Array_t& b, Array_t& x )
328  {
329  static_assert( is_array<Array_t>::value, "Must use an array" );
330  static_assert(
331  std::is_same<typename Array_t::entity_type, entity_type>::value,
332  "Array entity type mush match solver entity type" );
333  static_assert(
334  std::is_same<typename Array_t::device_type, DeviceType>::value,
335  "Array device type and solver device type are different." );
336 
337  static_assert(
338  std::is_same<typename Array_t::value_type, value_type>::value,
339  "Array value type and solver value type are different." );
340 
341  // This function is only valid for non-preconditioners.
342  if ( _is_preconditioner )
343  throw std::logic_error( "Cannot call solve() on preconditioners" );
344 
345  if ( b.layout()->dofsPerEntity() != 1 ||
346  x.layout()->dofsPerEntity() != 1 )
347  throw std::runtime_error(
348  "Structured solver only for scalar fields" );
349 
350  // Spatial dimension.
351  const std::size_t num_space_dim = Array_t::num_space_dim;
352 
353  // Initialize the RHS.
354  auto error = HYPRE_StructVectorInitialize( _b );
355  checkHypreError( error );
356 
357  // Get a local view of RHS on the host.
358  auto b_mirror = Kokkos::create_mirror_view_and_copy(
359  Kokkos::HostSpace(), b.view() );
360 
361  // Copy the RHS into HYPRE. The HYPRE layout is fixed as layout-right.
362  auto owned_space = b.layout()->indexSpace( Own(), Local() );
363  std::array<long, num_space_dim + 1> reorder_size;
364  for ( std::size_t d = 0; d < num_space_dim; ++d )
365  {
366  reorder_size[d] = owned_space.extent( d );
367  }
368  reorder_size.back() = 1;
369  IndexSpace<num_space_dim + 1> reorder_space( reorder_size );
370  auto vector_values =
371  createView<HYPRE_Complex, Kokkos::LayoutRight, Kokkos::HostSpace>(
372  "vector_values", reorder_space );
373  auto b_mirror_subv = createSubview( b_mirror, owned_space );
374  Kokkos::deep_copy( vector_values, b_mirror_subv );
375 
376  // Insert b values into the HYPRE vector.
377  error = HYPRE_StructVectorSetBoxValues(
378  _b, _lower.data(), _upper.data(), vector_values.data() );
379  checkHypreError( error );
380  error = HYPRE_StructVectorAssemble( _b );
381  checkHypreError( error );
382 
383  // Solve the problem
384  this->solveImpl( _A, _b, _x );
385 
386  // Extract the solution from the LHS
387  error = HYPRE_StructVectorGetBoxValues(
388  _x, _lower.data(), _upper.data(), vector_values.data() );
389  checkHypreError( error );
390 
391  // Get a local view of x on the host.
392  auto x_mirror =
393  Kokkos::create_mirror_view( Kokkos::HostSpace(), x.view() );
394 
395  // Copy the HYPRE solution to the LHS.
396  auto x_mirror_subv = createSubview( x_mirror, owned_space );
397  Kokkos::deep_copy( x_mirror_subv, vector_values );
398 
399  // Copy back to the device.
400  Kokkos::deep_copy( x.view(), x_mirror );
401  }
