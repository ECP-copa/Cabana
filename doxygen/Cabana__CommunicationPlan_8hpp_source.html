211  template<class ViewType>
212  void createFromExportsAndTopology(
213  const ViewType& element_export_ranks,
214  const std::vector<int>& neighbor_ranks,
215  const int mpi_tag = 1221 )
216  {
217  // Store the number of export elements.
218  _num_export_element = element_export_ranks.size();
219 
220  // Store the neighbors.
221  _neighbors = neighbor_ranks;
222  int num_n = _neighbors.size();
223 
224  // Get the MPI rank we are currently on.
225  int my_rank = -1;
226  MPI_Comm_rank( _comm, &my_rank );
227 
228  // If we are sending to ourself put that one first in the neighbor
229  // list.
230  for ( auto& n : _neighbors )
231  if ( n == my_rank )
232  {
233  std::swap( n, _neighbors[0] );
234  break;
235  }
236 
237  // Initialize import/export sizes.
238  _num_export.assign( num_n, 0 );
239  _num_import.assign( num_n, 0 );
240 
241  // Copy the topology to the device.
242  Kokkos::View<int*,Kokkos::HostSpace,Kokkos::MemoryUnmanaged>
243  topology_host( _neighbors.data(), num_n );
244  auto topology = Kokkos::create_mirror_view_and_copy(
245  memory_space(), topology_host );
246 
247  // Count the number of sends to each neighbor.
248  Kokkos::View<std::size_t*,Kokkos::HostSpace,Kokkos::MemoryUnmanaged>
249  num_export_host( _num_export.data(), num_n );
250  auto export_counts = Kokkos::create_mirror_view_and_copy(
251  memory_space(), num_export_host );
252  auto count_neighbor_func =
253  KOKKOS_LAMBDA( const int i )
254  {
255  for ( int n = 0; n < num_n; ++n )
256  if ( topology(n) == element_export_ranks(i) )
257  Kokkos::atomic_increment( &export_counts(n) );
258  };
259  Kokkos::RangePolicy<execution_space> count_neighbor_policy(
260  0, _num_export_element );
261  Kokkos::parallel_for( "Cabana::CommunicationPlan::count_neighbors",
262  count_neighbor_policy,
263  count_neighbor_func );
264  Kokkos::fence();
265 
266  // Copy counts back to the host.
267  Kokkos::deep_copy( num_export_host, export_counts );
268 
269  // Post receives for the number of imports we will get.
270  std::vector<MPI_Request> requests;
271  requests.reserve( num_n );
272  for ( int n = 0; n < num_n; ++n )
273  if ( my_rank != _neighbors[n] )
274  {
275  requests.push_back( MPI_Request() );
276  MPI_Irecv( &_num_import[n],
277  1,
278  MPI_UNSIGNED_LONG,
279  _neighbors[n],
280  mpi_tag,
281  _comm,
282  &(requests.back()) );
283  }
284  else
285  _num_import[n] = _num_export[n];
286 
287  // Send the number of exports to each of our neighbors.
288  for ( int n = 0; n < num_n; ++n )
289  if ( my_rank != _neighbors[n] )
290  MPI_Send( &_num_export[n],
291  1,
292  MPI_UNSIGNED_LONG,
293  _neighbors[n],
294  mpi_tag,
295  _comm );
296 
297  // Wait on receives.
298  std::vector<MPI_Status> status( requests.size() );
299  const int ec =
300  MPI_Waitall( requests.size(), requests.data(), status.data() );
301  assert( MPI_SUCCESS == ec );
302 
303  // Get the total number of imports/exports.
304  _total_num_export =
305  std::accumulate( _num_export.begin(), _num_export.end(), 0 );
306  _total_num_import =
307  std::accumulate( _num_import.begin(), _num_import.end(), 0 );
308 
309  // Barrier before continuing to ensure synchronization.
310  MPI_Barrier( _comm );
311  }

341  template<class ViewType>
342  void createFromExportsOnly( const ViewType& element_export_ranks,
343  const int mpi_tag = 1221 )
344  {
345  // Store the number of export elements.
346  _num_export_element = element_export_ranks.size();
347 
348  // Get the size of this communicator.
349  int comm_size = -1;
350  MPI_Comm_size( _comm, &comm_size );
351 
352  // Get the MPI rank we are currently on.
353  int my_rank = -1;
354  MPI_Comm_rank( _comm, &my_rank );
355 
356  // Count the number of sends this rank will do to other ranks.
357  Kokkos::View<int*,memory_space> neighbor_counts(
358  "neighbor_counts", comm_size );
359  auto count_sends_func =
360  KOKKOS_LAMBDA( const int i )
361  {
362  if ( element_export_ranks(i) >= 0 )
363  Kokkos::atomic_increment(
364  &neighbor_counts(element_export_ranks(i)) );
365  };
366  Kokkos::RangePolicy<execution_space> count_sends_policy(
367  0, _num_export_element );
368  Kokkos::parallel_for( "Cabana::CommunicationPlan::count_sends",
369  count_sends_policy,
370  count_sends_func );
371  Kokkos::fence();
372 
373  // Copy the counts to the host.
374  auto neighbor_counts_host = Kokkos::create_mirror_view_and_copy(
375  Kokkos::HostSpace(), neighbor_counts );
376 
377  // Extract the export ranks and number of exports and then flag the
378  // send ranks.
379  _neighbors.clear();
380  _num_export.clear();
381  _total_num_export = 0;
382  for ( int r = 0; r < comm_size; ++r )
383  if ( neighbor_counts_host(r) > 0 )
384  {
385  _neighbors.push_back( r );
386  _num_export.push_back( neighbor_counts_host(r) );
387  _total_num_export += neighbor_counts_host(r);
388  neighbor_counts_host(r) = 1;
389  }
390 
391  // Get the number of export ranks and initially allocate the import sizes.
392  int num_export_rank = _neighbors.size();
393  _num_import.assign( num_export_rank, 0 );
394 
395  // If we are sending to ourself put that one first in the neighbor
396  // list and assign the number of imports to be the number of exports.
397  bool self_send = false;
398  for ( int n = 0; n < num_export_rank; ++n )
399  if ( _neighbors[n] == my_rank )
400  {
401  std::swap( _neighbors[n], _neighbors[0] );
402  std::swap( _num_export[n], _num_export[0] );
403  _num_import[0] = _num_export[0];
404  self_send = true;
405  break;
406  }
407 
408  // Determine how many total import ranks each neighbor has.
409  std::vector<int> total_receives( comm_size );
410  int root_rank = 0;
411  MPI_Reduce( neighbor_counts_host.data(),
412  total_receives.data(),
413  comm_size,
414  MPI_INT,
415  MPI_SUM,
416  root_rank,
417  _comm );
418  int num_import_rank = -1;
419  MPI_Scatter( total_receives.data(),
420  1,
421  MPI_INT,
422  &num_import_rank,
423  1,
424  MPI_INT,
425  root_rank,
426  _comm );
427  if ( self_send ) --num_import_rank;
428 
429  // Post the expected number of receives and indicate we might get them
430  // from any rank.
431  std::vector<std::size_t> import_sizes( num_import_rank );
432  std::vector<MPI_Request> requests( num_import_rank );
433  for ( int n = 0; n < num_import_rank; ++n )
434  MPI_Irecv( &import_sizes[n],
435  1,
436  MPI_UNSIGNED_LONG,
437  MPI_ANY_SOURCE,
438  mpi_tag,
439  _comm,
440  &requests[n] );
441 
442  // Do blocking sends. Dont do any self sends.
443  int self_offset = (self_send) ? 1 : 0;
444  for ( int n = self_offset; n < num_export_rank; ++n )
445  MPI_Send( &_num_export[n],
446  1,
447  MPI_UNSIGNED_LONG,
448  _neighbors[n],
449  mpi_tag,
450  _comm );
451 
452  // Wait on non-blocking receives.
453  std::vector<MPI_Status> status( requests.size() );
454  const int ec =
455  MPI_Waitall( requests.size(), requests.data(), status.data() );
456  assert( MPI_SUCCESS == ec );
457 
458  // Compute the total number of imports.
459  _total_num_import = std::accumulate(
460  import_sizes.begin(), import_sizes.end(),
461  (self_send) ? _num_import[0] : 0 );
462 
463  // Extract the imports. If we did self sends we already know what
464  // imports we got from that.
465  for ( int i = 0; i < num_import_rank; ++i )
466  {
467  // Get the message source.
468  const auto source = status[i].MPI_SOURCE;
469 
470  // See if the neighbor we received stuff from was someone we also
471  // sent stuff to. If it was, just record what they sent us.
472  auto found_neighbor =
473  std::find( _neighbors.begin(), _neighbors.end(), source );
474 
475  // If this is a new neighbor (i.e. someone we didn't send anything
476  // to) record this. Otherwise add it to the one we found.
477  if ( found_neighbor == std::end(_neighbors) )
478  {
479  _neighbors.push_back( source );
480  _num_import.push_back( import_sizes[i] );
481  _num_export.push_back( 0 );
482  }
483  else
484  {
485  _num_import[i+self_offset] = import_sizes[i];
486  }
487  }
488 
489  // Barrier before continuing to ensure synchronization.
490  MPI_Barrier( _comm );
491  }

537  // Create the export steering vector.
538  template<class RankViewType,class IdViewType>
539  void createSteering(
540  const bool use_iota,
541  const RankViewType& element_export_ranks,
542  const IdViewType& element_export_ids )
543  {
544  if ( !use_iota &&
545  (element_export_ids.size() != element_export_ranks.size()) )
546  throw std::runtime_error("Export ids and ranks different sizes!");
547 
548  // Calculate the steering offsets via exclusive prefix sum for the
549  // exports.
550  int num_n = _neighbors.size();
551  Kokkos::View<std::size_t*,Kokkos::HostSpace>
552  offsets_host( "offsets", num_n );
553  for ( int n = 1; n < num_n; ++n )
554  offsets_host(n) = offsets_host(n-1) + _num_export[n-1];
555 
556  // Copy the offsets to the device.
557  auto offsets = Kokkos::create_mirror_view_and_copy(
558  memory_space(), offsets_host );
559 
560  // Copy the neighbors to the device.
561  Kokkos::View<int*,Kokkos::HostSpace,Kokkos::MemoryUnmanaged>
562  neighbor_ranks_host( _neighbors.data(), num_n );
563  auto neighbor_ranks = Kokkos::create_mirror_view_and_copy(
564  memory_space(), neighbor_ranks_host );
565 
566  // Create the export steering vector for writing local elements into
567  // the send buffer. Note we create a local, shallow copy - this is a
568  // CUDA workaround.
569  _export_steering = Kokkos::View<std::size_t*,memory_space>(
570  Kokkos::ViewAllocateWithoutInitializing("export_steering"),
571  _total_num_export );
572  auto steer_vec = _export_steering;
573  Kokkos::View<std::size_t*,memory_space> counts( "counts", num_n );
574  auto steer_func =
575  KOKKOS_LAMBDA( const int i )
576  {
577  for ( int n = 0; n < num_n; ++n )
578  if ( element_export_ranks(i) == neighbor_ranks(n) )
579  {
580  auto c = Kokkos::atomic_fetch_add( &counts(n), 1 );
581  steer_vec( offsets(n) + c ) =
582  (use_iota) ? i : element_export_ids(i);
583  break;
584  }
585  };
586  Kokkos::RangePolicy<execution_space> steer_policy(
587  0, element_export_ranks.size() );
588  Kokkos::parallel_for( "Cabana::createSteering",
589  steer_policy,
590  steer_func );
591  Kokkos::fence();
592  }
