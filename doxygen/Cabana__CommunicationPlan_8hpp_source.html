60 template<class ExportRankView>
61 auto countSendsAndCreateSteering(
62  const ExportRankView element_export_ranks,
63  const int comm_size,
64  CountSendsAndCreateSteeringAtomic )
65  -> std::pair<Kokkos::View<int*,
66  typename ExportRankView::device_type>,
67  Kokkos::View<typename ExportRankView::size_type*,
68  typename ExportRankView::device_type> >
69 {
70  using device_type = typename ExportRankView::device_type;
71  using execution_space = typename ExportRankView::execution_space;
72  using size_type = typename ExportRankView::size_type;
73 
74  // Create views.
75  Kokkos::View<int*,device_type> neighbor_counts(
76  "neighbor_counts", comm_size );
77  Kokkos::View<size_type*,device_type> neighbor_ids(
78  Kokkos::ViewAllocateWithoutInitializing("neighbor_ids"),
79  element_export_ranks.extent(0) );
80 
81  // Count the sends and create the steering vector.
82  Kokkos::parallel_for(
83  "Cabana::CommunicationPlan::countSendsAndCreateSteering",
84  Kokkos::RangePolicy<execution_space>(0,element_export_ranks.extent(0)),
85  KOKKOS_LAMBDA( const size_type i ){
86  if ( element_export_ranks(i) >= 0 )
87  neighbor_ids(i) = Kokkos::atomic_fetch_add(
88  &neighbor_counts(element_export_ranks(i)), 1 );
89  } );
90  Kokkos::fence();
91 
92  // Return the counts and ids.
93  return std::make_pair( neighbor_counts, neighbor_ids );
94 }

96 // Count sends and generate the steering vector. Duplicated version.
97 template<class ExportRankView>
98 auto countSendsAndCreateSteering(
99  const ExportRankView element_export_ranks,
100  const int comm_size,
101  CountSendsAndCreateSteeringDuplicated )
102  -> std::pair<Kokkos::View<int*,
103  typename ExportRankView::device_type>,
104  Kokkos::View<typename ExportRankView::size_type*,
105  typename ExportRankView::device_type> >
106 {
107  using device_type = typename ExportRankView::device_type;
108  using execution_space = typename ExportRankView::execution_space;
109  using size_type = typename ExportRankView::size_type;
110 
111  // Create a unique thread token.
112  Kokkos::Experimental::UniqueToken<
113  execution_space,Kokkos::Experimental::UniqueTokenScope::Global>
114  unique_token;
115 
116  // Create views.
117  Kokkos::View<int*,device_type> neighbor_counts(
118  Kokkos::ViewAllocateWithoutInitializing("neighbor_counts"),
119  comm_size );
120  Kokkos::View<size_type*,device_type> neighbor_ids(
121  Kokkos::ViewAllocateWithoutInitializing("neighbor_ids"),
122  element_export_ranks.extent(0) );
123  Kokkos::View<int**,device_type> neighbor_counts_dup(
124  "neighbor_counts", unique_token.size(), comm_size );
125  Kokkos::View<size_type**,device_type> neighbor_ids_dup(
126  "neighbor_ids", unique_token.size(),
127  element_export_ranks.extent(0) );
128 
129  // Compute initial duplicated sends and steering.
130  Kokkos::parallel_for(
131  "Cabana::CommunicationPlan::intialCount",
132  Kokkos::RangePolicy<execution_space>(0,element_export_ranks.extent(0)),
133  KOKKOS_LAMBDA( const size_type i ){
134  if ( element_export_ranks(i) >= 0 )
135  {
136  // Get the thread id.
137  auto thread_id = unique_token.acquire();
138 
139  // Do the duplicated fetch-add. If this is a valid element id
140  // increment the send count for this rank. Add the incremented
141  // count as the thread-local neighbor id. This is too big by
142  // one (because we use the prefix increment operator) but we
143  // want a non-zero value so we can later find which thread
144  // this element was located on because we are always
145  // guaranteed a non-zero value. We will subtract this value
146  // later.
147  neighbor_ids_dup(thread_id,i) =
148  ++neighbor_counts_dup( thread_id, element_export_ranks(i) );
149 
150  // Release the thread id.
151  unique_token.release( thread_id );
152  }
153  } );
154  Kokkos::fence();
155 
156  // Team policy
157  using team_policy =
158  Kokkos::TeamPolicy<execution_space,Kokkos::Schedule<Kokkos::Dynamic> >;
159  using index_type = typename team_policy::index_type;
160 
161  // Compute the send counts for each neighbor rank by reducing across
162  // the thread duplicates.
163  Kokkos::parallel_for(
164  "Cabana::CommunicationPlan::finalCount",
165  team_policy(neighbor_counts.extent(0),Kokkos::AUTO),
166  KOKKOS_LAMBDA( const typename team_policy::member_type& team ){
167  // Get the element id.
168  auto i = team.league_rank();
169 
170  // Add the thread results.
171  int thread_counts = 0;
172  Kokkos::parallel_reduce(
173  Kokkos::TeamThreadRange(team,neighbor_counts_dup.extent(0)),
174  [&]( const index_type thread_id, int& result ){
175  result += neighbor_counts_dup( thread_id, i );
176  },
177  thread_counts );
178  neighbor_counts(i) = thread_counts;
179  });
180  Kokkos::fence();
181 
182  // Compute the location of each export element in the send buffer of
183  // its destination rank.
184  Kokkos::parallel_for(
185  "Cabana::CommunicationPlan::createSteering",
186  team_policy(element_export_ranks.extent(0),Kokkos::AUTO),
187  KOKKOS_LAMBDA( const typename team_policy::member_type& team ){
188  // Get the element id.
189  auto i = team.league_rank();
190 
191  // Only operate on valid elements
192  if ( element_export_ranks(i) >= 0 )
193  {
194  // Compute the thread id in which we located the element
195  // during the count phase. Only the thread in which we
196  // located the element will contribute to the reduction.
197  index_type dup_thread = 0;
198  Kokkos::parallel_reduce(
199  Kokkos::TeamThreadRange(team,neighbor_ids_dup.extent(0)),
200  [&]( const index_type thread_id, index_type& result ){
201  if ( neighbor_ids_dup(thread_id,i) > 0 )
202  result += thread_id;
203  },
204  dup_thread );
205 
206  // Compute the offset of this element in the steering
207  // vector for its destination rank. Loop through the
208  // threads up to the thread that found this element in the
209  // count stage. All thread counts prior to that thread
210  // will contribute to the offset.
211  size_type thread_offset = 0;
212  Kokkos::parallel_reduce(
213  Kokkos::TeamThreadRange(team,dup_thread),
214  [&]( const index_type thread_id, size_type& result ){
215  result +=
216  neighbor_counts_dup(
217  thread_id,element_export_ranks(i));
218  },
219  thread_offset );
220 
221  // Add the thread-local value to the offset where we subtract
222  // the 1 that we added artificially when we were first
223  // counting.
224  neighbor_ids(i) = thread_offset +
225  neighbor_ids_dup(dup_thread,i) - 1;
226  }
227  });
228  Kokkos::fence();
229 
230  // Return the counts and ids.
231  return std::make_pair( neighbor_counts, neighbor_ids );
232 }

431  template<class ViewType>
432  Kokkos::View<size_type*,device_type>
433  createFromExportsAndTopology( const ViewType& element_export_ranks,
434  const std::vector<int>& neighbor_ranks,
435  const int mpi_tag = 1221 )
436  {
437  // Store the number of export elements.
438  _num_export_element = element_export_ranks.size();
439 
440  // Store the neighbors.
441  _neighbors = neighbor_ranks;
442  int num_n = _neighbors.size();
443 
444  // Get the size of this communicator.
445  int comm_size = -1;
446  MPI_Comm_size( _comm, &comm_size );
447 
448  // Get the MPI rank we are currently on.
449  int my_rank = -1;
450  MPI_Comm_rank( _comm, &my_rank );
451 
452  // If we are sending to ourself put that one first in the neighbor
453  // list.
454  for ( auto& n : _neighbors )
455  if ( n == my_rank )
456  {
457  std::swap( n, _neighbors[0] );
458  break;
459  }
460 
461  // Initialize import/export sizes.
462  _num_export.assign( num_n, 0 );
463  _num_import.assign( num_n, 0 );
464 
465  // Count the number of sends this rank will do to other ranks. Keep
466  // track of which slot we get in our neighbor's send buffer.
467  auto counts_and_ids =
468  Impl::countSendsAndCreateSteering(
469  element_export_ranks, comm_size,
470  typename Impl::CountSendsAndCreateSteeringAlgorithm<
471  execution_space>::type() );
472 
473  // Copy the counts to the host.
474  auto neighbor_counts_host = Kokkos::create_mirror_view_and_copy(
475  Kokkos::HostSpace(), counts_and_ids.first );
476 
477  // Get the export counts.
478  for ( int n = 0; n < num_n; ++n )
479  _num_export[n] = neighbor_counts_host( _neighbors[n] );
480 
481  // Post receives for the number of imports we will get.
482  std::vector<MPI_Request> requests;
483  requests.reserve( num_n );
484  for ( int n = 0; n < num_n; ++n )
485  if ( my_rank != _neighbors[n] )
486  {
487  requests.push_back( MPI_Request() );
488  MPI_Irecv( &_num_import[n],
489  1,
490  MPI_UNSIGNED_LONG,
491  _neighbors[n],
492  mpi_tag,
493  _comm,
494  &(requests.back()) );
495  }
496  else
497  _num_import[n] = _num_export[n];
498 
499  // Send the number of exports to each of our neighbors.
500  for ( int n = 0; n < num_n; ++n )
501  if ( my_rank != _neighbors[n] )
502  MPI_Send( &_num_export[n],
503  1,
504  MPI_UNSIGNED_LONG,
505  _neighbors[n],
506  mpi_tag,
507  _comm );
508 
509  // Wait on receives.
510  std::vector<MPI_Status> status( requests.size() );
511  const int ec =
512  MPI_Waitall( requests.size(), requests.data(), status.data() );
513  if ( MPI_SUCCESS != ec )
514  throw std::logic_error( "Failed MPI Communication" );
515 
516  // Get the total number of imports/exports.
517  _total_num_export =
518  std::accumulate( _num_export.begin(), _num_export.end(), 0 );
519  _total_num_import =
520  std::accumulate( _num_import.begin(), _num_import.end(), 0 );
521 
522  // Barrier before continuing to ensure synchronization.
523  MPI_Barrier( _comm );
524 
525  // Return the neighbor ids.
526  return counts_and_ids.second;
527  }

560  template<class ViewType>
561  Kokkos::View<size_type*,device_type>
562  createFromExportsOnly( const ViewType& element_export_ranks,
563  const int mpi_tag = 1221 )
564  {
565  // Store the number of export elements.
566  _num_export_element = element_export_ranks.size();
567 
568  // Get the size of this communicator.
569  int comm_size = -1;
570  MPI_Comm_size( _comm, &comm_size );
571 
572  // Get the MPI rank we are currently on.
573  int my_rank = -1;
574  MPI_Comm_rank( _comm, &my_rank );
575 
576  // Count the number of sends this rank will do to other ranks. Keep
577  // track of which slot we get in our neighbor's send buffer.
578  auto counts_and_ids =
579  Impl::countSendsAndCreateSteering(
580  element_export_ranks, comm_size,
581  typename Impl::CountSendsAndCreateSteeringAlgorithm<
582  execution_space>::type() );
583 
584  // Copy the counts to the host.
585  auto neighbor_counts_host = Kokkos::create_mirror_view_and_copy(
586  Kokkos::HostSpace(), counts_and_ids.first );
587 
588  // Extract the export ranks and number of exports and then flag the
589  // send ranks.
590  _neighbors.clear();
591  _num_export.clear();
592  _total_num_export = 0;
593  for ( int r = 0; r < comm_size; ++r )
594  if ( neighbor_counts_host(r) > 0 )
595  {
596  _neighbors.push_back( r );
597  _num_export.push_back( neighbor_counts_host(r) );
598  _total_num_export += neighbor_counts_host(r);
599  neighbor_counts_host(r) = 1;
600  }
601 
602  // Get the number of export ranks and initially allocate the import sizes.
603  int num_export_rank = _neighbors.size();
604  _num_import.assign( num_export_rank, 0 );
605 
606  // If we are sending to ourself put that one first in the neighbor
607  // list and assign the number of imports to be the number of exports.
608  bool self_send = false;
609  for ( int n = 0; n < num_export_rank; ++n )
610  if ( _neighbors[n] == my_rank )
611  {
612  std::swap( _neighbors[n], _neighbors[0] );
613  std::swap( _num_export[n], _num_export[0] );
614  _num_import[0] = _num_export[0];
615  self_send = true;
616  break;
617  }
618 
619  // Determine how many total import ranks each neighbor has.
620  int num_import_rank = -1;
621  std::vector<int> recv_counts( comm_size, 1 );
622  MPI_Reduce_scatter( neighbor_counts_host.data(),
623  &num_import_rank,
624  recv_counts.data(),
625  MPI_INT,
626  MPI_SUM,
627  _comm );
628  if ( self_send ) --num_import_rank;
629 
630  // Post the expected number of receives and indicate we might get them
631  // from any rank.
632  std::vector<std::size_t> import_sizes( num_import_rank );
633  std::vector<MPI_Request> requests( num_import_rank );
634  for ( int n = 0; n < num_import_rank; ++n )
635  MPI_Irecv( &import_sizes[n],
636  1,
637  MPI_UNSIGNED_LONG,
638  MPI_ANY_SOURCE,
639  mpi_tag,
640  _comm,
641  &requests[n] );
642 
643  // Do blocking sends. Dont do any self sends.
644  int self_offset = (self_send) ? 1 : 0;
645  for ( int n = self_offset; n < num_export_rank; ++n )
646  MPI_Send( &_num_export[n],
647  1,
648  MPI_UNSIGNED_LONG,
649  _neighbors[n],
650  mpi_tag,
651  _comm );
652 
653  // Wait on non-blocking receives.
654  std::vector<MPI_Status> status( requests.size() );
655  const int ec =
656  MPI_Waitall( requests.size(), requests.data(), status.data() );
657  if ( MPI_SUCCESS != ec )
658  throw std::logic_error( "Failed MPI Communication" );
659 
660  // Compute the total number of imports.
661  _total_num_import = std::accumulate(
662  import_sizes.begin(), import_sizes.end(),
663  (self_send) ? _num_import[0] : 0 );
664 
665  // Extract the imports. If we did self sends we already know what
666  // imports we got from that.
667  for ( int i = 0; i < num_import_rank; ++i )
668  {
669  // Get the message source.
670  const auto source = status[i].MPI_SOURCE;
671 
672  // See if the neighbor we received stuff from was someone we also
673  // sent stuff to. If it was, just record what they sent us.
674  auto found_neighbor =
675  std::find( _neighbors.begin(), _neighbors.end(), source );
676 
677  // If this is a new neighbor (i.e. someone we didn't send anything
678  // to) record this. Otherwise add it to the one we found.
679  if ( found_neighbor == std::end(_neighbors) )
680  {
681  _neighbors.push_back( source );
682  _num_import.push_back( import_sizes[i] );
683  _num_export.push_back( 0 );
684  }
685  else
686  {
687  _num_import[i+self_offset] = import_sizes[i];
688  }
689  }
690 
691  // Barrier before continuing to ensure synchronization.
692  MPI_Barrier( _comm );
693 
694  // Return the neighbor ids.
695  return counts_and_ids.second;
696  }

750  // Create the export steering vector.
751  template<class PackViewType, class RankViewType,class IdViewType>
752  void createSteering(
753  const bool use_iota,
754  const PackViewType& neighbor_ids,
755  const RankViewType& element_export_ranks,
756  const IdViewType& element_export_ids )
757  {
758  if ( !use_iota &&
759  (element_export_ids.size() != element_export_ranks.size()) )
760  throw std::runtime_error("Export ids and ranks different sizes!");
761 
762  // Get the size of this communicator.
763  int comm_size = -1;
764  MPI_Comm_size( _comm, &comm_size );
765 
766  // Calculate the steering offsets via exclusive prefix sum for the
767  // exports.
768  int num_n = _neighbors.size();
769  std::vector<std::size_t> offsets( num_n, 0.0 );
770  for ( int n = 1; n < num_n; ++n )
771  offsets[n] = offsets[n-1] + _num_export[n-1];
772 
773  // Map the offsets to the device.
774  Kokkos::View<std::size_t*,Kokkos::HostSpace> rank_offsets_host(
775  Kokkos::ViewAllocateWithoutInitializing("rank_map"), comm_size );
776  for ( int n = 0; n < num_n; ++n )
777  rank_offsets_host( _neighbors[n] ) = offsets[n];
778  auto rank_offsets = Kokkos::create_mirror_view_and_copy(
779  memory_space(), rank_offsets_host );
780 
781  // Create the export steering vector for writing local elements into
782  // the send buffer. Note we create a local, shallow copy - this is a
783  // CUDA workaround for handling class private data.
784  _export_steering = Kokkos::View<std::size_t*,memory_space>(
785  Kokkos::ViewAllocateWithoutInitializing("export_steering"),
786  _total_num_export );
787  auto steer_vec = _export_steering;
788  Kokkos::View<std::size_t*,memory_space> counts( "counts", num_n );
789  Kokkos::parallel_for(
790  "Cabana::createSteering",
791  Kokkos::RangePolicy<execution_space>(0,_num_export_element),
792  KOKKOS_LAMBDA( const int i ){
793  if ( element_export_ranks(i) >= 0 )
794  steer_vec( rank_offsets( element_export_ranks(i) ) +
795  neighbor_ids(i) ) =
796  (use_iota) ? i : element_export_ids(i);
797  });
798  Kokkos::fence();
799  }
800 
801  private:
802 
803  MPI_Comm _comm;
804  std::vector<int> _neighbors;
805  std::size_t _total_num_export;
806  std::size_t _total_num_import;
807  std::vector<std::size_t> _num_export;
808  std::vector<std::size_t> _num_import;
809  std::size_t _num_export_element;
810  Kokkos::View<std::size_t*,device_type> _export_steering;
811 };
