89 //---------------------------------------------------------------------------//
90 // Count sends and generate the steering vector. Atomic version.
91 template <class ExportRankView>
92 auto countSendsAndCreateSteering( const ExportRankView element_export_ranks,
93  const int comm_size,
94  CountSendsAndCreateSteeringAtomic )
95  -> std::pair<Kokkos::View<int*, typename ExportRankView::device_type>,
96  Kokkos::View<typename ExportRankView::size_type*,
97  typename ExportRankView::device_type>>
98 {
99  using device_type = typename ExportRankView::device_type;
100  using execution_space = typename ExportRankView::execution_space;
101  using size_type = typename ExportRankView::size_type;
102 
103  // Create views.
104  Kokkos::View<int*, device_type> neighbor_counts( "neighbor_counts",
105  comm_size );
106  Kokkos::View<size_type*, device_type> neighbor_ids(
107  Kokkos::ViewAllocateWithoutInitializing( "neighbor_ids" ),
108  element_export_ranks.size() );
109 
110  // Count the sends and create the steering vector.
111 
112  // For smaller values of comm_size, use the optimized scratch memory path
113  // Value of 64 is arbitrary, should be at least 27 to cover 3-d halos?
114  if ( comm_size <= 64 )
115  {
116  constexpr int team_size = 256;
117  Kokkos::TeamPolicy<execution_space> team_policy(
118  ( element_export_ranks.size() + team_size - 1 ) / team_size,
119  team_size );
120  team_policy = team_policy.set_scratch_size(
121  0,
122  Kokkos::PerTeam( sizeof( int ) * ( team_size + 2 * comm_size ) ) );
123  Kokkos::parallel_for(
124  "Cabana::CommunicationPlan::countSendsAndCreateSteeringShared",
125  team_policy,
126  KOKKOS_LAMBDA(
127  const typename Kokkos::TeamPolicy<execution_space>::member_type&
128  team ) {
129  // NOTE: `get_shmem` returns shared memory pointers *aligned to
130  // 8 bytes*, so if `comm_size` is odd we can get erroneously
131  // padded offsets if we call `get_shmem` separately for each
132  // shared memory intermediary. Acquiring all the needed scratch
133  // memory at once then computing pointer offsets by hand avoids
134  // this issue.
135  int* scratch = (int*)team.team_shmem().get_shmem(
136  ( team.team_size() + 2 * comm_size ) * sizeof( int ), 0 );
137 
138  // local neighbor_ids, gives the local offset relative to
139  // calculated global offset. Size team.team_size() * sizeof(int)
140  int* local_neighbor_ids = scratch;
141  // local histogram, `comm_size` in size
142  int* histo = local_neighbor_ids + team.team_size();
143  // offset into global array, `comm_size` in size
144  int* global_offset = histo + comm_size;
145  // overall element `tid`
146  const auto tid =
147  team.team_rank() + team.league_rank() * team.team_size();
148  // local element
149  const int local_id = team.team_rank();
150  // total number of elements, for convenience
151  const int num_elements = element_export_ranks.size();
152  // my element export rank
153  const int my_element_export_rank =
154  ( tid < num_elements ? element_export_ranks( tid ) : -1 );
155 
156  // cannot outright terminate early b/c threads share work
157  // if (tid >= num_elements) return;
158  const bool in_bounds =
159  tid < num_elements && my_element_export_rank >= 0;
160 
161  Kokkos::parallel_for(
162  Kokkos::TeamThreadRange( team, comm_size ),
163  [&]( const int i ) { histo[i] = 0; } );
164 
165  // synchronize zeroing
166  team.team_barrier();
167 
168  // build local histogram, need to encode num_elements check here
169  if ( in_bounds )
170  {
171  // shared memory atomic add, accumulate into local offset
172  local_neighbor_ids[local_id] = Kokkos::atomic_fetch_add(
173  &histo[my_element_export_rank], 1 );
174  }
175 
176  // synchronize local histogram build
177  team.team_barrier();
178 
179  // reserve space in global array via a loop over neighbor counts
180  Kokkos::parallel_for(
181  Kokkos::TeamThreadRange( team, comm_size ),
182  [&]( const int i ) {
183  // global memory atomic add, reserves space
184  global_offset[i] = Kokkos::atomic_fetch_add(
185  &neighbor_counts( i ), histo[i] );
186  } );
187 
188  // synchronize block-stride loop
189  team.team_barrier();
190 
191  // and now store to my location
192  if ( in_bounds )
193  {
194  neighbor_ids( tid ) =
195  global_offset[my_element_export_rank] +
196  local_neighbor_ids[local_id];
197  }
198  } );
199  }
200  else
201  {
202  // For larger numbers of export ranks (for ex, migration) we use the
203  // global memory version, though this is a point of future optimization
204 
205  Kokkos::parallel_for(
206  "Cabana::CommunicationPlan::countSendsAndCreateSteering",
207  Kokkos::RangePolicy<execution_space>( 0,
208  element_export_ranks.size() ),
209  KOKKOS_LAMBDA( const size_type i ) {
210  if ( element_export_ranks( i ) >= 0 )
211  neighbor_ids( i ) = Kokkos::atomic_fetch_add(
212  &neighbor_counts( element_export_ranks( i ) ), 1 );
213  } );
214  }
215  Kokkos::fence();
216 
217  // Return the counts and ids.
218  return std::make_pair( neighbor_counts, neighbor_ids );
219 }
220 //---------------------------------------------------------------------------//
221 // Count sends and generate the steering vector. Duplicated version.
222 template <class ExportRankView>
223 auto countSendsAndCreateSteering( const ExportRankView element_export_ranks,
224  const int comm_size,
225  CountSendsAndCreateSteeringDuplicated )
226  -> std::pair<Kokkos::View<int*, typename ExportRankView::device_type>,
227  Kokkos::View<typename ExportRankView::size_type*,
228  typename ExportRankView::device_type>>
229 {
230  using device_type = typename ExportRankView::device_type;
231  using execution_space = typename ExportRankView::execution_space;
232  using size_type = typename ExportRankView::size_type;
233 
234  // Create a unique thread token.
235  Kokkos::Experimental::UniqueToken<
236  execution_space, Kokkos::Experimental::UniqueTokenScope::Global>
237  unique_token;
238 
239  // Create views.
240  Kokkos::View<int*, device_type> neighbor_counts(
241  Kokkos::ViewAllocateWithoutInitializing( "neighbor_counts" ),
242  comm_size );
243  Kokkos::View<size_type*, device_type> neighbor_ids(
244  Kokkos::ViewAllocateWithoutInitializing( "neighbor_ids" ),
245  element_export_ranks.size() );
246  Kokkos::View<int**, device_type> neighbor_counts_dup(
247  "neighbor_counts", unique_token.size(), comm_size );
248  Kokkos::View<size_type**, device_type> neighbor_ids_dup(
249  "neighbor_ids", unique_token.size(), element_export_ranks.size() );
250 
251  // Compute initial duplicated sends and steering.
252  Kokkos::parallel_for(
253  "Cabana::CommunicationPlan::intialCount",
254  Kokkos::RangePolicy<execution_space>( 0, element_export_ranks.size() ),
255  KOKKOS_LAMBDA( const size_type i ) {
256  if ( element_export_ranks( i ) >= 0 )
257  {
258  // Get the thread id.
259  auto thread_id = unique_token.acquire();
260 
261  // Do the duplicated fetch-add. If this is a valid element id
262  // increment the send count for this rank. Add the incremented
263  // count as the thread-local neighbor id. This is too big by
264  // one (because we use the prefix increment operator) but we
265  // want a non-zero value so we can later find which thread
266  // this element was located on because we are always
267  // guaranteed a non-zero value. We will subtract this value
268  // later.
269  neighbor_ids_dup( thread_id, i ) = ++neighbor_counts_dup(
270  thread_id, element_export_ranks( i ) );
271 
272  // Release the thread id.
273  unique_token.release( thread_id );
274  }
275  } );
276  Kokkos::fence();
277 
278  // Team policy
279  using team_policy =
280  Kokkos::TeamPolicy<execution_space, Kokkos::Schedule<Kokkos::Dynamic>>;
281  using index_type = typename team_policy::index_type;
282 
283  // Compute the send counts for each neighbor rank by reducing across
284  // the thread duplicates.
285  Kokkos::parallel_for(
286  "Cabana::CommunicationPlan::finalCount",
287  team_policy( neighbor_counts.extent( 0 ), Kokkos::AUTO ),
288  KOKKOS_LAMBDA( const typename team_policy::member_type& team ) {
289  // Get the element id.
290  auto i = team.league_rank();
291 
292  // Add the thread results.
293  int thread_counts = 0;
294  Kokkos::parallel_reduce(
295  Kokkos::TeamThreadRange( team,
296  neighbor_counts_dup.extent( 0 ) ),
297  [&]( const index_type thread_id, int& result ) {
298  result += neighbor_counts_dup( thread_id, i );
299  },
300  thread_counts );
301  neighbor_counts( i ) = thread_counts;
302  } );
303  Kokkos::fence();
304 
305  // Compute the location of each export element in the send buffer of
306  // its destination rank.
307  Kokkos::parallel_for(
308  "Cabana::CommunicationPlan::createSteering",
309  team_policy( element_export_ranks.size(), Kokkos::AUTO ),
310  KOKKOS_LAMBDA( const typename team_policy::member_type& team ) {
311  // Get the element id.
312  auto i = team.league_rank();
313 
314  // Only operate on valid elements
315  if ( element_export_ranks( i ) >= 0 )
316  {
317  // Compute the thread id in which we located the element
318  // during the count phase. Only the thread in which we
319  // located the element will contribute to the reduction.
320  index_type dup_thread = 0;
321  Kokkos::parallel_reduce(
322  Kokkos::TeamThreadRange( team,
323  neighbor_ids_dup.extent( 0 ) ),
324  [&]( const index_type thread_id, index_type& result ) {
325  if ( neighbor_ids_dup( thread_id, i ) > 0 )
326  result += thread_id;
327  },
328  dup_thread );
329 
330  // Compute the offset of this element in the steering
331  // vector for its destination rank. Loop through the
332  // threads up to the thread that found this element in the
333  // count stage. All thread counts prior to that thread
334  // will contribute to the offset.
335  size_type thread_offset = 0;
336  Kokkos::parallel_reduce(
337  Kokkos::TeamThreadRange( team, dup_thread ),
338  [&]( const index_type thread_id, size_type& result ) {
339  result += neighbor_counts_dup(
340  thread_id, element_export_ranks( i ) );
341  },
342  thread_offset );
343 
344  // Add the thread-local value to the offset where we subtract
345  // the 1 that we added artificially when we were first
346  // counting.
347  neighbor_ids( i ) =
348  thread_offset + neighbor_ids_dup( dup_thread, i ) - 1;
349  }
350  } );
351  Kokkos::fence();
352 
353  // Return the counts and ids.
354  return std::make_pair( neighbor_counts, neighbor_ids );
355 }

357 //---------------------------------------------------------------------------//
358 // Return unique neighbor ranks, with the current rank first.
359 std::vector<int> getUniqueTopology( std::vector<int> topology )
360 {
361  auto remove_end = std::remove( topology.begin(), topology.end(), -1 );
362  std::sort( topology.begin(), remove_end );
363  auto unique_end = std::unique( topology.begin(), remove_end );
364  topology.resize( std::distance( topology.begin(), unique_end ) );
365 
366  // Put this rank first.
367  int my_rank = -1;
368  MPI_Comm_rank( MPI_COMM_WORLD, &my_rank );
369  for ( auto& n : topology )
370  {
371  if ( n == my_rank )
372  {
373  std::swap( n, topology[0] );
374  break;
375  }
376  }
377  return topology;
378 }

588  template <class ViewType>
589  Kokkos::View<size_type*, device_type>
590  createFromExportsAndTopology( const ViewType& element_export_ranks,
591  const std::vector<int>& neighbor_ranks )
592  {
593  // Store the number of export elements.
594  _num_export_element = element_export_ranks.size();
595 
596  // Store the unique neighbors (this rank first).
597  _neighbors = Impl::getUniqueTopology( neighbor_ranks );
598  int num_n = _neighbors.size();
599 
600  // Get the size of this communicator.
601  int comm_size = -1;
602  MPI_Comm_size( comm(), &comm_size );
603 
604  // Get the MPI rank we are currently on.
605  int my_rank = -1;
606  MPI_Comm_rank( comm(), &my_rank );
607 
608  // Pick an mpi tag for communication. This object has it's own
609  // communication space so any mpi tag will do.
610  const int mpi_tag = 1221;
611 
612  // Initialize import/export sizes.
613  _num_export.assign( num_n, 0 );
614  _num_import.assign( num_n, 0 );
615 
616  // Count the number of sends this rank will do to other ranks. Keep
617  // track of which slot we get in our neighbor's send buffer.
618  auto counts_and_ids = Impl::countSendsAndCreateSteering(
619  element_export_ranks, comm_size,
620  typename Impl::CountSendsAndCreateSteeringAlgorithm<
621  execution_space>::type() );
622 
623  // Copy the counts to the host.
624  auto neighbor_counts_host = Kokkos::create_mirror_view_and_copy(
625  Kokkos::HostSpace(), counts_and_ids.first );
626 
627  // Get the export counts.
628  for ( int n = 0; n < num_n; ++n )
629  _num_export[n] = neighbor_counts_host( _neighbors[n] );
630 
631  // Post receives for the number of imports we will get.
632  std::vector<MPI_Request> requests;
633  requests.reserve( num_n );
634  for ( int n = 0; n < num_n; ++n )
635  if ( my_rank != _neighbors[n] )
636  {
637  requests.push_back( MPI_Request() );
638  MPI_Irecv( &_num_import[n], 1, MPI_UNSIGNED_LONG, _neighbors[n],
639  mpi_tag, comm(), &( requests.back() ) );
640  }
641  else
642  _num_import[n] = _num_export[n];
643 
644  // Send the number of exports to each of our neighbors.
645  for ( int n = 0; n < num_n; ++n )
646  if ( my_rank != _neighbors[n] )
647  MPI_Send( &_num_export[n], 1, MPI_UNSIGNED_LONG, _neighbors[n],
648  mpi_tag, comm() );
649 
650  // Wait on receives.
651  std::vector<MPI_Status> status( requests.size() );
652  const int ec =
653  MPI_Waitall( requests.size(), requests.data(), status.data() );
654  if ( MPI_SUCCESS != ec )
655  throw std::logic_error( "Failed MPI Communication" );
656 
657  // Get the total number of imports/exports.
658  _total_num_export =
659  std::accumulate( _num_export.begin(), _num_export.end(), 0 );
660  _total_num_import =
661  std::accumulate( _num_import.begin(), _num_import.end(), 0 );
662 
663  // Barrier before continuing to ensure synchronization.
664  MPI_Barrier( comm() );
665 
666  // Return the neighbor ids.
667  return counts_and_ids.second;
668  }

698  template <class ViewType>
699  Kokkos::View<size_type*, device_type>
700  createFromExportsOnly( const ViewType& element_export_ranks )
701  {
702  // Store the number of export elements.
703  _num_export_element = element_export_ranks.size();
704 
705  // Get the size of this communicator.
706  int comm_size = -1;
707  MPI_Comm_size( comm(), &comm_size );
708 
709  // Get the MPI rank we are currently on.
710  int my_rank = -1;
711  MPI_Comm_rank( comm(), &my_rank );
712 
713  // Pick an mpi tag for communication. This object has it's own
714  // communication space so any mpi tag will do.
715  const int mpi_tag = 1221;
716 
717  // Count the number of sends this rank will do to other ranks. Keep
718  // track of which slot we get in our neighbor's send buffer.
719  auto counts_and_ids = Impl::countSendsAndCreateSteering(
720  element_export_ranks, comm_size,
721  typename Impl::CountSendsAndCreateSteeringAlgorithm<
722  execution_space>::type() );
723 
724  // Copy the counts to the host.
725  auto neighbor_counts_host = Kokkos::create_mirror_view_and_copy(
726  Kokkos::HostSpace(), counts_and_ids.first );
727 
728  // Extract the export ranks and number of exports and then flag the
729  // send ranks.
730  _neighbors.clear();
731  _num_export.clear();
732  _total_num_export = 0;
733  for ( int r = 0; r < comm_size; ++r )
734  if ( neighbor_counts_host( r ) > 0 )
735  {
736  _neighbors.push_back( r );
737  _num_export.push_back( neighbor_counts_host( r ) );
738  _total_num_export += neighbor_counts_host( r );
739  neighbor_counts_host( r ) = 1;
740  }
741 
742  // Get the number of export ranks and initially allocate the import
743  // sizes.
744  int num_export_rank = _neighbors.size();
745  _num_import.assign( num_export_rank, 0 );
746 
747  // If we are sending to ourself put that one first in the neighbor
748  // list and assign the number of imports to be the number of exports.
749  bool self_send = false;
750  for ( int n = 0; n < num_export_rank; ++n )
751  if ( _neighbors[n] == my_rank )
752  {
753  std::swap( _neighbors[n], _neighbors[0] );
754  std::swap( _num_export[n], _num_export[0] );
755  _num_import[0] = _num_export[0];
756  self_send = true;
757  break;
758  }
759 
760  // Determine how many total import ranks each neighbor has.
761  int num_import_rank = -1;
762  std::vector<int> recv_counts( comm_size, 1 );
763  MPI_Reduce_scatter( neighbor_counts_host.data(), &num_import_rank,
764  recv_counts.data(), MPI_INT, MPI_SUM, comm() );
765  if ( self_send )
766  --num_import_rank;
767 
768  // Post the expected number of receives and indicate we might get them
769  // from any rank.
770  std::vector<std::size_t> import_sizes( num_import_rank );
771  std::vector<MPI_Request> requests( num_import_rank );
772  for ( int n = 0; n < num_import_rank; ++n )
773  MPI_Irecv( &import_sizes[n], 1, MPI_UNSIGNED_LONG, MPI_ANY_SOURCE,
774  mpi_tag, comm(), &requests[n] );
775 
776  // Do blocking sends. Dont do any self sends.
777  int self_offset = ( self_send ) ? 1 : 0;
778  for ( int n = self_offset; n < num_export_rank; ++n )
779  MPI_Send( &_num_export[n], 1, MPI_UNSIGNED_LONG, _neighbors[n],
780  mpi_tag, comm() );
781 
782  // Wait on non-blocking receives.
783  std::vector<MPI_Status> status( requests.size() );
784  const int ec =
785  MPI_Waitall( requests.size(), requests.data(), status.data() );
786  if ( MPI_SUCCESS != ec )
787  throw std::logic_error( "Failed MPI Communication" );
788 
789  // Compute the total number of imports.
790  _total_num_import =
791  std::accumulate( import_sizes.begin(), import_sizes.end(),
792  ( self_send ) ? _num_import[0] : 0 );
793 
794  // Extract the imports. If we did self sends we already know what
795  // imports we got from that.
796  for ( int i = 0; i < num_import_rank; ++i )
797  {
798  // Get the message source.
799  const auto source = status[i].MPI_SOURCE;
800 
801  // See if the neighbor we received stuff from was someone we also
802  // sent stuff to.
803  auto found_neighbor =
804  std::find( _neighbors.begin(), _neighbors.end(), source );
805 
806  // If this is a new neighbor (i.e. someone we didn't send anything
807  // to) record this.
808  if ( found_neighbor == std::end( _neighbors ) )
809  {
810  _neighbors.push_back( source );
811  _num_import.push_back( import_sizes[i] );
812  _num_export.push_back( 0 );
813  }
814 
815  // Otherwise if we already sent something to this neighbor that
816  // means we already have a neighbor/export entry. Just assign the
817  // import entry for that neighbor.
818  else
819  {
820  auto n = std::distance( _neighbors.begin(), found_neighbor );
821  _num_import[n] = import_sizes[i];
822  }
823  }
824 
825  // Barrier before continuing to ensure synchronization.
826  MPI_Barrier( comm() );
827 
828  // Return the neighbor ids.
829  return counts_and_ids.second;
830  }

884  // Create the export steering vector.
885  template <class PackViewType, class RankViewType, class IdViewType>
886  void createSteering( const bool use_iota, const PackViewType& neighbor_ids,
887  const RankViewType& element_export_ranks,
888  const IdViewType& element_export_ids )
889  {
890  if ( !use_iota &&
891  ( element_export_ids.size() != element_export_ranks.size() ) )
892  throw std::runtime_error( "Export ids and ranks different sizes!" );
893 
894  // Get the size of this communicator.
895  int comm_size = -1;
896  MPI_Comm_size( *_comm_ptr, &comm_size );
897 
898  // Calculate the steering offsets via exclusive prefix sum for the
899  // exports.
900  int num_n = _neighbors.size();
901  std::vector<std::size_t> offsets( num_n, 0.0 );
902  for ( int n = 1; n < num_n; ++n )
903  offsets[n] = offsets[n - 1] + _num_export[n - 1];
904 
905  // Map the offsets to the device.
906  Kokkos::View<std::size_t*, Kokkos::HostSpace> rank_offsets_host(
907  Kokkos::ViewAllocateWithoutInitializing( "rank_map" ), comm_size );
908  for ( int n = 0; n < num_n; ++n )
909  rank_offsets_host( _neighbors[n] ) = offsets[n];
910  auto rank_offsets = Kokkos::create_mirror_view_and_copy(
911  memory_space(), rank_offsets_host );
912 
913  // Create the export steering vector for writing local elements into
914  // the send buffer. Note we create a local, shallow copy - this is a
915  // CUDA workaround for handling class private data.
916  _export_steering = Kokkos::View<std::size_t*, memory_space>(
917  Kokkos::ViewAllocateWithoutInitializing( "export_steering" ),
918  _total_num_export );
919  auto steer_vec = _export_steering;
920  Kokkos::parallel_for(
921  "Cabana::createSteering",
922  Kokkos::RangePolicy<execution_space>( 0, _num_export_element ),
923  KOKKOS_LAMBDA( const int i ) {
924  if ( element_export_ranks( i ) >= 0 )
925  steer_vec( rank_offsets( element_export_ranks( i ) ) +
926  neighbor_ids( i ) ) =
927  ( use_iota ) ? i : element_export_ids( i );
928  } );
929  Kokkos::fence();
930  }
932 
933  private:
934  std::shared_ptr<MPI_Comm> _comm_ptr;
935  std::vector<int> _neighbors;
936  std::size_t _total_num_export;
937  std::size_t _total_num_import;
938  std::vector<std::size_t> _num_export;
939  std::vector<std::size_t> _num_import;
940  std::size_t _num_export_element;
941  Kokkos::View<std::size_t*, device_type> _export_steering;
942 };
