84 //---------------------------------------------------------------------------//
85 // Count sends and generate the steering vector. Atomic version.
86 template <class ExportRankView>
87 auto countSendsAndCreateSteering( const ExportRankView element_export_ranks,
88  const int comm_size,
89  CountSendsAndCreateSteeringAtomic )
90  -> std::pair<Kokkos::View<int*, typename ExportRankView::device_type>,
91  Kokkos::View<typename ExportRankView::size_type*,
92  typename ExportRankView::device_type>>
93 {
94  using device_type = typename ExportRankView::device_type;
95  using execution_space = typename ExportRankView::execution_space;
96  using size_type = typename ExportRankView::size_type;
97 
98  // Create views.
99  Kokkos::View<int*, device_type> neighbor_counts( "neighbor_counts",
100  comm_size );
101  Kokkos::View<size_type*, device_type> neighbor_ids(
102  Kokkos::ViewAllocateWithoutInitializing( "neighbor_ids" ),
103  element_export_ranks.size() );
104 
105  // Count the sends and create the steering vector.
106 
107  // For smaller values of comm_size, use the optimized scratch memory path
108  // Value of 64 is arbitrary, should be at least 27 to cover 3-d halos?
109  if ( comm_size <= 64 )
110  {
111  constexpr int team_size = 256;
112  Kokkos::TeamPolicy<execution_space> team_policy(
113  ( element_export_ranks.size() + team_size - 1 ) / team_size,
114  team_size );
115  team_policy = team_policy.set_scratch_size(
116  0,
117  Kokkos::PerTeam( sizeof( int ) * ( team_size + 2 * comm_size ) ) );
118  Kokkos::parallel_for(
119  "Cabana::CommunicationPlan::countSendsAndCreateSteeringShared",
120  team_policy,
121  KOKKOS_LAMBDA(
122  const typename Kokkos::TeamPolicy<execution_space>::member_type&
123  team ) {
124  // NOTE: `get_shmem` returns shared memory pointers *aligned to
125  // 8 bytes*, so if `comm_size` is odd we can get erroneously
126  // padded offsets if we call `get_shmem` separately for each
127  // shared memory intermediary. Acquiring all the needed scratch
128  // memory at once then computing pointer offsets by hand avoids
129  // this issue.
130  int* scratch = (int*)team.team_shmem().get_shmem(
131  ( team.team_size() + 2 * comm_size ) * sizeof( int ), 0 );
132 
133  // local neighbor_ids, gives the local offset relative to
134  // calculated global offset. Size team.team_size() * sizeof(int)
135  int* local_neighbor_ids = scratch;
136  // local histogram, `comm_size` in size
137  int* histo = local_neighbor_ids + team.team_size();
138  // offset into global array, `comm_size` in size
139  int* global_offset = histo + comm_size;
140  // overall element `tid`
141  const auto tid =
142  team.team_rank() + team.league_rank() * team.team_size();
143  // local element
144  const int local_id = team.team_rank();
145  // total number of elements, for convenience
146  const int num_elements = element_export_ranks.size();
147  // my element export rank
148  const int my_element_export_rank =
149  ( tid < num_elements ? element_export_ranks( tid ) : -1 );
150 
151  // cannot outright terminate early b/c threads share work
152  // if (tid >= num_elements) return;
153  const bool in_bounds =
154  tid < num_elements && my_element_export_rank >= 0;
155 
156  Kokkos::parallel_for(
157  Kokkos::TeamThreadRange( team, comm_size ),
158  [&]( const int i ) { histo[i] = 0; } );
159 
160  // synchronize zeroing
161  team.team_barrier();
162 
163  // build local histogram, need to encode num_elements check here
164  if ( in_bounds )
165  {
166  // shared memory atomic add, accumulate into local offset
167  local_neighbor_ids[local_id] = Kokkos::atomic_fetch_add(
168  &histo[my_element_export_rank], 1 );
169  }
170 
171  // synchronize local histogram build
172  team.team_barrier();
173 
174  // reserve space in global array via a loop over neighbor counts
175  Kokkos::parallel_for(
176  Kokkos::TeamThreadRange( team, comm_size ),
177  [&]( const int i ) {
178  // global memory atomic add, reserves space
179  global_offset[i] = Kokkos::atomic_fetch_add(
180  &neighbor_counts( i ), histo[i] );
181  } );
182 
183  // synchronize block-stride loop
184  team.team_barrier();
185 
186  // and now store to my location
187  if ( in_bounds )
188  {
189  neighbor_ids( tid ) =
190  global_offset[my_element_export_rank] +
191  local_neighbor_ids[local_id];
192  }
193  } );
194  }
195  else
196  {
197  // For larger numbers of export ranks (for ex, migration) we use the
198  // global memory version, though this is a point of future optimization
199 
200  Kokkos::parallel_for(
201  "Cabana::CommunicationPlan::countSendsAndCreateSteering",
202  Kokkos::RangePolicy<execution_space>( 0,
203  element_export_ranks.size() ),
204  KOKKOS_LAMBDA( const size_type i ) {
205  if ( element_export_ranks( i ) >= 0 )
206  neighbor_ids( i ) = Kokkos::atomic_fetch_add(
207  &neighbor_counts( element_export_ranks( i ) ), 1 );
208  } );
209  }
210  Kokkos::fence();
211 
212  // Return the counts and ids.
213  return std::make_pair( neighbor_counts, neighbor_ids );
214 }
215 //---------------------------------------------------------------------------//
216 // Count sends and generate the steering vector. Duplicated version.
217 template <class ExportRankView>
218 auto countSendsAndCreateSteering( const ExportRankView element_export_ranks,
219  const int comm_size,
220  CountSendsAndCreateSteeringDuplicated )
221  -> std::pair<Kokkos::View<int*, typename ExportRankView::device_type>,
222  Kokkos::View<typename ExportRankView::size_type*,
223  typename ExportRankView::device_type>>
224 {
225  using device_type = typename ExportRankView::device_type;
226  using execution_space = typename ExportRankView::execution_space;
227  using size_type = typename ExportRankView::size_type;
228 
229  // Create a unique thread token.
230  Kokkos::Experimental::UniqueToken<
231  execution_space, Kokkos::Experimental::UniqueTokenScope::Global>
232  unique_token;
233 
234  // Create views.
235  Kokkos::View<int*, device_type> neighbor_counts(
236  Kokkos::ViewAllocateWithoutInitializing( "neighbor_counts" ),
237  comm_size );
238  Kokkos::View<size_type*, device_type> neighbor_ids(
239  Kokkos::ViewAllocateWithoutInitializing( "neighbor_ids" ),
240  element_export_ranks.size() );
241  Kokkos::View<int**, device_type> neighbor_counts_dup(
242  "neighbor_counts", unique_token.size(), comm_size );
243  Kokkos::View<size_type**, device_type> neighbor_ids_dup(
244  "neighbor_ids", unique_token.size(), element_export_ranks.size() );
245 
246  // Compute initial duplicated sends and steering.
247  Kokkos::parallel_for(
248  "Cabana::CommunicationPlan::intialCount",
249  Kokkos::RangePolicy<execution_space>( 0, element_export_ranks.size() ),
250  KOKKOS_LAMBDA( const size_type i ) {
251  if ( element_export_ranks( i ) >= 0 )
252  {
253  // Get the thread id.
254  auto thread_id = unique_token.acquire();
255 
256  // Do the duplicated fetch-add. If this is a valid element id
257  // increment the send count for this rank. Add the incremented
258  // count as the thread-local neighbor id. This is too big by
259  // one (because we use the prefix increment operator) but we
260  // want a non-zero value so we can later find which thread
261  // this element was located on because we are always
262  // guaranteed a non-zero value. We will subtract this value
263  // later.
264  neighbor_ids_dup( thread_id, i ) = ++neighbor_counts_dup(
265  thread_id, element_export_ranks( i ) );
266 
267  // Release the thread id.
268  unique_token.release( thread_id );
269  }
270  } );
271  Kokkos::fence();
272 
273  // Team policy
274  using team_policy =
275  Kokkos::TeamPolicy<execution_space, Kokkos::Schedule<Kokkos::Dynamic>>;
276  using index_type = typename team_policy::index_type;
277 
278  // Compute the send counts for each neighbor rank by reducing across
279  // the thread duplicates.
280  Kokkos::parallel_for(
281  "Cabana::CommunicationPlan::finalCount",
282  team_policy( neighbor_counts.extent( 0 ), Kokkos::AUTO ),
283  KOKKOS_LAMBDA( const typename team_policy::member_type& team ) {
284  // Get the element id.
285  auto i = team.league_rank();
286 
287  // Add the thread results.
288  int thread_counts = 0;
289  Kokkos::parallel_reduce(
290  Kokkos::TeamThreadRange( team,
291  neighbor_counts_dup.extent( 0 ) ),
292  [&]( const index_type thread_id, int& result ) {
293  result += neighbor_counts_dup( thread_id, i );
294  },
295  thread_counts );
296  neighbor_counts( i ) = thread_counts;
297  } );
298  Kokkos::fence();
299 
300  // Compute the location of each export element in the send buffer of
301  // its destination rank.
302  Kokkos::parallel_for(
303  "Cabana::CommunicationPlan::createSteering",
304  team_policy( element_export_ranks.size(), Kokkos::AUTO ),
305  KOKKOS_LAMBDA( const typename team_policy::member_type& team ) {
306  // Get the element id.
307  auto i = team.league_rank();
308 
309  // Only operate on valid elements
310  if ( element_export_ranks( i ) >= 0 )
311  {
312  // Compute the thread id in which we located the element
313  // during the count phase. Only the thread in which we
314  // located the element will contribute to the reduction.
315  index_type dup_thread = 0;
316  Kokkos::parallel_reduce(
317  Kokkos::TeamThreadRange( team,
318  neighbor_ids_dup.extent( 0 ) ),
319  [&]( const index_type thread_id, index_type& result ) {
320  if ( neighbor_ids_dup( thread_id, i ) > 0 )
321  result += thread_id;
322  },
323  dup_thread );
324 
325  // Compute the offset of this element in the steering
326  // vector for its destination rank. Loop through the
327  // threads up to the thread that found this element in the
328  // count stage. All thread counts prior to that thread
329  // will contribute to the offset.
330  size_type thread_offset = 0;
331  Kokkos::parallel_reduce(
332  Kokkos::TeamThreadRange( team, dup_thread ),
333  [&]( const index_type thread_id, size_type& result ) {
334  result += neighbor_counts_dup(
335  thread_id, element_export_ranks( i ) );
336  },
337  thread_offset );
338 
339  // Add the thread-local value to the offset where we subtract
340  // the 1 that we added artificially when we were first
341  // counting.
342  neighbor_ids( i ) =
343  thread_offset + neighbor_ids_dup( dup_thread, i ) - 1;
344  }
345  } );
346  Kokkos::fence();
347 
348  // Return the counts and ids.
349  return std::make_pair( neighbor_counts, neighbor_ids );
350 }

384 template <class DeviceType>
385 class CommunicationPlan

387  public:

389  using device_type = DeviceType;

391  // Memory space.

407  CommunicationPlan( MPI_Comm comm )
408  {
409  _comm_ptr.reset(
410  // Duplicate the communicator and store in a std::shared_ptr so that
411  // all copies point to the same object
412  [comm]() {
413  auto p = std::make_unique<MPI_Comm>();
414  MPI_Comm_dup( comm, p.get() );
415  return p.release();
416  }(),
417  // Custom deleter to mark the communicator for deallocation
418  []( MPI_Comm* p ) {
419  MPI_Comm_free( p );
420  delete p;
421  } );
422  }

523  // The functions in the public block below would normally be protected but
524  // we make them public to allow using private class data in CUDA kernels
525  // with lambda functions.
526  public:
562  template <class ViewType>
563  Kokkos::View<size_type*, device_type>
564  createFromExportsAndTopology( const ViewType& element_export_ranks,
565  const std::vector<int>& neighbor_ranks )
566  {
567  // Store the number of export elements.
568  _num_export_element = element_export_ranks.size();
569 
570  // Store the unique neighbors.
571  _neighbors = neighbor_ranks;
572  std::sort( _neighbors.begin(), _neighbors.end() );
573  auto unique_end = std::unique( _neighbors.begin(), _neighbors.end() );
574  _neighbors.resize( std::distance( _neighbors.begin(), unique_end ) );
575  int num_n = _neighbors.size();
576 
577  // Get the size of this communicator.
578  int comm_size = -1;
579  MPI_Comm_size( comm(), &comm_size );
580 
581  // Get the MPI rank we are currently on.
582  int my_rank = -1;
583  MPI_Comm_rank( comm(), &my_rank );
584 
585  // Pick an mpi tag for communication. This object has it's own
586  // communication space so any mpi tag will do.
587  const int mpi_tag = 1221;
588 
589  // If we are sending to ourself put that one first in the neighbor
590  // list.
591  for ( auto& n : _neighbors )
592  if ( n == my_rank )
593  {
594  std::swap( n, _neighbors[0] );
595  break;
596  }
597 
598  // Initialize import/export sizes.
599  _num_export.assign( num_n, 0 );
600  _num_import.assign( num_n, 0 );
601 
602  // Count the number of sends this rank will do to other ranks. Keep
603  // track of which slot we get in our neighbor's send buffer.
604  auto counts_and_ids = Impl::countSendsAndCreateSteering(
605  element_export_ranks, comm_size,
606  typename Impl::CountSendsAndCreateSteeringAlgorithm<
607  execution_space>::type() );
608 
609  // Copy the counts to the host.
610  auto neighbor_counts_host = Kokkos::create_mirror_view_and_copy(
611  Kokkos::HostSpace(), counts_and_ids.first );
612 
613  // Get the export counts.
614  for ( int n = 0; n < num_n; ++n )
615  _num_export[n] = neighbor_counts_host( _neighbors[n] );
616 
617  // Post receives for the number of imports we will get.
618  std::vector<MPI_Request> requests;
619  requests.reserve( num_n );
620  for ( int n = 0; n < num_n; ++n )
621  if ( my_rank != _neighbors[n] )
622  {
623  requests.push_back( MPI_Request() );
624  MPI_Irecv( &_num_import[n], 1, MPI_UNSIGNED_LONG, _neighbors[n],
625  mpi_tag, comm(), &( requests.back() ) );
626  }
627  else
628  _num_import[n] = _num_export[n];
629 
630  // Send the number of exports to each of our neighbors.
631  for ( int n = 0; n < num_n; ++n )
632  if ( my_rank != _neighbors[n] )
633  MPI_Send( &_num_export[n], 1, MPI_UNSIGNED_LONG, _neighbors[n],
634  mpi_tag, comm() );
635 
636  // Wait on receives.
637  std::vector<MPI_Status> status( requests.size() );
638  const int ec =
639  MPI_Waitall( requests.size(), requests.data(), status.data() );
640  if ( MPI_SUCCESS != ec )
641  throw std::logic_error( "Failed MPI Communication" );
642 
643  // Get the total number of imports/exports.
644  _total_num_export =
645  std::accumulate( _num_export.begin(), _num_export.end(), 0 );
646  _total_num_import =
647  std::accumulate( _num_import.begin(), _num_import.end(), 0 );
648 
649  // Barrier before continuing to ensure synchronization.
650  MPI_Barrier( comm() );
651 
652  // Return the neighbor ids.
653  return counts_and_ids.second;
654  }

684  template <class ViewType>
685  Kokkos::View<size_type*, device_type>
686  createFromExportsOnly( const ViewType& element_export_ranks )
687  {
688  // Store the number of export elements.
689  _num_export_element = element_export_ranks.size();
690 
691  // Get the size of this communicator.
692  int comm_size = -1;
693  MPI_Comm_size( comm(), &comm_size );
694 
695  // Get the MPI rank we are currently on.
696  int my_rank = -1;
697  MPI_Comm_rank( comm(), &my_rank );
698 
699  // Pick an mpi tag for communication. This object has it's own
700  // communication space so any mpi tag will do.
701  const int mpi_tag = 1221;
702 
703  // Count the number of sends this rank will do to other ranks. Keep
704  // track of which slot we get in our neighbor's send buffer.
705  auto counts_and_ids = Impl::countSendsAndCreateSteering(
706  element_export_ranks, comm_size,
707  typename Impl::CountSendsAndCreateSteeringAlgorithm<
708  execution_space>::type() );
709 
710  // Copy the counts to the host.
711  auto neighbor_counts_host = Kokkos::create_mirror_view_and_copy(
712  Kokkos::HostSpace(), counts_and_ids.first );
713 
714  // Extract the export ranks and number of exports and then flag the
715  // send ranks.
716  _neighbors.clear();
717  _num_export.clear();
718  _total_num_export = 0;
719  for ( int r = 0; r < comm_size; ++r )
720  if ( neighbor_counts_host( r ) > 0 )
721  {
722  _neighbors.push_back( r );
723  _num_export.push_back( neighbor_counts_host( r ) );
724  _total_num_export += neighbor_counts_host( r );
725  neighbor_counts_host( r ) = 1;
726  }
727 
728  // Get the number of export ranks and initially allocate the import
729  // sizes.
730  int num_export_rank = _neighbors.size();
731  _num_import.assign( num_export_rank, 0 );
732 
733  // If we are sending to ourself put that one first in the neighbor
734  // list and assign the number of imports to be the number of exports.
735  bool self_send = false;
736  for ( int n = 0; n < num_export_rank; ++n )
737  if ( _neighbors[n] == my_rank )
738  {
739  std::swap( _neighbors[n], _neighbors[0] );
740  std::swap( _num_export[n], _num_export[0] );
741  _num_import[0] = _num_export[0];
742  self_send = true;
743  break;
744  }
745 
746  // Determine how many total import ranks each neighbor has.
747  int num_import_rank = -1;
748  std::vector<int> recv_counts( comm_size, 1 );
749  MPI_Reduce_scatter( neighbor_counts_host.data(), &num_import_rank,
750  recv_counts.data(), MPI_INT, MPI_SUM, comm() );
751  if ( self_send )
752  --num_import_rank;
753 
754  // Post the expected number of receives and indicate we might get them
755  // from any rank.
756  std::vector<std::size_t> import_sizes( num_import_rank );
757  std::vector<MPI_Request> requests( num_import_rank );
758  for ( int n = 0; n < num_import_rank; ++n )
759  MPI_Irecv( &import_sizes[n], 1, MPI_UNSIGNED_LONG, MPI_ANY_SOURCE,
760  mpi_tag, comm(), &requests[n] );
761 
762  // Do blocking sends. Dont do any self sends.
763  int self_offset = ( self_send ) ? 1 : 0;
764  for ( int n = self_offset; n < num_export_rank; ++n )
765  MPI_Send( &_num_export[n], 1, MPI_UNSIGNED_LONG, _neighbors[n],
766  mpi_tag, comm() );
767 
768  // Wait on non-blocking receives.
769  std::vector<MPI_Status> status( requests.size() );
770  const int ec =
771  MPI_Waitall( requests.size(), requests.data(), status.data() );
772  if ( MPI_SUCCESS != ec )
773  throw std::logic_error( "Failed MPI Communication" );
774 
775  // Compute the total number of imports.
776  _total_num_import =
777  std::accumulate( import_sizes.begin(), import_sizes.end(),
778  ( self_send ) ? _num_import[0] : 0 );
779 
780  // Extract the imports. If we did self sends we already know what
781  // imports we got from that.
782  for ( int i = 0; i < num_import_rank; ++i )
783  {
784  // Get the message source.
785  const auto source = status[i].MPI_SOURCE;
786 
787  // See if the neighbor we received stuff from was someone we also
788  // sent stuff to.
789  auto found_neighbor =
790  std::find( _neighbors.begin(), _neighbors.end(), source );
791 
792  // If this is a new neighbor (i.e. someone we didn't send anything
793  // to) record this.
794  if ( found_neighbor == std::end( _neighbors ) )
795  {
796  _neighbors.push_back( source );
797  _num_import.push_back( import_sizes[i] );
798  _num_export.push_back( 0 );
799  }
800 
801  // Otherwise if we already sent something to this neighbor that
802  // means we already have a neighbor/export entry. Just assign the
803  // import entry for that neighbor.
804  else
805  {
806  auto n = std::distance( _neighbors.begin(), found_neighbor );
807  _num_import[n] = import_sizes[i];
808  }
809  }
810 
811  // Barrier before continuing to ensure synchronization.
812  MPI_Barrier( comm() );
813 
814  // Return the neighbor ids.
815  return counts_and_ids.second;
816  }

869  // Create the export steering vector.
870  template <class PackViewType, class RankViewType, class IdViewType>
871  void createSteering( const bool use_iota, const PackViewType& neighbor_ids,
872  const RankViewType& element_export_ranks,
873  const IdViewType& element_export_ids )
874  {
875  if ( !use_iota &&
876  ( element_export_ids.size() != element_export_ranks.size() ) )
877  throw std::runtime_error( "Export ids and ranks different sizes!" );
878 
879  // Get the size of this communicator.
880  int comm_size = -1;
881  MPI_Comm_size( *_comm_ptr, &comm_size );
882 
883  // Calculate the steering offsets via exclusive prefix sum for the
884  // exports.
885  int num_n = _neighbors.size();
886  std::vector<std::size_t> offsets( num_n, 0.0 );
887  for ( int n = 1; n < num_n; ++n )
888  offsets[n] = offsets[n - 1] + _num_export[n - 1];
889 
890  // Map the offsets to the device.
891  Kokkos::View<std::size_t*, Kokkos::HostSpace> rank_offsets_host(
892  Kokkos::ViewAllocateWithoutInitializing( "rank_map" ), comm_size );
893  for ( int n = 0; n < num_n; ++n )
894  rank_offsets_host( _neighbors[n] ) = offsets[n];
895  auto rank_offsets = Kokkos::create_mirror_view_and_copy(
896  memory_space(), rank_offsets_host );
897 
898  // Create the export steering vector for writing local elements into
899  // the send buffer. Note we create a local, shallow copy - this is a
900  // CUDA workaround for handling class private data.
901  _export_steering = Kokkos::View<std::size_t*, memory_space>(
902  Kokkos::ViewAllocateWithoutInitializing( "export_steering" ),
903  _total_num_export );
904  auto steer_vec = _export_steering;
905  Kokkos::parallel_for(
906  "Cabana::createSteering",
907  Kokkos::RangePolicy<execution_space>( 0, _num_export_element ),
908  KOKKOS_LAMBDA( const int i ) {
909  if ( element_export_ranks( i ) >= 0 )
910  steer_vec( rank_offsets( element_export_ranks( i ) ) +
911  neighbor_ids( i ) ) =
912  ( use_iota ) ? i : element_export_ids( i );
913  } );
914  Kokkos::fence();
915  }
916 
917  private:
918  std::shared_ptr<MPI_Comm> _comm_ptr;
919  std::vector<int> _neighbors;
920  std::size_t _total_num_export;
921  std::size_t _total_num_import;
922  std::vector<std::size_t> _num_export;
923  std::vector<std::size_t> _num_import;
924  std::size_t _num_export_element;
925  Kokkos::View<std::size_t*, device_type> _export_steering;
926 };
