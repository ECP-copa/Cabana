90 // Count sends and generate the steering vector. Atomic version.
91 template <class ExportRankView>
92 auto countSendsAndCreateSteering( const ExportRankView element_export_ranks,
93  const int comm_size,
94  CountSendsAndCreateSteeringAtomic )
95  -> std::pair<Kokkos::View<int*, typename ExportRankView::device_type>,
96  Kokkos::View<typename ExportRankView::size_type*,
97  typename ExportRankView::device_type>>
98 {
99  using device_type = typename ExportRankView::device_type;
100  using execution_space = typename ExportRankView::execution_space;
101  using size_type = typename ExportRankView::size_type;
102 
103  // Create views.
104  Kokkos::View<int*, device_type> neighbor_counts( "neighbor_counts",
105  comm_size );
106  Kokkos::View<size_type*, device_type> neighbor_ids(
107  Kokkos::ViewAllocateWithoutInitializing( "neighbor_ids" ),
108  element_export_ranks.size() );
109 
110  // Count the sends and create the steering vector.
111 
112  // For smaller values of comm_size, use the optimized scratch memory path
113  // Value of 64 is arbitrary, should be at least 27 to cover 3-d halos?
114  if ( comm_size <= 64 )
115  {
116  constexpr int team_size = 256;
117  Kokkos::TeamPolicy<execution_space> team_policy(
118  ( element_export_ranks.size() + team_size - 1 ) / team_size,
119  team_size );
120  team_policy = team_policy.set_scratch_size(
121  0,
122  Kokkos::PerTeam( sizeof( int ) * ( team_size + 2 * comm_size ) ) );
123  Kokkos::parallel_for(
124  "Cabana::CommunicationPlan::countSendsAndCreateSteeringShared",
125  team_policy,
126  KOKKOS_LAMBDA(
127  const typename Kokkos::TeamPolicy<execution_space>::member_type&
128  team ) {
129  // NOTE: `get_shmem` returns shared memory pointers *aligned to
130  // 8 bytes*, so if `comm_size` is odd we can get erroneously
131  // padded offsets if we call `get_shmem` separately for each
132  // shared memory intermediary. Acquiring all the needed scratch
133  // memory at once then computing pointer offsets by hand avoids
134  // this issue.
135  int* scratch = (int*)team.team_shmem().get_shmem(
136  ( team.team_size() + 2 * comm_size ) * sizeof( int ), 0 );
137 
138  // local neighbor_ids, gives the local offset relative to
139  // calculated global offset. Size team.team_size() * sizeof(int)
140  int* local_neighbor_ids = scratch;
141  // local histogram, `comm_size` in size
142  int* histo = local_neighbor_ids + team.team_size();
143  // offset into global array, `comm_size` in size
144  int* global_offset = histo + comm_size;
145  // overall element `tid`
146  const auto tid =
147  team.team_rank() + team.league_rank() * team.team_size();
148  // local element
149  const int local_id = team.team_rank();
150  // total number of elements, for convenience
151  const int num_elements = element_export_ranks.size();
152  // my element export rank
153  const int my_element_export_rank =
154  ( tid < num_elements ? element_export_ranks( tid ) : -1 );
155 
156  // cannot outright terminate early b/c threads share work
157  // if (tid >= num_elements) return;
158  const bool in_bounds =
159  tid < num_elements && my_element_export_rank >= 0;
160 
161  Kokkos::parallel_for(
162  Kokkos::TeamThreadRange( team, comm_size ),
163  [&]( const int i ) { histo[i] = 0; } );
164 
165  // synchronize zeroing
166  team.team_barrier();
167 
168  // build local histogram, need to encode num_elements check here
169  if ( in_bounds )
170  {
171  // shared memory atomic add, accumulate into local offset
172  local_neighbor_ids[local_id] = Kokkos::atomic_fetch_add(
173  &histo[my_element_export_rank], 1 );
174  }
175 
176  // synchronize local histogram build
177  team.team_barrier();
178 
179  // reserve space in global array via a loop over neighbor counts
180  Kokkos::parallel_for(
181  Kokkos::TeamThreadRange( team, comm_size ),
182  [&]( const int i )
183  {
184  // global memory atomic add, reserves space
185  global_offset[i] = Kokkos::atomic_fetch_add(
186  &neighbor_counts( i ), histo[i] );
187  } );
188 
189  // synchronize block-stride loop
190  team.team_barrier();
191 
192  // and now store to my location
193  if ( in_bounds )
194  {
195  neighbor_ids( tid ) =
196  global_offset[my_element_export_rank] +
197  local_neighbor_ids[local_id];
198  }
199  } );
200  }
201  else
202  {
203  // For larger numbers of export ranks (for ex, migration) we use the
204  // global memory version, though this is a point of future optimization
205 
206  Kokkos::parallel_for(
207  "Cabana::CommunicationPlan::countSendsAndCreateSteering",
208  Kokkos::RangePolicy<execution_space>( 0,
209  element_export_ranks.size() ),
210  KOKKOS_LAMBDA( const size_type i ) {
211  if ( element_export_ranks( i ) >= 0 )
212  neighbor_ids( i ) = Kokkos::atomic_fetch_add(
213  &neighbor_counts( element_export_ranks( i ) ), 1 );
214  } );
215  }
216  Kokkos::fence();
217 
218  // Return the counts and ids.
219  return std::make_pair( neighbor_counts, neighbor_ids );
220 }

222 // Count sends and generate the steering vector. Duplicated version.
223 template <class ExportRankView>
224 auto countSendsAndCreateSteering( const ExportRankView element_export_ranks,
225  const int comm_size,
226  CountSendsAndCreateSteeringDuplicated )
227  -> std::pair<Kokkos::View<int*, typename ExportRankView::device_type>,
228  Kokkos::View<typename ExportRankView::size_type*,
229  typename ExportRankView::device_type>>
230 {
231  using device_type = typename ExportRankView::device_type;
232  using execution_space = typename ExportRankView::execution_space;
233  using size_type = typename ExportRankView::size_type;
234 
235  // Create a unique thread token.
236  Kokkos::Experimental::UniqueToken<
237  execution_space, Kokkos::Experimental::UniqueTokenScope::Global>
238  unique_token;
239 
240  // Create views.
241  Kokkos::View<int*, device_type> neighbor_counts(
242  Kokkos::ViewAllocateWithoutInitializing( "neighbor_counts" ),
243  comm_size );
244  Kokkos::View<size_type*, device_type> neighbor_ids(
245  Kokkos::ViewAllocateWithoutInitializing( "neighbor_ids" ),
246  element_export_ranks.size() );
247  Kokkos::View<int**, device_type> neighbor_counts_dup(
248  "neighbor_counts", unique_token.size(), comm_size );
249  Kokkos::View<size_type**, device_type> neighbor_ids_dup(
250  "neighbor_ids", unique_token.size(), element_export_ranks.size() );
251 
252  // Compute initial duplicated sends and steering.
253  Kokkos::parallel_for(
254  "Cabana::CommunicationPlan::intialCount",
255  Kokkos::RangePolicy<execution_space>( 0, element_export_ranks.size() ),
256  KOKKOS_LAMBDA( const size_type i ) {
257  if ( element_export_ranks( i ) >= 0 )
258  {
259  // Get the thread id.
260  auto thread_id = unique_token.acquire();
261 
262  // Do the duplicated fetch-add. If this is a valid element id
263  // increment the send count for this rank. Add the incremented
264  // count as the thread-local neighbor id. This is too big by
265  // one (because we use the prefix increment operator) but we
266  // want a non-zero value so we can later find which thread
267  // this element was located on because we are always
268  // guaranteed a non-zero value. We will subtract this value
269  // later.
270  neighbor_ids_dup( thread_id, i ) = ++neighbor_counts_dup(
271  thread_id, element_export_ranks( i ) );
272 
273  // Release the thread id.
274  unique_token.release( thread_id );
275  }
276  } );
277  Kokkos::fence();
278 
279  // Team policy
280  using team_policy =
281  Kokkos::TeamPolicy<execution_space, Kokkos::Schedule<Kokkos::Dynamic>>;
282  using index_type = typename team_policy::index_type;
283 
284  // Compute the send counts for each neighbor rank by reducing across
285  // the thread duplicates.
286  Kokkos::parallel_for(
287  "Cabana::CommunicationPlan::finalCount",
288  team_policy( neighbor_counts.extent( 0 ), Kokkos::AUTO ),
289  KOKKOS_LAMBDA( const typename team_policy::member_type& team ) {
290  // Get the element id.
291  auto i = team.league_rank();
292 
293  // Add the thread results.
294  int thread_counts = 0;
295  Kokkos::parallel_reduce(
296  Kokkos::TeamThreadRange( team,
297  neighbor_counts_dup.extent( 0 ) ),
298  [&]( const index_type thread_id, int& result )
299  { result += neighbor_counts_dup( thread_id, i ); },
300  thread_counts );
301  neighbor_counts( i ) = thread_counts;
302  } );
303  Kokkos::fence();
304 
305  // Compute the location of each export element in the send buffer of
306  // its destination rank.
307  Kokkos::parallel_for(
308  "Cabana::CommunicationPlan::createSteering",
309  team_policy( element_export_ranks.size(), Kokkos::AUTO ),
310  KOKKOS_LAMBDA( const typename team_policy::member_type& team ) {
311  // Get the element id.
312  auto i = team.league_rank();
313 
314  // Only operate on valid elements
315  if ( element_export_ranks( i ) >= 0 )
316  {
317  // Compute the thread id in which we located the element
318  // during the count phase. Only the thread in which we
319  // located the element will contribute to the reduction.
320  index_type dup_thread = 0;
321  Kokkos::parallel_reduce(
322  Kokkos::TeamThreadRange( team,
323  neighbor_ids_dup.extent( 0 ) ),
324  [&]( const index_type thread_id, index_type& result )
325  {
326  if ( neighbor_ids_dup( thread_id, i ) > 0 )
327  result += thread_id;
328  },
329  dup_thread );
330 
331  // Compute the offset of this element in the steering
332  // vector for its destination rank. Loop through the
333  // threads up to the thread that found this element in the
334  // count stage. All thread counts prior to that thread
335  // will contribute to the offset.
336  size_type thread_offset = 0;
337  Kokkos::parallel_reduce(
338  Kokkos::TeamThreadRange( team, dup_thread ),
339  [&]( const index_type thread_id, size_type& result ) {
340  result += neighbor_counts_dup(
341  thread_id, element_export_ranks( i ) );
342  },
343  thread_offset );
344 
345  // Add the thread-local value to the offset where we subtract
346  // the 1 that we added artificially when we were first
347  // counting.
348  neighbor_ids( i ) =
349  thread_offset + neighbor_ids_dup( dup_thread, i ) - 1;
350  }
351  } );
352  Kokkos::fence();
353 
354  // Return the counts and ids.
355  return std::make_pair( neighbor_counts, neighbor_ids );
356 }

358 //---------------------------------------------------------------------------//
359 // Return unique neighbor ranks, with the current rank first.
360 inline std::vector<int> getUniqueTopology( std::vector<int> topology )
361 {
362  auto remove_end = std::remove( topology.begin(), topology.end(), -1 );
363  std::sort( topology.begin(), remove_end );
364  auto unique_end = std::unique( topology.begin(), remove_end );
365  topology.resize( std::distance( topology.begin(), unique_end ) );
366 
367  // Put this rank first.
368  int my_rank = -1;
369  MPI_Comm_rank( MPI_COMM_WORLD, &my_rank );
370  for ( auto& n : topology )
371  {
372  if ( n == my_rank )
373  {
374  std::swap( n, topology[0] );
375  break;
376  }
377  }
378  return topology;
379 }

591  template <class ViewType>
592  Kokkos::View<size_type*, device_type>
593  createFromExportsAndTopology( const ViewType& element_export_ranks,
594  const std::vector<int>& neighbor_ranks )
595  {
596  // Store the number of export elements.
597  _num_export_element = element_export_ranks.size();
598 
599  // Store the unique neighbors (this rank first).
600  _neighbors = Impl::getUniqueTopology( neighbor_ranks );
601  int num_n = _neighbors.size();
602 
603  // Get the size of this communicator.
604  int comm_size = -1;
605  MPI_Comm_size( comm(), &comm_size );
606 
607  // Get the MPI rank we are currently on.
608  int my_rank = -1;
609  MPI_Comm_rank( comm(), &my_rank );
610 
611  // Pick an mpi tag for communication. This object has it's own
612  // communication space so any mpi tag will do.
613  const int mpi_tag = 1221;
614 
615  // Initialize import/export sizes.
616  _num_export.assign( num_n, 0 );
617  _num_import.assign( num_n, 0 );
618 
619  // Count the number of sends this rank will do to other ranks. Keep
620  // track of which slot we get in our neighbor's send buffer.
621  auto counts_and_ids = Impl::countSendsAndCreateSteering(
622  element_export_ranks, comm_size,
623  typename Impl::CountSendsAndCreateSteeringAlgorithm<
624  execution_space>::type() );
625 
626  // Copy the counts to the host.
627  auto neighbor_counts_host = Kokkos::create_mirror_view_and_copy(
628  Kokkos::HostSpace(), counts_and_ids.first );
629 
630  // Get the export counts.
631  for ( int n = 0; n < num_n; ++n )
632  _num_export[n] = neighbor_counts_host( _neighbors[n] );
633 
634  // Post receives for the number of imports we will get.
635  std::vector<MPI_Request> requests;
636  requests.reserve( num_n );
637  for ( int n = 0; n < num_n; ++n )
638  if ( my_rank != _neighbors[n] )
639  {
640  requests.push_back( MPI_Request() );
641  MPI_Irecv( &_num_import[n], 1, MPI_UNSIGNED_LONG, _neighbors[n],
642  mpi_tag, comm(), &( requests.back() ) );
643  }
644  else
645  _num_import[n] = _num_export[n];
646 
647  // Send the number of exports to each of our neighbors.
648  for ( int n = 0; n < num_n; ++n )
649  if ( my_rank != _neighbors[n] )
650  MPI_Send( &_num_export[n], 1, MPI_UNSIGNED_LONG, _neighbors[n],
651  mpi_tag, comm() );
652 
653  // Wait on receives.
654  std::vector<MPI_Status> status( requests.size() );
655  const int ec =
656  MPI_Waitall( requests.size(), requests.data(), status.data() );
657  if ( MPI_SUCCESS != ec )
658  throw std::logic_error( "Failed MPI Communication" );
659 
660  // Get the total number of imports/exports.
661  _total_num_export =
662  std::accumulate( _num_export.begin(), _num_export.end(), 0 );
663  _total_num_import =
664  std::accumulate( _num_import.begin(), _num_import.end(), 0 );
665 
666  // Barrier before continuing to ensure synchronization.
667  MPI_Barrier( comm() );
668 
669  // Return the neighbor ids.
670  return counts_and_ids.second;
671  }

701  template <class ViewType>
702  Kokkos::View<size_type*, device_type>
703  createFromExportsOnly( const ViewType& element_export_ranks )
704  {
705  // Store the number of export elements.
706  _num_export_element = element_export_ranks.size();
707 
708  // Get the size of this communicator.
709  int comm_size = -1;
710  MPI_Comm_size( comm(), &comm_size );
711 
712  // Get the MPI rank we are currently on.
713  int my_rank = -1;
714  MPI_Comm_rank( comm(), &my_rank );
715 
716  // Pick an mpi tag for communication. This object has it's own
717  // communication space so any mpi tag will do.
718  const int mpi_tag = 1221;
719 
720  // Count the number of sends this rank will do to other ranks. Keep
721  // track of which slot we get in our neighbor's send buffer.
722  auto counts_and_ids = Impl::countSendsAndCreateSteering(
723  element_export_ranks, comm_size,
724  typename Impl::CountSendsAndCreateSteeringAlgorithm<
725  execution_space>::type() );
726 
727  // Copy the counts to the host.
728  auto neighbor_counts_host = Kokkos::create_mirror_view_and_copy(
729  Kokkos::HostSpace(), counts_and_ids.first );
730 
731  // Extract the export ranks and number of exports and then flag the
732  // send ranks.
733  _neighbors.clear();
734  _num_export.clear();
735  _total_num_export = 0;
736  for ( int r = 0; r < comm_size; ++r )
737  if ( neighbor_counts_host( r ) > 0 )
738  {
739  _neighbors.push_back( r );
740  _num_export.push_back( neighbor_counts_host( r ) );
741  _total_num_export += neighbor_counts_host( r );
742  neighbor_counts_host( r ) = 1;
743  }
744 
745  // Get the number of export ranks and initially allocate the import
746  // sizes.
747  int num_export_rank = _neighbors.size();
748  _num_import.assign( num_export_rank, 0 );
749 
750  // If we are sending to ourself put that one first in the neighbor
751  // list and assign the number of imports to be the number of exports.
752  bool self_send = false;
753  for ( int n = 0; n < num_export_rank; ++n )
754  if ( _neighbors[n] == my_rank )
755  {
756  std::swap( _neighbors[n], _neighbors[0] );
757  std::swap( _num_export[n], _num_export[0] );
758  _num_import[0] = _num_export[0];
759  self_send = true;
760  break;
761  }
762 
763  // Determine how many total import ranks each neighbor has.
764  int num_import_rank = -1;
765  std::vector<int> recv_counts( comm_size, 1 );
766  MPI_Reduce_scatter( neighbor_counts_host.data(), &num_import_rank,
767  recv_counts.data(), MPI_INT, MPI_SUM, comm() );
768  if ( self_send )
769  --num_import_rank;
770 
771  // Post the expected number of receives and indicate we might get them
772  // from any rank.
773  std::vector<std::size_t> import_sizes( num_import_rank );
774  std::vector<MPI_Request> requests( num_import_rank );
775  for ( int n = 0; n < num_import_rank; ++n )
776  MPI_Irecv( &import_sizes[n], 1, MPI_UNSIGNED_LONG, MPI_ANY_SOURCE,
777  mpi_tag, comm(), &requests[n] );
778 
779  // Do blocking sends. Dont do any self sends.
780  int self_offset = ( self_send ) ? 1 : 0;
781  for ( int n = self_offset; n < num_export_rank; ++n )
782  MPI_Send( &_num_export[n], 1, MPI_UNSIGNED_LONG, _neighbors[n],
783  mpi_tag, comm() );
784 
785  // Wait on non-blocking receives.
786  std::vector<MPI_Status> status( requests.size() );
787  const int ec =
788  MPI_Waitall( requests.size(), requests.data(), status.data() );
789  if ( MPI_SUCCESS != ec )
790  throw std::logic_error( "Failed MPI Communication" );
791 
792  // Compute the total number of imports.
793  _total_num_import =
794  std::accumulate( import_sizes.begin(), import_sizes.end(),
795  ( self_send ) ? _num_import[0] : 0 );
796 
797  // Extract the imports. If we did self sends we already know what
798  // imports we got from that.
799  for ( int i = 0; i < num_import_rank; ++i )
800  {
801  // Get the message source.
802  const auto source = status[i].MPI_SOURCE;
803 
804  // See if the neighbor we received stuff from was someone we also
805  // sent stuff to.
806  auto found_neighbor =
807  std::find( _neighbors.begin(), _neighbors.end(), source );
808 
809  // If this is a new neighbor (i.e. someone we didn't send anything
810  // to) record this.
811  if ( found_neighbor == std::end( _neighbors ) )
812  {
813  _neighbors.push_back( source );
814  _num_import.push_back( import_sizes[i] );
815  _num_export.push_back( 0 );
816  }
817 
818  // Otherwise if we already sent something to this neighbor that
819  // means we already have a neighbor/export entry. Just assign the
820  // import entry for that neighbor.
821  else
822  {
823  auto n = std::distance( _neighbors.begin(), found_neighbor );
824  _num_import[n] = import_sizes[i];
825  }
826  }
827 
828  // Barrier before continuing to ensure synchronization.
829  MPI_Barrier( comm() );
830 
831  // Return the neighbor ids.
832  return counts_and_ids.second;
833  }

887  // Create the export steering vector.
888  template <class PackViewType, class RankViewType, class IdViewType>
889  void createSteering( const bool use_iota, const PackViewType& neighbor_ids,
890  const RankViewType& element_export_ranks,
891  const IdViewType& element_export_ids )
892  {
893  if ( !use_iota &&
894  ( element_export_ids.size() != element_export_ranks.size() ) )
895  throw std::runtime_error( "Export ids and ranks different sizes!" );
896 
897  // Get the size of this communicator.
898  int comm_size = -1;
899  MPI_Comm_size( *_comm_ptr, &comm_size );
900 
901  // Calculate the steering offsets via exclusive prefix sum for the
902  // exports.
903  int num_n = _neighbors.size();
904  std::vector<std::size_t> offsets( num_n, 0.0 );
905  for ( int n = 1; n < num_n; ++n )
906  offsets[n] = offsets[n - 1] + _num_export[n - 1];
907 
908  // Map the offsets to the device.
909  Kokkos::View<std::size_t*, Kokkos::HostSpace> rank_offsets_host(
910  Kokkos::ViewAllocateWithoutInitializing( "rank_map" ), comm_size );
911  for ( int n = 0; n < num_n; ++n )
912  rank_offsets_host( _neighbors[n] ) = offsets[n];
913  auto rank_offsets = Kokkos::create_mirror_view_and_copy(
914  memory_space(), rank_offsets_host );
915 
916  // Create the export steering vector for writing local elements into
917  // the send buffer. Note we create a local, shallow copy - this is a
918  // CUDA workaround for handling class private data.
919  _export_steering = Kokkos::View<std::size_t*, memory_space>(
920  Kokkos::ViewAllocateWithoutInitializing( "export_steering" ),
921  _total_num_export );
922  auto steer_vec = _export_steering;
923  Kokkos::parallel_for(
924  "Cabana::createSteering",
925  Kokkos::RangePolicy<execution_space>( 0, _num_export_element ),
926  KOKKOS_LAMBDA( const int i ) {
927  if ( element_export_ranks( i ) >= 0 )
928  steer_vec( rank_offsets( element_export_ranks( i ) ) +
929  neighbor_ids( i ) ) =
930  ( use_iota ) ? i : element_export_ids( i );
931  } );
932  Kokkos::fence();
933  }
935 
936  private:
937  std::shared_ptr<MPI_Comm> _comm_ptr;
938  std::vector<int> _neighbors;
939  std::size_t _total_num_export;
940  std::size_t _total_num_import;
941  std::vector<std::size_t> _num_export;
942  std::vector<std::size_t> _num_import;
943  std::size_t _num_export_element;
944  Kokkos::View<std::size_t*, device_type> _export_steering;
945 };
