219 template <class Scalar, class EntityType, class MeshType, class... Params>
220 class Array
221 {
222  public:
224  using value_type = Scalar;
225 
227  using entity_type = EntityType;
228 
230  using mesh_type = MeshType;
231 
233  static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;
234 
236  using array_layout = ArrayLayout<entity_type, mesh_type>;
237 
239  using view_type = std::conditional_t<
240  3 == num_space_dim, Kokkos::View<value_type****, Params...>,
241  std::conditional_t<2 == num_space_dim,
242  Kokkos::View<value_type***, Params...>, void>>;
243 
245  using device_type = typename view_type::device_type;
246 
248  using memory_space = typename view_type::memory_space;
249 
251  using execution_space = typename view_type::execution_space;
252 
259  Array( const std::string& label,
260  const std::shared_ptr<array_layout>& layout )
261  : _layout( layout )
262  , _data( createView<value_type, Params...>(
263  label, layout->indexSpace( Ghost(), Local() ) ) )
264  {
265  }
266 
273  Array( const std::shared_ptr<array_layout>& layout, const view_type& view )
274  : _layout( layout )
275  , _data( view )
276  {
277  for ( std::size_t d = 0; d < num_space_dim + 1; ++d )
278  if ( (long)view.extent( d ) !=
279  layout->indexSpace( Ghost(), Local() ).extent( d ) )
280  throw std::runtime_error(
281  "Layout and view dimensions do not match" );
282  }
283 
285  std::shared_ptr<array_layout> layout() const { return _layout; }
286 
288  view_type view() const { return _data; }
289 
291  std::string label() const { return _data.label(); }
292 
293  private:
294  std::shared_ptr<array_layout> _layout;
295  view_type _data;
296 
297  public:
299  using subview_type = decltype(
300  createSubview( _data, _layout->indexSpace( Ghost(), Local() ) ) );
302  using subview_layout = typename subview_type::array_layout;
304  using subview_memory_traits = typename subview_type::memory_traits;
305 };

346 //---------------------------------------------------------------------------//
347 // Subarray creation.
348 //---------------------------------------------------------------------------//
356 template <class Scalar, class EntityType, class MeshType, class... Params>
357 std::shared_ptr<Array<
358  Scalar, EntityType, MeshType,
359  typename Array<Scalar, EntityType, MeshType, Params...>::subview_layout,
360  typename Array<Scalar, EntityType, MeshType, Params...>::device_type,
361  typename Array<Scalar, EntityType, MeshType,
362  Params...>::subview_memory_traits>>
363 createSubarray( const Array<Scalar, EntityType, MeshType, Params...>& array,
364  const int dof_min, const int dof_max )
365 {
366  if ( dof_min < 0 || dof_max > array.layout()->dofsPerEntity() )
367  throw std::logic_error( "Subarray dimensions out of bounds" );
368 
369  auto space = array.layout()->indexSpace( Ghost(), Local() );
370  std::array<long, MeshType::num_space_dim + 1> min;
371  std::array<long, MeshType::num_space_dim + 1> max;
372  for ( std::size_t d = 0; d < MeshType::num_space_dim; ++d )
373  {
374  min[d] = space.min( d );
375  max[d] = space.max( d );
376  }
377  min.back() = dof_min;
378  max.back() = dof_max;
379  IndexSpace<MeshType::num_space_dim + 1> sub_space( min, max );
380  auto sub_view = createSubview( array.view(), sub_space );
381  auto sub_layout = createArrayLayout( array.layout()->localGrid(),
382  dof_max - dof_min, EntityType() );
383  return std::make_shared<Array<
384  Scalar, EntityType, MeshType,
385  typename Array<Scalar, EntityType, MeshType, Params...>::subview_layout,
386  typename Array<Scalar, EntityType, MeshType, Params...>::device_type,
387  typename Array<Scalar, EntityType, MeshType,
388  Params...>::subview_memory_traits>>( sub_layout,
389  sub_view );
390 }

674 template <class ViewType, std::size_t NumSpaceDim>
675 struct DotFunctor
676 {
678  static constexpr std::size_t num_space_dim = NumSpaceDim;
680  typedef typename ViewType::value_type value_type[];
682  typedef typename ViewType::size_type size_type;
684  size_type value_count;
686  ViewType _a;
688  ViewType _b;
689 
691  DotFunctor( const ViewType& a, const ViewType& b )
692  : value_count( a.extent( NumSpaceDim ) )
693  , _a( a )
694  , _b( b )
695  {
696  }
697 
699  template <std::size_t NSD = num_space_dim>
700  KOKKOS_INLINE_FUNCTION std::enable_if_t<3 == NSD, void>
701  operator()( const size_type i, const size_type j, const size_type k,
702  const size_type l, value_type sum ) const
703  {
704  sum[l] += _a( i, j, k, l ) * _b( i, j, k, l );
705  }
706 
708  template <std::size_t NSD = num_space_dim>
709  KOKKOS_INLINE_FUNCTION std::enable_if_t<2 == NSD, void>
710  operator()( const size_type i, const size_type j, const size_type l,
711  value_type sum ) const
712  {
713  sum[l] += _a( i, j, l ) * _b( i, j, l );
714  }
715 
717  KOKKOS_INLINE_FUNCTION
718  void join( volatile value_type dst, const volatile value_type src ) const
719  {
720  for ( size_type j = 0; j < value_count; ++j )
721  dst[j] += src[j];
722  }
723 
725  KOKKOS_INLINE_FUNCTION void init( value_type sum ) const
726  {
727  for ( size_type j = 0; j < value_count; ++j )
728  sum[j] = 0.0;
729  }
730 };

740 template <class Array_t>
741 void dot( const Array_t& a, const Array_t& b,
742  std::vector<typename Array_t::value_type>& products )
743 {
744  static_assert( is_array<Array_t>::value, "Cajita::Array required" );
745  if ( products.size() !=
746  static_cast<unsigned>( a.layout()->dofsPerEntity() ) )
747  throw std::runtime_error( "Incorrect vector size" );
748 
749  for ( auto& p : products )
750  p = 0.0;
751 
752  DotFunctor<typename Array_t::view_type, Array_t::num_space_dim> functor(
753  a.view(), b.view() );
754  Kokkos::parallel_reduce(
755  "ArrayOp::dot",
756  createExecutionPolicy( a.layout()->indexSpace( Own(), Local() ),
757  typename Array_t::execution_space() ),
758  functor, products.data() );
759 
760  MPI_Allreduce( MPI_IN_PLACE, products.data(), products.size(),
761  MpiTraits<typename Array_t::value_type>::type(), MPI_SUM,
762  a.layout()->localGrid()->globalGrid().comm() );
763 }
