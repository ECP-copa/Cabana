34 template <class EntityType, class MeshType>
35 class ArrayLayout
36 {
37  public:
38  // Entity type.
39  using entity_type = EntityType;
40 
41  // Mesh type.
42  using mesh_type = MeshType;
43 
44  // Spatial dimension.
45  static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;
46 
53  ArrayLayout( const std::shared_ptr<LocalGrid<MeshType>>& local_grid,
54  const int dofs_per_entity )
55  : _local_grid( local_grid )
56  , _dofs_per_entity( dofs_per_entity )
57  {
58  }
59 
60  // Get the local grid over which this layout is defined.
61  const std::shared_ptr<LocalGrid<MeshType>> localGrid() const
62  {
63  return _local_grid;
64  }
65 
66  // Get the number of degrees-of-freedom on each grid entity.
67  int dofsPerEntity() const { return _dofs_per_entity; }
68 
69  // Get the index space of the array elements in the given
70  // decomposition.
71  template <class DecompositionTag, class IndexType>
72  IndexSpace<num_space_dim + 1>
73  indexSpace( DecompositionTag decomposition_tag, IndexType index_type ) const
74  {
75  return appendDimension( _local_grid->indexSpace( decomposition_tag,
76  EntityType(),
77  index_type ),
78  _dofs_per_entity );
79  }
80 
81  // Get the local index space of the array elements we shared with the
82  // given neighbor in the given decomposition. Optionally provide a halo
83  // width for the shared space. This halo width must be less than or equal
84  // to the halo width of the local grid. The default behavior is to use the
85  // halo width of the local grid.
86  template <class DecompositionTag>
87  IndexSpace<num_space_dim + 1>
88  sharedIndexSpace( DecompositionTag decomposition_tag,
89  const std::array<int, num_space_dim>& off_ijk,
90  const int halo_width = -1 ) const
91  {
92  return appendDimension(
93  _local_grid->sharedIndexSpace( decomposition_tag, EntityType(),
94  off_ijk, halo_width ),
95  _dofs_per_entity );
96  }
97 
98  template <class DecompositionTag, std::size_t NSD = num_space_dim>
99  std::enable_if_t<3 == NSD, IndexSpace<4>>
100  sharedIndexSpace( DecompositionTag decomposition_tag, const int off_i,
101  const int off_j, const int off_k,
102  const int halo_width = -1 ) const
103  {
104  std::array<int, 3> off_ijk = { off_i, off_j, off_k };
105  return sharedIndexSpace( decomposition_tag, off_ijk, halo_width );
106  }
107 
108  template <class DecompositionTag, std::size_t NSD = num_space_dim>
109  std::enable_if_t<2 == NSD, IndexSpace<3>>
110  sharedIndexSpace( DecompositionTag decomposition_tag, const int off_i,
111  const int off_j, const int halo_width = -1 ) const
112  {
113  std::array<int, 2> off_ijk = { off_i, off_j };
114  return sharedIndexSpace( decomposition_tag, off_ijk, halo_width );
115  }
116 
117  private:
118  std::shared_ptr<LocalGrid<MeshType>> _local_grid;
119  int _dofs_per_entity;
120 };

182 template <class Scalar, class EntityType, class MeshType, class... Params>
183 class Array
184 {
185  public:
186  // Value type.
187  using value_type = Scalar;
188 
189  // Entity type.
190  using entity_type = EntityType;
191 
192  // Mesh type.
193  using mesh_type = MeshType;
194 
195  // Spatial dimension.
196  static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;
197 
198  // Array layout type.
199  using array_layout = ArrayLayout<entity_type, mesh_type>;
200 
201  // View type.
202  using view_type = std::conditional_t<
203  3 == num_space_dim, Kokkos::View<value_type****, Params...>,
204  std::conditional_t<2 == num_space_dim,
205  Kokkos::View<value_type***, Params...>, void>>;
206 
207  // Device type.
208  using device_type = typename view_type::device_type;
209 
210  // Memory space.
211  using memory_space = typename view_type::memory_space;
212 
213  // Execution space.
214  using execution_space = typename view_type::execution_space;
215 
222  Array( const std::string& label,
223  const std::shared_ptr<array_layout>& layout )
224  : _layout( layout )
225  , _data( createView<value_type, Params...>(
226  label, layout->indexSpace( Ghost(), Local() ) ) )
227  {
228  }
229 
236  Array( const std::shared_ptr<array_layout>& layout, const view_type& view )
237  : _layout( layout )
238  , _data( view )
239  {
240  for ( std::size_t d = 0; d < num_space_dim + 1; ++d )
241  if ( (long)view.extent( d ) !=
242  layout->indexSpace( Ghost(), Local() ).extent( d ) )
243  throw std::runtime_error(
244  "Layout and view dimensions do not match" );
245  }
246 
248  std::shared_ptr<array_layout> layout() const { return _layout; }
249 
251  view_type view() const { return _data; }
252 
254  std::string label() const { return _data.label(); }
255 
256  private:
257  std::shared_ptr<array_layout> _layout;
258  view_type _data;
259 
260  public:
261  // Subview types.
262  using subview_type = decltype(
263  createSubview( _data, _layout->indexSpace( Ghost(), Local() ) ) );
264  using subview_layout = typename subview_type::array_layout;
265  using subview_memory_traits = typename subview_type::memory_traits;
266 };

307 //---------------------------------------------------------------------------//
308 // Subarray creation.
309 //---------------------------------------------------------------------------//
317 template <class Scalar, class EntityType, class MeshType, class... Params>
318 std::shared_ptr<Array<
319  Scalar, EntityType, MeshType,
320  typename Array<Scalar, EntityType, MeshType, Params...>::subview_layout,
321  typename Array<Scalar, EntityType, MeshType, Params...>::device_type,
322  typename Array<Scalar, EntityType, MeshType,
323  Params...>::subview_memory_traits>>
324 createSubarray( const Array<Scalar, EntityType, MeshType, Params...>& array,
325  const int dof_min, const int dof_max )
326 {
327  if ( dof_min < 0 || dof_max > array.layout()->dofsPerEntity() )
328  throw std::logic_error( "Subarray dimensions out of bounds" );
329 
330  auto space = array.layout()->indexSpace( Ghost(), Local() );
331  std::array<long, MeshType::num_space_dim + 1> min;
332  std::array<long, MeshType::num_space_dim + 1> max;
333  for ( std::size_t d = 0; d < MeshType::num_space_dim; ++d )
334  {
335  min[d] = space.min( d );
336  max[d] = space.max( d );
337  }
338  min.back() = dof_min;
339  max.back() = dof_max;
340  IndexSpace<MeshType::num_space_dim + 1> sub_space( min, max );
341  auto sub_view = createSubview( array.view(), sub_space );
342  auto sub_layout = createArrayLayout( array.layout()->localGrid(),
343  dof_max - dof_min, EntityType() );
344  return std::make_shared<Array<
345  Scalar, EntityType, MeshType,
346  typename Array<Scalar, EntityType, MeshType, Params...>::subview_layout,
347  typename Array<Scalar, EntityType, MeshType, Params...>::device_type,
348  typename Array<Scalar, EntityType, MeshType,
349  Params...>::subview_memory_traits>>( sub_layout,
350  sub_view );
351 }

625 // Dot product
626 template <class ViewType, std::size_t NumSpaceDim>
627 struct DotFunctor
628 {
629  static constexpr std::size_t num_space_dim = NumSpaceDim;
630  typedef typename ViewType::value_type value_type[];
631  typedef typename ViewType::size_type size_type;
632  size_type value_count;
633  ViewType _a;
634  ViewType _b;
635 
636  DotFunctor( const ViewType& a, const ViewType& b )
637  : value_count( a.extent( NumSpaceDim ) )
638  , _a( a )
639  , _b( b )
640  {
641  }
642 
643  template <std::size_t NSD = num_space_dim>
644  KOKKOS_INLINE_FUNCTION std::enable_if_t<3 == NSD, void>
645  operator()( const size_type i, const size_type j, const size_type k,
646  const size_type l, value_type sum ) const
647  {
648  sum[l] += _a( i, j, k, l ) * _b( i, j, k, l );
649  }
650 
651  template <std::size_t NSD = num_space_dim>
652  KOKKOS_INLINE_FUNCTION std::enable_if_t<2 == NSD, void>
653  operator()( const size_type i, const size_type j, const size_type l,
654  value_type sum ) const
655  {
656  sum[l] += _a( i, j, l ) * _b( i, j, l );
657  }
658 
659  KOKKOS_INLINE_FUNCTION
660  void join( volatile value_type dst, const volatile value_type src ) const
661  {
662  for ( size_type j = 0; j < value_count; ++j )
663  dst[j] += src[j];
664  }
665 
666  KOKKOS_INLINE_FUNCTION void init( value_type sum ) const
667  {
668  for ( size_type j = 0; j < value_count; ++j )
669  sum[j] = 0.0;
670  }
671 };

681 template <class Array_t>
682 void dot( const Array_t& a, const Array_t& b,
683  std::vector<typename Array_t::value_type>& products )
684 {
685  static_assert( is_array<Array_t>::value, "Cajita::Array required" );
686  if ( products.size() !=
687  static_cast<unsigned>( a.layout()->dofsPerEntity() ) )
688  throw std::runtime_error( "Incorrect vector size" );
689 
690  for ( auto& p : products )
691  p = 0.0;
692 
693  DotFunctor<typename Array_t::view_type, Array_t::num_space_dim> functor(
694  a.view(), b.view() );
695  Kokkos::parallel_reduce(
696  "ArrayOp::dot",
697  createExecutionPolicy( a.layout()->indexSpace( Own(), Local() ),
698  typename Array_t::execution_space() ),
699  functor, products.data() );
700 
701  MPI_Allreduce( MPI_IN_PLACE, products.data(), products.size(),
702  MpiTraits<typename Array_t::value_type>::type(), MPI_SUM,
703  a.layout()->localGrid()->globalGrid().comm() );
704 }

707 // Infinity norm
708 template <class ViewType, std::size_t NumSpaceDim>
709 struct NormInfFunctor
710 {
711  static constexpr std::size_t num_space_dim = NumSpaceDim;
712  typedef typename ViewType::value_type value_type[];
713  typedef typename ViewType::size_type size_type;
714  size_type value_count;
715  ViewType _view;
716 
717  NormInfFunctor( const ViewType& view )
718  : value_count( view.extent( NumSpaceDim ) )
719  , _view( view )
720  {
721  }
722 
723  template <std::size_t NSD = num_space_dim>
724  KOKKOS_INLINE_FUNCTION std::enable_if_t<3 == NSD, void>
725  operator()( const size_type i, const size_type j, const size_type k,
