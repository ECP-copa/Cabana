49  ArrayLayout( const std::shared_ptr<LocalGrid<MeshType>>& local_grid,
50  const int dofs_per_entity )
51  : _local_grid( local_grid )
52  , _dofs_per_entity( dofs_per_entity )
53  {
54  }
55 
56  // Get the local grid over which this layout is defined.
57  const std::shared_ptr<LocalGrid<MeshType>> localGrid() const
58  {
59  return _local_grid;
60  }
61 
62  // Get the number of degrees-of-freedom on each grid entity.
63  int dofsPerEntity() const { return _dofs_per_entity; }
64 
65  // Get the index space of the array elements in the given
66  // decomposition.
67  template <class DecompositionTag, class IndexType>
68  IndexSpace<4> indexSpace( DecompositionTag decomposition_tag,
69  IndexType index_type ) const
70  {
71  return appendDimension( _local_grid->indexSpace( decomposition_tag,
72  EntityType(),
73  index_type ),
74  _dofs_per_entity );
75  }
76 
77  // Get the local index space of the array elements we shared with the
78  // given neighbor in the given decomposition. Optionally provide a halo
79  // width for the shared space. This halo width must be less than or equal
80  // to the halo width of the local grid. The default behavior is to use the
81  // halo width of the local grid.
82  template <class DecompositionTag>
83  IndexSpace<4> sharedIndexSpace( DecompositionTag decomposition_tag,
84  const int off_i, const int off_j,
85  const int off_k,
86  const int halo_width = -1 ) const
87  {
88  return appendDimension(
89  _local_grid->sharedIndexSpace( decomposition_tag, EntityType(),
90  off_i, off_j, off_k, halo_width ),
91  _dofs_per_entity );
92  }
93 
94  private:
95  std::shared_ptr<LocalGrid<MeshType>> _local_grid;
96  int _dofs_per_entity;
97 };

155 template <class Scalar, class EntityType, class MeshType, class... Params>
156 class Array
157 {
158  public:
159  // Value type.
160  using value_type = Scalar;
161 
162  // Entity type.
163  using entity_type = EntityType;
164 
165  // Mesh type.
166  using mesh_type = MeshType;
167 
168  // Array layout type.
169  using array_layout = ArrayLayout<entity_type, mesh_type>;
170 
171  // View type.
172  using view_type = Kokkos::View<value_type****, Params...>;
173 
174  // Device type.
175  using device_type = typename view_type::device_type;
176 
177  // Memory space.
178  using memory_space = typename view_type::memory_space;
179 
180  // Execution space.
181  using execution_space = typename view_type::execution_space;
182 
189  Array( const std::string& label,
190  const std::shared_ptr<array_layout>& layout )
191  : _layout( layout )
192  , _data( createView<value_type, Params...>(
193  label, layout->indexSpace( Ghost(), Local() ) ) )
194  {
195  }
196 
203  Array( const std::shared_ptr<array_layout>& layout, const view_type& view )
204  : _layout( layout )
205  , _data( view )
206  {
207  if ( (long)view.extent( 0 ) !=
208  layout->indexSpace( Ghost(), Local() ).extent( 0 ) ||
209  (long)view.extent( 1 ) !=
210  layout->indexSpace( Ghost(), Local() ).extent( 1 ) ||
211  (long)view.extent( 2 ) !=
212  layout->indexSpace( Ghost(), Local() ).extent( 2 ) ||
213  (long)view.extent( 3 ) !=
214  layout->indexSpace( Ghost(), Local() ).extent( 3 ) )
215  throw std::runtime_error(
216  "Layout and view dimensions do not match" );
217  }
218 
220  std::shared_ptr<array_layout> layout() const { return _layout; }
221 
223  view_type view() const { return _data; }
224 
226  std::string label() const { return _data.label(); }
227 
228  private:
229  std::shared_ptr<array_layout> _layout;
230  view_type _data;
231 
232  public:
233  // Subview types.
234  using subview_type = decltype(
235  createSubview( _data, _layout->indexSpace( Ghost(), Local() ) ) );
236  using subview_layout = typename subview_type::array_layout;
237  using subview_memory_traits = typename subview_type::memory_traits;
238 };

279 //---------------------------------------------------------------------------//
280 // Subarray creation.
281 //---------------------------------------------------------------------------//
289 template <class Scalar, class EntityType, class MeshType, class... Params>
290 std::shared_ptr<Array<
291  Scalar, EntityType, MeshType,
292  typename Array<Scalar, EntityType, MeshType, Params...>::subview_layout,
293  typename Array<Scalar, EntityType, MeshType, Params...>::device_type,
294  typename Array<Scalar, EntityType, MeshType,
295  Params...>::subview_memory_traits>>
296 createSubarray( const Array<Scalar, EntityType, MeshType, Params...>& array,
297  const int dof_min, const int dof_max )
298 {
299  if ( dof_min < 0 || dof_max > array.layout()->dofsPerEntity() )
300  throw std::logic_error( "Subarray dimensions out of bounds" );
301 
302  auto space = array.layout()->indexSpace( Ghost(), Local() );
303  IndexSpace<4> sub_space(
304  { space.min( 0 ), space.min( 1 ), space.min( 2 ), dof_min },
305  { space.max( 0 ), space.max( 1 ), space.max( 2 ), dof_max } );
306  auto sub_view = createSubview( array.view(), sub_space );
307  auto sub_layout = createArrayLayout( array.layout()->localGrid(),
308  dof_max - dof_min, EntityType() );
309  return std::make_shared<Array<
310  Scalar, EntityType, MeshType,
311  typename Array<Scalar, EntityType, MeshType, Params...>::subview_layout,
312  typename Array<Scalar, EntityType, MeshType, Params...>::device_type,
313  typename Array<Scalar, EntityType, MeshType,
314  Params...>::subview_memory_traits>>( sub_layout,
315  sub_view );
316 }

361 template <class Array_t, class DecompositionTag>
362 void scale( Array_t& array, const typename Array_t::value_type alpha,
363  DecompositionTag tag )
364 {
365  static_assert( is_array<Array_t>::value, "Cajita::Array required" );
366  auto view = array.view();
367  Kokkos::parallel_for(
368  "ArrayOp::scale",
369  createExecutionPolicy( array.layout()->indexSpace( tag, Local() ),
370  typename Array_t::execution_space() ),
371  KOKKOS_LAMBDA( const int i, const int j, const int k, const int l ) {
372  view( i, j, k, l ) *= alpha;
373  } );
374 }

385 template <class Array_t, class DecompositionTag>
386 void scale( Array_t& array,
387  const std::vector<typename Array_t::value_type>& alpha,
388  DecompositionTag tag )
389 {
390  static_assert( is_array<Array_t>::value, "Cajita::Array required" );
391  if ( alpha.size() !=
392  static_cast<unsigned>( array.layout()->dofsPerEntity() ) )
393  throw std::runtime_error( "Incorrect vector size" );
394 
395  Kokkos::View<const typename Array_t::value_type*, Kokkos::HostSpace,
396  Kokkos::MemoryUnmanaged>
397  alpha_view_host( alpha.data(), alpha.size() );
398  auto alpha_view = Kokkos::create_mirror_view_and_copy(
399  typename Array_t::device_type(), alpha_view_host );
400 
401  auto array_view = array.view();
402  Kokkos::parallel_for(
403  "ArrayOp::scale",
404  createExecutionPolicy( array.layout()->indexSpace( tag, Local() ),
405  typename Array_t::execution_space() ),
406  KOKKOS_LAMBDA( const int i, const int j, const int k, const int l ) {
407  array_view( i, j, k, l ) *= alpha_view( l );
408  } );
409 }

419 template <class Array_t, class DecompositionTag>
420 void copy( Array_t& a, const Array_t& b, DecompositionTag tag )
421 {
422  static_assert( is_array<Array_t>::value, "Cajita::Array required" );
423  auto a_space = a.layout()->indexSpace( tag, Local() );
424  auto b_space = b.layout()->indexSpace( tag, Local() );
425  if ( a_space != b_space )
426  throw std::logic_error( "Incompatible index spaces" );
427  auto subview_a = createSubview( a.view(), a_space );
428  auto subview_b = createSubview( b.view(), b_space );
429  Kokkos::deep_copy( subview_a, subview_b );
430 }

476 template <class ViewType>
477 struct DotFunctor
478 {
479  typedef typename ViewType::value_type value_type[];
480  typedef typename ViewType::size_type size_type;
481  size_type value_count;
482  ViewType _a;
483  ViewType _b;
484 
485  DotFunctor( const ViewType& a, const ViewType& b )
486  : value_count( a.extent( 3 ) )
487  , _a( a )
488  , _b( b )
489  {
490  }
491 
492  KOKKOS_INLINE_FUNCTION
493  void operator()( const size_type i, const size_type j, const size_type k,
494  const size_type l, value_type sum ) const
495  {
496  sum[l] += _a( i, j, k, l ) * _b( i, j, k, l );
497  }
498 
499  KOKKOS_INLINE_FUNCTION
500  void join( volatile value_type dst, const volatile value_type src ) const
501  {
502  for ( size_type j = 0; j < value_count; ++j )
503  dst[j] += src[j];
504  }
505 
506  KOKKOS_INLINE_FUNCTION void init( value_type sum ) const
507  {
508  for ( size_type j = 0; j < value_count; ++j )
509  sum[j] = 0.0;
510  }
511 };

548 // Infinity norm
549 template <class ViewType>
550 struct NormInfFunctor
551 {
552  typedef typename ViewType::value_type value_type[];
553  typedef typename ViewType::size_type size_type;
554  size_type value_count;
555  ViewType _view;
556 
557  NormInfFunctor( const ViewType& view )
558  : value_count( view.extent( 3 ) )
559  , _view( view )
560  {
561  }
562 
563  KOKKOS_INLINE_FUNCTION
564  void operator()( const size_type i, const size_type j, const size_type k,
565  const size_type l, value_type norm ) const
566  {
567  auto v_abs = fabs( _view( i, j, k, l ) );
568  if ( v_abs > norm[l] )
569  norm[l] = v_abs;
570  }
571 
572  KOKKOS_INLINE_FUNCTION
573  void join( volatile value_type dst, const volatile value_type src ) const
574  {
575  for ( size_type j = 0; j < value_count; ++j )
576  if ( src[j] > dst[j] )
577  dst[j] = src[j];
578  }
579 
580  KOKKOS_INLINE_FUNCTION void init( value_type norm ) const
581  {
582  for ( size_type j = 0; j < value_count; ++j )
583  norm[j] = 0.0;
584  }
