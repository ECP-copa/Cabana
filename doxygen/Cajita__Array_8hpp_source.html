223template <class Scalar, class EntityType, class MeshType, class... Params>
224class Array
225{
226 public:
228 using value_type = Scalar;
229
231 using entity_type = EntityType;
232
234 using mesh_type = MeshType;
235
237 static constexpr std::size_t num_space_dim = mesh_type::num_space_dim;
238
240 using array_layout = ArrayLayout<entity_type, mesh_type>;
241
243 using view_type = std::conditional_t<
244 3 == num_space_dim, Kokkos::View<value_type****, Params...>,
245 std::conditional_t<2 == num_space_dim,
246 Kokkos::View<value_type***, Params...>, void>>;
247
249 using device_type = typename view_type::device_type;
250
252 using memory_space = typename view_type::memory_space;
253
255 using execution_space = typename view_type::execution_space;
256
263 Array( const std::string& label,
264 const std::shared_ptr<array_layout>& layout )
265 : _layout( layout )
266 , _data( createView<value_type, Params...>(
267 label, layout->indexSpace( Ghost(), Local() ) ) )
268 {
269 }
270
277 Array( const std::shared_ptr<array_layout>& layout, const view_type& view )
278 : _layout( layout )
279 , _data( view )
280 {
281 for ( std::size_t d = 0; d < num_space_dim + 1; ++d )
282 if ( (long)view.extent( d ) !=
283 layout->indexSpace( Ghost(), Local() ).extent( d ) )
284 throw std::runtime_error(
285 "Layout and view dimensions do not match" );
286 }
287
289 std::shared_ptr<array_layout> layout() const { return _layout; }
290
292 view_type view() const { return _data; }
293
295 std::string label() const { return _data.label(); }
296
297 private:
298 std::shared_ptr<array_layout> _layout;
299 view_type _data;
300
301 public:
303 using subview_type = decltype( createSubview(
304 _data, _layout->indexSpace( Ghost(), Local() ) ) );
306 using subview_layout = typename subview_type::array_layout;
308 using subview_memory_traits = typename subview_type::memory_traits;
309};

350//---------------------------------------------------------------------------//
351// Subarray creation.
352//---------------------------------------------------------------------------//
360template <class Scalar, class EntityType, class MeshType, class... Params>
361std::shared_ptr<Array<
362 Scalar, EntityType, MeshType,
363 typename Array<Scalar, EntityType, MeshType, Params...>::subview_layout,
364 typename Array<Scalar, EntityType, MeshType, Params...>::device_type,
365 typename Array<Scalar, EntityType, MeshType,
366 Params...>::subview_memory_traits>>
367createSubarray( const Array<Scalar, EntityType, MeshType, Params...>& array,
368 const int dof_min, const int dof_max )
369{
370 if ( dof_min < 0 || dof_max > array.layout()->dofsPerEntity() )
371 throw std::logic_error( "Subarray dimensions out of bounds" );
372
373 auto space = array.layout()->indexSpace( Ghost(), Local() );
374 std::array<long, MeshType::num_space_dim + 1> min;
375 std::array<long, MeshType::num_space_dim + 1> max;
376 for ( std::size_t d = 0; d < MeshType::num_space_dim; ++d )
377 {
378 min[d] = space.min( d );
379 max[d] = space.max( d );
380 }
381 min.back() = dof_min;
382 max.back() = dof_max;
383 IndexSpace<MeshType::num_space_dim + 1> sub_space( min, max );
384 auto sub_view = createSubview( array.view(), sub_space );
385 auto sub_layout = createArrayLayout( array.layout()->localGrid(),
386 dof_max - dof_min, EntityType() );
387 return std::make_shared<Array<
388 Scalar, EntityType, MeshType,
389 typename Array<Scalar, EntityType, MeshType, Params...>::subview_layout,
390 typename Array<Scalar, EntityType, MeshType, Params...>::device_type,
391 typename Array<Scalar, EntityType, MeshType,
392 Params...>::subview_memory_traits>>( sub_layout,
393 sub_view );
394}

699struct DotFunctor
700{
702 static constexpr std::size_t num_space_dim = NumSpaceDim;
704 typedef typename ViewType::value_type value_type[];
706 typedef typename ViewType::size_type size_type;
708 size_type value_count;
710 ViewType _a;
712 ViewType _b;
713
715 DotFunctor( const ViewType& a, const ViewType& b )
716 : value_count( a.extent( NumSpaceDim ) )
717 , _a( a )
718 , _b( b )
719 {
720 }
721
723 template <std::size_t NSD = num_space_dim>
724 KOKKOS_INLINE_FUNCTION std::enable_if_t<3 == NSD, void>

736 {
737 sum[l] += _a( i, j, l ) * _b( i, j, l );
738 }
739
741 KOKKOS_INLINE_FUNCTION
742 void join( value_type dst, const value_type src ) const
743 {
744 for ( size_type j = 0; j < value_count; ++j )
745 dst[j] += src[j];
746 }
747
749 KOKKOS_INLINE_FUNCTION
750 void join( volatile value_type dst, const volatile value_type src ) const
751 {
752 for ( size_type j = 0; j < value_count; ++j )
753 dst[j] += src[j];
754 }

772template <class Array_t>
773void dot( const Array_t& a, const Array_t& b,
774 std::vector<typename Array_t::value_type>& products )
775{
776 static_assert( is_array<Array_t>::value, "Cajita::Array required" );
777 if ( products.size() !=
778 static_cast<unsigned>( a.layout()->dofsPerEntity() ) )
779 throw std::runtime_error( "Incorrect vector size" );
780
781 for ( auto& p : products )
782 p = 0.0;
783
784 DotFunctor<typename Array_t::view_type, Array_t::num_space_dim> functor(
785 a.view(), b.view() );
786 typename Array_t::execution_space exec_space;
787 Kokkos::parallel_reduce(
788 "ArrayOp::dot",
789 createExecutionPolicy( a.layout()->indexSpace( Own(), Local() ),
790 exec_space ),
791 functor,
792 Kokkos::View<typename Array_t::value_type*, Kokkos::HostSpace>(
793 products.data(), products.size() ) );
794 exec_space.fence( "ArrayOp::dot before MPI_Allreduce" );
795
796 MPI_Allreduce( MPI_IN_PLACE, products.data(), products.size(),
797 MpiTraits<typename Array_t::value_type>::type(), MPI_SUM,
798 a.layout()->localGrid()->globalGrid().comm() );
799}
