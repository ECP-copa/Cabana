299  view_type _data;

301  public:

360 template <class Scalar, class EntityType, class MeshType, class... Params>
361 std::shared_ptr<Array<
362  Scalar, EntityType, MeshType,
363  typename Array<Scalar, EntityType, MeshType, Params...>::subview_layout,
364  typename Array<Scalar, EntityType, MeshType, Params...>::device_type,
365  typename Array<Scalar, EntityType, MeshType,
366  Params...>::subview_memory_traits>>
367 createSubarray( const Array<Scalar, EntityType, MeshType, Params...>& array,
368  const int dof_min, const int dof_max )
369 {
370  if ( dof_min < 0 || dof_max > array.layout()->dofsPerEntity() )
371  throw std::logic_error( "Subarray dimensions out of bounds" );
372 
373  auto space = array.layout()->indexSpace( Ghost(), Local() );
374  std::array<long, MeshType::num_space_dim + 1> min;
375  std::array<long, MeshType::num_space_dim + 1> max;
376  for ( std::size_t d = 0; d < MeshType::num_space_dim; ++d )
377  {
378  min[d] = space.min( d );
379  max[d] = space.max( d );
380  }
381  min.back() = dof_min;
382  max.back() = dof_max;
383  IndexSpace<MeshType::num_space_dim + 1> sub_space( min, max );
384  auto sub_view = createSubview( array.view(), sub_space );
385  auto sub_layout = createArrayLayout( array.layout()->localGrid(),
386  dof_max - dof_min, EntityType() );
387  return std::make_shared<Array<
388  Scalar, EntityType, MeshType,
389  typename Array<Scalar, EntityType, MeshType, Params...>::subview_layout,
390  typename Array<Scalar, EntityType, MeshType, Params...>::device_type,
391  typename Array<Scalar, EntityType, MeshType,
392  Params...>::subview_memory_traits>>( sub_layout,
393  sub_view );
394 }

724  KOKKOS_INLINE_FUNCTION std::enable_if_t<3 == NSD, void>

736  {

738  }

742  void join( value_type dst, const value_type src ) const
743  {

789  typename Array_t::execution_space() ),

793  MpiTraits<typename Array_t::value_type>::type(), MPI_SUM,
