87// Rank-0 field
88template <class SliceType>
89void writeFields(
90 DBfile* silo_file, const std::string& mesh_name, const std::size_t begin,
91 const std::size_t end, const SliceType& slice,
92
93 typename std::enable_if<
94 2 == SliceType::kokkos_view::traits::dimension::rank, int*>::type = 0 )
95{
96 // Reorder in a contiguous blocked format.
97 Kokkos::View<typename SliceType::value_type*,
98 typename SliceType::device_type>
99 view( Kokkos::ViewAllocateWithoutInitializing( "scalar_field" ),
100 end - begin );
101 Kokkos::parallel_for(
102 "Cabana::SiloParticleOutput::writeFieldRank0",
103 Kokkos::RangePolicy<typename SliceType::execution_space>( begin, end ),
104 KOKKOS_LAMBDA( const int i ) { view( i - begin ) = slice( i ); } );
105
106 // Mirror the field to the host.
107 auto host_view =
108 Kokkos::create_mirror_view_and_copy( Kokkos::HostSpace(), view );
109
110 // Write the field.
111 DBPutPointvar1( silo_file, slice.label().c_str(), mesh_name.c_str(),
112 host_view.data(), host_view.extent( 0 ),
113 SiloTraits<typename SliceType::value_type>::type(),
114 nullptr );
115}

117// Rank-1 field
118template <class SliceType>
119void writeFields(
120 DBfile* silo_file, const std::string& mesh_name, const std::size_t begin,
121 const std::size_t end, const SliceType& slice,
122 typename std::enable_if<
123 3 == SliceType::kokkos_view::traits::dimension::rank, int*>::type = 0 )
124{
125 // Reorder in a contiguous blocked format.
126 Kokkos::View<typename SliceType::value_type**, Kokkos::LayoutLeft,
127 typename SliceType::device_type>
128 view( Kokkos::ViewAllocateWithoutInitializing( "vector_field" ),
129 end - begin, slice.extent( 2 ) );
130 Kokkos::parallel_for(
131 "Cabana::SiloParticleOutput::writeFieldRank1",
132 Kokkos::RangePolicy<typename SliceType::execution_space>( begin, end ),
133 KOKKOS_LAMBDA( const int i ) {
134 for ( std::size_t d0 = 0; d0 < slice.extent( 2 ); ++d0 )
135 view( i - begin, d0 ) = slice( i, d0 );
136 } );
137
138 // Mirror the field to the host.
139 auto host_view =
140 Kokkos::create_mirror_view_and_copy( Kokkos::HostSpace(), view );
141
142 // Get the data pointers per dimension.
143 std::vector<typename SliceType::value_type*> ptrs( host_view.extent( 1 ) );
144 for ( std::size_t d0 = 0; d0 < host_view.extent( 1 ); ++d0 )
145 ptrs[d0] = Kokkos::subview( host_view, Kokkos::ALL(), d0 ).data();
146
147 // Write the field.
148 DBPutPointvar( silo_file, slice.label().c_str(), mesh_name.c_str(),
149 host_view.extent( 1 ), ptrs.data(), host_view.extent( 0 ),
150 SiloTraits<typename SliceType::value_type>::type(),
151 nullptr );
152}

154// Rank-2 field
155template <class SliceType>
156void writeFields(
157 DBfile* silo_file, const std::string& mesh_name, const std::size_t begin,
158 const std::size_t end, const SliceType& slice,
159 typename std::enable_if<
160 4 == SliceType::kokkos_view::traits::dimension::rank, int*>::type = 0 )
161{
162 // Reorder in a contiguous blocked format.
163 Kokkos::View<typename SliceType::value_type***, Kokkos::LayoutLeft,
164 typename SliceType::device_type>
165 view( Kokkos::ViewAllocateWithoutInitializing( "matrix_field" ),
166 end - begin, slice.extent( 2 ), slice.extent( 3 ) );
167 Kokkos::parallel_for(
168 "Cabana::SiloParticleOutput::writeFieldRank2",
169 Kokkos::RangePolicy<typename SliceType::execution_space>( begin, end ),
170 KOKKOS_LAMBDA( const int i ) {
171 for ( std::size_t d0 = 0; d0 < slice.extent( 2 ); ++d0 )
172 for ( std::size_t d1 = 0; d1 < slice.extent( 3 ); ++d1 )
173 view( i - begin, d0, d1 ) = slice( i, d0, d1 );
174 } );
175
176 // Mirror the field to the host.
177 auto host_view =
178 Kokkos::create_mirror_view_and_copy( Kokkos::HostSpace(), view );
179
180 // Get the data pointers per dimension.
181 std::vector<typename SliceType::value_type*> ptrs;
182 ptrs.reserve( host_view.extent( 1 ) * host_view.extent( 2 ) );
183 for ( unsigned d0 = 0; d0 < host_view.extent( 1 ); ++d0 )
184 for ( unsigned d1 = 0; d1 < host_view.extent( 2 ); ++d1 )
185 ptrs.push_back(
186 Kokkos::subview( host_view, Kokkos::ALL(), d0, d1 ).data() );
187
188 // Write the field.
189 DBPutPointvar( silo_file, slice.label().c_str(), mesh_name.c_str(),
190 host_view.extent( 1 ) * host_view.extent( 2 ), ptrs.data(),
191 host_view.extent( 0 ),
192 SiloTraits<typename SliceType::value_type>::type(),
193 nullptr );
194}

300template <class... FieldSliceTypes>
301void writeMultiMesh( PMPIO_baton_t* baton, DBfile* silo_file,
302 const int comm_size, const std::string& prefix,
303 const std::string& mesh_name, const int time_step_index,
304 const double time, FieldSliceTypes&&... fields )
305{
306 // Go to the root directory of the file.
307 DBSetDir( silo_file, "/" );
308
309 // Create the mesh block names.
310 std::vector<std::string> mb_names;
311 for ( int r = 0; r < comm_size; ++r )
312 {
313 int group_rank = PMPIO_GroupRank( baton, r );
314 if ( 0 == group_rank )
315 {
316 std::stringstream bname;
317 bname << "rank_" << r << "/" << mesh_name;
318 mb_names.push_back( bname.str() );
319 }
320 else
321 {
322 std::stringstream bname;
323 bname << prefix << "_" << time_step_index << "_group_" << group_rank
324 << ".silo:/rank_" << r << "/" << mesh_name;
325 mb_names.push_back( bname.str() );
326 }
327 }
328 char** mesh_block_names = (char**)malloc( comm_size * sizeof( char* ) );
329 for ( int r = 0; r < comm_size; ++r )
330 mesh_block_names[r] = const_cast<char*>( mb_names[r].c_str() );
331
332 std::vector<int> mesh_block_types( comm_size, DB_POINTMESH );
333
334 // Get the names of the fields.
335 std::vector<std::string> field_names = getFieldNames( fields... );
336
337 // Create the field block names.
338 int num_field = field_names.size();
339 std::vector<std::vector<std::string>> fb_names( num_field );
340 for ( int f = 0; f < num_field; ++f )
341 {
342 for ( int r = 0; r < comm_size; ++r )
343 {
344 int group_rank = PMPIO_GroupRank( baton, r );
345 if ( 0 == group_rank )
346 {
347 std::stringstream bname;
348 bname << "rank_" << r << "/" << field_names[f];
349 fb_names[f].push_back( bname.str() );
350 }
351 else
352 {
353 std::stringstream bname;
354 bname << prefix << "_" << time_step_index << "_group_"
355 << group_rank << ".silo:/rank_" << r << "/"
356 << field_names[f];
357 fb_names[f].push_back( bname.str() );
358 }
359 }
360 }
361
362 std::vector<char**> field_block_names( num_field );
363 for ( int f = 0; f < num_field; ++f )
364 {
365 field_block_names[f] = (char**)malloc( comm_size * sizeof( char* ) );
366 for ( int r = 0; r < comm_size; ++r )
367 field_block_names[f][r] =
368 const_cast<char*>( fb_names[f][r].c_str() );
369 }
370
371 std::vector<int> field_block_types( comm_size, DB_POINTVAR );
372
373 // Create options.
374 DBoptlist* options = DBMakeOptlist( 1 );
375 DBAddOption( options, DBOPT_DTIME, (void*)&time );
376 DBAddOption( options, DBOPT_CYCLE, (void*)&time_step_index );
377
378 // Add the multiblock mesh.
379 std::stringstream mbname;
380 mbname << "multi_" << mesh_name;
381 DBPutMultimesh( silo_file, mbname.str().c_str(), comm_size,
382 mesh_block_names, mesh_block_types.data(), options );
383
384 // Add the multiblock fields.
385 for ( int f = 0; f < num_field; ++f )
386 {
387 std::stringstream mfname;
388 mfname << "multi_" << field_names[f];
389 DBPutMultivar( silo_file, mfname.str().c_str(), comm_size,
390 field_block_names[f], field_block_types.data(),
391 options );
392 }
393
394 // Cleanup.
395 free( mesh_block_names );
396 for ( auto& f_name : field_block_names )
397 free( f_name );
398 DBFreeOptlist( options );
399}

414template <class CoordSliceType, class... FieldSliceTypes>
415void writePartialRangeTimeStep( const std::string& prefix, MPI_Comm comm,
416 const int num_group, const int time_step_index,
417 const double time, const std::size_t begin,
418 const std::size_t end,
419 const CoordSliceType& coords,
420 FieldSliceTypes&&... fields )
421{
422 // Create the parallel baton.
423 int mpi_tag = 1948;
424 PMPIO_baton_t* baton =
425 PMPIO_Init( num_group, PMPIO_WRITE, comm, mpi_tag, createFile, openFile,
426 closeFile, nullptr );
427
428 // Allow empty.
429 DBSetAllowEmptyObjects( 1 );
430
431 // Compose a data file name.
432 int comm_rank;
433 MPI_Comm_rank( comm, &comm_rank );
434 int group_rank = PMPIO_GroupRank( baton, comm_rank );
435 std::stringstream file_name;
436
437 // Group 0 writes a master file for the time step.
438 if ( 0 == group_rank )
439 file_name << prefix << "_" << time_step_index << ".silo";
440 // The other groups write auxiliary files.
441 else
442 file_name << prefix << "_" << time_step_index << "_group_" << group_rank
443 << ".silo";
444
445 // Compose a directory name.
446 std::stringstream dir_name;
447 dir_name << "rank_" << comm_rank;
448
449 // Wait for our turn to write to the file.
450 DBfile* silo_file = (DBfile*)PMPIO_WaitForBaton(
451 baton, file_name.str().c_str(), dir_name.str().c_str() );
452
453 // Reorder the coordinates in a blocked format.
454 Kokkos::View<typename CoordSliceType::value_type**, Kokkos::LayoutLeft,
455 typename CoordSliceType::device_type>
456 view( Kokkos::ViewAllocateWithoutInitializing( "coords" ), end - begin,
457 coords.extent( 2 ) );
458 Kokkos::parallel_for(
459 "Cabana::SiloParticleOutput::writeCoords",
460 Kokkos::RangePolicy<typename CoordSliceType::execution_space>( begin,
461 end ),
462 KOKKOS_LAMBDA( const int i ) {
463 for ( std::size_t d0 = 0; d0 < coords.extent( 2 ); ++d0 )
464 view( i - begin, d0 ) = coords( i, d0 );
465 } );
466
467 // Mirror the coordinates to the host.
468 auto host_coords =
469 Kokkos::create_mirror_view_and_copy( Kokkos::HostSpace(), view );
470
471 // Add the point mesh.
472 std::string mesh_name = prefix;
473 auto host_x = Kokkos::subview( host_coords, Kokkos::ALL(), 0 );
474 auto host_y = Kokkos::subview( host_coords, Kokkos::ALL(), 1 );
475 auto host_z = Kokkos::subview( host_coords, Kokkos::ALL(), 2 );
476 typename CoordSliceType::value_type* ptrs[3] = {
477 host_x.data(),
478 host_y.data(),
479 host_z.data(),
480 };
481 DBPutPointmesh( silo_file, mesh_name.c_str(), host_coords.extent( 1 ), ptrs,
482 host_coords.extent( 0 ),
483 SiloTraits<typename CoordSliceType::value_type>::type(),
484 nullptr );
485
486 // Add variables.
487 writeFields( silo_file, mesh_name, begin, end, fields... );
488
489 // Root rank writes the global multimesh hierarchy for parallel
490 // simulations.
491 int comm_size;
492 MPI_Comm_size( comm, &comm_size );
493 if ( 0 == comm_rank && comm_size > 1 )
494 writeMultiMesh( baton, silo_file, comm_size, prefix, mesh_name,
495 time_step_index, time, fields... );
496
497 // Hand off the baton.
498 PMPIO_HandOffBaton( baton, silo_file );
499
500 // Finish.
501 PMPIO_Finish( baton );
502}
