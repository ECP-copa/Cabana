87 // Rank-0 field
88 template <class SliceType>
89 void writeFields(
90  DBfile* silo_file, const std::string& mesh_name, const std::size_t begin,
91  const std::size_t end, const SliceType& slice,
92 
93  typename std::enable_if<
94  2 == SliceType::kokkos_view::traits::dimension::rank, int*>::type = 0 )
95 {
96  // Reorder in a contiguous blocked format.
97  Kokkos::View<typename SliceType::value_type*,
98  typename SliceType::device_type>
99  view( "field", end - begin );
100  Kokkos::parallel_for(
101  "Cabana::SiloParticleOutput::writeFieldRank0",
102  Kokkos::RangePolicy<typename SliceType::execution_space>( begin, end ),
103  KOKKOS_LAMBDA( const int i ) { view( i - begin ) = slice( i ); } );
104 
105  // Mirror the field to the host.
106  auto host_view =
107  Kokkos::create_mirror_view_and_copy( Kokkos::HostSpace(), view );
108 
109  // Write the field.
110  DBPutPointvar1( silo_file, slice.label().c_str(), mesh_name.c_str(),
111  host_view.data(), host_view.extent( 0 ),
112  SiloTraits<typename SliceType::value_type>::type(),
113  nullptr );
114 }

116 // Rank-1 field
117 template <class SliceType>
118 void writeFields(
119  DBfile* silo_file, const std::string& mesh_name, const std::size_t begin,
120  const std::size_t end, const SliceType& slice,
121  typename std::enable_if<
122  3 == SliceType::kokkos_view::traits::dimension::rank, int*>::type = 0 )
123 {
124  // Reorder in a contiguous blocked format.
125  Kokkos::View<typename SliceType::value_type**, Kokkos::LayoutLeft,
126  typename SliceType::device_type>
127  view( "field", end - begin, slice.extent( 2 ) );
128  Kokkos::parallel_for(
129  "Cabana::SiloParticleOutput::writeFieldRank1",
130  Kokkos::RangePolicy<typename SliceType::execution_space>( begin, end ),
131  KOKKOS_LAMBDA( const int i ) {
132  for ( std::size_t d0 = 0; d0 < slice.extent( 2 ); ++d0 )
133  view( i - begin, d0 ) = slice( i, d0 );
134  } );
135 
136  // Mirror the field to the host.
137  auto host_view =
138  Kokkos::create_mirror_view_and_copy( Kokkos::HostSpace(), view );
139 
140  // Get the data pointers.
141  std::vector<typename SliceType::value_type*> ptrs( host_view.extent( 1 ) );
142  for ( std::size_t d0 = 0; d0 < host_view.extent( 1 ); ++d0 )
143  ptrs[d0] = &host_view( 0, d0 );
144 
145  // Write the field.
146  DBPutPointvar( silo_file, slice.label().c_str(), mesh_name.c_str(),
147  host_view.extent( 1 ), ptrs.data(), host_view.extent( 0 ),
148  SiloTraits<typename SliceType::value_type>::type(),
149  nullptr );
150 }

152 // Rank-2 field
153 template <class SliceType>
154 void writeFields(
155  DBfile* silo_file, const std::string& mesh_name, const std::size_t begin,
156  const std::size_t end, const SliceType& slice,
157  typename std::enable_if<
158  4 == SliceType::kokkos_view::traits::dimension::rank, int*>::type = 0 )
159 {
160  // Reorder in a contiguous blocked format.
161  Kokkos::View<typename SliceType::value_type***, Kokkos::LayoutLeft,
162  typename SliceType::device_type>
163  view( "field", end - begin, slice.extent( 2 ), slice.extent( 3 ) );
164  Kokkos::parallel_for(
165  "Cabana::SiloParticleOutput::writeFieldRank2",
166  Kokkos::RangePolicy<typename SliceType::execution_space>( begin, end ),
167  KOKKOS_LAMBDA( const int i ) {
168  for ( std::size_t d0 = 0; d0 < slice.extent( 2 ); ++d0 )
169  for ( std::size_t d1 = 0; d1 < slice.extent( 3 ); ++d1 )
170  view( i - begin, d0, d1 ) = slice( i, d0, d1 );
171  } );
172 
173  // Mirror the field to the host.
174  auto host_view =
175  Kokkos::create_mirror_view_and_copy( Kokkos::HostSpace(), view );
176 
177  // Get the data pointers.
178  std::vector<typename SliceType::value_type*> ptrs;
179  ptrs.reserve( host_view.extent( 1 ) * host_view.extent( 2 ) );
180  for ( unsigned d0 = 0; d0 < host_view.extent( 1 ); ++d0 )
181  for ( unsigned d1 = 0; d1 < host_view.extent( 2 ); ++d1 )
182  ptrs.push_back( &host_view( 0, d0, d1 ) );
183 
184  // Write the field.
185  DBPutPointvar( silo_file, slice.label().c_str(), mesh_name.c_str(),
186  host_view.extent( 1 ) * host_view.extent( 2 ), ptrs.data(),
187  host_view.extent( 0 ),
188  SiloTraits<typename SliceType::value_type>::type(),
189  nullptr );
190 }

296 template <class... FieldSliceTypes>
297 void writeMultiMesh( PMPIO_baton_t* baton, DBfile* silo_file,
298  const int comm_size, const std::string& mesh_name,
299  const int time_step_index, const double time,
300  FieldSliceTypes&&... fields )
301 {
302  // Go to the root directory of the file.
303  DBSetDir( silo_file, "/" );
304 
305  // Create the mesh block names.
306  std::vector<std::string> mb_names;
307  for ( int r = 0; r < comm_size; ++r )
308  {
309  int group_rank = PMPIO_GroupRank( baton, r );
310  if ( 0 == group_rank )
311  {
312  std::stringstream bname;
313  bname << "rank_" << r << "/" << mesh_name;
314  mb_names.push_back( bname.str() );
315  }
316  else
317  {
318  std::stringstream bname;
319  bname << "particles_" << time_step_index << "_group_" << group_rank
320  << ".silo:/rank_" << r << "/" << mesh_name;
321  mb_names.push_back( bname.str() );
322  }
323  }
324  char** mesh_block_names = (char**)malloc( comm_size * sizeof( char* ) );
325  for ( int r = 0; r < comm_size; ++r )
326  mesh_block_names[r] = const_cast<char*>( mb_names[r].c_str() );
327 
328  std::vector<int> mesh_block_types( comm_size, DB_POINTMESH );
329 
330  // Get the names of the fields.
331  std::vector<std::string> field_names = getFieldNames( fields... );
332 
333  // Create the field block names.
334  int num_field = field_names.size();
335  std::vector<std::vector<std::string>> fb_names( num_field );
336  for ( int f = 0; f < num_field; ++f )
337  {
338  for ( int r = 0; r < comm_size; ++r )
339  {
340  int group_rank = PMPIO_GroupRank( baton, r );
341  if ( 0 == group_rank )
342  {
343  std::stringstream bname;
344  bname << "rank_" << r << "/" << field_names[f];
345  fb_names[f].push_back( bname.str() );
346  }
347  else
348  {
349  std::stringstream bname;
350  bname << "particles_" << time_step_index << "_group_"
351  << group_rank << ".silo:/rank_" << r << "/"
352  << field_names[f];
353  fb_names[f].push_back( bname.str() );
354  }
355  }
356  }
357 
358  std::vector<char**> field_block_names( num_field );
359  for ( int f = 0; f < num_field; ++f )
360  {
361  field_block_names[f] = (char**)malloc( comm_size * sizeof( char* ) );
362  for ( int r = 0; r < comm_size; ++r )
363  field_block_names[f][r] =
364  const_cast<char*>( fb_names[f][r].c_str() );
365  }
366 
367  std::vector<int> field_block_types( comm_size, DB_POINTVAR );
368 
369  // Create options.
370  DBoptlist* options = DBMakeOptlist( 1 );
371  DBAddOption( options, DBOPT_DTIME, (void*)&time );
372  DBAddOption( options, DBOPT_CYCLE, (void*)&time_step_index );
373 
374  // Add the multiblock mesh.
375  std::stringstream mbname;
376  mbname << "multi_" << mesh_name;
377  DBPutMultimesh( silo_file, mbname.str().c_str(), comm_size,
378  mesh_block_names, mesh_block_types.data(), options );
379 
380  // Add the multiblock fields.
381  for ( int f = 0; f < num_field; ++f )
382  {
383  std::stringstream mfname;
384  mfname << "multi_" << field_names[f];
385  DBPutMultivar( silo_file, mfname.str().c_str(), comm_size,
386  field_block_names[f], field_block_types.data(),
387  options );
388  }
389 
390  // Cleanup.
391  free( mesh_block_names );
392  for ( auto& f_name : field_block_names )
393  free( f_name );
394  DBFreeOptlist( options );
395 }

410 template <class CoordSliceType, class... FieldSliceTypes>
411 void writePartialRangeTimeStep( const std::string& prefix, MPI_Comm comm,
412  const int num_group, const int time_step_index,
413  const double time, const std::size_t begin,
414  const std::size_t end,
415  const CoordSliceType& coords,
416  FieldSliceTypes&&... fields )
417 {
418  // Create the parallel baton.
419  int mpi_tag = 1948;
420  PMPIO_baton_t* baton =
421  PMPIO_Init( num_group, PMPIO_WRITE, comm, mpi_tag, createFile, openFile,
422  closeFile, nullptr );
423 
424  // Compose a data file name.
425  int comm_rank;
426  MPI_Comm_rank( comm, &comm_rank );
427  int group_rank = PMPIO_GroupRank( baton, comm_rank );
428  std::stringstream file_name;
429 
430  // Group 0 writes a master file for the time step.
431  if ( 0 == group_rank )
432  file_name << prefix << "_" << time_step_index << ".silo";
433  // The other groups write auxiliary files.
434  else
435  file_name << prefix << "_" << time_step_index << "_group_" << group_rank
436  << ".silo";
437 
438  // Compose a directory name.
439  std::stringstream dir_name;
440  dir_name << "rank_" << comm_rank;
441 
442  // Wait for our turn to write to the file.
443  DBfile* silo_file = (DBfile*)PMPIO_WaitForBaton(
444  baton, file_name.str().c_str(), dir_name.str().c_str() );
445 
446  // Reorder the coordinates in a blocked format.
447  Kokkos::View<typename CoordSliceType::value_type**, Kokkos::LayoutLeft,
448  typename CoordSliceType::device_type>
449  view( "coords", end - begin, coords.extent( 2 ) );
450  Kokkos::parallel_for(
451  "Cabana::SiloParticleOutput::writeCoords",
452  Kokkos::RangePolicy<typename CoordSliceType::execution_space>( begin,
453  end ),
454  KOKKOS_LAMBDA( const int i ) {
455  for ( std::size_t d0 = 0; d0 < coords.extent( 2 ); ++d0 )
456  view( i - begin, d0 ) = coords( i, d0 );
457  } );
458 
459  // Mirror the coordinates to the host.
460  auto host_coords =
461  Kokkos::create_mirror_view_and_copy( Kokkos::HostSpace(), view );
462 
463  // Add the point mesh.
464  std::string mesh_name = "particles";
465  double* ptrs[3] = { &host_coords( 0, 0 ), &host_coords( 0, 1 ),
466  &host_coords( 0, 2 ) };
467  DBPutPointmesh( silo_file, mesh_name.c_str(), host_coords.extent( 1 ), ptrs,
468  host_coords.extent( 0 ),
469  SiloTraits<typename CoordSliceType::value_type>::type(),
470  nullptr );
471 
472  // Add variables.
473  writeFields( silo_file, mesh_name, begin, end, fields... );
474 
475  // Root rank writes the global multimesh hierarchy for parallel
476  // simulations.
477  int comm_size;
478  MPI_Comm_size( comm, &comm_size );
479  if ( 0 == comm_rank && comm_size > 1 )
480  writeMultiMesh( baton, silo_file, comm_size, mesh_name, time_step_index,
481  time, fields... );
482 
483  // Hand off the baton.
484  PMPIO_HandOffBaton( baton, silo_file );
485 
486  // Finish.
487  PMPIO_Finish( baton );
488 }
